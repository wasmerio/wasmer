Creates a new uninitialized vector of [wasm_importtype_t *].

int main() {
    // Creates an empty vector of `wasm_importtype_t *`.
    wasm_importtype_vec_t vector;
    wasm_importtype_vec_new_uninitialized(&vector, 3);

    // Check that it contains 3 items.
    assert(vector.size == 3);

    // Free it.
    wasm_importtype_vec_delete(&vector);
