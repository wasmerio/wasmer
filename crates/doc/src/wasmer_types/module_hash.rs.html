20/// The hash of a WebAssembly module.
21#[derive(
22    Debug,
23    Copy,

44#[cfg(feature = "artifact-size")]
45impl loupe::MemoryUsage for ModuleHash {
46    fn size_of_val(&self, _tracker: &mut dyn loupe::MemoryUsageTracker) -> usize {
47        match self {
48            Self::XXHash(_) => 8 * 8,
49            Self::Sha256(_) => 8 * 32,
50        }
51    }
52}
53
54impl ModuleHash {
55    /// Create a new [`ModuleHash`] from the raw xxhash hash.
56    pub fn xxhash_from_bytes(key: [u8; 8]) -> Self {
57        Self::XXHash(key)
58    }
59
60    /// Create a new [`ModuleHash`] from the raw sha256 hash.
61    pub fn sha256_from_bytes(key: [u8; 32]) -> Self {
62        Self::Sha256(key)
63    }
64
65    /// Parse a XXHash hash from a hex-encoded string.
66    pub fn xxhash_parse_hex(hex_str: &str) -> Result<Self, hex::FromHexError> {
67        let mut hash = [0_u8; 8];
68        hex::decode_to_slice(hex_str, &mut hash)?;
69        Ok(Self::xxhash_from_bytes(hash))
70    }
71
72    /// Parse a Sha256 hash from a hex-encoded string.
73    pub fn sha256_parse_hex(hex_str: &str) -> Result<Self, hex::FromHexError> {
74        let mut hash = [0_u8; 32];
75        hex::decode_to_slice(hex_str, &mut hash)?;
76        Ok(Self::sha256_from_bytes(hash))
77    }
78
79    /// Generate a new [`ModuleHash`] based on the XXHash hash of some bytes.
80    pub fn xxhash(wasm: impl AsRef<[u8]>) -> Self {
81        let wasm = wasm.as_ref();
82
83        let hash = xxhash_rust::xxh64::xxh64(wasm, 0);
84
85        Self::XXHash(hash.to_ne_bytes())
86    }
87
88    /// Generate a new [`ModuleHash`] based on the Sha256 hash of some bytes.
89    pub fn sha256(wasm: impl AsRef<[u8]>) -> Self {
90        let wasm = wasm.as_ref();
91
92        let hash: [u8; 32] = sha2::Sha256::digest(wasm).into();
93
94        Self::Sha256(hash)
95    }
96
97    /// Generate a random [`ModuleHash`]. For when you don't care about caches.
98    pub fn random() -> Self {
99        let mut bytes = [0_u8; 8];
100        getrandom::getrandom(&mut bytes).unwrap();
101        Self::XXHash(bytes)
102    }
103
104    /// Get the raw hash.
105    pub fn as_bytes(&self) -> &[u8] {
106        match self {
107            Self::XXHash(bytes) => bytes.as_slice(),
108            Self::Sha256(bytes) => bytes.as_slice(),
109        }
110    }
111
112    /// Build a short hex representation of the hash (first 4 bytes).
113    pub fn short_hash(&self) -> String {
114        hex::encode_upper(&self.as_bytes()[..4])
115    }
116}

118impl Display for ModuleHash {
119    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
120        fn format<const N: usize>(f: &mut Formatter<'_>, bytes: &[u8; N]) -> fmt::Result {
121            write!(f, "{}", hex::encode_upper(bytes))
122        }
123
124        match self {
