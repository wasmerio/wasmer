39impl Display for WasmCAPIVersion {

55/// Returns the host-provided WebAssembly C API version imported by a module.
56pub fn module_wasm_c_api_version_used(module: &Module) -> Option<WasmCAPIVersion> {
57    let mut version = None;
58
59    for import in module.imports() {
60        let Some(detected_version) = wasm_c_api_version_from_namespace(import.module()) else {
61            continue;
62        };
63
64        version = Some(match version {
65            None => detected_version,
66            Some(existing) if existing == detected_version => existing,
67            Some(_) => WasmCAPIVersion::Unknown,
68        });
69    }
70
71    version
72}

74fn wasm_c_api_version_from_namespace(namespace: &str) -> Option<WasmCAPIVersion> {
75    if namespace == WASM_C_API_MODULE_NAME {
76        return Some(WasmCAPIVersion::V0);
77    }
78
79    let suffix = namespace.strip_prefix(WASM_C_API_MODULE_PREFIX)?;
80    Some(match suffix {
81        "0" => WasmCAPIVersion::V0,
82        _ => WasmCAPIVersion::Unknown,
83    })
84}

88struct WasmCapiSession {
89    version: Option<WasmCAPIVersion>,
90    imported_memory_type: Option<MemoryType>,
91    imported_table_type: Option<wasmer_api::TableType>,

93    func_env: Mutex<Option<FunctionEnv<WasmCapiEnv>>>,

96impl WasmCapiSession {
97    fn new(module: &Module, resolve_module_sync: Option<ResolveModuleSync>) -> Self {
98        let imported_memory_type = module.imports().find_map(|import| {
99            if import.module() == "env"
100                && import.name() == "memory"
101                && let ExternType::Memory(ty) = import.ty()
102            {
103                return Some(*ty);
104            }
105            None
106        });
107
108        let imported_table_type = module.imports().find_map(|import| {
109            if import.module() == "env"
110                && import.name() == "__indirect_function_table"
111                && let ExternType::Table(ty) = import.ty()
112            {
113                return Some(*ty);
114            }
115            None
116        });
117
118        Self {
119            version: module_wasm_c_api_version_used(module),
120            imported_memory_type,
121            imported_table_type,
122            resolve_module_sync,
123            func_env: Mutex::new(None),
124        }
125    }
126
127    fn needs_imports(&self) -> bool {
128        self.version.is_some()
129    }
130
131    fn validate_supported(&self) -> Result<()> {
132        if let Some(version) = self.version
133            && !WasmCAPIVersion::V0.is_compatible_with(version)
134        {
135            bail!("unsupported Wasm C API import version: {version:?}");
136        }
137        Ok(())
138    }
139
140    fn register_imports(&self, store: &mut StoreMut<'_>, io: &mut Imports) -> Result<()> {
141        self.validate_supported()?;
142        if self.version.is_none() {
143            return Ok(());
144        }
145
146        let func_env = FunctionEnv::new(
147            &mut *store,
148            WasmCapiEnv {
149                resolve_module_sync: self.resolve_module_sync.clone(),
150                ..WasmCapiEnv::default()
151            },
152        );
153        *self
154            .func_env
155            .lock()
156            .expect("poisoned WasmCapiSession mutex") = Some(func_env.clone());
157        register_wasm_c_api_imports(store, &func_env, io);
158
159        if let Some(memory_type) = self.imported_memory_type {
160            if let Some(existing) = io.get_export("env", "memory") {
161                let Extern::Memory(memory) = existing else {
162                    bail!("env.memory import for Wasm C API module is not a memory");
163                };
164                self.set_memory(store, &memory)?;
165            } else {
166                let memory = Memory::new(&mut *store, memory_type)?;
167                io.define("env", "memory", memory.clone());
168                self.set_memory(store, &memory)?;
169            }
170        }
171
172        if let Some(table_type) = self.imported_table_type {
173            if let Some(existing) = io.get_export("env", "__indirect_function_table") {
174                let Extern::Table(table) = existing else {
175                    bail!(
176                        "env.__indirect_function_table import for Wasm C API module is not a table"
177                    );
178                };
179                self.set_table(store, &table)?;
180            } else {
181                let table = Table::new(&mut *store, table_type, Value::FuncRef(None))?;
182                io.define("env", "__indirect_function_table", table.clone());
183                self.set_table(store, &table)?;
184            }
185        }
186
187        Ok(())
188    }
189
190    fn set_memory(&self, store: &mut StoreMut<'_>, memory: &Memory) -> Result<()> {
191        let Some(func_env) = self.func_env()? else {
192            return Ok(());
193        };
194        func_env.as_mut(&mut *store).memory = Some(memory.clone());
195        Ok(())
196    }
197
198    fn set_table(&self, store: &mut StoreMut<'_>, table: &Table) -> Result<()> {
199        let Some(func_env) = self.func_env()? else {
200            return Ok(());
201        };
202        func_env.as_mut(&mut *store).table = Some(table.clone());
203        Ok(())
204    }
205
206    fn configure_instance(
207        &self,
208        store: &mut StoreMut<'_>,
209        instance: &Instance,
210        imported_memory: Option<&Memory>,
211    ) -> Result<()> {
212        let Some(func_env) = self.func_env()? else {
213            return Ok(());
214        };
215
216        if let Some(memory) = imported_memory {
217            func_env.as_mut(&mut *store).memory = Some(memory.clone());
218        } else if let Ok(memory) = instance.exports.get_memory("memory") {
219            func_env.as_mut(&mut *store).memory = Some(memory.clone());
220        }
221
222        for export_name in ["unofficial_napi_guest_malloc", "malloc"] {
223            if let Ok(malloc) = instance
224                .exports
225                .get_typed_function::<i32, i32>(&store, export_name)
226            {
227                func_env.as_mut(&mut *store).malloc_fn = Some(malloc);
228                break;
229            }
230        }
231
232        for export_name in ["unofficial_napi_guest_free", "free"] {
233            if let Ok(free) = instance
234                .exports
235                .get_typed_function::<i32, ()>(&store, export_name)
236            {
237                func_env.as_mut(&mut *store).free_fn = Some(free);
238                break;
239            }
240        }
241
242        if let Ok(table) = instance.exports.get_table("__indirect_function_table") {
243            func_env.as_mut(&mut *store).table = Some(table.clone());
244        }
245
246        Ok(())
247    }
248
249    fn func_env(&self) -> Result<Option<FunctionEnv<WasmCapiEnv>>> {
250        if self.version.is_none() {
251            return Ok(None);
252        }
253
254        self.func_env
255            .lock()
256            .expect("poisoned WasmCapiSession mutex")
257            .clone()
258            .context("missing Wasm C API function env during instance setup")
259            .map(Some)
260    }
261}

272/// Runtime hooks that provide `wasm_c_api_v0` imports for WASIX guests.
273#[derive(Clone, Default)]
274pub struct WasmCapiRuntimeHooks {
275    /// Pending per-instantiation sessions, grouped by compiled module.
276    ///
277    /// `add_imports` creates host functions before instantiation, while
278    /// `configure_instance` can only discover exports like guest malloc/free
279    /// after instantiation. The queue preserves that pairing when the same
280    /// module is instantiated multiple times.
281    sessions: Arc<Mutex<HashMap<ModuleKey, VecDeque<WasmCapiSession>>>>,
282    resolve_module_sync: Option<ResolveModuleSync>,
283}
284
285impl WasmCapiRuntimeHooks {
286    /// Creates an empty hook set.
287    pub fn new() -> Self {
288        Self::default()
289    }
290
291    /// Uses an embedder-provided resolver for `wasm_module_new`.
292    pub fn with_resolve_module_sync(
293        mut self,
294        resolve: impl Fn(Vec<u8>) -> Result<Module> + Send + Sync + 'static,
295    ) -> Self {
296        self.resolve_module_sync = Some(Arc::new(resolve));
297        self
298    }
299
300    fn module_key(module: &Module) -> ModuleKey {
301        ModuleKey::new(module)
302    }
303
304    /// Creates `wasm_c_api_v0` imports when `module` requests them.
305    pub fn additional_imports(&self, module: &Module, store: &mut StoreMut<'_>) -> Result<Imports> {
306        let mut imports = Imports::new();
307        self.add_imports(module, store, &mut imports)?;
308        Ok(imports)
309    }
310
311    /// Merges `wasm_c_api_v0` imports into an existing import object when needed.
312    pub fn add_imports(
313        &self,
314        module: &Module,
315        store: &mut StoreMut<'_>,
316        imports: &mut Imports,
317    ) -> Result<()> {
318        let session = WasmCapiSession::new(module, self.resolve_module_sync.clone());
319        if !session.needs_imports() {
320            return Ok(());
321        }
322
323        session.register_imports(store, imports)?;
324        let mut sessions = self
325            .sessions
326            .lock()
327            .expect("poisoned WasmCapiRuntimeHooks session queue");
328        sessions
329            .entry(Self::module_key(module))
330            .or_default()
331            .push_back(session);
332        Ok(())
333    }
334
335    /// Completes memory, table, and guest allocation wiring after instantiation.
336    pub fn configure_instance(
337        &self,
338        module: &Module,
339        store: &mut StoreMut<'_>,
340        instance: &Instance,
341        imported_memory: Option<&Memory>,
342    ) -> Result<()> {
343        if module_wasm_c_api_version_used(module).is_none() {
344            return Ok(());
345        }
346
347        let session = {
348            let mut sessions = self
349                .sessions
350                .lock()
351                .expect("poisoned WasmCapiRuntimeHooks session queue");
352            let key = Self::module_key(module);
353            let Some(queue) = sessions.get_mut(&key) else {
354                bail!("missing pending Wasm C API session for module instance setup");
355            };
356            let session = queue
357                .pop_front()
358                .context("missing queued Wasm C API session for module instance setup")?;
359            if queue.is_empty() {
360                sessions.remove(&key);
361            }
362            session
363        };
364
365        session.configure_instance(store, instance, imported_memory)
366    }
367}

462/// Host-side storage for opaque WebAssembly C API handles.
463///
464/// The guest sees pointers from `wasm.h` as positive i32 handles. Handles are
465/// never reused during one instance lifetime, and invalid/unknown handles
466/// become lookup failures that callers map to the C API's null/false result.
467struct WasmCapiState {
468    /// Next positive i32 handle to expose to the guest.
469    next_handle: i32,
470    /// Opaque objects addressed by guest-visible handles.
471    objects: HashMap<i32, WasmObject>,
472    /// Guest shadow allocations for `wasm_memory_data`, keyed by memory handle.
473    memory_shadows: HashMap<i32, MemoryShadow>,
474}
475
476impl Default for WasmCapiState {
477    fn default() -> Self {
478        Self {
479            next_handle: 1,
480            objects: HashMap::new(),
481            memory_shadows: HashMap::new(),
482        }
483    }
484}
485
486impl WasmCapiState {
487    fn insert(&mut self, object: WasmObject) -> i32 {
488        let handle = self.next_handle;
489        if handle <= INVALID_HANDLE {
490            return INVALID_HANDLE;
491        }
492
493        self.next_handle = handle.checked_add(1).unwrap_or(INVALID_HANDLE);
494        self.objects.insert(handle, object);
495        handle
496    }
497
498    fn get(&self, handle: i32) -> Option<&WasmObject> {
499        if handle <= INVALID_HANDLE {
500            return None;
501        }
502        self.objects.get(&handle)
503    }
504
505    fn remove(&mut self, handle: i32) -> Option<MemoryShadow> {
506        if handle > INVALID_HANDLE {
507            self.objects.remove(&handle);
508            return self.memory_shadows.remove(&handle);
509        }
510        None
511    }
512}

533fn write_guest_bytes(env: &mut FunctionEnvMut<WasmCapiEnv>, guest_ptr: i32, data: &[u8]) -> bool {
534    let Some(ptr) = guest_byte_ptr(guest_ptr) else {
535        return false;
536    };

538        return false;
539    };

541    let Some(memory) = state.memory.clone() else {
542        return false;
543    };
544    let view = memory.view(&store);
545    ptr.slice(&view, len)
546        .and_then(|slice| slice.write_slice(data))
547        .is_ok()
548}

620fn read_guest_i32_vec(
621    env: &mut FunctionEnvMut<WasmCapiEnv>,
622    guest_ptr: i32,
623    len: usize,
624) -> Option<Vec<i32>> {
625    let offset = guest_memory_offset(guest_ptr)?;
626    if len == 0 {
627        return Some(Vec::new());
628    }
629    let byte_len = len.checked_mul(size_of::<i32>())?;
630
631    let (state, store) = env.data_and_store_mut();
632    let memory = state.memory.clone()?;
633    let view = memory.view(&store);
634
635    let mut out = Vec::<i32>::with_capacity(len);
636    unsafe {
637        let spare = out.spare_capacity_mut();
638        let bytes = slice::from_raw_parts_mut(spare.as_mut_ptr().cast::<u8>(), byte_len);
639        view.read(offset, bytes).ok()?;
640        out.set_len(len);
641    }
642
643    // Guest ABI vectors are little-endian, matching the write path's
644    // `to_le_bytes` encoding. This is a no-op on little-endian hosts.
645    for value in &mut out {
646        *value = i32::from_le(*value);
647    }
648
649    Some(out)
650}

652fn write_guest_i32_slice(
653    env: &mut FunctionEnvMut<WasmCapiEnv>,
654    guest_ptr: i32,
655    values: &[i32],
656) -> bool {
657    let Some(byte_len) = values.len().checked_mul(size_of::<i32>()) else {
658        return false;
659    };
660    if byte_len == 0 {
661        return true;
662    }
663
664    for (index, value) in values.iter().enumerate() {
665        let Ok(index) = i32::try_from(index) else {
666            return false;
667        };
668        let Some(offset) = index.checked_mul(size_of::<i32>() as i32) else {
669            return false;
670        };
671        if !write_guest_bytes_offset(env, guest_ptr, offset, &value.to_le_bytes()) {
672            return false;
673        }
674    }
675    true
676}

678fn allocate_guest_memory(env: &mut FunctionEnvMut<WasmCapiEnv>, len: usize) -> Option<i32> {
679    if len == 0 {
680        return Some(0);
681    }
682
683    let malloc_fn = env.data().malloc_fn.clone()?;
684    let len = i32::try_from(len).ok()?;
685    let guest_ptr: i32 = {
686        let (_, mut store_ref) = env.data_and_store_mut();
687        malloc_fn.call(&mut store_ref, len).ok()?
688    };
689    if guest_ptr <= INVALID_HANDLE {
690        return None;
691    }
692    Some(guest_ptr)
693}
694
695fn free_guest_memory(env: &mut FunctionEnvMut<WasmCapiEnv>, guest_ptr: i32) {
696    if guest_ptr <= INVALID_HANDLE {
697        return;
698    }
699
700    let Some(free_fn) = env.data().free_fn.clone() else {
701        return;
702    };
703    let (_, mut store_ref) = env.data_and_store_mut();
704    let _ = free_fn.call(&mut store_ref, guest_ptr);
705}

707struct GuestAllocation<'env, 'store> {
708    env: &'env mut FunctionEnvMut<'store, WasmCapiEnv>,
709    ptr: i32,
710}
711
712impl<'env, 'store> GuestAllocation<'env, 'store> {
713    fn new(env: &'env mut FunctionEnvMut<'store, WasmCapiEnv>, len: usize) -> Option<Self> {
714        let ptr = allocate_guest_memory(env, len)?;
715        non_null_guest_ptr(ptr)?;
716        Some(Self { env, ptr })
717    }
718
719    fn write_bytes(&mut self, data: &[u8]) -> bool {
720        write_guest_bytes(self.env, self.ptr, data)
721    }
722
723    fn write_i32_slice(&mut self, values: &[i32]) -> bool {
724        write_guest_i32_slice(self.env, self.ptr, values)
725    }
726
727    fn write_wasm_val_at(&mut self, offset: usize, value: &Value) -> bool {
728        let Ok(offset) = i32::try_from(offset) else {
729            return false;
730        };
731        let Some(ptr) = self.ptr.checked_add(offset) else {
732            return false;
733        };
734        write_wasm_val(self.env, ptr, value)
735    }
736
737    fn write_vec_header(&mut self, vec_ptr: NonZeroI32, len: u32) -> bool {
738        write_guest_vec_header(self.env, vec_ptr.get(), len, self.ptr)
739    }
740
741    fn write_self_vec_header(&mut self, len: u32, data_ptr: i32) -> bool {
742        write_guest_vec_header(self.env, self.ptr, len, data_ptr)
743    }
744
745    fn write_handles(&mut self, handles: &[i32]) -> bool {
746        write_handle_vec(self.env, self.ptr, handles)
747    }
748
749    fn allocate_vec_header(&mut self, len: u32) -> Option<i32> {
750        allocate_guest_vec_header(self.env, len, self.ptr)
751    }
752
753    fn copy_from_wasmer_memory(&mut self, memory: &Memory, len: usize) -> bool {
754        copy_wasmer_memory_to_guest(self.env, memory, self.ptr, len)
755    }
756
757    fn into_raw(mut self) -> i32 {
758        let ptr = self.ptr;
759        self.ptr = INVALID_HANDLE;
760        ptr
761    }
762}

767            free_guest_memory(self.env, self.ptr);

769    }

818fn write_byte_vec(env: &mut FunctionEnvMut<WasmCapiEnv>, vec_ptr: i32, bytes: &[u8]) -> bool {
819    let Some(vec_ptr) = non_null_guest_ptr(vec_ptr) else {
820        return false;
821    };
822    let Ok(len) = u32::try_from(bytes.len()) else {
823        return false;
824    };

826    if bytes.is_empty() {
827        return write_guest_vec_header(env, vec_ptr.get(), len, 0);
828    }
829
830    let Some(mut data) = GuestAllocation::new(env, bytes.len()) else {
831        return false;
832    };
833    if !data.write_bytes(bytes) || !data.write_vec_header(vec_ptr, len) {
834        return false;
835    }
836    let _ = data.into_raw();
837    true

840fn allocate_name(env: &mut FunctionEnvMut<WasmCapiEnv>, name: &str) -> i32 {

842        return INVALID_HANDLE;

853        return INVALID_HANDLE;
854    }

856        return INVALID_HANDLE;

862fn write_handle_vec(env: &mut FunctionEnvMut<WasmCapiEnv>, out_ptr: i32, handles: &[i32]) -> bool {
863    let Some(out_ptr) = non_null_guest_ptr(out_ptr) else {
864        return false;
865    };

867        return false;
868    };
869    let Some(byte_len) = handles.len().checked_mul(size_of::<i32>()) else {
870        return false;
871    };
872
873    if handles.is_empty() {
874        return write_guest_vec_header(env, out_ptr.get(), len, 0);
875    }
876
877    let Some(mut data) = GuestAllocation::new(env, byte_len) else {
878        return false;
879    };
880    if !data.write_i32_slice(handles) || !data.write_vec_header(out_ptr, len) {
881        return false;
882    }
883    let _ = data.into_raw();
884    true
885}

933fn read_wasm_val(env: &mut FunctionEnvMut<WasmCapiEnv>, val_ptr: i32, ty: Type) -> Option<Value> {
934    match ty {
935        Type::I32 => Some(Value::I32(read_i32(
936            env,
937            val_ptr + WASM_VAL_PAYLOAD_OFFSET,
938        )?)),
939        Type::I64 => Some(Value::I64(
940            read_u64(env, val_ptr + WASM_VAL_PAYLOAD_OFFSET)? as i64,
941        )),
942        Type::F32 => {
943            let raw = read_u32(env, val_ptr + WASM_VAL_PAYLOAD_OFFSET)?;
944            Some(Value::F32(f32::from_bits(raw)))
945        }
946        Type::F64 => {
947            let raw = read_u64(env, val_ptr + WASM_VAL_PAYLOAD_OFFSET)?;
948            Some(Value::F64(f64::from_bits(raw)))
949        }
950        // The guest stores a `wasm_ref_t*` as an i32 handle in the payload.
951        Type::FuncRef | Type::ExternRef => {
952            let handle = read_i32(env, val_ptr + WASM_VAL_PAYLOAD_OFFSET)?;
953            Some(ref_value_from_handle(env, handle, ty))
954        }
955        Type::V128 | Type::ExceptionRef => None,
956    }
957}

959/// Resolve a guest `wasm_ref_t*` handle into a reference [`Value`] of the
960/// expected reference kind. A null handle (`0`) is the null reference.
961fn ref_value_from_handle(env: &FunctionEnvMut<WasmCapiEnv>, handle: i32, ty: Type) -> Value {
962    let null = match ty {
963        Type::FuncRef => Value::FuncRef(None),
964        _ => Value::ExternRef(None),
965    };
966    if handle <= INVALID_HANDLE {
967        return null;
968    }
969    match env.data().state.get(handle) {
970        Some(WasmObject::Ref(value)) => value.clone(),
971        // A funcref may also be handed over as a `wasm_func_t*` handle.
972        Some(WasmObject::Func(f)) | Some(WasmObject::Extern(WasmExtern::Func(f))) => {
973            Value::FuncRef(Some(f.clone()))
974        }
975        _ => null,
976    }
977}
978
979/// Materialize a reference [`Value`] into a guest `wasm_ref_t*` handle, or `0`
980/// for the null reference.
981fn ref_value_to_handle(env: &mut FunctionEnvMut<WasmCapiEnv>, value: &Value) -> i32 {
982    match value {
983        Value::ExternRef(None) | Value::FuncRef(None) => INVALID_HANDLE,
984        Value::ExternRef(Some(_)) | Value::FuncRef(Some(_)) => {
985            insert(env, WasmObject::Ref(value.clone()))
986        }
987        _ => INVALID_HANDLE,
988    }
989}

991fn write_wasm_val(env: &mut FunctionEnvMut<WasmCapiEnv>, val_ptr: i32, value: &Value) -> bool {
992    let Some(val_ptr) = non_null_guest_ptr(val_ptr) else {
993        return false;
994    };
995    let val_ptr = val_ptr.get();
996    let Some(kind) = type_to_wasm_kind(value.ty()) else {
997        return false;
998    };
999    // For references, mint the guest handle before writing (mutates state).
1000    let ref_handle = matches!(value, Value::FuncRef(_) | Value::ExternRef(_))
1001        .then(|| ref_value_to_handle(env, value));
1002    if !write_guest_bytes(env, val_ptr, &[kind]) {
1003        return false;
1004    }
1005    match value {
1006        Value::I32(v) => {
1007            write_guest_bytes_offset(env, val_ptr, WASM_VAL_PAYLOAD_OFFSET, &v.to_le_bytes())
1008        }
1009        Value::I64(v) => {
1010            write_guest_bytes_offset(env, val_ptr, WASM_VAL_PAYLOAD_OFFSET, &v.to_le_bytes())
1011        }
1012        Value::F32(v) => write_guest_bytes_offset(
1013            env,
1014            val_ptr,
1015            WASM_VAL_PAYLOAD_OFFSET,
1016            &v.to_bits().to_le_bytes(),
1017        ),
1018        Value::F64(v) => write_guest_bytes_offset(
1019            env,
1020            val_ptr,
1021            WASM_VAL_PAYLOAD_OFFSET,
1022            &v.to_bits().to_le_bytes(),
1023        ),
1024        Value::FuncRef(_) | Value::ExternRef(_) => write_guest_bytes_offset(
1025            env,
1026            val_ptr,
1027            WASM_VAL_PAYLOAD_OFFSET,
1028            &ref_handle.unwrap_or(INVALID_HANDLE).to_le_bytes(),
1029        ),
1030        Value::V128(_) | Value::ExceptionRef(_) => false,
1031    }
1032}

1323fn copy_wasmer_memory_to_guest(
1324    env: &mut FunctionEnvMut<WasmCapiEnv>,
1325    memory: &Memory,
1326    guest_ptr: i32,
1327    len: usize,
1328) -> bool {
1329    if len == 0 {
1330        return true;
1331    }
1332    let Some(guest_memory) = env.data().memory.clone() else {
1333        return false;
1334    };
1335    let source_view = memory.view(&*env);
1336    let guest_view = guest_memory.view(&*env);
1337    let Some(source_offset) = checked_memory_offset(0, len, source_view.data_size()) else {
1338        return false;
1339    };
1340    let Some(guest_offset) = checked_memory_offset(guest_ptr, len, guest_view.data_size()) else {
1341        return false;
1342    };
1343    let source_base = source_view.data_ptr();
1344    let guest_base = guest_view.data_ptr();
1345    if ptr::eq(source_base, guest_base) {
1346        return false;
1347    }
1348    unsafe {
1349        // Both ranges are bounds-checked above and same-memory copies are rejected.
1350        ptr::copy_nonoverlapping(
1351            source_base.add(source_offset),
1352            guest_base.add(guest_offset),
1353            len,
1354        );
1355    }
1356    true
1357}

1359fn copy_guest_memory_to_wasmer(
1360    env: &mut FunctionEnvMut<WasmCapiEnv>,
1361    guest_ptr: i32,
1362    memory: &Memory,
1363    len: usize,
1364) -> bool {
1365    if len == 0 {
1366        return true;
1367    }
1368    let Some(guest_memory) = env.data().memory.clone() else {
1369        return false;
1370    };
1371    let guest_view = guest_memory.view(&*env);
1372    let target_view = memory.view(&*env);
1373    let Some(guest_offset) = checked_memory_offset(guest_ptr, len, guest_view.data_size()) else {
1374        return false;
1375    };
1376    let Some(target_offset) = checked_memory_offset(0, len, target_view.data_size()) else {
1377        return false;
1378    };
1379    let guest_base = guest_view.data_ptr();
1380    let target_base = target_view.data_ptr();
1381    if ptr::eq(guest_base, target_base) {
1382        return false;
1383    }
1384    unsafe {
1385        // Both ranges are bounds-checked above and same-memory copies are rejected.
1386        ptr::copy_nonoverlapping(
1387            guest_base.add(guest_offset),
1388            target_base.add(target_offset),
1389            len,
1390        );
1391    }
1392    true
1393}

1405fn wasm_memory_data(mut env: FunctionEnvMut<WasmCapiEnv>, memory_handle: i32) -> i32 {
1406    let Some(memory) = memory_from_handle(&env, memory_handle) else {
1407        return INVALID_HANDLE;
1408    };
1409
1410    if !memory_supports_shadow(&env, &memory) {
1411        return INVALID_HANDLE;
1412    };
1413    let size = {
1414        let view = memory.view(&env);
1415        let Ok(size) = usize::try_from(view.data_size()) else {
1416            return INVALID_HANDLE;
1417        };
1418        size
1419    };
1420    if size == 0 {
1421        return INVALID_HANDLE;
1422    }
1423
1424    let existing = env.data().state.memory_shadows.get(&memory_handle).copied();
1425    let shadow = match existing {
1426        // Size unchanged: the shadow is kept coherent by the sync/refresh
1427        // brackets around every wasm call and guest callback, and re-copying
1428        // here would DESTROY guest writes that have not been synced yet (the
1429        // guest may query the data pointer between writing to the shadow and
1430        // the call that flushes it).
1431        Some(existing) if existing.len >= size => MemoryShadow {
1432            guest_ptr: existing.guest_ptr,
1433            len: size,
1434        },
1435        _ => {
1436            let Some(mut allocation) = GuestAllocation::new(&mut env, size) else {
1437                return INVALID_HANDLE;
1438            };
1439            if !allocation.copy_from_wasmer_memory(&memory, size) {
1440                return INVALID_HANDLE;
1441            }
1442            let guest_ptr = allocation.into_raw();
1443            if let Some(existing) = existing {
1444                free_guest_memory(&mut env, existing.guest_ptr);
1445            }
1446            MemoryShadow {
1447                guest_ptr,
1448                len: size,
1449            }
1450        }
1451    };
1452
1453    env.data_mut()
1454        .state
1455        .memory_shadows
1456        .insert(memory_handle, shadow);
1457    shadow.guest_ptr
1458}

1460fn sync_memory_shadows_to_wasmer(env: &mut FunctionEnvMut<WasmCapiEnv>) {
1461    let shadows: Vec<_> = env
1462        .data()
1463        .state
1464        .memory_shadows
1465        .iter()
1466        .map(|(&handle, &shadow)| (handle, shadow))
1467        .collect();
1468
1469    for (handle, shadow) in shadows {
1470        let Some(memory) = memory_from_handle(env, handle) else {
1471            continue;
1472        };
1473        if !memory_supports_shadow(env, &memory) {
1474            continue;
1475        };
1476        let _ = copy_guest_memory_to_wasmer(env, shadow.guest_ptr, &memory, shadow.len);
1477    }
1478}

1481    let shadows: Vec<_> = env
1482        .data()
1483        .state
1484        .memory_shadows
1485        .iter()
1486        .map(|(&handle, &shadow)| (handle, shadow))
1487        .collect();
1488
1489    for (handle, shadow) in shadows {

1492        };

1495        }

1598    if let Some(trap_out_ptr) = non_null_guest_ptr(trap_out_ptr) {

1782fn wasm_func_call(
1783    mut env: FunctionEnvMut<WasmCapiEnv>,
1784    func_handle: i32,
1785    args_vec_ptr: i32,
1786    results_vec_ptr: i32,
1787) -> i32 {
1788    let func = match env.data().state.get(func_handle) {
1789        Some(WasmObject::Func(func)) => func.clone(),
1790        Some(WasmObject::Extern(WasmExtern::Func(func))) => func.clone(),
1791        _ => return insert(&mut env, WasmObject::Trap("invalid function".to_string())),
1792    };
1793    let ty = func.ty(&env);
1794    let arg_data_ptr = guest_ptr_with_offset(args_vec_ptr, WASM_VEC_DATA_OFFSET)
1795        .and_then(|ptr| read_i32(&mut env, ptr))
1796        .unwrap_or(INVALID_HANDLE);
1797    let mut args = Vec::with_capacity(ty.params().len());
1798    for (index, ty) in ty.params().iter().enumerate() {
1799        let val_ptr = arg_data_ptr + (index * WASM_VAL_SIZE) as i32;
1800        let Some(value) = read_wasm_val(&mut env, val_ptr, *ty) else {
1801            return insert(
1802                &mut env,
1803                WasmObject::Trap("invalid function argument".to_string()),
1804            );
1805        };
1806        args.push(value);
1807    }
1808    sync_memory_shadows_to_wasmer(&mut env);
1809    match func.call(&mut env, &args) {
1810        Ok(results) => {
1811            refresh_memory_shadows_from_wasmer(&mut env);
1812            let result_data_ptr = guest_ptr_with_offset(results_vec_ptr, WASM_VEC_DATA_OFFSET)
1813                .and_then(|ptr| read_i32(&mut env, ptr))
1814                .unwrap_or(INVALID_HANDLE);
1815            for (index, value) in results.iter().enumerate() {
1816                let val_ptr = result_data_ptr + (index * WASM_VAL_SIZE) as i32;
1817                write_wasm_val(&mut env, val_ptr, value);
1818            }
1819            INVALID_HANDLE
1820        }
1821        Err(err) => {
1822            refresh_memory_shadows_from_wasmer(&mut env);
1823            insert(&mut env, WasmObject::Trap(err.to_string()))
1824        }
1825    }
1826}

1829    env: &mut FunctionEnvMut<WasmCapiEnv>,

1833    let byte_len = values.len().checked_mul(WASM_VAL_SIZE)?;
1834

1838    }

1841    for (index, value) in values.iter().enumerate() {

1844            return None;
1845        }
1846    }

1848    let vec_ptr = data.allocate_vec_header(len)?;
1849    let data_ptr = data.into_raw();

1878fn call_guest_wasm_callback(
1879    env: &mut FunctionEnvMut<WasmCapiEnv>,
1880    callback: u32,
1881    callback_env: i32,
1882    args_vec_ptr: i32,
1883    results_vec_ptr: i32,
1884) -> i32 {
1885    let Some(table) = env.data().table.clone() else {
1886        return insert(
1887            env,
1888            WasmObject::Trap("missing guest indirect function table".to_string()),
1889        );
1890    };
1891    let Some(elem) = table.get(&mut *env, callback) else {
1892        return insert(
1893            env,
1894            WasmObject::Trap("invalid guest WebAssembly callback pointer".to_string()),
1895        );
1896    };
1897    let func = match elem {
1898        Value::FuncRef(Some(func)) => func,
1899        _ => {
1900            return insert(
1901                env,
1902                WasmObject::Trap("invalid guest WebAssembly callback reference".to_string()),
1903            );
1904        }
1905    };
1906    // Mirror of the wasm_func_call bracketing, in the opposite direction: the
1907    // guest callback reads the target module's memory through its shadow (so
1908    // it must see mutations made since the last refresh), and its writes into
1909    // the shadow (e.g. wasm-bindgen retptr stores) must land back in the real
1910    // memory before the target module resumes.
1911    refresh_memory_shadows_from_wasmer(env);
1912    let call_result = func.call(
1913        env,
1914        &[
1915            Value::I32(callback_env),
1916            Value::I32(args_vec_ptr),
1917            Value::I32(results_vec_ptr),
1918        ],
1919    );
1920    sync_memory_shadows_to_wasmer(env);
1921    match call_result {
1922        Ok(values) => match values.first() {
1923            Some(Value::I32(value)) => *value,
1924            Some(Value::I64(value)) => *value as i32,
1925            _ => INVALID_HANDLE,
1926        },
1927        Err(err) => insert(env, WasmObject::Trap(err.to_string())),
1928    }
1929}

1931fn wasm_func_new_with_env(
1932    mut env: FunctionEnvMut<WasmCapiEnv>,
1933    _store: i32,
1934    type_handle: i32,
1935    callback: i32,
1936    callback_env: i32,
1937    _finalizer: i32,
1938) -> i32 {
1939    if callback <= INVALID_HANDLE {
1940        return INVALID_HANDLE;
1941    }
1942    let ty = match env.data().state.get(type_handle) {
1943        Some(WasmObject::FuncType(ty)) => ty.clone(),
1944        _ => return INVALID_HANDLE,
1945    };
1946    let Some(func_env) = env.data().func_env.clone() else {
1947        return INVALID_HANDLE;
1948    };
1949    let Ok(callback) = u32::try_from(callback) else {
1950        return INVALID_HANDLE;
1951    };
1952    let result_types = ty.results().to_vec();
1953    let func = WasmerFunction::new_with_env(&mut env, &func_env, ty, move |mut env, args| {
1954        let Some((args_vec_ptr, _args_data_ptr)) = allocate_wasm_val_vec_for_values(&mut env, args)
1955        else {
1956            return Err(RuntimeError::new(
1957                "failed to allocate WebAssembly callback arguments",
1958            ));
1959        };
1960        let Some((results_vec_ptr, results_data_ptr)) =
1961            allocate_uninitialized_wasm_val_vec(&mut env, result_types.len())
1962        else {
1963            return Err(RuntimeError::new(
1964                "failed to allocate WebAssembly callback results",
1965            ));
1966        };
1967
1968        let trap = call_guest_wasm_callback(
1969            &mut env,
1970            callback,
1971            callback_env,
1972            args_vec_ptr,
1973            results_vec_ptr,
1974        );
1975        if trap != INVALID_HANDLE {
1976            return Err(RuntimeError::new(trap_message_from_handle(&env, trap)));
1977        }
1978
1979        let mut results = Vec::with_capacity(result_types.len());
1980        for (index, ty) in result_types.iter().enumerate() {
1981            let val_ptr = results_data_ptr + (index * WASM_VAL_SIZE) as i32;
1982            let Some(value) = read_wasm_val(&mut env, val_ptr, *ty) else {
1983                return Err(RuntimeError::new(
1984                    "guest WebAssembly callback returned an invalid value",
1985                ));
1986            };
1987            results.push(value);
1988        }
1989        Ok(results)
1990    });
1991    insert(&mut env, WasmObject::Func(func))
1992}
