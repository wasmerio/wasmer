15/// Represents the kind of compiled function or module, used for debugging and identification
16/// purposes across multiple compiler backends (e.g., LLVM, Cranelift).
17#[derive(Debug, Clone)]
18pub enum CompiledKind {
19    /// A locally-defined function in the Wasm file.
20    Local(LocalFunctionIndex, String),
21    /// A function call trampoline for a given signature.
22    FunctionCallTrampoline(FunctionType),
23    /// A dynamic function trampoline for a given signature.
24    DynamicFunctionTrampoline(FunctionType),
25    /// An entire Wasm module.
26    Module,
27}

29/// Converts a slice of `Type` into a string signature, mapping each type to a specific character.
30/// Used to represent function signatures in a compact string form.
31pub fn types_to_signature(types: &[Type]) -> String {
32    let tokens = types
33        .iter()
34        .map(|ty| match ty {
35            Type::I32 => "i",
36            Type::I64 => "I",
37            Type::F32 => "f",
38            Type::F64 => "F",
39            Type::V128 => "v",
40            Type::ExternRef => "e",
41            Type::FuncRef => "r",
42            Type::ExceptionRef => "x",
43        })
44        .collect_vec();
45    // Apparently, LLVM has issues if the filename is too long, thus we compact it.
46    tokens
47        .chunk_by(|a, b| a == b)
48        .map(|chunk| {
49            if chunk.len() >= 8 {
50                format!("{}x{}", chunk.len(), chunk[0])
51            } else {
52                chunk.to_owned().join("")
53            }
54        })
55        .join("")
56}

58/// Sanitizes a string so it can be safely used as a filename.
59fn sanitize_filename(name: &str) -> String {
60    name.chars()
61        .map(|c| {
62            if c.is_alphanumeric() || c == '_' || c == '-' {
63                c
64            } else {
65                '_'
66            }
67        })
68        .collect()
69}

71/// Converts a kind into a filename, that we will use to dump
72/// the contents of the IR object file to.
73pub fn function_kind_to_filename(kind: &CompiledKind, suffix: &str) -> String {
74    match kind {
75        CompiledKind::Local(local_func_index, name) => {
76            let mut name = sanitize_filename(name);
77
78            // Limit to 255 characters to comply with common filesystem path component restrictions.
79            const PATH_LIMIT: usize = 255;
80
81            if name.len() + suffix.len() > PATH_LIMIT {
82                let id_string = local_func_index.as_u32().to_string();
83                name.truncate(PATH_LIMIT - id_string.len() - suffix.len() - 1);
84                name.push('_');
85                name.push_str(&id_string);
86                name.push_str(suffix);
87            } else {
88                name.push_str(suffix);
89            }
90
91            debug_assert!(name.len() <= PATH_LIMIT);
92            name
93        }
94        CompiledKind::FunctionCallTrampoline(func_type) => format!(
95            "trampoline_call_{}_{}{suffix}",
96            types_to_signature(func_type.params()),
97            types_to_signature(func_type.results())
98        ),
99        CompiledKind::DynamicFunctionTrampoline(func_type) => format!(
100            "trampoline_dynamic_{}_{}{suffix}",
101            types_to_signature(func_type.params()),
102            types_to_signature(func_type.results())
103        ),
104        CompiledKind::Module => "module".into(),
105    }
106}

108/// Saves disassembled assembly code to a file with optional comments at specific offsets.
109///
110/// This function takes raw machine code bytes, disassembles them using `objdump`, and writes
111/// the annotated assembly to a file in the specified debug directory.
112#[cfg(not(target_arch = "wasm32"))]
113pub fn save_assembly_to_file<C: Display>(
114    arch: Architecture,
115    path: PathBuf,
116    body: &[u8],
117    assembly_comments: HashMap<usize, C>,
118) -> Result<(), CompileError> {
119    use std::{fs::File, io::Write};
120    use which::which;
121
122    #[derive(Debug)]
123    struct DecodedInsn<'a> {
124        offset: usize,
125        insn: &'a str,
126    }
127
128    fn parse_instructions(content: &str) -> Result<Vec<DecodedInsn<'_>>, CompileError> {
129        content
130            .lines()
131            .map(|line| line.trim())
132            .skip_while(|l| !l.starts_with("0000000000000000"))
133            .skip(1)
134            .filter(|line| line.trim() != "...")
135            .map(|line| -> Result<DecodedInsn<'_>, CompileError> {
136                let (offset, insn_part) = line.split_once(':').ok_or(CompileError::Codegen(
137                    format!("cannot parse objdump line: '{line}'"),
138                ))?;
139                // instruction content can be empty
140                let insn = insn_part
141                    .trim()
142                    .split_once('\t')
143                    .map_or("", |(_data, insn)| insn)
144                    .trim();
145                Ok(DecodedInsn {
146                    offset: usize::from_str_radix(offset, 16).map_err(|err| {
147                        CompileError::Codegen(format!("hex number expected: {err}"))
148                    })?,
149                    insn,
150                })
151            })
152            .collect()
153    }
154
155    // Note objdump cannot read from stdin.
156    let mut tmpfile = NamedTempFile::new()
157        .map_err(|err| CompileError::Codegen(format!("cannot create temporary file: {err}")))?;
158    tmpfile
159        .write_all(body)
160        .map_err(|err| CompileError::Codegen(format!("assembly dump write failed: {err}")))?;
161    tmpfile
162        .flush()
163        .map_err(|err| CompileError::Codegen(format!("flush failed: {err}")))?;
164
165    let (objdump_arch, objdump_binary) = match arch {
166        Architecture::X86_64 => ("i386:x86-64", "x86_64-linux-gnu-objdump"),
167        Architecture::Aarch64(..) => ("aarch64", "aarch64-linux-gnu-objdump"),
168        Architecture::Riscv64(..) => ("riscv:rv64", "riscv64-linux-gnu-objdump"),
169        _ => {
170            return Err(CompileError::Codegen(
171                "Assembly dumping is not supported for this architecture".to_string(),
172            ));
173        }
174    };
175
176    let bins = [objdump_binary, "objdump"];
177    let objdump_binary = bins.iter().find(|bin| which(bin).is_ok());
178    let Some(objdump_binary) = objdump_binary else {
179        // Objdump is an optional dependency, do not fail if not present.
180        return Ok(());
181    };
182
183    let command = Command::new(objdump_binary)
184        .arg("-b")
185        .arg("binary")
186        .arg("-m")
187        .arg(objdump_arch)
188        .arg("-D")
189        .arg(tmpfile.path())
190        .stdout(Stdio::piped())
191        .stderr(Stdio::null())
192        .spawn();
193
194    let Ok(command) = command else {
195        // The target might not be supported, do not fail in that case.
196        return Ok(());
197    };
198
199    let output = command
200        .wait_with_output()
201        .map_err(|err| CompileError::Codegen(format!("failed to read stdout: {err}")))?;
202    let content = String::from_utf8_lossy(&output.stdout);
203
204    let parsed_instructions = parse_instructions(content.as_ref())?;
205
206    let mut file = File::create(path).map_err(|err| {
207        CompileError::Codegen(format!("debug object file creation failed: {err}"))
208    })?;
209
210    // Dump the instruction annotated with the comments.
211    for insn in parsed_instructions {
212        if let Some(comment) = assembly_comments.get(&insn.offset) {
213            file.write_all(format!("      \t\t;; {comment}\n").as_bytes())
214                .map_err(|err| {
215                    CompileError::Codegen(format!("cannot write content to object file: {err}"))
216                })?;
217        }
218        file.write_all(format!("{:6x}:\t\t{}\n", insn.offset, insn.insn).as_bytes())
219            .map_err(|err| {
220                CompileError::Codegen(format!("cannot write content to object file: {err}"))
221            })?;
222    }
223
224    Ok(())
225}
