4//! Helper functions to gather information for each of the non-function sections of a
5//! WebAssembly module.
6//!
7//! The code of these helper functions is straightforward since they only read metadata
8//! about linear memories, tables, globals, etc. and store them for later use.
9//!
10//! The special case of the initialize expressions for table elements offsets or global variables
11//! is handled, according to the semantics of WebAssembly, to only specific expressions that are
12//! interpreted on the fly.
13

35/// Helper function translating wasmparser types to Wasm Type.
36pub fn wptype_to_type(ty: wasmparser::ValType) -> WasmResult<Type> {
37    match ty {
38        wasmparser::ValType::I32 => Ok(Type::I32),
39        wasmparser::ValType::I64 => Ok(Type::I64),
40        wasmparser::ValType::F32 => Ok(Type::F32),
41        wasmparser::ValType::F64 => Ok(Type::F64),
42        wasmparser::ValType::V128 => Ok(Type::V128),
43        wasmparser::ValType::Ref(ty) => wpreftype_to_type(ty),
44    }
45}

47/// Converts a wasmparser ref type to a Wasm Type.
48pub fn wpreftype_to_type(ty: wasmparser::RefType) -> WasmResult<Type> {
49    if ty.is_extern_ref() {
50        Ok(Type::ExternRef)
51    } else if ty.is_func_ref() {
52        Ok(Type::FuncRef)
53    } else if ty == wasmparser::RefType::EXNREF || ty == wasmparser::RefType::EXN {
54        // no `.is_exnref` yet
55        Ok(Type::ExceptionRef)
56    } else {
57        Err(wasm_unsupported!("unsupported reference type: {:?}", ty))
58    }
59}

61/// Converts a wasmparser heap type to a Wasm Type.
62pub fn wpheaptype_to_type(ty: wasmparser::HeapType) -> WasmResult<Type> {
63    match ty {
64        wasmparser::HeapType::Abstract { ty, .. } => match ty {
65            wasmparser::AbstractHeapType::Func => Ok(Type::FuncRef),
66            wasmparser::AbstractHeapType::Extern => Ok(Type::ExternRef),
67            wasmparser::AbstractHeapType::Exn => Ok(Type::ExceptionRef),
68            other => Err(wasm_unsupported!("unsupported reference type: {other:?}")),
69        },
70        other => Err(wasm_unsupported!("unsupported reference type: {other:?}")),
71    }
72}

74/// Parses the Type section of the wasm module.
75pub fn parse_type_section(
76    types: TypeSectionReader,
77    module_translation_state: &mut ModuleTranslationState,
78    environ: &mut ModuleEnvironment,
79) -> WasmResult<()> {
80    let count = types.count();
81    environ.reserve_signatures(count)?;
82
83    for res in types.into_iter_err_on_gc_types() {
84        let functype = res.map_err(from_binaryreadererror_wasmerror)?;
85
86        let params = functype.params();
87        let returns = functype.results();
88        let sig_params: Box<[Type]> = params
89            .iter()
90            .map(|ty| {
91                wptype_to_type(*ty)
92                    .expect("only numeric types are supported in function signatures")
93            })
94            .collect();
95        let sig_returns: Box<[Type]> = returns
96            .iter()
97            .map(|ty| {
98                wptype_to_type(*ty)
99                    .expect("only numeric types are supported in function signatures")
100            })
101            .collect();
102        let sig = FunctionType::new(sig_params, sig_returns);
103        environ.declare_signature(sig)?;
104        module_translation_state
105            .wasm_types
106            .push((params.to_vec().into(), returns.to_vec().into()));
107    }
108
109    Ok(())
110}

112/// Parses the Import section of the wasm module.
113pub fn parse_import_section<'data>(
114    imports: ImportSectionReader<'data>,
115    environ: &mut ModuleEnvironment<'data>,
116) -> WasmResult<()> {
117    environ.reserve_imports(imports.count())?;
118
119    for entry in imports {
120        let import = entry.map_err(from_binaryreadererror_wasmerror)?;
121        let module_name = import.module;
122        let field_name = import.name;
123
124        match import.ty {
125            TypeRef::Func(sig) => {
126                environ.declare_func_import(
127                    SignatureIndex::from_u32(sig),
128                    module_name,
129                    field_name,
130                )?;
131            }
132            TypeRef::FuncExact(_) => {
133                return Err(WasmError::Generic(
134                    "custom-descriptors not implemented yet".to_string(),
135                ));
136            }
137            TypeRef::Tag(t) => {
138                environ.declare_tag_import(t, module_name, field_name)?;
139            }
140            TypeRef::Memory(WPMemoryType {
141                shared,
142                memory64,
143                initial,
144                maximum,
145                ..
146            }) => {
147                if memory64 {
148                    unimplemented!("64bit memory not implemented yet");
149                }
150                environ.declare_memory_import(
151                    MemoryType {
152                        minimum: Pages(initial as u32),
153                        maximum: maximum.map(|p| Pages(p as u32)),
154                        shared,
155                    },
156                    module_name,
157                    field_name,
158                )?;
159            }
160            TypeRef::Global(ref ty) => {
161                environ.declare_global_import(
162                    GlobalType {
163                        ty: wptype_to_type(ty.content_type)?,
164                        mutability: ty.mutable.into(),
165                    },
166                    module_name,
167                    field_name,
168                )?;
169            }
170            TypeRef::Table(ref tab) => {
171                environ.declare_table_import(
172                    TableType {
173                        ty: wpreftype_to_type(tab.element_type)?,
174                        minimum: tab.initial as u32,
175                        maximum: tab.maximum.map(|v| v as u32),
176                    },
177                    module_name,
178                    field_name,
179                )?;
180            }
181        }
182    }
183
184    environ.finish_imports()?;
185    Ok(())
186}

188/// Parses the Function section of the wasm module.
189pub fn parse_function_section(
190    functions: FunctionSectionReader,
191    environ: &mut ModuleEnvironment,
192) -> WasmResult<()> {
193    let num_functions = functions.count();
194    if num_functions == u32::MAX {
195        // We reserve `u32::MAX` for our own use.
196        return Err(WasmError::ImplLimitExceeded);
197    }
198
199    environ.reserve_func_types(num_functions)?;
200
201    for entry in functions {
202        let sigindex = entry.map_err(from_binaryreadererror_wasmerror)?;
203        environ.declare_func_type(SignatureIndex::from_u32(sigindex))?;
204    }
205
206    Ok(())
207}

209/// Parses the Table section of the wasm module.
210pub fn parse_table_section(
211    tables: TableSectionReader,
212    environ: &mut ModuleEnvironment,
213) -> WasmResult<()> {
214    environ.reserve_tables(tables.count())?;
215
216    for entry in tables {
217        let table = entry.map_err(from_binaryreadererror_wasmerror)?;
218        environ.declare_table(TableType {
219            ty: wpreftype_to_type(table.ty.element_type).unwrap(),
220            minimum: table.ty.initial as u32,
221            maximum: table.ty.maximum.map(|v| v as u32),
222        })?;
223    }
224
225    Ok(())
226}

271/// Parses the Global section of the wasm module.
272pub fn parse_global_section(
273    globals: GlobalSectionReader,
274    environ: &mut ModuleEnvironment,
275) -> WasmResult<()> {
276    environ.reserve_globals(globals.count())?;
277
278    for entry in globals {
279        let wasmparser::Global {
280            ty:
281                WPGlobalType {
282                    content_type,
283                    mutable,
284                    ..
285                },
286            init_expr,
287        } = entry.map_err(from_binaryreadererror_wasmerror)?;
288        let mut init_expr_reader = init_expr.get_operators_reader();
289        let initializer = match init_expr_reader
290            .read()
291            .map_err(from_binaryreadererror_wasmerror)?
292        {
293            Operator::I32Const { value } => GlobalInit::I32Const(value),
294            Operator::I64Const { value } => GlobalInit::I64Const(value),
295            Operator::F32Const { value } => GlobalInit::F32Const(f32::from_bits(value.bits())),
296            Operator::F64Const { value } => GlobalInit::F64Const(f64::from_bits(value.bits())),
297            Operator::V128Const { value } => GlobalInit::V128Const(V128::from(*value.bytes())),
298            Operator::RefNull { hty: _ } => {
299                // TODO: Do we need to handle different heap types here?
300                GlobalInit::RefNullConst
301            }
302            Operator::RefFunc { function_index } => {
303                GlobalInit::RefFunc(FunctionIndex::from_u32(function_index))
304            }
305            Operator::GlobalGet { global_index } => {
306                GlobalInit::GetGlobal(GlobalIndex::from_u32(global_index))
307            }
308            ref s => {
309                return Err(wasm_unsupported!(
310                    "unsupported init expr in global section: {:?}",
311                    s
312                ));
313            }
314        };
315        let global = GlobalType {
316            ty: wptype_to_type(content_type).unwrap(),
317            mutability: mutable.into(),
318        };
319        environ.declare_global(global, initializer)?;
320    }
321
322    Ok(())
323}

325/// Parses the Export section of the wasm module.
326pub fn parse_export_section<'data>(
327    exports: ExportSectionReader<'data>,
328    environ: &mut ModuleEnvironment<'data>,
329) -> WasmResult<()> {
330    environ.reserve_exports(exports.count())?;
331
332    for entry in exports {
333        let Export {
334            name: field,
335            ref kind,
336            index,
337        } = entry.map_err(from_binaryreadererror_wasmerror)?;
338
339        // The input has already been validated, so we should be able to
340        // assume valid UTF-8 and use `from_utf8_unchecked` if performance
341        // becomes a concern here.
342        let index = index as usize;
343        match *kind {
344            ExternalKind::Func => environ.declare_func_export(FunctionIndex::new(index), field)?,
345            ExternalKind::Table => environ.declare_table_export(TableIndex::new(index), field)?,
346            ExternalKind::Memory => {
347                environ.declare_memory_export(MemoryIndex::new(index), field)?
348            }
349            ExternalKind::Global => {
350                environ.declare_global_export(GlobalIndex::new(index), field)?
351            }
352            ExternalKind::Tag => environ.declare_tag_export(TagIndex::new(index), field)?,
353            ExternalKind::FuncExact => unimplemented!("custom-descriptors not implemented yet"),
354        }
355    }
356
357    environ.finish_exports()?;
358    Ok(())
359}

367fn read_elems(items: &ElementItems) -> WasmResult<Box<[FunctionIndex]>> {
368    let mut out = Vec::new();
369
370    match items {
371        ElementItems::Functions(funcs) => {
372            for res in funcs.clone().into_iter() {
373                let func_index = res.map_err(from_binaryreadererror_wasmerror)?;
374                out.push(FunctionIndex::from_u32(func_index));
375            }
376        }
377        ElementItems::Expressions(ty, section) => {
378            // TODO: check type is supported
379            if !(ty.is_extern_ref() || ty.is_func_ref()) {
380                return Err(wasm_unsupported!(
381                    "unsupported element type in element section: {:?}",
382                    ty
383                ));
384            }
385
386            for res in section.clone().into_iter() {
387                let expr = res.map_err(from_binaryreadererror_wasmerror)?;
388
389                let op = expr
390                    .get_operators_reader()
391                    .read()
392                    .map_err(from_binaryreadererror_wasmerror)?;
393                match op {
394                    Operator::RefNull { .. } => out.push(FunctionIndex::reserved_value()),
395                    Operator::RefFunc { function_index } => {
396                        out.push(FunctionIndex::from_u32(function_index))
397                    }
398                    other => {
399                        return Err(WasmError::Unsupported(format!(
400                            "unsupported init expr in element section: {other:?}",
401                        )));
402                    }
403                }
404            }
405        }
406    }
407
408    Ok(out.into_boxed_slice())
409}

411/// Parses the Element section of the wasm module.
412pub fn parse_element_section(
413    elements: ElementSectionReader<'_>,
414    environ: &mut ModuleEnvironment,
415) -> WasmResult<()> {
416    environ.reserve_table_initializers(elements.count())?;
417
418    for (index, elem) in elements.into_iter().enumerate() {
419        let Element {
420            kind,
421            items,
422            range: _,
423        } = elem.map_err(from_binaryreadererror_wasmerror)?;
424
425        let segments = read_elems(&items)?;
426        match kind {
427            ElementKind::Active {
428                table_index,
429                offset_expr,
430            } => {
431                let table_index = TableIndex::from_u32(table_index.unwrap_or(0));
432
433                let mut init_expr_reader = offset_expr.get_operators_reader();
434                let (base, offset) = match init_expr_reader
435                    .read()
436                    .map_err(from_binaryreadererror_wasmerror)?
437                {
438                    Operator::I32Const { value } => (None, value as u32 as usize),
439                    Operator::GlobalGet { global_index } => {
440                        (Some(GlobalIndex::from_u32(global_index)), 0)
441                    }
442                    ref s => {
443                        return Err(wasm_unsupported!(
444                            "unsupported init expr in element section: {:?}",
445                            s
446                        ));
447                    }
448                };
449                environ.declare_table_initializers(table_index, base, offset, segments)?
450            }
451            ElementKind::Passive => {
452                let index = ElemIndex::from_u32(index as u32);
453                environ.declare_passive_element(index, segments)?;
454            }
455            ElementKind::Declared => (),
456        }
457    }
458    Ok(())
459}

461/// Parses the Data section of the wasm module.
462pub fn parse_data_section<'data>(
463    data: DataSectionReader<'data>,
464    environ: &mut ModuleEnvironment<'data>,
465) -> WasmResult<()> {
466    environ.reserve_data_initializers(data.count())?;
467
468    for (index, entry) in data.into_iter().enumerate() {
469        let Data {
470            kind,
471            data,
472            range: _,
473        } = entry.map_err(from_binaryreadererror_wasmerror)?;
474        match kind {
475            DataKind::Active {
476                memory_index,
477                offset_expr,
478            } => {
479                let mut init_expr_reader = offset_expr.get_operators_reader();
480                let (base, offset) = match init_expr_reader
481                    .read()
482                    .map_err(from_binaryreadererror_wasmerror)?
483                {
484                    Operator::I32Const { value } => (None, value as u32 as usize),
485                    Operator::GlobalGet { global_index } => {
486                        (Some(GlobalIndex::from_u32(global_index)), 0)
487                    }
488                    ref s => {
489                        return Err(wasm_unsupported!(
490                            "unsupported init expr in data section: {:?}",
491                            s
492                        ));
493                    }
494                };
495                environ.declare_data_initialization(
496                    MemoryIndex::from_u32(memory_index),
497                    base,
498                    offset,
499                    data,
500                )?;
501            }
502            DataKind::Passive => {
503                let index = DataIndex::from_u32(index as u32);
504                environ.declare_passive_data(index, data)?;
505            }
506        }
507    }
508
509    Ok(())
510}

512/// Parses the Name section of the wasm module.
513pub fn parse_name_section<'data>(
514    names: NameSectionReader<'data>,
515    environ: &mut ModuleEnvironment<'data>,
516) -> WasmResult<()> {
517    let mut functions = HashMap::new();
518    let mut names_set = HashSet::new();
519
520    for res in names {
521        let subsection = if let Ok(subsection) = res {
522            subsection
523        } else {
524            // Should we log / warn here?
525            continue;
526        };
527        match subsection {
528            wasmparser::Name::Function(function_subsection) => {
529                let mut unique_name_map = HashMap::new();
530                for naming in function_subsection.into_iter().flatten() {
531                    if naming.index != u32::MAX {
532                        let mut name = naming.name.to_string();
533                        // In very rare cases a function name can have duplicates.
534                        if names_set.contains(&name) {
535                            let index = unique_name_map
536                                .entry(name.clone())
537                                .and_modify(|e| *e += 1)
538                                .or_insert(0);
539                            let alternative = format!("{name}.{index}");
540                            name = alternative;
541                        }
542                        names_set.insert(name.clone());
543                        functions.insert(FunctionIndex::from_u32(naming.index), name);
544                    }
545                }
546            }
547            wasmparser::Name::Module {
548                name,
549                name_range: _,
550            } => {
551                environ.declare_module_name(name)?;
552            }
553            wasmparser::Name::Local(_) => {}
554            wasmparser::Name::Label(_)
555            | wasmparser::Name::Type(_)
556            | wasmparser::Name::Table(_)
557            | wasmparser::Name::Memory(_)
558            | wasmparser::Name::Global(_)
559            | wasmparser::Name::Element(_)
560            | wasmparser::Name::Data(_)
561            | wasmparser::Name::Unknown { .. }
562            | wasmparser::Name::Field(_)
563            | wasmparser::Name::Tag(..) => {}
564        }
565    }
566
567    environ.declare_function_names(functions)
568}
