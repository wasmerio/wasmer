18#[allow(clippy::too_many_arguments)]
19fn apply_relocation(
20    body: usize,
21    r: &impl RelocationLike,
22    allocated_functions: &PrimaryMap<LocalFunctionIndex, FunctionExtent>,
23    allocated_sections: &PrimaryMap<SectionIndex, SectionBodyPtr>,
24    libcall_trampolines_sec_idx: SectionIndex,
25    libcall_trampoline_len: usize,
26    riscv_pcrel_hi20s: &mut HashMap<usize, u32>,
27    get_got_address: &dyn Fn(RelocationTarget) -> Option<usize>,
28) {
29    let reloc_target = r.reloc_target();
30
31    // Note: if the relocation needs GOT and its addend is not zero we will relax the
32    // relocation and, instead of making it use the GOT entry, we will fixup the assembly to
33    // use the final pointer directly, without any indirection. Also, see the comment in
34    // compiler-llvm/src/object_file.rs:288.
35    let target_func_address: usize = if r.kind().needs_got() && r.addend() == 0 {
36        if let Some(got_address) = get_got_address(reloc_target) {
37            got_address
38        } else {
39            panic!("No GOT entry for reloc target {reloc_target:?}")
40        }
41    } else {
42        match reloc_target {
43            RelocationTarget::LocalFunc(index) => *allocated_functions[index].ptr as usize,
44            RelocationTarget::LibCall(libcall) => {
45                // Use the direct target of the libcall if the relocation supports
46                // a full 64-bit address. Otherwise use a trampoline.
47                if matches!(
48                    r.kind(),
49                    RelocationKind::Abs8
50                        | RelocationKind::X86PCRel8
51                        | RelocationKind::MachoArm64RelocUnsigned
52                        | RelocationKind::MachoX86_64RelocUnsigned
53                ) {
54                    function_pointer(libcall)
55                } else {
56                    get_libcall_trampoline(
57                        libcall,
58                        allocated_sections[libcall_trampolines_sec_idx].0 as usize,
59                        libcall_trampoline_len,
60                    )
61                }
62            }
63            RelocationTarget::CustomSection(custom_section) => {
64                *allocated_sections[custom_section] as usize
65            }
66        }
67    };
68
69    // A set of addresses at which a SUBTRACTOR relocation was applied.
70    let mut macho_aarch64_subtractor_addresses = HashSet::new();
71
72    match r.kind() {
73        RelocationKind::Abs8 => unsafe {
74            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
75            write_unaligned(reloc_address as *mut u64, reloc_delta);
76        },
77        RelocationKind::X86PCRel4 => unsafe {
78            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
79            write_unaligned(reloc_address as *mut u32, reloc_delta as _);
80        },
81        RelocationKind::X86PCRel8 => unsafe {
82            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
83            write_unaligned(reloc_address as *mut u64, reloc_delta);
84        },
85        RelocationKind::X86CallPCRel4 => unsafe {
86            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
87            write_unaligned(reloc_address as *mut u32, reloc_delta as _);
88        },
89        RelocationKind::Arm64Call => unsafe {
90            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
91            if (reloc_delta as i64).abs() >= 0x1000_0000 {
92                panic!(
93                    "Relocation to big for {:?} for {:?} with {:x}, current val {:x}",
94                    r.kind(),
95                    r.reloc_target(),
96                    reloc_delta,
97                    read_unaligned(reloc_address as *mut u32)
98                )
99            }
100            let reloc_delta = (((reloc_delta / 4) as u32) & 0x3ff_ffff)
101                | (read_unaligned(reloc_address as *mut u32) & 0xfc00_0000);
102            write_unaligned(reloc_address as *mut u32, reloc_delta);
103        },
104        RelocationKind::Arm64Movw0 => unsafe {
105            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
106            let reloc_delta =
107                (((reloc_delta & 0xffff) as u32) << 5) | read_unaligned(reloc_address as *mut u32);
108            write_unaligned(reloc_address as *mut u32, reloc_delta);
109        },
110        RelocationKind::Arm64Movw1 => unsafe {
111            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
112            let reloc_delta = ((((reloc_delta >> 16) & 0xffff) as u32) << 5)
113                | read_unaligned(reloc_address as *mut u32);
114            write_unaligned(reloc_address as *mut u32, reloc_delta);
115        },
116        RelocationKind::Arm64Movw2 => unsafe {
117            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
118            let reloc_delta = ((((reloc_delta >> 32) & 0xffff) as u32) << 5)
119                | read_unaligned(reloc_address as *mut u32);
120            write_unaligned(reloc_address as *mut u32, reloc_delta);
121        },
122        RelocationKind::Arm64Movw3 => unsafe {
123            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
124            let reloc_delta = ((((reloc_delta >> 48) & 0xffff) as u32) << 5)
125                | read_unaligned(reloc_address as *mut u32);
126            write_unaligned(reloc_address as *mut u32, reloc_delta);
127        },
128        RelocationKind::RiscvPCRelHi20 => unsafe {
129            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
130
131            // save for later reference with RiscvPCRelLo12I
132            riscv_pcrel_hi20s.insert(reloc_address, reloc_delta as u32);
133
134            let reloc_delta = ((reloc_delta.wrapping_add(0x800) & 0xfffff000) as u32)
135                | read_unaligned(reloc_address as *mut u32);
136            write_unaligned(reloc_address as *mut u32, reloc_delta);
137        },
138        RelocationKind::RiscvPCRelLo12I => unsafe {
139            let (reloc_address, reloc_abs) = r.for_address(body, target_func_address as u64);
140            let reloc_delta = ((riscv_pcrel_hi20s.get(&(reloc_abs as usize)).expect(
141                "R_RISCV_PCREL_LO12_I relocation target must be a symbol with R_RISCV_PCREL_HI20",
142            ) & 0xfff)
143                << 20)
144                | read_unaligned(reloc_address as *mut u32);
145            write_unaligned(reloc_address as *mut u32, reloc_delta);
146        },
147        RelocationKind::RiscvCall => unsafe {
148            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
149            let reloc_delta = ((reloc_delta & 0xfff) << 52)
150                | (reloc_delta.wrapping_add(0x800) & 0xfffff000)
151                | read_unaligned(reloc_address as *mut u64);
152            write_unaligned(reloc_address as *mut u64, reloc_delta);
153        },
154        RelocationKind::LArchAbsHi20 | RelocationKind::LArchPCAlaHi20 => unsafe {
155            let (reloc_address, reloc_abs) = r.for_address(body, target_func_address as u64);
156            let reloc_abs = ((((reloc_abs >> 12) & 0xfffff) as u32) << 5)
157                | read_unaligned(reloc_address as *mut u32);
158            write_unaligned(reloc_address as *mut u32, reloc_abs);
159        },
160        RelocationKind::LArchAbsLo12 | RelocationKind::LArchPCAlaLo12 => unsafe {
161            let (reloc_address, reloc_abs) = r.for_address(body, target_func_address as u64);
162            let reloc_abs =
163                (((reloc_abs & 0xfff) as u32) << 10) | read_unaligned(reloc_address as *mut u32);
164            write_unaligned(reloc_address as *mut u32, reloc_abs);
165        },
166        RelocationKind::LArchAbs64Hi12 | RelocationKind::LArchPCAla64Hi12 => unsafe {
167            let (reloc_address, reloc_abs) = r.for_address(body, target_func_address as u64);
168            let reloc_abs = ((((reloc_abs >> 52) & 0xfff) as u32) << 10)
169                | read_unaligned(reloc_address as *mut u32);
170            write_unaligned(reloc_address as *mut u32, reloc_abs);
171        },
172        RelocationKind::LArchAbs64Lo20 | RelocationKind::LArchPCAla64Lo20 => unsafe {
173            let (reloc_address, reloc_abs) = r.for_address(body, target_func_address as u64);
174            let reloc_abs = ((((reloc_abs >> 32) & 0xfffff) as u32) << 5)
175                | read_unaligned(reloc_address as *mut u32);
176            write_unaligned(reloc_address as *mut u32, reloc_abs);
177        },
178        RelocationKind::LArchCall36 => unsafe {
179            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
180            let reloc_delta1 = ((((reloc_delta >> 18) & 0xfffff) as u32) << 5)
181                | read_unaligned(reloc_address as *mut u32);
182            write_unaligned(reloc_address as *mut u32, reloc_delta1);
183            let reloc_delta2 = ((((reloc_delta >> 2) & 0xffff) as u32) << 10)
184                | read_unaligned((reloc_address + 4) as *mut u32);
185            write_unaligned((reloc_address + 4) as *mut u32, reloc_delta2);
186        },
187        RelocationKind::Aarch64AdrPrelPgHi21 => unsafe {
188            let (reloc_address, delta) = r.for_address(body, target_func_address as u64);
189
190            let delta = delta as isize;
191            assert!(
192                ((-1 << 32)..(1 << 32)).contains(&delta),
193                "can't generate page-relative relocation with ±4GB `adrp` instruction"
194            );
195
196            let op = read_unaligned(reloc_address as *mut u32);
197            let delta = delta >> 12;
198            let immlo = ((delta as u32) & 0b11) << 29;
199            let immhi = (((delta as u32) >> 2) & 0x7ffff) << 5;
200            let mask = !((0x7ffff << 5) | (0b11 << 29));
201            let op = (op & mask) | immlo | immhi;
202
203            write_unaligned(reloc_address as *mut u32, op);
204        },
205        RelocationKind::Aarch64AdrPrelLo21 => unsafe {
206            let (reloc_address, delta) = r.for_address(body, target_func_address as u64);
207
208            let delta = delta as isize;
209            assert!(
210                ((-1 << 20)..(1 << 20)).contains(&delta),
211                "can't generate an ADR_PREL_LO21 relocation with an immediate larger than 20 bits"
212            );
213
214            let op = read_unaligned(reloc_address as *mut u32);
215            let immlo = ((delta as u32) & 0b11) << 29;
216            let immhi = (((delta as u32) >> 2) & 0x7ffff) << 5;
217            let mask = !((0x7ffff << 5) | (0b11 << 29));
218            let op = (op & mask) | immlo | immhi;
219
220            write_unaligned(reloc_address as *mut u32, op);
221        },
222        RelocationKind::Aarch64AddAbsLo12Nc => unsafe {
223            let (reloc_address, delta) = r.for_address(body, target_func_address as u64);
224
225            let delta = delta as isize;
226            let op = read_unaligned(reloc_address as *mut u32);
227            let imm = ((delta as u32) & 0xfff) << 10;
228            let mask = !((0xfff) << 10);
229            let op = (op & mask) | imm;
230
231            write_unaligned(reloc_address as *mut u32, op);
232        },
233        RelocationKind::Aarch64Ldst128AbsLo12Nc => unsafe {
234            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
235            let reloc_delta = ((reloc_delta as u32 & 0xfff) >> 4) << 10
236                | (read_unaligned(reloc_address as *mut u32) & 0xFFC003FF);
237            write_unaligned(reloc_address as *mut u32, reloc_delta);
238        },
239        RelocationKind::Aarch64Ldst64AbsLo12Nc => unsafe {
240            let (reloc_address, reloc_delta) = r.for_address(body, target_func_address as u64);
241            let reloc_delta = ((reloc_delta as u32 & 0xfff) >> 3) << 10
242                | (read_unaligned(reloc_address as *mut u32) & 0xFFC003FF);
243            write_unaligned(reloc_address as *mut u32, reloc_delta);
244        },
245        RelocationKind::MachoArm64RelocSubtractor | RelocationKind::MachoX86_64RelocSubtractor => unsafe {
246            let (reloc_address, reloc_sub) = r.for_address(body, target_func_address as u64);
247            macho_aarch64_subtractor_addresses.insert(reloc_address);
248            write_unaligned(reloc_address as *mut u64, reloc_sub);
249        },
250        RelocationKind::MachoArm64RelocGotLoadPage21
251        | RelocationKind::MachoArm64RelocTlvpLoadPage21 => unsafe {
252            let (reloc_address, _) = r.for_address(body, target_func_address as u64);
253            let target_func_page = target_func_address & !0xfff;
254            let reloc_at_page = reloc_address & !0xfff;
255            let pcrel = (target_func_page as isize)
256                .checked_sub(reloc_at_page as isize)
257                .unwrap();
258            assert!(
259                (-1 << 32) <= (pcrel as i64) && (pcrel as i64) < (1 << 32),
260                "can't reach GOT page with ±4GB `adrp` instruction"
261            );
262            let val = pcrel >> 12;
263
264            let immlo = ((val as u32) & 0b11) << 29;
265            let immhi = (((val as u32) >> 2) & 0x7ffff) << 5;
266            let mask = !((0x7ffff << 5) | (0b11 << 29));
267            let op = read_unaligned(reloc_address as *mut u32);
268            write_unaligned(reloc_address as *mut u32, (op & mask) | immlo | immhi);
269        },
270
271        RelocationKind::MachoArm64RelocPage21 => unsafe {
272            let target_page: u64 =
273                ((target_func_address.wrapping_add(r.addend() as _)) & !0xfff) as u64;
274            let reloc_address = body.wrapping_add(r.offset() as _);
275            let pc_page: u64 = (reloc_address & !0xfff) as u64;
276            let page_delta = target_page - pc_page;
277            let raw_instr = read_unaligned(reloc_address as *mut u32);
278            assert_eq!(
279                (raw_instr & 0xffffffe0),
280                0x90000000,
281                "raw_instr isn't an ADRP instruction"
282            );
283
284            let immlo: u32 = ((page_delta >> 12) & 0x3) as _;
285            let immhi: u32 = ((page_delta >> 14) & 0x7ffff) as _;
286            let fixed_instr = raw_instr | (immlo << 29) | (immhi << 5);
287            write_unaligned(reloc_address as *mut u32, fixed_instr);
288        },
289        RelocationKind::MachoArm64RelocPageoff12 => unsafe {
290            let target_offset: u64 =
291                ((target_func_address.wrapping_add(r.addend() as _)) & 0xfff) as u64;
292
293            let reloc_address = body.wrapping_add(r.offset() as _);
294            let raw_instr = read_unaligned(reloc_address as *mut u32);
295            let imm_shift = {
296                const VEC128_MASK: u32 = 0x04800000;
297
298                const LOAD_STORE_IMM12_MASK: u32 = 0x3b000000;
299                let is_load_store_imm12 = (raw_instr & LOAD_STORE_IMM12_MASK) == 0x39000000;
300
301                if is_load_store_imm12 {
302                    let mut implicit_shift = raw_instr >> 30;
303
304                    if implicit_shift == 0 && (raw_instr & VEC128_MASK) == VEC128_MASK {
305                        implicit_shift = 4;
306                    }
307
308                    implicit_shift
309                } else {
310                    0
311                }
312            };
313
314            assert_eq!(
315                target_offset & ((1 << imm_shift) - 1),
316                0,
317                "PAGEOFF12 target is not aligned"
318            );
319
320            let encoded_imm: u32 = ((target_offset as u32) >> imm_shift) << 10;
321            let fixed_instr: u32 = raw_instr | encoded_imm;
322            write_unaligned(reloc_address as *mut u32, fixed_instr);
323        },
324
325        RelocationKind::MachoArm64RelocGotLoadPageoff12 => unsafe {
326            // See comment at the top of the function. TLDR: if addend != 0 we can't really use the
327            // GOT entry. We fixup this relocation to use a `add` rather than a `ldr` instruction,
328            // skipping the indirection from the GOT.
329            if r.addend() == 0 {
330                let (reloc_address, _) = r.for_address(body, target_func_address as u64);
331                assert_eq!(target_func_address & 0b111, 0);
332                let val = target_func_address >> 3;
333                let imm9 = ((val & 0x1ff) << 10) as u32;
334                let mask = !(0x1ff << 10);
335                let op = read_unaligned(reloc_address as *mut u32);
336                write_unaligned(reloc_address as *mut u32, (op & mask) | imm9);
337            } else {
338                let fixup_ptr = body + r.offset() as usize;
339                let target_address: usize = target_func_address + r.addend() as usize;
340
341                let raw_instr = read_unaligned(fixup_ptr as *mut u32);
342
343                assert_eq!(
344                    raw_instr & 0xfffffc00,
345                    0xf9400000,
346                    "raw_instr isn't a 64-bit LDR immediate (bits: {raw_instr:032b}, hex: {raw_instr:x})"
347                );
348
349                let reg: u32 = raw_instr & 0b11111;
350
351                let mut fixup_ldr = 0x91000000 | (reg << 5) | reg;
352                fixup_ldr |= ((target_address & 0xfff) as u32) << 10;
353
354                write_unaligned(fixup_ptr as *mut u32, fixup_ldr);
355            }
356        },
357        RelocationKind::MachoArm64RelocUnsigned | RelocationKind::MachoX86_64RelocUnsigned => unsafe {
358            let (reloc_address, mut reloc_delta) = r.for_address(body, target_func_address as u64);
359
360            if macho_aarch64_subtractor_addresses.contains(&reloc_address) {
361                reloc_delta -= read_unaligned(reloc_address as *mut u64);
362            }
363
364            write_unaligned(reloc_address as *mut u64, reloc_delta);
365        },
366
367        RelocationKind::MachoArm64RelocPointerToGot => unsafe {
368            let at = body + r.offset() as usize;
369            let pcrel = i32::try_from((target_func_address as isize) - (at as isize)).unwrap();
370            write_unaligned(at as *mut i32, pcrel);
371        },
372
373        RelocationKind::MachoArm64RelocBranch26 => unsafe {
374            let fixup_ptr = body + r.offset() as usize;
375            assert_eq!(fixup_ptr & 0x3, 0, "Branch-inst is not 32-bit aligned");
376            let value = i32::try_from((target_func_address as isize) - (fixup_ptr as isize))
377                .unwrap()
378                .wrapping_add(r.addend() as _);
379            assert!(
380                value & 0x3 == 0,
381                "BranchPCRel26 target is not 32-bit aligned"
382            );
383
384            assert!(
385                (-(1 << 27)..=((1 << 27) - 1)).contains(&value),
386                "out of range BranchPCRel26 target"
387            );
388
389            let raw_instr = read_unaligned(fixup_ptr as *mut u32);
390
391            assert_eq!(
392                raw_instr & 0x7fffffff,
393                0x14000000,
394                "RawInstr isn't a B or BR immediate instruction"
395            );
396            let imm: u32 = ((value as u32) & ((1 << 28) - 1)) >> 2;
397            let fixed_instr: u32 = raw_instr | imm;
398
399            write_unaligned(fixup_ptr as *mut u32, fixed_instr);
400        },
401        kind => panic!("Relocation kind unsupported in the current architecture: {kind}"),
402    }
403}

405/// Links a module, patching the allocated functions with the
406/// required relocations and jump tables.
407#[allow(clippy::too_many_arguments)]
408pub fn link_module<'a>(
409    _module: &ModuleInfo,
410    allocated_functions: &PrimaryMap<LocalFunctionIndex, FunctionExtent>,
411    function_relocations: impl Iterator<
412        Item = (
413            LocalFunctionIndex,
414            impl Iterator<Item = &'a (impl RelocationLike + 'a)>,
415        ),
416    >,
417    allocated_sections: &PrimaryMap<SectionIndex, SectionBodyPtr>,
418    section_relocations: impl Iterator<
419        Item = (
420            SectionIndex,
421            impl Iterator<Item = &'a (impl RelocationLike + 'a)>,
422        ),
423    >,
424    libcall_trampolines: SectionIndex,
425    trampoline_len: usize,
426    get_got_address: &'a dyn Fn(RelocationTarget) -> Option<usize>,
427) {
428    let mut riscv_pcrel_hi20s: HashMap<usize, u32> = HashMap::new();
429
430    for (i, section_relocs) in section_relocations {
431        let body = *allocated_sections[i] as usize;
432        for r in section_relocs {
433            apply_relocation(
434                body,
435                r,
436                allocated_functions,
437                allocated_sections,
438                libcall_trampolines,
439                trampoline_len,
440                &mut riscv_pcrel_hi20s,
441                get_got_address,
442            );
443        }
444    }
445    for (i, function_relocs) in function_relocations {
446        let body = *allocated_functions[i].ptr as usize;
447        for r in function_relocs {
448            apply_relocation(
449                body,
450                r,
451                allocated_functions,
452                allocated_sections,
453                libcall_trampolines,
454                trampoline_len,
455                &mut riscv_pcrel_hi20s,
456                get_got_address,
457            );
458        }
459    }
