22/// The `gimli` reader type used for DWARF sections loaded from an ELF image.
23///
24/// Each section's bytes are copied out of the source image into their own
25/// `Arc<[u8]>`, so the reader is independent of the lifetime of the
26/// `object::File` (or the buffer it was parsed from) used to load it.
27pub type DwarfReader = gimli::EndianArcSlice<gimli::RunTimeEndian>;
28
29/// Lazily-loaded DWARF debug info for an ELF-backed artifact.
30///
31/// Building an `addr2line::Context` parses the DWARF sections eagerly, which
32/// is wasted work for modules that are never symbolicated (e.g. no trap or
33/// backtrace ever occurs). This defers that work until the first lookup.
34#[derive(Clone)]
35pub(crate) enum DebugInfoSource {
36    Bytes(Arc<[u8]>),
37    File(Arc<File>),
38}
39
40pub(crate) struct DebugInfo {
41    /// The ELF image, kept around (or reopened) so the DWARF sections can be
42    /// loaded on first use. `None` for non-ELF artifacts.
43    elf_data: Option<DebugInfoSource>,
44    /// `None` until first accessed; `Some(None)` once loading was attempted
45    /// and failed (or there was no ELF image to load from).
46    ///
47    /// `addr2line::Context` caches parsed DWARF units behind plain
48    /// `OnceCell`s internally, so it is `Send` but not `Sync` — a `Mutex`
49    /// serializes lookups from concurrent backtraces/traps instead of
50    /// exposing a `&Context` that could be read from multiple threads at
51    /// once.
52    context: Mutex<Option<Option<addr2line::Context<DwarfReader>>>>,
53}
54
55impl DebugInfo {
56    pub(crate) fn new(elf_data: Option<DebugInfoSource>) -> Self {
57        Self {
58            elf_data,
59            context: Mutex::new(None),
60        }
61    }
62
63    /// Runs `f` with the DWARF context, building it from the ELF image on
64    /// first access. `f` receives `None` if there is no ELF image, or the
65    /// image has no (or malformed) DWARF debug info.
66    pub(crate) fn with_context<T>(
67        &self,
68        f: impl FnOnce(Option<&addr2line::Context<DwarfReader>>) -> T,
69    ) -> T {
70        let mut context = self.context.lock().unwrap();
71        let context = context.get_or_insert_with(|| {
72            let elf_data = match self.elf_data.as_ref()? {
73                DebugInfoSource::Bytes(data) => data.clone(),
74                DebugInfoSource::File(file) => {
75                    let mut file = file.try_clone().ok()?;
76                    use std::io::{Read as _, Seek as _};
77                    file.rewind().ok()?;
78                    let mut data = Vec::new();
79                    file.read_to_end(&mut data).ok()?;
80                    Arc::from(data)
81                }
82            };
83            let object_file = object::File::parse(&elf_data[..]).ok()?;
84            load_dwarf_context(&object_file).ok()
85        });
86        f(context.as_ref())
87    }
88}
89
90fn load_dwarf_context(
91    object_file: &object::File<'_>,
92) -> Result<addr2line::Context<DwarfReader>, gimli::Error> {
93    let endian = if object_file.is_little_endian() {
94        gimli::RunTimeEndian::Little
95    } else {
96        gimli::RunTimeEndian::Big
97    };
98
99    let load_section = |id: gimli::SectionId| -> Result<DwarfReader, gimli::Error> {
100        let data: Vec<u8> = object_file
101            .section_by_name(id.name())
102            .and_then(|section| section.uncompressed_data().ok())
103            .map(|data| data.into_owned())
104            .unwrap_or_default();
105        Ok(gimli::EndianReader::new(Arc::from(data), endian))
106    };
107
108    let dwarf = gimli::Dwarf::load(load_section)?;
109    addr2line::Context::from_dwarf(dwarf)
110}

179struct ImageSegment {
180    pub(crate) mem_address: usize,
181    pub(crate) mem_size: usize,
182    pub(crate) file_address: usize,
183    pub(crate) file_size: usize,
184    pub(crate) page_size: usize,
185    pub(crate) flags: SegmentFlags,
186}
187
188#[cfg(unix)]
189impl ImageSegment {
190    fn protection(&self) -> Result<i32, String> {
191        let (read, write, exec) = match self.flags {
192            SegmentFlags::Elf { p_flags } => (
193                p_flags & elf::PF_R != 0,
194                p_flags & elf::PF_W != 0,
195                p_flags & elf::PF_X != 0,
196            ),
197            _ => return Err(format!("unsupported segment flags: {:?}", self.flags)),
198        };
199
200        let mut protection = 0;
201        if read {
202            protection |= libc::PROT_READ;
203        }
204        if write {
205            protection |= libc::PROT_WRITE;
206        }
207        if exec {
208            protection |= libc::PROT_EXEC;
209        }
210        Ok(protection)
211    }
212
213    fn mem_size_page_aligned(&self) -> usize {
214        (self.mem_size + (self.mem_address - self.mem_address_page_aligned()))
215            .next_multiple_of(self.page_size)
216    }
217
218    fn mem_address_page_aligned(&self) -> usize {
219        self.mem_address & !(self.page_size - 1)
220    }
221
222    fn file_size_page_aligned(&self) -> usize {
223        (self.file_size + (self.file_address - self.file_address_page_aligned()))
224            .next_multiple_of(self.page_size)
225    }
226
227    fn file_address_page_aligned(&self) -> usize {
228        self.file_address & !(self.page_size - 1)
229    }
230}

265    /// Maps an ELF image's load segments directly from an open file.
266    pub(crate) fn try_from_file<'a, R: ReadRef<'a>>(
267        object_file: &object::File<'a, R>,

272
273    fn try_from_source<'a, R: ReadRef<'a>>(
274        object_file: &object::File<'a, R>,
275        data: Option<&[u8]>,
276        file: Option<RawFd>,
277    ) -> Result<Self, String> {
278        let page_size = unsafe { libc::sysconf(libc::_SC_PAGESIZE) };
279        if page_size == -1 {
280            return Err("Cannot get page size".to_string());
281        }
282        let page_size = page_size as usize;
283
284        let segments = object_file
285            .segments()
286            .map(|segment| {
287                let mem_address = segment.address() as usize;
288                let mem_size = segment.size() as usize;
289                let (file_address, file_size) = segment.file_range();
290                let file_address = file_address as usize;
291                let file_size = file_size as usize;
292                ImageSegment {
293                    mem_address,
294                    mem_size,
295                    file_address,
296                    file_size,
297                    page_size,
298                    flags: segment.flags(),
299                }
300            })
301            .collect_vec();
302        let last_segment = segments
303            .last()
304            .ok_or("at least one segment is mandatory".to_string())?;
305        let total_memory_size =
306            last_segment.mem_address_page_aligned() + last_segment.mem_size_page_aligned();
307
308        // Create a contiguous virtual address memory map that will be populated
309        // per-partes with the individual protection flags.
310        let map = Self::new_mmap(total_memory_size)?;
311        let base = map.base();
312
313        // Mmap individual load segments
314        for load_segment in segments {
315            // The virtual offset does not need to start at a page boundary.
316            if load_segment.file_address % page_size != load_segment.mem_address % page_size {
317                return Err(format!(
318                    "Load segment file offset 0x{:x} and virtual address 0x{:x} have incompatible page alignment",
319                    load_segment.file_address, load_segment.mem_address
320                ));
321            }
322
323            let protection = load_segment.protection()?;
324
325            let offset = load_segment.mem_address_page_aligned();
326            let size = load_segment.file_size_page_aligned();
327            let file_offset = load_segment.file_address_page_aligned();
328            let result = if let Some(file) = file {
329                map.map_file(offset, size, protection, file, file_offset)
330            } else {
331                map.map_copy(
332                    offset,
333                    size,
334                    protection,
335                    data.expect("byte-backed mapping requires image data"),
336                    file_offset,
337                )
338            };
339            result.map_err(|error| {
340                format!(
341                    "Cannot map load segment at virtual address 0x{:x}: {error}",
342                    load_segment.mem_address_page_aligned()
343                )
344            })?;
345
346            if load_segment.mem_size_page_aligned() > load_segment.file_size_page_aligned() {
347                map.map_zero(
348                    load_segment.mem_address_page_aligned() + load_segment.file_size_page_aligned(),
349                    load_segment.mem_size_page_aligned() - load_segment.file_size_page_aligned(),
350                    protection,
351                )
352                .map_err(|error| format!("Cannot map zero-fill segment tail: {error}"))?;
353            }
354            if load_segment.mem_size_page_aligned() < load_segment.file_size_page_aligned() {
355                return Err("invalid memory segment with larger file representation".to_string());
356            }
357        }
358
359        // Apply dynamic relocations for the libcalls
360        if let Some(dynamic_relocations) = object_file.dynamic_relocations() {
361            let dynamic_symbols = object_file.dynamic_symbol_table().unwrap();
362
363            for (offset, relocation) in dynamic_relocations {
364                let rel_flags = relocation.flags();
365                if matches!(
366                    rel_flags,
367                    object::RelocationFlags::Elf {
368                        r_type: elf::R_X86_64_RELATIVE,
369                    }
370                ) {
371                    unsafe {
372                        ptr::write_unaligned(
373                            base.add(offset as usize) as *mut usize,
374                            (base as usize).wrapping_add(relocation.addend() as usize),
375                        );
376                    }
377                    continue;
378                }
379
380                let object::RelocationTarget::Symbol(symbol_index) = relocation.target() else {
381                    return Err("unsupported dynamic relocation target".to_string());
382                };
383                let symbol = dynamic_symbols.symbol_by_index(symbol_index).unwrap();
384                let symbol_name = symbol.name().unwrap();
385                let Some(&libcall) = LIBCALLS_ELF.get(symbol_name) else {
386                    return Err(format!(
387                        "unsupported dynamic relocation symbol {symbol_name}"
388                    ));
389                };
390
391                let apply_absolute_relocation = || unsafe {
392                    ptr::write_unaligned(
393                        base.add(offset as usize) as *mut usize,
394                        function_pointer(libcall).wrapping_add(relocation.addend() as usize),
395                    );
396                };
397                match (relocation.kind(), rel_flags) {
398                    (object::RelocationKind::Absolute, _) => apply_absolute_relocation(),
399                    (
400                        object::RelocationKind::Unknown,
401                        object::RelocationFlags::Elf {
402                            r_type: elf::R_X86_64_GLOB_DAT,
403                        },
404                    ) => apply_absolute_relocation(),
405                    (
406                        object::RelocationKind::Unknown,
407                        object::RelocationFlags::Elf {
408                            r_type: elf::R_X86_64_JUMP_SLOT,
409                        },
410                    ) => apply_absolute_relocation(),
411                    kind => return Err(format!("unsupported dynamic relocation kind {kind:?}")),
412                }
413            }
414        }
415
416        Ok(map)
417    }

419    fn new_mmap(size: usize) -> Result<Self, String> {
420        let base = unsafe {
421            libc::mmap(
422                ptr::null_mut(),
423                size,
424                libc::PROT_NONE,
425                libc::MAP_PRIVATE | libc::MAP_ANONYMOUS,
426                -1,
427                0,
428            )
429        };
430        if base == libc::MAP_FAILED {
431            return Err("Cannot create a memory map for built Artifact".to_string());
432        }
433
434        Ok(Self {
435            base,
436            size,
437            unwind_registry: Some(UnwindRegistry::new()),
438            frame_info_registration: None,
439        })
440    }

488    /// Maps an anonymous zero-filled region at `offset` with the given
489    /// protection (used for a segment's BSS tail).
490    fn map_zero(&self, offset: usize, size: usize, protection: i32) -> Result<(), String> {
491        if offset + size > self.size {
492            return Err("Segment will overwrite allocated range".to_string());
493        }
494        let result = unsafe {
495            libc::mmap(
496                self.base.add(offset),
497                size,
498                protection,
499                libc::MAP_PRIVATE | libc::MAP_ANONYMOUS | libc::MAP_FIXED,
500                -1,
501                0,
502            )
503        };
504        if result == libc::MAP_FAILED {
505            return Err(std::io::Error::last_os_error().to_string());
506        }
507        Ok(())
508    }

510    /// Maps a region at `offset` directly from a file.
511    fn map_file(
512        &self,
513        offset: usize,
514        size: usize,
515        protection: i32,
516        file: RawFd,
517        file_offset: usize,
518    ) -> Result<(), String> {
519        if offset + size > self.size {
520            return Err("Segment will overwrite allocated range".to_string());
521        }
522        let result = unsafe {
523            libc::mmap(
524                self.base.add(offset),
525                size,
526                protection,
527                libc::MAP_PRIVATE | libc::MAP_FIXED,
528                file,
529                file_offset as libc::off_t,
530            )
531        };
532        if result == libc::MAP_FAILED {
533            return Err(std::io::Error::last_os_error().to_string());
534        }
535        Ok(())
536    }
537
538    /// Maps an anonymous region at `offset` and copies `size` bytes from
539    /// `data[file_offset..]` into it, then applies the final protection.
540    ///
541    /// Copying (rather than mapping the backing file directly) keeps this
542    /// portable: on macOS/Mach-O a file-backed `MAP_FIXED` mapping cannot be
543    /// created with executable protection, and here we don't need a real
544    /// file descriptor for the image at all.
545    fn map_copy(
546        &self,
547        offset: usize,
548        size: usize,
549        protection: i32,
550        data: &[u8],
551        file_offset: usize,
552    ) -> Result<(), String> {
553        if offset + size > self.size {
554            return Err("Segment will overwrite allocated range".to_string());
555        }
556        let dest = unsafe { self.base.add(offset) };
557        let result = unsafe {
558            libc::mmap(
559                dest,
560                size,
561                libc::PROT_READ | libc::PROT_WRITE,
562                libc::MAP_PRIVATE | libc::MAP_ANONYMOUS | libc::MAP_FIXED,
563                -1,
564                0,
565            )
566        };
567        if result == libc::MAP_FAILED {
568            return Err(std::io::Error::last_os_error().to_string());
569        }
570
571        let available = data.len().saturating_sub(file_offset).min(size);
572        unsafe {
573            ptr::copy_nonoverlapping(data.as_ptr().add(file_offset), dest as *mut u8, available);
574        }
575
576        if protection != (libc::PROT_READ | libc::PROT_WRITE)
577            && unsafe { libc::mprotect(dest, size, protection) } != 0
578        {
579            return Err(std::io::Error::last_os_error().to_string());
580        }
581        Ok(())
582    }

609#[cfg(unix)]
610impl Drop for MemoryMappedBinary {
611    fn drop(&mut self) {
612        // The registered `.eh_frame` records point into this mmap, so deregister
613        // them while the mapping is still live.
614        drop(self.unwind_registry.take());
615
616        if !self.base.is_null() && self.size != 0 {
617            unsafe {
