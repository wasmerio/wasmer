58    /// Compile a data buffer into a `ArtifactBuild`, which may then be instantiated.
59    #[cfg(feature = "compiler")]
60    pub fn new(
61        inner_engine: &mut EngineInner,
62        data: &[u8],
63        target: &Target,
64        memory_styles: PrimaryMap<MemoryIndex, MemoryStyle>,
65        table_styles: PrimaryMap<TableIndex, TableStyle>,
66        hash_algorithm: Option<HashAlgorithm>,
67        progress_callback: Option<&CompilationProgressCallback>,
68    ) -> Result<Self, CompileError> {
69        let environ = ModuleEnvironment::new();
70        let features = inner_engine.features().clone();
71
72        let translation = environ.translate(data).map_err(CompileError::Wasm)?;
73
74        let compiler = inner_engine.compiler()?;
75
76        // We try to apply the middleware first
77        let mut module = translation.module;
78        let middlewares = compiler.get_middlewares();
79        middlewares
80            .apply_on_module_info(&mut module)
81            .map_err(|err| CompileError::MiddlewareError(err.to_string()))?;
82
83        if let Some(hash_algorithm) = hash_algorithm {
84            let hash = match hash_algorithm {
85                HashAlgorithm::Sha256 => ModuleHash::sha256(data),
86                HashAlgorithm::XXHash => ModuleHash::xxhash(data),
87            };
88
89            module.hash = Some(hash);
90        }
91
92        let compile_info = CompileModuleInfo {
93            module: Arc::new(module),
94            features,
95            memory_styles,
96            table_styles,
97        };
98
99        // Compile the Module
100        let compilation = compiler.compile_module(
101            target,
102            &compile_info,
103            // SAFETY: Calling `unwrap` is correct since
104            // `environ.translate()` above will write some data into
105            // `module_translation_state`.
106            translation.module_translation_state.as_ref().unwrap(),
107            translation.function_body_inputs,
108            progress_callback,
109        )?;
110
111        let data_initializers = translation
112            .data_initializers
113            .iter()
114            .map(OwnedDataInitializer::new)
115            .collect::<Vec<_>>()
116            .into_boxed_slice();
117
118        // Synthesize a custom section to hold the libcall trampolines.
119        let mut function_frame_info = PrimaryMap::with_capacity(compilation.functions.len());
120        let mut function_bodies = PrimaryMap::with_capacity(compilation.functions.len());
121        let mut function_relocations = PrimaryMap::with_capacity(compilation.functions.len());
122        for (_, func) in compilation.functions.into_iter() {
123            function_bodies.push(func.body);
124            function_relocations.push(func.relocations);
125            function_frame_info.push(func.frame_info);
126        }
127        let mut custom_sections = compilation.custom_sections.clone();
128        let mut custom_section_relocations = compilation
129            .custom_sections
130            .iter()
131            .map(|(_, section)| section.relocations.clone())
132            .collect::<PrimaryMap<SectionIndex, _>>();
133        let libcall_trampolines_section = make_libcall_trampolines(target);
134        custom_section_relocations.push(libcall_trampolines_section.relocations.clone());
135        let libcall_trampolines = custom_sections.push(libcall_trampolines_section);
136        let libcall_trampoline_len = libcall_trampoline_len(target) as u32;
137        let cpu_features = compiler.get_cpu_features_used(target.cpu_features());
138
139        let serializable_compilation = SerializableCompilation {
140            function_bodies,
141            function_relocations,
142            function_frame_info,
143            function_call_trampolines: compilation.function_call_trampolines,
144            dynamic_function_trampolines: compilation.dynamic_function_trampolines,
145            custom_sections,
146            custom_section_relocations,
147            unwind_info: compilation.unwind_info,
148            libcall_trampolines,
149            libcall_trampoline_len,
150            got: compilation.got,
151        };
152        let serializable = SerializableModule {
153            compilation: serializable_compilation,
154            compile_info,
155            data_initializers,
156            cpu_features: cpu_features.as_u64(),
157        };
158        Ok(Self { serializable })
159    }

323impl ArtifactBuildFromArchive {
324    #[allow(unused)]
325    pub(crate) fn try_new(
326        buffer: OwnedBuffer,
327        module_builder: impl FnOnce(
328            &OwnedBuffer,
329        ) -> Result<&ArchivedSerializableModule, DeserializeError>,
330    ) -> Result<Self, DeserializeError> {
331        let mut compile_info = MaybeUninit::uninit();
332
333        let cell = ArtifactBuildFromArchiveCell::try_new(buffer, |buffer| {
334            let module = module_builder(buffer)?;
335            compile_info = MaybeUninit::new(
336                rkyv::deserialize::<_, RkyvError>(&module.compile_info)
337                    .map_err(|e| DeserializeError::CorruptedBinary(format!("{e:?}")))?,
338            );
339            ModuleFromArchive::from_serializable_module(module)
340        })?;
341
342        // Safety: we know the lambda will execute before getting here and assign both values
343        let compile_info = unsafe { compile_info.assume_init() };
344        Ok(Self {
345            cell: Arc::new(cell),
346            compile_info,
347        })
348    }

487    }

491    }

494        &self.compile_info.table_styles
495    }

497    fn serialize(&self) -> Result<Vec<u8>, SerializeError> {
498        // We could have stored the original bytes, but since the module info name
499        // is mutable, we have to assume the data may have changed and serialize
500        // everything all over again. Also, to be able to serialize, first we have
501        // to deserialize completely. Luckily, serializing a module that was already
502        // deserialized from a file makes little sense, so hopefully, this is not a
503        // common use-case.
504
505        let mut module: SerializableModule =
506            rkyv::deserialize::<_, RkyvError>(self.cell.borrow_dependent().original_module)
507                .map_err(|e| SerializeError::Generic(e.to_string()))?;
508        module.compile_info = self.compile_info.clone();
509        serialize_module(&module)
510    }
511}

513fn serialize_module(module: &SerializableModule) -> Result<Vec<u8>, SerializeError> {
514    let serialized_data = module.serialize()?;
515    assert!(std::mem::align_of::<SerializableModule>() <= MetadataHeader::ALIGN);
516
517    let mut metadata_binary = vec![];
518    metadata_binary.extend(ArtifactBuild::MAGIC_HEADER);
519    metadata_binary.extend(MetadataHeader::new(serialized_data.len()).into_bytes());
520    metadata_binary.extend(serialized_data);
521    Ok(metadata_binary)
522}
