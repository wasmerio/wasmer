14/// Represents the memory layout of the parts that the thread itself uses
15#[derive(Debug, Clone, Default, PartialEq, Eq, Hash)]
16#[cfg_attr(feature = "enable-serde", derive(Serialize, Deserialize))]
17#[cfg_attr(feature = "enable-serde", serde(rename_all = "snake_case"))]
18pub struct WasiMemoryLayout {
19    /// This is the top part of the stack (stacks go backwards)
20    pub stack_upper: u64,
21    /// This is the bottom part of the stack (anything more below this is a stack overflow)
22    pub stack_lower: u64,
23    /// Piece of memory that is marked as none readable/writable so stack overflows cause an exception
24    /// TODO: This field will need to be used to mark the guard memory as inaccessible
25    #[allow(dead_code)]
26    pub guard_size: u64,
27    /// Total size of the stack
28    pub stack_size: u64,
29    /// Pointer to the thread's TLS (Thread Local Storage) memory, only filled in when it's known.
30    /// This is mainly used by the dynamic linker; especially so since we can't even find the TLS
31    /// area for a non-PIE module that doesn't export its __tls_base. For DL modules, this is the
32    /// TLS base of the main module.
33    pub tls_base: Option<u64>,
34}
