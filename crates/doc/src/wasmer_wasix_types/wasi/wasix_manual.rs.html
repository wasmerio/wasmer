108#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
109#[repr(C)]
110pub struct StackSnapshot {
111    pub user: u64,
112    // The hash is defined as two u64s in wasix-libc, so we must do the same
113    // here to make sure the alignment and padding are the same.
114    pub hash_lower: u64,
115    pub hash_upper: u64,
116}
117
118impl StackSnapshot {
119    pub fn new(user: u64, hash: u128) -> Self {
120        Self {
121            user,
122            hash_lower: (hash & 0xffff_ffff_ffff_ffff) as u64,
123            hash_upper: (hash >> 64) as u64,
124        }
125    }
126
127    pub fn hash(&self) -> u128 {
128        ((self.hash_upper as u128) << 64) | (self.hash_lower as u128)
129    }
130}
131
132#[test]
133fn snapshot_hash_roundtrip() {
134    let hash = 0x1234_5678_90ab_cdef_0987_6543_fedc_ba12;
135    let snapshot = StackSnapshot::new(0, hash);
136    assert_eq!(snapshot.hash(), hash);
137}
