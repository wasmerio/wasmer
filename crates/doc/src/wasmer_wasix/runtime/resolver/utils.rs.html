9/// Polyfill for [`Url::from_file_path()`] that works on `wasm32-unknown-unknown`.
10pub(crate) fn url_from_file_path(path: impl AsRef<Path>) -> Option<Url> {
11    let path = path.as_ref();
12
13    if !path.is_absolute() {
14        return None;
15    }
16
17    let mut buffer = String::new();
18
19    for component in path {
20        if !buffer.ends_with('/') {
21            buffer.push('/');
22        }
23
24        buffer.push_str(component.to_str()?);
25    }
26
27    buffer.insert_str(0, "file://");
28
29    buffer.parse().ok()
30}

39pub(crate) fn http_error(response: &HttpResponse) -> Error {
40    let status = response.status;
41
42    if status == StatusCode::SERVICE_UNAVAILABLE {
43        if let Some(retry_after) = response
44            .headers
45            .get("Retry-After")
46            .and_then(|retry_after| retry_after.to_str().ok())
47        {
48            tracing::debug!(
49                %retry_after,
50                "Received 503 Service Unavailable while looking up a package. The backend may still be generating the *.webc file.",
51            );
52            return anyhow::anyhow!("{status} (Retry After: {retry_after})");
53        }
54    }
55
56    Error::msg(status)
57}

59pub(crate) fn file_path_from_url(url: &Url) -> Result<PathBuf, Error> {
60    debug_assert_eq!(url.scheme(), "file");
61
62    // Note: The Url::to_file_path() method is platform-specific
63    cfg_if::cfg_if! {
64        if #[cfg(any(unix, windows, target_os = "redox", target_os = "wasi"))] {
65            use anyhow::Context;
66
67            if let Ok(path) = url.to_file_path() {
68                return Ok(path);
69            }
70
71            // Sometimes we'll get a UNC-like path (e.g.
72            // "file:///?\\C:/\\/path/to/file.txt") and Url::to_file_path()
73            // won't be able to handle the "\\?" so we try to "massage" the URL
74            // a bit.
75            // See <https://github.com/servo/rust-url/issues/450> for more.
76            let modified = url.as_str().replace(r"\\?", "").replace("//?", "").replace('\\', "/");
77            Url::parse(&modified)
78                .ok()
79                .and_then(|url| url.to_file_path().ok())
80                .context("Unable to extract the file path")
81        } else {
82            anyhow::bail!("Url::to_file_path() is not supported on this platform");
83        }
84    }
85}
