8/// A cache for compiled WebAssembly modules.
9///
10/// ## Deterministic ID
11///
12/// Implementations are encouraged to take the [`Engine::deterministic_id()`]
13/// into account when saving and loading cached a [`Module`].
14///
15/// ## Assumptions
16///
17/// Implementations can assume that cache keys are unique and that using the
18/// same key to load or save will always result in the "same" module.
19///
20/// Implementations can also assume that [`ModuleCache::load()`] will
21/// be called more often than [`ModuleCache::save()`] and optimise
22/// their caching strategy accordingly.
23///
24#[async_trait::async_trait]
25pub trait ModuleCache: Debug {
26    /// Load a module based on its hash.
27    async fn load(&self, key: ModuleHash, engine: &Engine) -> Result<Module, CacheError>;
28
29    /// Check if a module is present in the cache.
30    async fn contains(&self, key: ModuleHash, engine: &Engine) -> Result<bool, CacheError>;
31
32    /// Save a module so it can be retrieved with [`ModuleCache::load()`] at a
33    /// later time.
34    ///
35    /// # Panics
36    ///
37    /// Implementations are free to assume the [`Module`] being passed in was
38    /// compiled using the provided [`Engine`], and may panic if this isn't the
39    /// case.
40    async fn save(
41        &self,
42        key: ModuleHash,
43        engine: &Engine,
44        module: &Module,
45    ) -> Result<(), CacheError>;
46
47    /// Chain a second [`ModuleCache`] that will be used as a fallback if
48    /// lookups on the primary cache fail.
49    ///
50    /// The general assumption is that each subsequent cache in the chain will
51    /// be significantly slower than the previous one.
52    ///
53    /// ```rust
54    /// use wasmer_wasix::runtime::module_cache::{
55    ///     ModuleCache, ThreadLocalCache, FileSystemCache, SharedCache,
56    /// };
57    /// use wasmer_wasix::runtime::task_manager::tokio::{RuntimeOrHandle, TokioTaskManager};
58    ///
59    /// let runtime = tokio::runtime::Runtime::new().unwrap();
60    /// let rt_handle = RuntimeOrHandle::from(runtime);
61    /// let task_manager = std::sync::Arc::new(TokioTaskManager::new(rt_handle));
62    ///
63    /// let cache = SharedCache::default()
64    ///     .with_fallback(FileSystemCache::new("~/.local/cache", task_manager));
65    /// ```
66    fn with_fallback<C>(self, other: C) -> FallbackCache<Self, C>
67    where
68        Self: Sized,
69        C: ModuleCache,
70    {
71        FallbackCache::new(self, other)
72    }
73}

75#[async_trait::async_trait]
76impl<D, C> ModuleCache for D
77where
78    D: Deref<Target = C> + Debug + Send + Sync,
79    C: ModuleCache + Send + Sync + ?Sized,
80{
81    async fn load(&self, key: ModuleHash, engine: &Engine) -> Result<Module, CacheError> {
82        (**self).load(key, engine).await
83    }
84
85    async fn contains(&self, key: ModuleHash, engine: &Engine) -> Result<bool, CacheError> {
86        (**self).contains(key, engine).await
87    }
88
89    async fn save(
90        &self,
91        key: ModuleHash,
92        engine: &Engine,
93        module: &Module,
94    ) -> Result<(), CacheError> {
95        (**self).save(key, engine, module).await
96    }
97}

99/// Possible errors that may occur during [`ModuleCache`] operations.
100#[derive(Debug, thiserror::Error)]
101pub enum CacheError {
102    #[error("Unable to serialize the module")]
103    Serialize(#[from] wasmer::SerializeError),
104    #[error("Unable to deserialize the module")]
105    Deserialize(#[from] wasmer::DeserializeError),
106    #[error("Unable to read from \"{}\"", path.display())]
107    FileRead {
108        path: PathBuf,
109        #[source]
110        error: std::io::Error,
111    },
112    #[error("Unable to write to \"{}\"", path.display())]
113    FileWrite {
114        path: PathBuf,
115        #[source]
116        error: std::io::Error,
117    },
118    /// The item was not found.
119    #[error("Not found")]
120    NotFound,
121    /// A catch-all variant for any other errors that may occur.
122    #[error(transparent)]
123    Other(Box<dyn std::error::Error + Send + Sync>),
124}
125
126impl CacheError {
127    pub fn other(error: impl std::error::Error + Send + Sync + 'static) -> Self {
128        CacheError::Other(Box::new(error))
129    }
130}
