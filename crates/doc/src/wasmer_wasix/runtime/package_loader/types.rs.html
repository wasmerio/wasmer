20
21    /// Load a resolved package into memory so it can be executed.

23    /// A good default implementation is to just call

33#[async_trait::async_trait]
34impl<D, P> PackageLoader for D
35where
36    D: Deref<Target = P> + Debug + Send + Sync,
37    P: PackageLoader + ?Sized + 'static,
38{
39    async fn load(&self, summary: &PackageSummary) -> Result<Container, Error> {
40        (**self).load(summary).await
41    }
42
43    async fn load_package_tree(
44        &self,
45        root: &Container,
46        resolution: &Resolution,
47        root_is_local_dir: bool,
48    ) -> Result<BinaryPackage, Error> {
49        (**self)
50            .load_package_tree(root, resolution, root_is_local_dir)
51            .await
52    }
