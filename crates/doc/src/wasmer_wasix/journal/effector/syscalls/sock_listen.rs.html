3impl JournalEffector {

6        fd: Fd,

13                backlog: backlog as u32,

18    pub fn apply_sock_listen(
19        ctx: &mut FunctionEnvMut<'_, WasiEnv>,
20        fd: Fd,
21        backlog: usize,
22    ) -> anyhow::Result<()> {
23        crate::syscalls::sock_listen_internal(ctx, fd, backlog)
24            .map(|r| r.map_err(|err| err.to_string()))
25            .unwrap_or_else(|err| Err(err.to_string()))
26            .map_err(|err| {
27                anyhow::format_err!(
28                    "journal restore error: failed to listen on socket (fd={fd}, backlog={backlog}) - {err}")
29            })?;
30        Ok(())
31    }
