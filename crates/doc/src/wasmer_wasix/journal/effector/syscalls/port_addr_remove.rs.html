1use std::net::IpAddr;
2
3use super::*;
4
5impl JournalEffector {
6    pub fn save_port_addr_remove(
7        ctx: &mut FunctionEnvMut<'_, WasiEnv>,
8        addr: IpAddr,
9    ) -> anyhow::Result<()> {
10        Self::save_event(ctx, JournalEntry::PortDelAddrV1 { addr })
11    }
12
13    pub fn apply_port_addr_remove(
14        ctx: &mut FunctionEnvMut<'_, WasiEnv>,
15        addr: IpAddr,
16    ) -> anyhow::Result<()> {
17        crate::syscalls::port_addr_remove_internal(ctx, addr)
18            .map(|r| r.map_err(|err| err.to_string()))
19            .unwrap_or_else(|err| Err(err.to_string()))
20            .map_err(|err| {
21                anyhow::format_err!(
22                    "journal restore error: failed to remove address from port (ip={addr}) - {err}"
23                )
24            })?;
25        Ok(())
26    }
27}
