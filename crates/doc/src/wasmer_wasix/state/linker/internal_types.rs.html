21#[derive(Debug)]
22pub(super) enum UnresolvedGlobal {
23    // A GOT.mem entry, should be resolved to an exported global from another module.
24    Mem(NeededSymbolResolutionKey, Global),
25    // A GOT.func entry, should be resolved to the address of an exported function
26    // from another module (e.g. an index into __indirect_function_table).
27    Func(NeededSymbolResolutionKey, Global),
28}
29
30impl UnresolvedGlobal {
31    pub(super) fn key(&self) -> &NeededSymbolResolutionKey {
32        match self {
33            Self::Func(key, _) => key,
34            Self::Mem(key, _) => key,
35        }
36    }
37
38    pub(super) fn global(&self) -> &Global {
39        match self {
40            Self::Func(_, global) => global,
41            Self::Mem(_, global) => global,
42        }
43    }
44
45    pub(super) fn import_module(&self) -> &str {
46        match self {
47            Self::Func(..) => "GOT.func",
48            Self::Mem(..) => "GOT.mem",
49        }
50    }
51}

67// Used only when processing a module load operation from another instance group.
68// Note: Non-TLS globals are constant across instance groups, and thus we store
69// their value, feeding it into new instance groups directly. In the case of TLS
70// symbols, they need to get new values based on each specific instance's __tls_base,
71// so they need to be tracked.
72// __wasm_apply_data_relocs operates on memory addresses, and so it needs to run
73// only once. __wasm_apply_tls_relocs does need to run once per instance group, but
74// it's run as part of __wasm_init_tls, which itself is called by __wasix_init_tls.
75// In either case, there's no need to call any relocation functions when spawning
76// further instances of a module that was already loaded and instantiated once.
77#[derive(Debug, Default)]
78pub(super) struct PendingResolutionsFromLinker {
79    pub(super) functions: Vec<PendingFunctionResolutionFromLinkerState>,
80    pub(super) tls: Vec<PendingTlsPointer>,
81}

142#[derive(Debug)]
143pub(super) enum SymbolResolutionResult {
144    // The symbol was resolved to a global address. We don't resolve again because
145    // the value of globals and the memory_base for each module and all of its instances
146    // is fixed.
147    // The case of unresolved globals is not mentioned in this enum, since it can't exist
148    // once a link operation is finalized.
149    Memory(u64),
150    // The symbol was resolved to a global address, but the global is a TLS variable.
151    // Each instance of each module has a different TLS area, and TLS symbols must be
152    // resolved again every time.
153    Tls {
154        resolved_from: ModuleHandle,
155        offset: u64,
156    },
157    // The symbol was resolved to a function export with the same name from this module.
158    // it is expected that the symbol resolves to an export of the correct type.
159    Function {
160        ty: FunctionType,
161        resolved_from: ModuleHandle,
162    },
163    // Same deal as above, but a pointer was generated and placed in the function table.
164    FunctionPointer {
165        resolved_from: ModuleHandle,
166        function_table_index: u32,
167    },
168    // The symbol failed to resolve, but it's a function so we can create a stub. The
169    // first call to any stub associated with this symbol must update the resolution
170    // record to point to the module the function was resolved from.
171    StubFunction(FunctionType),
172}
