4/// ### `sock_listen()`
5/// Listen for connections on a socket
6///
7/// Polling the socket handle will wait until a connection
8/// attempt is made
9///
10/// Note: This is similar to `listen`
11///
12/// ## Parameters
13///
14/// * `fd` - File descriptor of the socket to be bind
15/// * `backlog` - Maximum size of the queue for pending connections
16#[instrument(level = "trace", skip_all, fields(%sock, %backlog), ret)]
17pub fn sock_listen<M: MemorySize>(
18    mut ctx: FunctionEnvMut<'_, WasiEnv>,
19    sock: WasiFd,
20    backlog: M::Offset,
21) -> Result<Errno, WasiError> {
22    WasiEnv::do_pending_operations(&mut ctx)?;
23
24    ctx = wasi_try_ok!(maybe_snapshot_once::<M>(ctx, SnapshotTrigger::FirstListen)?);
25
26    let env = ctx.data();
27    let backlog: usize = wasi_try_ok!(backlog.try_into().map_err(|_| Errno::Inval));
28
29    wasi_try_ok!(sock_listen_internal(&mut ctx, sock, backlog)?);
30
31    #[cfg(feature = "journal")]
32    if ctx.data().enable_journal {
33        JournalEffector::save_sock_listen(&mut ctx, sock, backlog).map_err(|err| {
34            tracing::error!("failed to save sock_listen event - {}", err);
35            WasiError::Exit(ExitCode::from(Errno::Fault))
36        })?;
37    }
38
39    Ok(Errno::Success)
40}
41
42pub(crate) fn sock_listen_internal(
43    ctx: &mut FunctionEnvMut<'_, WasiEnv>,
44    sock: WasiFd,
45    backlog: usize,
46) -> Result<Result<(), Errno>, WasiError> {
47    let env = ctx.data();
48    let net = env.net().clone();
49    let tasks = ctx.data().tasks().clone();
50    wasi_try_ok_ok!(__sock_upgrade(
51        ctx,
52        sock,
53        Rights::SOCK_LISTEN,
54        |socket, _| async move { socket.listen(tasks.deref(), net.deref(), backlog).await }
55    ));
56
57    Ok(Ok(()))
58}
