51impl CommonWasiOptions {
52    pub(crate) fn prepare_webc_env(
53        &self,
54        builder: &mut WasiEnvBuilder,
55        container_fs: Option<Arc<dyn FileSystem + Send + Sync>>,
56        wasi: &WasiAnnotation,
57        root_fs: Option<TmpFileSystem>,
58    ) -> Result<(), anyhow::Error> {
59        if let Some(ref entry_function) = self.entry_function {
60            builder.set_entry_function(entry_function);
61        }
62
63        let root_fs = root_fs.unwrap_or_else(|| {
64            RootFileSystemBuilder::default()
65                .with_tmp(!self.is_tmp_mapped)
66                .build()
67        });
68        let fs = prepare_filesystem(root_fs, &self.mounts, container_fs)?;
69
70        // TODO: What's a preopen for '.' supposed to mean anyway? Why do we need it?
71        if self.mounts.iter().all(|m| m.guest != ".") {
72            // The user hasn't mounted "." to anything, so let's map it to "/"
73            let path = builder.get_current_dir().unwrap_or(PathBuf::from("/"));
74            builder.add_map_dir(".", path)?;
75        }
76
77        builder.add_preopen_dir("/")?;
78
79        builder.set_fs(Box::new(fs));
80
81        for pkg in &self.injected_packages {
82            builder.add_webc(pkg.clone());
83        }
84
85        let mapped_cmds = self
86            .mapped_host_commands
87            .iter()
88            .map(|c| (c.alias.as_str(), c.target.as_str()));
89        builder.add_mapped_commands(mapped_cmds);
90
91        self.populate_env(wasi, builder);
92        self.populate_args(wasi, builder);
93
94        *builder.capabilities_mut() = self.capabilities.clone();
95
96        #[cfg(feature = "journal")]
97        {
98            for journal in &self.read_only_journals {
99                builder.add_read_only_journal(journal.clone());
100            }
101            for journal in &self.writable_journals {
102                builder.add_writable_journal(journal.clone());
103            }
104            for trigger in &self.snapshot_on {
105                builder.add_snapshot_trigger(*trigger);
106            }
107            if let Some(interval) = self.snapshot_interval {
108                builder.with_snapshot_interval(interval);
109            }
110            builder.with_stop_running_after_snapshot(self.stop_running_after_snapshot);
111        }
112
113        Ok(())
114    }
115
116    fn populate_env(&self, wasi: &WasiAnnotation, builder: &mut WasiEnvBuilder) {
117        for item in wasi.env.as_deref().unwrap_or_default() {
118            // TODO(Michael-F-Bryan): Convert "wasi.env" in the webc crate from an
119            // Option<Vec<String>> to a HashMap<String, String> so we avoid this
120            // string.split() business
121            match item.split_once('=') {
122                Some((k, v)) => {
123                    builder.add_env(k, v);
124                }
125                None => {
126                    builder.add_env(item, String::new());
127                }
128            }
129        }
130
131        if self.forward_host_env {
132            builder.add_envs(std::env::vars());
133        }
134
135        builder.add_envs(self.env.clone());
136    }
137
138    fn populate_args(&self, wasi: &WasiAnnotation, builder: &mut WasiEnvBuilder) {
139        if let Some(main_args) = &wasi.main_args {
140            builder.add_args(main_args);
141        }
142
143        builder.add_args(&self.args);
144    }
145}

148//     OverlayFileSystem<TmpFileSystem, [RelativeOrAbsolutePathHack<Arc<dyn FileSystem>>; 1]>;

150fn build_directory_mappings(
151    root_fs: &mut TmpFileSystem,
152    mounted_dirs: &[MountedDirectory],
153) -> Result<(), anyhow::Error> {
154    for dir in mounted_dirs {
155        let MountedDirectory {
156            guest: guest_path,
157            fs,
158        } = dir;
159        let mut guest_path = PathBuf::from(guest_path);
160        tracing::debug!(
161            guest=%guest_path.display(),
162            "Mounting",
163        );
164
165        if guest_path.is_relative() {
166            guest_path = apply_relative_path_mounting_hack(&guest_path);
167        }
168
169        let guest_path = root_fs
170            .canonicalize_unchecked(&guest_path)
171            .with_context(|| {
172                format!(
173                    "Unable to canonicalize guest path '{}'",
174                    guest_path.display()
175                )
176            })?;
177
178        if guest_path == Path::new("/") {
179            root_fs
180                .mount_directory_entries(&guest_path, fs, "/".as_ref())
181                .context("Unable to mount to root")?;
182        } else {
183            if let Some(parent) = guest_path.parent() {
184                create_dir_all(&*root_fs, parent).with_context(|| {
185                    format!("Unable to create the \"{}\" directory", parent.display())
186                })?;
187            }
188
189            TmpFileSystem::mount(root_fs, guest_path.clone(), fs, "/".into())
190                .with_context(|| format!("Unable to mount \"{}\"", guest_path.display()))?;
191        }
192    }
193
194    Ok(())
195}

197fn prepare_filesystem(
198    mut root_fs: TmpFileSystem,
199    mounted_dirs: &[MountedDirectory],
200    container_fs: Option<Arc<dyn FileSystem + Send + Sync>>,
201) -> Result<Box<dyn FileSystem + Send + Sync>, Error> {
202    if !mounted_dirs.is_empty() {
203        build_directory_mappings(&mut root_fs, mounted_dirs)?;
204    }
205
206    // HACK(Michael-F-Bryan): The WebcVolumeFileSystem only accepts relative
207    // paths, but our Python executable will try to access its standard library
208    // with relative paths assuming that it is being run from the root
209    // directory (i.e. it does `open("lib/python3.6/io.py")` instead of
210    // `open("/lib/python3.6/io.py")`).
211    // Until the FileSystem trait figures out whether relative paths should be
212    // supported or not, we'll add an adapter that automatically retries
213    // operations using an absolute path if it failed using a relative path.
214
215    let fs = if let Some(container) = container_fs {
216        let container = RelativeOrAbsolutePathHack(container);
217        let fs = OverlayFileSystem::new(root_fs, [container]);
218        Box::new(fs) as Box<dyn FileSystem + Send + Sync>
219    } else {
220        let fs = RelativeOrAbsolutePathHack(root_fs);
221        Box::new(fs) as Box<dyn FileSystem + Send + Sync>
222    };
223
224    Ok(fs)
225}
226
227/// HACK: We need this so users can mount host directories at relative paths.
228/// This assumes that the current directory when a runner starts will be "/", so
229/// instead of mounting to a relative path, we just mount to "/$path".
230///
231/// This isn't really a long-term solution because there is no guarantee what
232/// the current directory will be. The WASI spec also doesn't require the
233/// current directory to be part of the "main" filesystem at all, we really
234/// *should* be mounting to a relative directory but that isn't supported by our
235/// virtual fs layer.
236///
237/// See <https://github.com/wasmerio/wasmer/issues/3794> for more.
238fn apply_relative_path_mounting_hack(original: &Path) -> PathBuf {
239    debug_assert!(original.is_relative());
240
241    let root = Path::new("/");
242    let mapped_path = if original == Path::new(".") {
243        root.to_path_buf()
244    } else {
245        root.join(original)
246    };
247
248    tracing::debug!(
249        original_path=%original.display(),
250        remapped_path=%mapped_path.display(),
251        "Remapping a relative path"
252    );
253
254    mapped_path
255}

257fn create_dir_all(fs: &dyn FileSystem, path: &Path) -> Result<(), Error> {
258    if fs.metadata(path).is_ok() {
259        return Ok(());
260    }
261
262    if let Some(parent) = path.parent() {
263        create_dir_all(fs, parent)?;
264    }
265
266    fs.create_dir(path)?;
267
268    Ok(())
269}

277/// A directory that should be mapped from the host filesystem into a WASI
278/// instance (the "guest").
279///
280/// # Panics
281///
282/// Converting a [`MappedDirectory`] to a [`MountedDirectory`] requires enabling
283/// the `host-fs` feature flag. Using the [`From`] implementation without
284/// enabling this feature will result in a runtime panic.
285#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
286pub struct MappedDirectory {
287    /// The absolute path for a directory on the host filesystem.
288    pub host: std::path::PathBuf,
289    /// The absolute path specifying where the host directory should be mounted
290    /// inside the guest.
291    pub guest: String,
292}
293
294impl From<MappedDirectory> for MountedDirectory {
295    fn from(value: MappedDirectory) -> Self {
296        cfg_if::cfg_if! {
297            if #[cfg(feature = "host-fs")] {
298                let MappedDirectory { host, guest } = value;
299                let fs: Arc<dyn FileSystem + Send + Sync> =
300                    Arc::new(virtual_fs::host_fs::FileSystem::new(Handle::current(), host).unwrap());
301
302                MountedDirectory { guest, fs }
303            } else {
304                unreachable!("The `host-fs` feature needs to be enabled to map {value:?}")
305            }
306        }
307    }
308}

311struct RelativeOrAbsolutePathHack<F>(F);

313impl<F: FileSystem> RelativeOrAbsolutePathHack<F> {
314    fn execute<Func, Ret>(&self, path: &Path, operation: Func) -> Result<Ret, FsError>
315    where
316        Func: Fn(&F, &Path) -> Result<Ret, FsError>,
317    {
318        // First, try it with the path we were given
319        let result = operation(&self.0, path);
320
321        if result.is_err() && !path.is_absolute() {
322            // we were given a relative path, but maybe the operation will work
323            // using absolute paths instead.
324            let path = Path::new("/").join(path);
325            operation(&self.0, &path)
326        } else {
327            result
328        }
329    }
330}

332impl<F: FileSystem> virtual_fs::FileSystem for RelativeOrAbsolutePathHack<F> {
333    fn readlink(&self, path: &Path) -> virtual_fs::Result<PathBuf> {
334        self.execute(path, |fs, p| fs.readlink(p))
335    }
336
337    fn read_dir(&self, path: &Path) -> virtual_fs::Result<virtual_fs::ReadDir> {
338        self.execute(path, |fs, p| fs.read_dir(p))
339    }
340
341    fn create_dir(&self, path: &Path) -> virtual_fs::Result<()> {
342        self.execute(path, |fs, p| fs.create_dir(p))
343    }
344
345    fn remove_dir(&self, path: &Path) -> virtual_fs::Result<()> {
346        self.execute(path, |fs, p| fs.remove_dir(p))
347    }
348
349    fn rename<'a>(&'a self, from: &Path, to: &Path) -> BoxFuture<'a, virtual_fs::Result<()>> {
350        let from = from.to_owned();
351        let to = to.to_owned();
352        Box::pin(async move { self.0.rename(&from, &to).await })
353    }
354
355    fn metadata(&self, path: &Path) -> virtual_fs::Result<virtual_fs::Metadata> {
356        self.execute(path, |fs, p| fs.metadata(p))
357    }
358
359    fn symlink_metadata(&self, path: &Path) -> virtual_fs::Result<virtual_fs::Metadata> {
360        self.execute(path, |fs, p| fs.symlink_metadata(p))
361    }
362
363    fn remove_file(&self, path: &Path) -> virtual_fs::Result<()> {
364        self.execute(path, |fs, p| fs.remove_file(p))
365    }
366
367    fn new_open_options(&self) -> virtual_fs::OpenOptions<'_> {
368        virtual_fs::OpenOptions::new(self)
369    }
370
371    fn mount(
372        &self,
373        name: String,
374        path: &Path,
375        fs: Box<dyn FileSystem + Send + Sync>,
376    ) -> virtual_fs::Result<()> {
377        let name_ref = &name;
378        let f_ref = &Arc::new(fs);
379        self.execute(path, move |f, p| {
380            f.mount(name_ref.clone(), p, Box::new(f_ref.clone()))
381        })
382    }
383}

385impl<F: FileSystem> virtual_fs::FileOpener for RelativeOrAbsolutePathHack<F> {

388        path: &Path,
