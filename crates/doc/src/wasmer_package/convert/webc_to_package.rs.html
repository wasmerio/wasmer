1use std::path::Path;
2
3use wasmer_config::package::{ModuleReference, SuggestedCompilerOptimizations, UserAnnotations};

7use super::ConversionError;

9/// Convert a webc image into a directory with a wasmer.toml file that can
10/// be used for generating a new pacakge.
11pub fn webc_to_package_dir(webc: &Container, target_dir: &Path) -> Result<(), ConversionError> {
12    let mut pkg_manifest = wasmer_config::package::Manifest::new_empty();
13
14    let manifest = webc.manifest();
15    // Convert the package annotation.
16
17    let pkg_annotation = manifest
18        .wapm()
19        .map_err(|err| ConversionError::msg(format!("could not read package annotation: {err}")))?;
20    if let Some(ann) = pkg_annotation {
21        let mut pkg = wasmer_config::package::Package::new_empty();
22
23        pkg.name = ann.name;
24        pkg.version = if let Some(raw) = ann.version {
25            let v = raw
26                .parse()
27                .map_err(|e| ConversionError::with_cause("invalid package version", e))?;
28            Some(v)
29        } else {
30            None
31        };
32
33        pkg.description = ann.description;
34        pkg.license = ann.license;
35
36        // TODO: map license_file and README (paths!)
37
38        pkg.homepage = ann.homepage;
39        pkg.repository = ann.repository;
40        pkg.private = ann.private;
41        pkg.entrypoint = manifest.entrypoint.clone();
42
43        pkg_manifest.package = Some(pkg);
44    }
45
46    // Map dependencies.
47    for (_name, target) in &manifest.use_map {
48        match target {
49            webc::metadata::UrlOrManifest::Url(_url) => {
50                // Not supported.
51            }
52            webc::metadata::UrlOrManifest::Manifest(_) => {
53                // Not supported.
54            }
55            webc::metadata::UrlOrManifest::RegistryDependentUrl(raw) => {
56                let (name, version) = if let Some((name, version_raw)) = raw.split_once('@') {
57                    let version = version_raw.parse().map_err(|err| {
58                        ConversionError::with_cause(
59                            format!("Could not parse version of dependency: '{raw}'"),
60                            err,
61                        )
62                    })?;
63                    (name.to_string(), version)
64                } else {
65                    (raw.to_string(), "*".parse().unwrap())
66                };
67
68                pkg_manifest.dependencies.insert(name, version);
69            }
70        }
71    }
72
73    // Convert filesystem mappings.
74
75    let fs_annotation = manifest
76        .filesystem()
77        .map_err(|err| ConversionError::msg(format!("could not read fs annotation: {err}")))?;
78    if let Some(ann) = fs_annotation {
79        for mapping in ann.0 {
80            if mapping.from.is_some() {
81                // wasmer.toml does not allow specifying dependency mounts.
82                continue;
83            }
84
85            // Extract the volume to "<target-dir>/<volume-name>".
86            let volume = webc.get_volume(&mapping.volume_name).ok_or_else(|| {
87                ConversionError::msg(format!(
88                    "Package annotations specify a volume that does not exist: '{}'",
89                    mapping.volume_name
90                ))
91            })?;
92
93            let volume_path = target_dir.join(mapping.volume_name.trim_start_matches('/'));
94
95            std::fs::create_dir_all(&volume_path).map_err(|err| {
96                ConversionError::with_cause(
97                    format!(
98                        "could not create volume directory '{}'",
99                        volume_path.display()
100                    ),
101                    err,
102                )
103            })?;
104
105            volume.unpack("/", &volume_path).map_err(|err| {
106                ConversionError::with_cause("could not unpack volume to filesystemt", err)
107            })?;
108
109            let mut source_path = mapping
110                .volume_name
111                .trim_start_matches('/')
112                .trim_end_matches('/')
113                .to_string();
114            if let Some(subpath) = mapping.host_path {
115                if !source_path.ends_with('/') {
116                    source_path.push('/');
117                }
118                source_path.push_str(&subpath);
119            }
120            source_path.insert_str(0, "./");
121
122            pkg_manifest
123                .fs
124                .insert(mapping.mount_path, source_path.into());
125        }
126    }
127
128    // Convert modules.
129
130    let module_dir_name = "modules";
131    let module_dir = target_dir.join(module_dir_name);
132
133    let atoms = webc.atoms();
134    if !atoms.is_empty() {
135        std::fs::create_dir_all(&module_dir).map_err(|err| {
136            ConversionError::with_cause(
137                format!("Could not create directory '{}'", module_dir.display()),
138                err,
139            )
140        })?;
141        for (atom_name, data) in atoms {
142            let atom_path = module_dir.join(&atom_name);
143
144            std::fs::write(&atom_path, &data).map_err(|err| {
145                ConversionError::with_cause(
146                    format!("Could not write atom to path '{}'", atom_path.display()),
147                    err,
148                )
149            })?;
150
151            let relative_path = format!("./{module_dir_name}/{atom_name}");
152
153            let mut annotations = None;
154
155            if let Some(manifest_atom) = manifest.atoms.get(&atom_name) {
156                if let Some(sco) = manifest_atom
157                    .annotations
158                    .get(SuggestedCompilerOptimizations::KEY)
159                {
160                    if let Some((_, v)) = sco.as_map().and_then(|v| {
161                        v.iter().find(|(k, _)| {
162                            k.as_text().is_some_and(|v| {
163                                v == SuggestedCompilerOptimizations::PASS_PARAMS_KEY
164                            })
165                        })
166                    }) {
167                        annotations = Some(UserAnnotations {
168                            suggested_compiler_optimizations: SuggestedCompilerOptimizations {
169                                pass_params: Some(v.as_bool().unwrap_or_default()),
170                            },
171                        });
172                    }
173                }
174            }
175
176            pkg_manifest.modules.push(wasmer_config::package::Module {
177                name: atom_name,
178                source: relative_path.into(),
179                abi: wasmer_config::package::Abi::None,
180                kind: None,
181                interfaces: None,
182                bindings: None,
183                annotations,
184            });
185        }
186    }
187
188    // Convert commands.
189    for (name, spec) in &manifest.commands {
190        let mut annotations = toml::Table::new();
191        for (key, value) in &spec.annotations {
192            if key == webc::metadata::annotations::Atom::KEY {
193                continue;
194            }
195
196            let raw_toml = toml::to_string(&value).unwrap();
197            let toml_value = toml::from_str::<toml::Value>(&raw_toml).unwrap();
198            annotations.insert(key.into(), toml_value);
199        }
200
201        let atom_annotation = spec
202            .annotation::<webc::metadata::annotations::Atom>(webc::metadata::annotations::Atom::KEY)
203            .map_err(|err| {
204                ConversionError::msg(format!(
205                    "could not read atom annotation for command '{name}': {err}"
206                ))
207            })?
208            .ok_or_else(|| {
209                ConversionError::msg(format!(
210                    "Command '{name}' is missing the required atom annotation"
211                ))
212            })?;
213
214        let module = if let Some(dep) = atom_annotation.dependency {
215            ModuleReference::Dependency {
216                dependency: dep,
217                module: atom_annotation.name,
218            }
219        } else {
220            ModuleReference::CurrentPackage {
221                module: atom_annotation.name,
222            }
223        };
224
225        let cmd = wasmer_config::package::Command::V2(wasmer_config::package::CommandV2 {
226            name: name.clone(),
227            module,
228            runner: spec.runner.clone(),
229            annotations: Some(wasmer_config::package::CommandAnnotations::Raw(
230                annotations.into(),
231            )),
232        });
233
234        pkg_manifest.commands.push(cmd);
235    }
236
237    // Write out the manifest.
238    let manifest_toml = toml::to_string(&pkg_manifest)
239        .map_err(|err| ConversionError::with_cause("could not serialize package manifest", err))?;
240    std::fs::write(target_dir.join("wasmer.toml"), manifest_toml)
241        .map_err(|err| ConversionError::with_cause("could not write wasmer.toml", err))?;
242
243    Ok(())
244}

256    // Build a webc from a pacakge directory, and then restore the directory
257    // from the webc.
258    #[test]
259    fn test_wasmer_package_webc_roundtrip() {
260        let tmpdir = tempfile::tempdir().unwrap();
261        let dir = tmpdir.path();
262
263        let webc = {
264            let dir_input = dir.join("input");
265            let dir_public = dir_input.join("public");
266
267            create_dir_all(&dir_public).unwrap();
268
269            std::fs::write(dir_public.join("index.html"), "INDEX").unwrap();
270
271            std::fs::write(dir_input.join("mywasm.wasm"), "()").unwrap();
272
273            std::fs::write(
274                dir_input.join("wasmer.toml"),
275                r#"
276[package]
277name = "testns/testpkg"
278version = "0.0.1"
279description = "descr1"
280license = "MIT"
281
282[dependencies]
283"wasmer/python" = "8.12.0"
284
285[fs]
286public = "./public"
287
288[[module]]
289name = "mywasm"
290source = "./mywasm.wasm"
291
292[[command]]
293name = "run"
294module = "mywasm"
295runner = "wasi"
296
297[command.annotations.wasi]
298env =  ["A=B"]
299main-args = ["/mounted/script.py"]
300"#,
301            )
302            .unwrap();
303
304            let pkg = Package::from_manifest(dir_input.join("wasmer.toml")).unwrap();
305            let raw = pkg.serialize().unwrap();
306            from_bytes(raw).unwrap()
307        };
308
309        let dir_output = dir.join("output");
310        webc_to_package_dir(&webc, &dir_output).unwrap();
311
312        assert_eq!(
313            std::fs::read_to_string(dir_output.join("public/index.html")).unwrap(),
314            "INDEX",
315        );
316
317        assert_eq!(
318            std::fs::read_to_string(dir_output.join("modules/mywasm")).unwrap(),
319            "()",
320        );
321
322        assert_eq!(
323            std::fs::read_to_string(dir_output.join("wasmer.toml"))
324                .unwrap()
325                .trim(),
326            r#"
