1//! Thread-local storage for storing the current store context,
2//! i.e. the currently active [`Store`](crate::Store)(s). When
3//! a function is called, a pointer to the [`StoreInner`] in placed
4//! inside the store context so it can be retrieved when needed.
5//! This lets code that needs access to the store get it with
6//! just the store ID.
7//!
8//! The currently active store context can be a sync or async
9//! context.
10//!
11//! For sync contexts, we just store a raw pointer
12//! to the `StoreInner`, which is owned by the embedder's stack.
13//!
14//! For async contexts, we store a write guard taken from the
15//! [`StoreAsync`](crate::StoreAsync); This achieves two goals:
16//!   * Makes the [`StoreAsync`](crate::StoreAsync) available
17//!     to whoever needs it, including when code needs to spawn
18//!     new coroutines
19//!   * Makes sure a write lock is held on the store as long as
20//!     the context is active, preventing other tasks from
21//!     accessing the store concurrently.
22//!
23//! We maintain a stack because it is technically possible to
24//! have nested `Function::call` invocations that use different
25//! stores, such as:
26//!     call(store1, func1) -> wasm code -> imported func ->
27//!     call(store2, func2)
28//!
29//! Note that this stack is maintained by both function
30//! calls and the async_runtime to reflect the exact WASM
31//! functions running on a given thread at any moment in
32//! time. If a function suspends, its store context is
33//! cleared and later reinstalled when it resumes. This lets
34//! us use thread-local storage for the context without
35//! requiring that async tasks are tied to specific threads.
36//!
37//! When something needs the "currently active" store context,
38//! they will only look at the top entry in the stack. It is
39//! always an error for code to try to access a store that's
40//! "paused", i.e. not the top entry. This should be impossible
41//! due to how the function call code is structured, but we
42//! guard against it anyway.
43

57enum StoreContextEntry {
58    Sync(*mut StoreInner),
59
60    #[cfg(feature = "experimental-async")]
61    Async(LocalRwLockWriteGuard<Box<StoreInner>>),
62}
63
64impl StoreContextEntry {
65    fn as_ptr(&self) -> *mut StoreInner {
66        match self {
67            Self::Sync(ptr) => *ptr,
68            #[cfg(feature = "experimental-async")]
69            Self::Async(guard) => &***guard as *const _ as *mut _,
70        }
71    }
72}

74pub(crate) struct StoreContext {
75    id: StoreId,
76
77    // StoreContexts can be used recursively when Function::call
78    // is used in an imported function. In the scenario, we're
79    // essentially passing a mutable borrow of the store into
80    // Function::call. However, entering the WASM code loses the
81    // reference, and it needs to be re-acquired from the
82    // StoreContext. This is why we use an UnsafeCell to allow
83    // multiple mutable references to the StoreMut; we do however
84    // keep track of how many borrows there are so we don't drop
85    // it prematurely.
86    borrow_count: u32,
87    entry: UnsafeCell<StoreContextEntry>,
88}

121thread_local! {
122    static STORE_CONTEXT_STACK: RefCell<Vec<StoreContext>> = const { RefCell::new(Vec::new()) };
123}
124
125impl StoreContext {
126    fn is_active(id: StoreId) -> bool {
127        STORE_CONTEXT_STACK.with(|cell| {
128            let stack = cell.borrow();
129            stack.last().is_some_and(|ctx| ctx.id == id)
130        })
131    }
132
133    fn is_suspended(id: StoreId) -> bool {
134        !Self::is_active(id)
135            && STORE_CONTEXT_STACK.with(|cell| {
136                let stack = cell.borrow();
137                stack.iter().rev().skip(1).any(|ctx| ctx.id == id)
138            })
139    }
140
141    fn install(id: StoreId, entry: StoreContextEntry) {
142        STORE_CONTEXT_STACK.with(|cell| {
143            let mut stack = cell.borrow_mut();
144            stack.push(Self {
145                id,
146                borrow_count: 0,
147                entry: UnsafeCell::new(entry),
148            });
149        })
150    }
151
152    /// Returns true if there are no active store context entries.
153    pub(crate) fn is_empty() -> bool {
154        STORE_CONTEXT_STACK.with(|cell| {
155            let stack = cell.borrow();
156            stack.is_empty()
157        })
158    }
159
160    /// The write guard ensures this is the only reference to the store,
161    /// so installation can never fail.
162    #[cfg(feature = "experimental-async")]
163    pub(crate) fn install_async(
164        guard: LocalRwLockWriteGuard<Box<StoreInner>>,
165    ) -> ForcedStoreInstallGuard {
166        let store_id = guard.objects.id();
167        Self::install(store_id, StoreContextEntry::Async(guard));
168        ForcedStoreInstallGuard { store_id }
169    }
170
171    /// Install the store context as sync if it is not already installed.
172    ///
173    /// # Safety
174    /// The pointer must be dereferenceable and remain valid until the
175    /// store context is uninstalled.
176    pub(crate) unsafe fn ensure_installed(store_ptr: *mut StoreInner) -> StoreInstallGuard {
177        let store_id = unsafe { store_ptr.as_ref().unwrap().objects.id() };
178        if Self::is_active(store_id) {
179            let current_ptr = STORE_CONTEXT_STACK.with(|cell| {
180                let stack = cell.borrow();
181                unsafe { stack.last().unwrap().entry.get().as_ref().unwrap().as_ptr() }
182            });
183            assert_eq!(store_ptr, current_ptr, "Store context pointer mismatch");
184            StoreInstallGuard::NotInstalled
185        } else {
186            Self::install(store_id, StoreContextEntry::Sync(store_ptr));
187            StoreInstallGuard::Installed(store_id)
188        }
189    }
190
191    /// "Pause" one borrow of the store context.
192    ///
193    /// # Safety
194    /// Code must ensure it does not use the StorePtrWrapper or
195    /// StoreAsyncGuardWrapper that it owns, or any StoreRef/StoreMut
196    /// derived from them, while the store context is paused.
197    ///
198    /// The safe, correct use-case for this method is to
199    /// pause the store context while executing WASM code, which
200    /// cannot use the store context directly. This allows an async
201    /// context to uninstall the store context when suspending if it's
202    /// called from a sync imported function. The imported function
203    /// will have borrowed the store context in its trampoline, which
204    /// will prevent the async context from uninstalling the store.
205    /// However, since the imported function passes a mutable borrow
206    /// of its store into `Function::call`, it will expect the store
207    /// to change before the call returns.
208    pub(crate) unsafe fn pause(id: StoreId) -> StorePtrPauseGuard {
209        STORE_CONTEXT_STACK.with(|cell| {
210            let mut stack = cell.borrow_mut();
211            let top = stack
212                .last_mut()
213                .expect("No store context installed on this thread");
214            assert_eq!(top.id, id, "Mismatched store context access");
215            let ref_count_decremented = if top.borrow_count > 0 {
216                top.borrow_count -= 1;
217                true
218            } else {
219                false
220            };
221            StorePtrPauseGuard {
222                store_id: id,
223                ptr: unsafe { top.entry.get().as_ref().unwrap().as_ptr() },
224                ref_count_decremented,
225            }
226        })
227    }
228
229    /// Safety: This method lets you borrow multiple mutable references
230    /// to the currently active store context. The caller must ensure that:
231    ///   * there is only one mutable reference alive, or
232    ///   * all but one mutable reference are inaccessible and passed
233    ///     into a function that lost the reference (e.g. into WASM code)
234    ///
235    /// The intended, valid use-case for this method is from within
236    /// imported function trampolines.
237    pub(crate) unsafe fn get_current(id: StoreId) -> StorePtrWrapper {
238        STORE_CONTEXT_STACK.with(|cell| {
239            let mut stack = cell.borrow_mut();
240            let top = stack
241                .last_mut()
242                .expect("No store context installed on this thread");
243            assert_eq!(top.id, id, "Mismatched store context access");
244            top.borrow_count += 1;
245            StorePtrWrapper {
246                store_ptr: unsafe { top.entry.get().as_mut().unwrap().as_ptr() },
247            }
248        })
249    }
250
251    /// Safety: In addition to the safety requirements of [`Self::get_current`],
252    /// the pointer returned from this function will become invalid if
253    /// the store context is changed in any way (via installing or uninstalling
254    /// a store context). The caller must ensure that the store context
255    /// remains unchanged as long as the pointer is being accessed.
256    pub(crate) unsafe fn get_current_transient(id: StoreId) -> *mut StoreInner {
257        STORE_CONTEXT_STACK.with(|cell| {
258            let mut stack = cell.borrow_mut();
259            let top = stack
260                .last_mut()
261                .expect("No store context installed on this thread");
262            assert_eq!(top.id, id, "Mismatched store context access");
263            unsafe { top.entry.get().as_mut().unwrap().as_ptr() }
264        })
265    }
266
267    /// Safety: See [`Self::get_current`].
268    pub(crate) unsafe fn try_get_current(id: StoreId) -> Option<StorePtrWrapper> {
269        STORE_CONTEXT_STACK.with(|cell| {
270            let mut stack = cell.borrow_mut();
271            let top = stack.last_mut()?;
272            if top.id != id {
273                return None;
274            }
275            top.borrow_count += 1;
276            Some(StorePtrWrapper {
277                store_ptr: unsafe { top.entry.get().as_mut().unwrap().as_ptr() },
278            })
279        })
280    }
281
282    /// Safety: See [`Self::get_current`].
283    #[cfg(feature = "experimental-async")]
284    pub(crate) unsafe fn try_get_current_async(id: StoreId) -> GetStoreAsyncGuardResult {
285        STORE_CONTEXT_STACK.with(|cell| {
286            let mut stack = cell.borrow_mut();
287            let Some(top) = stack.last_mut() else {
288                return GetStoreAsyncGuardResult::NotInstalled;
289            };
290            if top.id != id {
291                return GetStoreAsyncGuardResult::NotInstalled;
292            }
293            top.borrow_count += 1;
294            match unsafe { top.entry.get().as_mut().unwrap() } {
295                StoreContextEntry::Async(guard) => {
296                    GetStoreAsyncGuardResult::Ok(StoreAsyncGuardWrapper {
297                        guard: guard as *mut _,
298                    })
299                }
300                StoreContextEntry::Sync(ptr) => {
301                    GetStoreAsyncGuardResult::NotAsync(StorePtrWrapper { store_ptr: *ptr })
302                }
303            }
304        })
305    }
306}

308impl StorePtrWrapper {
309    pub(crate) fn as_ref(&self) -> StoreRef<'_> {
310        // Safety: the store_mut is always initialized unless the StoreMutWrapper
311        // is dropped, at which point it's impossible to call this function
312        unsafe { self.store_ptr.as_ref().unwrap().as_store_ref() }
313    }
314
315    pub(crate) fn as_mut(&mut self) -> StoreMut<'_> {
316        // Safety: the store_mut is always initialized unless the StoreMutWrapper
317        // is dropped, at which point it's impossible to call this function
318        unsafe { self.store_ptr.as_mut().unwrap().as_store_mut() }
319    }
320}

322impl Clone for StorePtrWrapper {
323    fn clone(&self) -> Self {
324        STORE_CONTEXT_STACK.with(|cell| {
325            let mut stack = cell.borrow_mut();
326            let top = stack
327                .last_mut()
328                .expect("No store context installed on this thread");
329            match unsafe { top.entry.get().as_ref().unwrap() } {
330                StoreContextEntry::Sync(ptr) if *ptr == self.store_ptr => (),
331                _ => panic!("Mismatched store context access"),
332            }
333            top.borrow_count += 1;
334            Self {
335                store_ptr: self.store_ptr,
336            }
337        })
338    }
339}

341impl Drop for StorePtrWrapper {
342    fn drop(&mut self) {
343        if std::thread::panicking() {
344            return;
345        }
346        let id = self.as_mut().objects_mut().id();
347        STORE_CONTEXT_STACK.with(|cell| {
348            let mut stack = cell.borrow_mut();
349            let top = stack
350                .last_mut()
351                .expect("No store context installed on this thread");
352            assert_eq!(top.id, id, "Mismatched store context reinstall");
353            top.borrow_count -= 1;
354        })
355    }
356}

376impl Drop for StoreInstallGuard {
377    fn drop(&mut self) {
378        if let Self::Installed(store_id) = self {
379            STORE_CONTEXT_STACK.with(|cell| {
380                let mut stack = cell.borrow_mut();
381                match (stack.pop(), std::thread::panicking()) {
382                    (Some(top), false) => {
383                        assert_eq!(top.id, *store_id, "Mismatched store context uninstall");
384                        assert_eq!(
385                            top.borrow_count, 0,
386                            "Cannot uninstall store context while it is still borrowed"
387                        );
388                    }
389                    (Some(top), true) => {
390                        // If we're panicking and there's a store ID mismatch, just
391                        // put the store back in the hope that its own install guard
392                        // take care of uninstalling it later.
393                        if top.id != *store_id {
394                            stack.push(top);
395                        }
396                    }
397                    (None, false) => panic!("Store context stack underflow"),
398                    (None, true) => {
399                        // Nothing to do if we're panicking; panics can put the context
400                        // in an invalid state, and we don't to cause another panic here.
401                    }
402                }
403            })
404        }
405    }
406}

438impl Drop for StorePtrPauseGuard {
439    fn drop(&mut self) {
440        if std::thread::panicking() {
441            return;
442        }
443        STORE_CONTEXT_STACK.with(|cell| {
444            let mut stack = cell.borrow_mut();
445            let top = stack
446                .last_mut()
447                .expect("No store context installed on this thread");
448            assert_eq!(top.id, self.store_id, "Mismatched store context access");
449            assert_eq!(
450                unsafe { top.entry.get().as_ref().unwrap() }.as_ptr(),
451                self.ptr,
452                "Mismatched store context access"
453            );
454            if self.ref_count_decremented {
455                top.borrow_count += 1;
456            }
457        })
458    }
