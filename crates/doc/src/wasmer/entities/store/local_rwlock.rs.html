1//! A single-threaded async-aware RwLock implementation.
2//!
3//! This module provides [`LocalRwLock`], which is similar to `async_lock::RwLock`
4//! but optimized for single-threaded async runtimes. It avoids atomic operations
5//! and is `!Send + !Sync`, making it more efficient when thread safety is not needed.
6//!
7//! Like `async_lock::RwLock`, it provides `read_rc()` and `write_rc()` methods
8//! that allow callers to asynchronously wait for the lock to become available,
9//! and return guards with `'static` lifetimes by holding an `Rc` to the lock.
10
11use std::cell::{Cell, RefCell, UnsafeCell};
12use std::future::Future;
13use std::marker::PhantomData;
14use std::ops::{Deref, DerefMut};
15use std::pin::Pin;
16use std::rc::Rc;
17use std::task::{Context, Poll, Waker};
18
19/// The main lock type.
20pub struct LocalRwLock<T> {
21    inner: Rc<LocalRwLockInner<T>>,
22}
23
24struct LocalRwLockInner<T> {
25    value: UnsafeCell<T>,
26    state: Cell<LockState>,
27    read_waiters: RefCell<Vec<Option<Waker>>>,
28    write_waiters: RefCell<Vec<Option<Waker>>>,
29}
30
31#[derive(Clone, Copy, PartialEq, Debug)]
32enum LockState {
33    Unlocked,
34    Reading(usize), // count of readers
35    Writing,
36}
37
38impl<T> LocalRwLock<T> {
39    /// Creates a new `LocalRwLock` with the given value.
40    pub fn new(value: T) -> Self {
41        Self {
42            inner: Rc::new(LocalRwLockInner {
43                value: UnsafeCell::new(value),
44                state: Cell::new(LockState::Unlocked),
45                read_waiters: RefCell::new(Vec::new()),
46                write_waiters: RefCell::new(Vec::new()),
47            }),
48        }
49    }
50
51    /// Acquires a read lock, waiting asynchronously if necessary.
52    ///
53    /// The returned guard holds an `Rc` clone, allowing it to have a `'static` lifetime.
54    pub fn read(&self) -> ReadFuture<T> {
55        ReadFuture {
56            inner: self.inner.clone(),
57            waiter_index: Cell::new(None),
58        }
59    }
60
61    /// Acquires a write lock, waiting asynchronously if necessary.
62    ///
63    /// The returned guard holds an `Rc` clone, allowing it to have a `'static` lifetime.
64    pub fn write(&self) -> WriteFuture<T> {
65        WriteFuture {
66            inner: self.inner.clone(),
67            waiter_index: Cell::new(None),
68        }
69    }
70
71    /// Attempts to acquire a read lock with a `'static` lifetime without waiting.
72    pub fn try_read(&self) -> Option<LocalRwLockReadGuard<T>> {
73        if self.inner.try_read() {
74            Some(LocalRwLockReadGuard {
75                inner: self.inner.clone(),
76            })
77        } else {
78            None
79        }
80    }
81
82    /// Attempts to acquire a write lock with a `'static` lifetime without waiting.
83    pub fn try_write(&self) -> Option<LocalRwLockWriteGuard<T>> {
84        if self.inner.try_write() {
85            Some(LocalRwLockWriteGuard {
86                inner: self.inner.clone(),
87            })
88        } else {
89            None
90        }
91    }
92
93    /// Attempts to consume the lock if there are no active or waiting
94    /// readers or writers, and returns the inner value if successful.
95    pub fn consume(self) -> Result<T, Self> {
96        if self.inner.state.get() == LockState::Unlocked
97            && self.inner.read_waiters.borrow().is_empty()
98            && self.inner.write_waiters.borrow().is_empty()
99        {
100            match Rc::try_unwrap(self.inner) {
101                Ok(inner) => Ok(inner.value.into_inner()),
102                Err(rc) => Err(Self { inner: rc }),
103            }
104        } else {
105            Err(self)
106        }
107    }
108}
109
110impl<T> Clone for LocalRwLock<T> {
111    fn clone(&self) -> Self {
112        Self {
113            inner: self.inner.clone(),
114        }
115    }
116}
117
118impl<T> LocalRwLockInner<T> {
119    fn try_read(&self) -> bool {
120        match self.state.get() {
121            LockState::Unlocked => {
122                self.state.set(LockState::Reading(1));
123                true
124            }
125            LockState::Reading(n) => {
126                self.state.set(LockState::Reading(n + 1));
127                true
128            }
129            LockState::Writing => false,
130        }
131    }
132
133    fn try_write(&self) -> bool {
134        match self.state.get() {
135            LockState::Unlocked => {
136                self.state.set(LockState::Writing);
137                true
138            }
139            _ => false,
140        }
141    }
142
143    fn release_read(&self) {
144        match self.state.get() {
145            LockState::Reading(1) => {
146                self.state.set(LockState::Unlocked);
147                self.wake_waiters();
148            }
149            LockState::Reading(n) if n > 1 => {
150                self.state.set(LockState::Reading(n - 1));
151            }
152            _ => panic!("LocalRwLock: release_read called but not in Reading state"),
153        }
154    }
155
156    fn release_write(&self) {
157        match self.state.get() {
158            LockState::Writing => {
159                self.state.set(LockState::Unlocked);
160                self.wake_waiters();
161            }
162            _ => panic!("LocalRwLock: release_write called but not in Writing state"),
163        }
164    }
165
166    fn wake_waiters(&self) {
167        // Wake writers first (priority)
168        let mut write_waiters = self.write_waiters.borrow_mut();
169        let has_writers = write_waiters.iter().any(|w| w.is_some());
170
171        if has_writers {
172            // If there are waiting writers, only wake them (they need exclusive access)
173            for waker in write_waiters.drain(..).flatten() {
174                waker.wake();
175            }
176        } else {
177            // No writers waiting, wake all readers (they can share the lock)
178            drop(write_waiters); // Release borrow before borrowing read_waiters
179            let mut read_waiters = self.read_waiters.borrow_mut();
180            for waker in read_waiters.drain(..).flatten() {
181                waker.wake();
182            }
183        }
184    }
185
186    /// Shared polling logic for all futures.
187    /// Returns Poll::Ready(()) if the lock was acquired, Poll::Pending otherwise.
188    fn poll_lock(
189        &self,
190        waiter_index: &Cell<Option<usize>>,
191        cx: &mut Context<'_>,
192        try_lock: impl FnOnce(&Self) -> bool,
193        is_write: bool,
194    ) -> Poll<()> {
195        if try_lock(self) {
196            // If we successfully acquired the lock, remove our waiter slot if we registered one
197            if let Some(index) = waiter_index.get() {
198                let waiters = if is_write {
199                    &self.write_waiters
200                } else {
201                    &self.read_waiters
202                };
203                let mut waiters = waiters.borrow_mut();
204                if index < waiters.len() {
205                    waiters[index] = None;
206                }
207            }
208            return Poll::Ready(());
209        }
210
211        // Register or update our waker in the appropriate waiters list
212        let waiters = if is_write {
213            &self.write_waiters
214        } else {
215            &self.read_waiters
216        };
217        let mut waiters = waiters.borrow_mut();
218
219        if let Some(index) = waiter_index.get() {
220            // We already have a slot, check if we need to update it
221            if index < waiters.len() {
222                if let Some(existing) = &waiters[index] {
223                    if !existing.will_wake(cx.waker()) {
224                        waiters[index] = Some(cx.waker().clone());
225                    }
226                } else {
227                    waiters[index] = Some(cx.waker().clone());
228                }
229            } else {
230                // Our slot was somehow removed, register a new one
231                let new_index = waiters.len();
232                waiters.push(Some(cx.waker().clone()));
233                waiter_index.set(Some(new_index));
234            }
235        } else {
236            // First time registering
237            let index = waiters.len();
238            waiters.push(Some(cx.waker().clone()));
239            waiter_index.set(Some(index));
240        }
241
242        Poll::Pending
243    }
244
245    /// Cleanup waiter slot on drop
246    fn cleanup_waiter(&self, waiter_index: &Cell<Option<usize>>, is_write: bool) {
247        if let Some(index) = waiter_index.get() {
248            let waiters = if is_write {
249                &self.write_waiters
250            } else {
251                &self.read_waiters
252            };
253            let mut waiters = waiters.borrow_mut();
254            if index < waiters.len() {
255                waiters[index] = None;
256            }
257        }
258    }
259}

263/// A read guard with a `'static` lifetime, holding an `Rc` to the lock.
264pub struct LocalRwLockReadGuard<T> {
265    inner: Rc<LocalRwLockInner<T>>,
266}
267
268impl<T> LocalRwLockReadGuard<T> {
269    /// Rebuild a handle to the lock from this [`LocalRwLockReadGuard`].
270    pub fn lock_handle(me: &Self) -> LocalRwLock<T> {
271        LocalRwLock {
272            inner: me.inner.clone(),
273        }
274    }
275}
276
277impl<T> Deref for LocalRwLockReadGuard<T> {
278    type Target = T;
279
280    fn deref(&self) -> &T {
281        unsafe { &*self.inner.value.get() }
282    }
283}
284
285impl<T> Drop for LocalRwLockReadGuard<T> {
286    fn drop(&mut self) {
287        self.inner.release_read();
288    }
289}

327/// Future returned by `read_rc()`.
328pub struct ReadFuture<T> {
329    inner: Rc<LocalRwLockInner<T>>,
330    waiter_index: Cell<Option<usize>>,
331}
332
333impl<T> Future for ReadFuture<T> {
334    type Output = LocalRwLockReadGuard<T>;
335
336    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
337        if self
338            .inner
339            .poll_lock(&self.waiter_index, cx, |inner| inner.try_read(), false)
340            .is_ready()
341        {
342            Poll::Ready(LocalRwLockReadGuard {
343                inner: self.inner.clone(),
344            })
345        } else {
346            Poll::Pending
347        }
348    }
349}
350
351impl<T> Drop for ReadFuture<T> {
352    fn drop(&mut self) {
353        self.inner.cleanup_waiter(&self.waiter_index, false);
354    }
355}

391    #[test]
392    fn test_basic_read_write() {
393        let lock = LocalRwLock::new(42);
394
395        // Can acquire read lock
396        let guard = lock.try_read().unwrap();
397        assert_eq!(*guard, 42);
398        drop(guard);
399
400        // Can acquire write lock
401        let mut guard = lock.try_write().unwrap();
402        *guard = 100;
403        drop(guard);
404
405        // Value was updated
406        let guard = lock.try_read().unwrap();
407        assert_eq!(*guard, 100);
408    }

472    #[test]
473    fn test_writer_priority() {
474        use std::sync::{Arc, Mutex};
475        use std::task::{Poll, Wake};
476
477        // Custom waker that tracks when it's woken
478        struct TrackingWaker {
479            name: &'static str,
480            wake_order: Arc<Mutex<Vec<&'static str>>>,
481        }
482
483        impl Wake for TrackingWaker {
484            fn wake(self: Arc<Self>) {
485                self.wake_order.lock().unwrap().push(self.name);
486            }
487        }
488
489        let lock = LocalRwLock::new(0);
490        let wake_order = Arc::new(Mutex::new(Vec::new()));
491
492        // Acquire write lock
493        let write_guard = lock.try_write().unwrap();
494
495        // Create read and write futures that will need to wait
496        let mut read_future = Box::pin(lock.read());
497        let mut write_future = Box::pin(lock.write());
498
499        // Create tracking wakers
500        let read_waker = Arc::new(TrackingWaker {
501            name: "reader",
502            wake_order: wake_order.clone(),
503        })
504        .into();
505        let mut read_cx = std::task::Context::from_waker(&read_waker);
506
507        let write_waker = Arc::new(TrackingWaker {
508            name: "writer",
509            wake_order: wake_order.clone(),
510        })
511        .into();
512        let mut write_cx = std::task::Context::from_waker(&write_waker);
513
514        // Poll both to register them as waiters
515        assert!(matches!(
516            write_future.as_mut().poll(&mut write_cx),
517            Poll::Pending
518        ));
519        assert!(matches!(
520            read_future.as_mut().poll(&mut read_cx),
521            Poll::Pending
522        ));
523
524        // Verify they're in separate queues
525        assert_eq!(
526            lock.inner
527                .write_waiters
528                .borrow()
529                .iter()
530                .filter(|w| w.is_some())
531                .count(),
532            1
533        );
534        assert_eq!(
535            lock.inner
536                .read_waiters
537                .borrow()
538                .iter()
539                .filter(|w| w.is_some())
540                .count(),
541            1
542        );
543
544        // No wakers called yet
545        assert_eq!(wake_order.lock().unwrap().len(), 0);
546
547        // Release the write lock - this should wake waiters
548        drop(write_guard);
549
550        // Verify ONLY writer was woken (not reader, since writer has priority)
551        let order = wake_order.lock().unwrap();
552        assert_eq!(
553            order.len(),
554            1,
555            "Only writer should be woken when writers are waiting"
556        );
557        assert_eq!(order[0], "writer", "Writer should be woken, not reader");
558    }
