13/// We require the context to have a fixed memory address for its lifetime since
14/// various bits of the VM have raw pointers that point back to it. Hence we
15/// wrap the actual context in a box.
16pub(crate) struct StoreInner {
17    pub(crate) objects: StoreObjects,
18    pub(crate) store: BackendStore,
19    pub(crate) on_called: Option<OnCalledHandler>,
20}
21
22impl std::fmt::Debug for StoreInner {
23    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
24        f.debug_struct("StoreInner")
25            .field("objects", &self.objects)
26            .field("store", &self.store)
27            .field("on_called", &"<...>")
28            .finish()
29    }
30}
