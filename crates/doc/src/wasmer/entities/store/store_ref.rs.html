11/// A temporary handle to a [`crate::Store`].
12#[derive(Debug)]
13pub struct StoreRef<'a> {
14    pub(crate) inner: &'a StoreInner,
15}
16
17impl<'a> StoreRef<'a> {
18    pub(crate) fn objects(&self) -> &'a StoreObjects {
19        &self.inner.objects
20    }
21
22    /// Returns the [`Engine`].
23    pub fn engine(&self) -> &Engine {
24        self.inner.store.engine()
25    }
26
27    /// Checks whether two stores are identical. A store is considered
28    /// equal to another store if both have the same engine.
29    pub fn same(a: &Self, b: &Self) -> bool {
30        StoreObjects::same(&a.inner.objects, &b.inner.objects)
31    }
32
33    /// The signal handler
34    #[cfg(feature = "sys")]
35    #[inline]
36    pub fn signal_handler(&self) -> Option<*const TrapHandlerFn<'static>> {
37        use crate::backend::sys::entities::store::NativeStoreExt;
38        self.inner.store.as_sys().signal_handler()
39    }
40}

42/// A temporary handle to a [`crate::Store`].
43pub struct StoreMut<'a> {
44    pub(crate) inner: &'a mut StoreInner,
45}
46
47impl StoreMut<'_> {
48    /// Returns the [`Engine`].
49    pub fn engine(&self) -> &Engine {
50        self.inner.store.engine()
51    }
52
53    /// Checks whether two stores are identical. A store is considered
54    /// equal to another store if both have the same engine.
55    pub fn same(a: &Self, b: &Self) -> bool {
56        StoreObjects::same(&a.inner.objects, &b.inner.objects)
57    }
58
59    #[allow(unused)]
60    pub(crate) fn as_raw(&self) -> *mut StoreInner {
61        self.inner as *const StoreInner as *mut StoreInner
62    }
63
64    #[allow(unused)]
65    pub(crate) unsafe fn from_raw(raw: *mut StoreInner) -> Self {
66        Self {
67            inner: unsafe { &mut *raw },
68        }
69    }
70
71    #[allow(unused)]
72    pub(crate) fn engine_and_objects_mut(&mut self) -> (&Engine, &mut StoreObjects) {
73        (self.inner.store.engine(), &mut self.inner.objects)
74    }
75
76    // TODO: OnCalledAction is needed for asyncify. It will be refactored with https://github.com/wasmerio/wasmer/issues/3451
77    /// Sets the unwind callback which will be invoked when the call finishes
78    pub fn on_called<F>(&mut self, callback: F)
79    where
80        F: FnOnce(StoreMut<'_>) -> Result<OnCalledAction, Box<dyn std::error::Error + Send + Sync>>
81            + Send
82            + Sync
83            + 'static,
84    {
85        self.inner.on_called.replace(Box::new(callback));
86    }
87}

89/// Helper trait for a value that is convertible to a [`StoreRef`].
90pub trait AsStoreRef {
91    /// Returns a `StoreRef` pointing to the underlying context.
92    fn as_store_ref(&self) -> StoreRef<'_>;
93}
94
95/// Helper trait for a value that is convertible to a [`StoreMut`].
96pub trait AsStoreMut: AsStoreRef {
97    /// Returns a `StoreMut` pointing to the underlying context.
98    fn as_store_mut(&mut self) -> StoreMut<'_>;
99
100    /// Returns the ObjectMutable
101    fn objects_mut(&mut self) -> &mut StoreObjects;
102}
103
104impl AsStoreRef for StoreRef<'_> {
105    fn as_store_ref(&self) -> StoreRef<'_> {
106        StoreRef { inner: self.inner }
107    }
108}
109
110impl AsEngineRef for StoreRef<'_> {

113    }
114}

116impl AsStoreRef for StoreMut<'_> {
117    fn as_store_ref(&self) -> StoreRef<'_> {
118        StoreRef { inner: self.inner }
119    }
120}
121impl AsStoreMut for StoreMut<'_> {
122    fn as_store_mut(&mut self) -> StoreMut<'_> {
123        StoreMut { inner: self.inner }
124    }
125
126    fn objects_mut(&mut self) -> &mut StoreObjects {
