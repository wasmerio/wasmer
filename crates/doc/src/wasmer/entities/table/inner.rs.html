10/// A WebAssembly `table` instance.
11///
12/// The `Table` struct is an array-like structure representing a WebAssembly Table,
13/// which stores function references.
14///
15/// A table created by the host or in WebAssembly code will be accessible and
16/// mutable from both host and WebAssembly.
17///
18/// Spec: <https://webassembly.github.io/spec/core/exec/runtime.html#table-instances>
19gen_rt_ty! {
20    #[cfg_attr(feature = "artifact-size", derive(loupe::MemoryUsage))]
21    #[derive(Debug, Clone, PartialEq, Eq, derive_more::From)]
22    pub BackendTable(entities::table::Table);
23}
24
25impl BackendTable {
26    /// Creates a new table with the provided [`TableType`] definition.
27    ///
28    /// All the elements in the table will be set to the `init` value.
29    ///
30    /// This function will construct the table using the store `BaseTunables`.
31    #[inline]
32    pub fn new(
33        store: &mut impl AsStoreMut,
34        ty: TableType,
35        init: Value,
36    ) -> Result<Self, RuntimeError> {
37        match &store.as_store_mut().inner.store {
38            #[cfg(feature = "sys")]
39            BackendStore::Sys(_) => Ok(Self::Sys(
40                crate::backend::sys::entities::table::Table::new(store, ty, init)?,
41            )),
42            #[cfg(feature = "v8")]
43            BackendStore::V8(_) => Ok(Self::V8(crate::backend::v8::entities::table::Table::new(
44                store, ty, init,
45            )?)),
46            #[cfg(feature = "js")]
47            BackendStore::Js(_) => Ok(Self::Js(crate::backend::js::entities::table::Table::new(
48                store, ty, init,
49            )?)),
50            #[cfg(feature = "jsc")]
51            BackendStore::Jsc(_) => Ok(Self::Jsc(
52                crate::backend::jsc::entities::table::Table::new(store, ty, init)?,
53            )),
54        }
55    }
56
57    /// Returns the [`TableType`] of the table.

94    /// Grows the size of the `Table` by `delta`, initializating
95    /// the elements with the provided `init` value.
96    ///
97    /// It returns the previous size of the `Table` in case is able
98    /// to grow the Table successfully.
99    ///
100    /// # Errors
101    ///
102    /// Returns an error if the `delta` is out of bounds for the table.
103    #[inline]
104    pub fn grow(
105        &self,
106        store: &mut impl AsStoreMut,
107        delta: u32,
108        init: Value,
109    ) -> Result<u32, RuntimeError> {
110        match_rt!(on self => s {
111            s.grow(store, delta, init)
112        })
113    }

115    /// Copies the `len` elements of `src_table` starting at `src_index`

210#[cfg(test)]
211mod test {
212    /// Check the example from <https://github.com/wasmerio/wasmer/issues/3197>.
213    #[test]
214    #[cfg_attr(
215        feature = "v8",
216        ignore = "growing tables in v8 is not currently supported"
217    )]
218    fn table_grow_issue_3197() {
219        use crate::{Instance, Module, Store, Table, TableType, Type, Value, imports};
220
221        const WAT: &str = r#"(module (table (import "env" "table") 100 funcref))"#;
222
223        // Tests that the table type of `table` is compatible with the export in the WAT
224        // This tests that `wasmer_types::types::is_table_compatible` works as expected.
225        let mut store = Store::default();
226        let module = Module::new(&store, WAT).unwrap();
227        let ty = TableType::new(Type::FuncRef, 0, None);
228        let table = Table::new(&mut store, ty, Value::FuncRef(None)).unwrap();
229        table.grow(&mut store, 100, Value::FuncRef(None)).unwrap();
230        assert_eq!(table.ty(&store).minimum, 0);
231        let imports = imports! {"env" => {"table" => table}};
232        let _instance = Instance::new(&mut store, &module, &imports).unwrap();
233    }
