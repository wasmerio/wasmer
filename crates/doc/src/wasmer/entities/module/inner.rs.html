21/// A WebAssembly Module contains stateless WebAssembly
22/// code that has already been compiled and can be instantiated
23/// multiple times.
24///
25/// ## Cloning a module
26///
27/// Cloning a module is cheap: it does a shallow copy of the compiled
28/// contents rather than a deep copy.
29gen_rt_ty!(Module
30    @cfg feature = "artifact-size" => derive(loupe::MemoryUsage)
31    @derives Clone, PartialEq, Eq, derive_more::From
32);
33
34impl BackendModule {
35    #[inline]
36    pub fn new(engine: &impl AsEngineRef, bytes: impl AsRef<[u8]>) -> Result<Self, CompileError> {
37        #[cfg(feature = "wat")]
38        let bytes = wat::parse_bytes(bytes.as_ref()).map_err(|e| {
39            CompileError::Wasm(WasmError::Generic(format!(
40                "Error when converting wat: {e}",
41            )))
42        })?;
43        Self::from_binary(engine, bytes.as_ref())
44    }
45
46    #[inline]
47    pub fn new_with_progress(
48        engine: &impl AsEngineRef,
49        bytes: impl AsRef<[u8]>,
50        callback: CompilationProgressCallback,
51    ) -> Result<Self, CompileError> {
52        #[cfg(feature = "wat")]
53        let bytes = wat::parse_bytes(bytes.as_ref()).map_err(|e| {
54            CompileError::Wasm(WasmError::Generic(format!(
55                "Error when converting wat: {e}",
56            )))
57        })?;
58
59        #[cfg(feature = "sys")]
60        return crate::backend::sys::entities::module::Module::from_binary_with_progress(
61            engine,
62            bytes.as_ref(),
63            callback,
64        )
65        .map(Self::Sys);
66        #[cfg(not(feature = "sys"))]
67        return Self::from_binary(engine, bytes.as_ref());
68    }

70    /// Creates a new WebAssembly module from a file path.
71    #[inline]
72    pub fn from_file(
73        engine: &impl AsEngineRef,
74        file: impl AsRef<Path>,
75    ) -> Result<Self, super::IoCompileError> {
76        let file_ref = file.as_ref();
77        let canonical = file_ref.canonicalize()?;
78        let wasm_bytes = std::fs::read(file_ref)?;
79        let mut module = Self::new(engine, wasm_bytes)?;
80        // Set the module name to the absolute path of the filename.
81        // This is useful for debugging the stack traces.
82        let filename = canonical.as_path().to_str().unwrap();
83        module.set_name(filename);
84        Ok(module)
85    }

674    /// Returns an iterator over the imported types in the Module.
675    ///
676    /// The order of the imports is guaranteed to be the same as in the
677    /// WebAssembly bytecode.
678    ///
679    /// # Example
680    ///
681    /// ```
682    /// # use wasmer::*;
683    /// # fn main() -> anyhow::Result<()> {
684    /// # let mut store = Store::default();
685    /// let wat = r#"(module
686    ///     (import "host" "func1" (func))
687    ///     (import "host" "func2" (func))
688    /// )"#;
689    /// let module = Module::new(&store, wat)?;
690    /// for import in module.imports() {
691    ///     assert_eq!(import.module(), "host");
692    ///     assert!(import.name().contains("func"));
693    ///     import.ty();
694    /// }
695    /// # Ok(())
696    /// # }
697    /// ```
698    #[inline]
699    pub fn imports(&self) -> ImportsIterator<Box<dyn Iterator<Item = ImportType> + '_>> {
700        match_rt!(on self => s {

705    /// Returns an iterator over the exported types in the Module.
706    ///
707    /// The order of the exports is guaranteed to be the same as in the
708    /// WebAssembly bytecode.
709    ///
710    /// # Example
711    ///
712    /// ```
713    /// # use wasmer::*;
714    /// # fn main() -> anyhow::Result<()> {
715    /// # let mut store = Store::default();
716    /// let wat = r#"(module
717    ///     (func (export "namedfunc"))
718    ///     (memory (export "namedmemory") 1)
719    /// )"#;
720    /// let module = Module::new(&store, wat)?;
721    /// for export_ in module.exports() {
722    ///     assert!(export_.name().contains("named"));
723    ///     export_.ty();
724    /// }
725    /// # Ok(())
726    /// # }
727    /// ```
728    #[inline]
729    pub fn exports(&self) -> ExportsIterator<Box<dyn Iterator<Item = ExportType> + '_>> {
730        match_rt!(on self => s {
731            s.exports()
732        })
733    }

736    ///
737    /// # Important
738    ///
739    /// Following the WebAssembly spec, one name can have multiple
740    /// custom sections. That's why an iterator (rather than one element)
741    /// is returned.
742    #[inline]
743    pub fn custom_sections<'a>(&'a self, name: &'a str) -> impl Iterator<Item = Box<[u8]>> + 'a {
