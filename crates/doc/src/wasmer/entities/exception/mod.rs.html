13/// A WebAssembly `exception` instance.
14///
15/// An exception is an internal construct in WebAssembly that represents a runtime object that can
16/// be thrown. A WebAssembly exception consists of an exception tag and its runtime arguments.
17///
18/// Spec: <https://github.com/WebAssembly/exception-handling/blob/main/proposals/exception-handling/Exceptions.md#exceptions>
19#[derive(Debug, Clone, PartialEq, Eq, derive_more::From)]
20#[cfg_attr(feature = "artifact-size", derive(loupe::MemoryUsage))]
21pub struct Exception(pub(crate) BackendException);
22
23impl Exception {
24    /// Create a new exception with the given tag type and payload.
25    pub fn new(store: &mut impl AsStoreMut, tag: Tag, payload: &[Value]) -> Self {
26        Self(BackendException::new(store, tag, payload))
27    }
28
29    /// Checks whether this `Exception` comes from the given store.
30    pub fn is_from_store(&self, store: &impl AsStoreRef) -> bool {
31        self.0.is_from_store(store)
32    }
33}
