11    /// Make a new extern reference
12    pub fn new<T>(store: &mut impl AsStoreMut, value: T) -> Self
13    where
14        T: Any + Send + Sync + 'static + Sized,
15    {
16        match &store.as_store_mut().inner.store {
17            #[cfg(feature = "sys")]

19                crate::backend::sys::entities::exception::ExceptionRef::new(store, value),
20            ),
21            #[cfg(feature = "wamr")]
22            crate::BackendStore::Wamr(s) => Self::Wamr(
23                crate::backend::wamr::entities::exception::ExceptionRef::new(store, value),
24            ),
25            #[cfg(feature = "wasmi")]
26            crate::BackendStore::Wasmi(s) => Self::Wasmi(
27                crate::backend::wasmi::entities::exception::ExceptionRef::new(store, value),
28            ),
29            #[cfg(feature = "v8")]
30            crate::BackendStore::V8(s) => Self::V8(
31                crate::backend::v8::entities::exception::ExceptionRef::new(store, value),
32            ),
33            #[cfg(feature = "js")]
34            crate::BackendStore::Js(s) => Self::Js(

44    /// Try to downcast to the given value.
45    pub fn downcast<'a, T>(&self, store: &'a impl AsStoreRef) -> Option<&'a T>
46    where
47        T: Any + Send + Sync + 'static + Sized,
48    {
49        match_rt!(on self => r {
50            r.downcast::<T>(store)
51        })
52    }

54    /// Create a [`VMExceptionRef`] from [`Self`].
55    pub(crate) fn vm_exceptionref(&self) -> VMExceptionRef {
56        match self {
57            #[cfg(feature = "sys")]
58            Self::Sys(r) => VMExceptionRef::Sys(r.vm_exceptionref()),
59            #[cfg(feature = "wamr")]
60            Self::Wamr(r) => VMExceptionRef::Wamr(r.vm_exceptionref()),
61            #[cfg(feature = "wasmi")]
62            Self::Wasmi(r) => VMExceptionRef::Wasmi(r.vm_exceptionref()),
63            #[cfg(feature = "v8")]
64            Self::V8(r) => VMExceptionRef::V8(r.vm_exceptionref()),
65            #[cfg(feature = "js")]
66            Self::Js(r) => VMExceptionRef::Js(r.vm_exceptionref()),
67            #[cfg(feature = "jsc")]
68            Self::Jsc(r) => VMExceptionRef::Jsc(r.vm_exceptionref()),
69        }
70    }

123    /// Checks whether this `ExceptionRef` can be used with the given context.
124    ///
125    /// Primitive (`i32`, `i64`, etc) and null funcref/externref values are not
126    /// tied to a context and can be freely shared between contexts.
127    ///
128    /// Externref and funcref values are tied to a context and can only be used
129    /// with that context.
130    pub fn is_from_store(&self, store: &impl AsStoreRef) -> bool {
131        match_rt!(on self => r {
132            r.is_from_store(store)
133        })
134    }
