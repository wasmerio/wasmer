9/// A WebAssembly `memory` view.
10///
11/// A memory view is used to read and write to the linear memory.
12///
13/// After a memory is grown a view must not be used anymore. Views are
14/// created using the Memory.view() method.
15#[derive(Debug, derive_more::From)]
16pub struct MemoryView<'a>(pub(crate) BackendMemoryView<'a>);
17
18impl<'a> MemoryView<'a> {
19    pub(crate) fn new(memory: &Memory, store: &'a (impl AsStoreRef + ?Sized)) -> Self {
20        Self(BackendMemoryView::new(memory, store))
21    }
22
23    /// Returns the pointer to the raw bytes of the `Memory`.
24    //
25    // This used by wasmer-c-api, but should be treated
26    // as deprecated and not used in future code.
27    #[doc(hidden)]
28    pub fn data_ptr(&self) -> *mut u8 {
29        self.0.data_ptr()
30    }
31
32    /// Returns the size (in bytes) of the `Memory`.
33    pub fn data_size(&self) -> u64 {
34        self.0.data_size()
35    }
36
37    /// Retrieve a slice of the memory contents.
38    ///
39    /// # Safety
40    ///
41    /// Until the returned slice is dropped, it is undefined behaviour to
42    /// modify the memory contents in any way including by calling a wasm
43    /// function that writes to the memory or by resizing the memory.
44    #[doc(hidden)]
45    pub unsafe fn data_unchecked(&self) -> &[u8] {
46        unsafe { self.0.data_unchecked() }
47    }
48
49    /// Retrieve a mutable slice of the memory contents.
50    ///
51    /// # Safety
52    ///
53    /// This method provides interior mutability without an UnsafeCell. Until
54    /// the returned value is dropped, it is undefined behaviour to read or
55    /// write to the pointed-to memory in any way except through this slice,
56    /// including by calling a wasm function that reads the memory contents or
57    /// by resizing this Memory.
58    #[allow(clippy::mut_from_ref)]
59    #[doc(hidden)]
60    pub unsafe fn data_unchecked_mut(&self) -> &mut [u8] {
61        unsafe { self.0.data_unchecked_mut() }
62    }
63
64    /// Returns the size (in [`Pages`]) of the `Memory`.
65    ///
66    /// # Example
67    ///
68    /// ```
69    /// # use wasmer::{Memory, MemoryType, Pages, Store, Type, Value};
70    /// # let mut store = Store::default();
71    /// #
72    /// let m = Memory::new(&mut store, MemoryType::new(1, None, false)).unwrap();
73    ///
74    /// assert_eq!(m.view(&mut store).size(), Pages(1));
75    /// ```
76    pub fn size(&self) -> Pages {
77        self.0.size()
78    }
79
80    #[inline]
81    pub(crate) fn buffer(&'a self) -> MemoryBuffer<'a> {
82        self.0.buffer()
83    }
84
85    /// Safely reads bytes from the memory at the given offset.
86    ///
87    /// The full buffer will be filled, otherwise a `MemoryAccessError` is returned
88    /// to indicate an out-of-bounds access.
89    ///
90    /// This method is guaranteed to be safe (from the host side) in the face of
91    /// concurrent writes.
92    pub fn read(&self, offset: u64, buf: &mut [u8]) -> Result<(), MemoryAccessError> {
93        self.0.read(offset, buf)
94    }
95
96    /// Safely reads a single byte from memory at the given offset
97    ///
98    /// This method is guaranteed to be safe (from the host side) in the face of
99    /// concurrent writes.
100    pub fn read_u8(&self, offset: u64) -> Result<u8, MemoryAccessError> {
101        self.0.read_u8(offset)
102    }
103
104    /// Safely reads bytes from the memory at the given offset.
105    ///
106    /// This method is similar to `read` but allows reading into an
107    /// uninitialized buffer. An initialized view of the buffer is returned.
108    ///
109    /// The full buffer will be filled, otherwise a `MemoryAccessError` is returned
110    /// to indicate an out-of-bounds access.
111    ///
112    /// This method is guaranteed to be safe (from the host side) in the face of
113    /// concurrent writes.
114    pub fn read_uninit<'b>(
115        &self,
116        offset: u64,
117        buf: &'b mut [MaybeUninit<u8>],
118    ) -> Result<&'b mut [u8], MemoryAccessError> {
119        self.0.read_uninit(offset, buf)
120    }
121
122    /// Safely writes bytes to the memory at the given offset.
123    ///
124    /// If the write exceeds the bounds of the memory then a `MemoryAccessError` is
125    /// returned.
126    ///
127    /// This method is guaranteed to be safe (from the host side) in the face of
128    /// concurrent reads/writes.
129    pub fn write(&self, offset: u64, data: &[u8]) -> Result<(), MemoryAccessError> {
130        self.0.write(offset, data)
131    }
132
133    /// Safely writes a single byte from memory at the given offset
134    ///
135    /// This method is guaranteed to be safe (from the host side) in the face of
136    /// concurrent writes.
137    pub fn write_u8(&self, offset: u64, val: u8) -> Result<(), MemoryAccessError> {
138        self.0.write_u8(offset, val)
139    }
140
141    /// Copies the memory and returns it as a vector of bytes
142    pub fn copy_to_vec(&self) -> Result<Vec<u8>, MemoryAccessError> {
143        self.0.copy_to_vec()
144    }
145
146    /// Copies a range of the memory and returns it as a vector of bytes
147    pub fn copy_range_to_vec(&self, range: Range<u64>) -> Result<Vec<u8>, MemoryAccessError> {
148        self.0.copy_range_to_vec(range)
149    }
150
151    /// Copies the memory to another new memory object
152    pub fn copy_to_memory(&self, amount: u64, new_memory: &Self) -> Result<(), MemoryAccessError> {
153        self.0.copy_to_memory(amount, &new_memory.0)
154    }
