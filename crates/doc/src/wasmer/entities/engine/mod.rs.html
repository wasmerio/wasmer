31/// The [`Engine`] is the entrypoint type for the runtime. It defines the kind of steps the runtime must take to execute
32/// the WebAssembly module (compile, interpret..) and the place of execution (in-browser, host,
33/// ..).
34#[derive(Debug, Clone)]
35pub struct Engine {
36    pub(crate) be: BackendEngine,
37    pub(crate) id: u64,
38}
39
40impl Default for Engine {
41    fn default() -> Self {
42        Self {
43            be: Default::default(),
44            id: Self::atomic_next_engine_id(),
45        }
46    }
47}
48
49impl Engine {
50    pub(crate) fn atomic_next_engine_id() -> u64 {
51        #[cfg(target_has_atomic = "64")]
52        {
53            static ENGINE_ID_COUNTER: std::sync::atomic::AtomicU64 =
54                std::sync::atomic::AtomicU64::new(0);
55            return ENGINE_ID_COUNTER.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
56        }
57
58        #[allow(unreachable_code)]
59        #[cfg(target_has_atomic = "32")]
60        {
61            static ENGINE_ID_COUNTER: std::sync::atomic::AtomicU32 =
62                std::sync::atomic::AtomicU32::new(0);
63            ENGINE_ID_COUNTER.fetch_add(1, std::sync::atomic::Ordering::SeqCst) as _
64        }
65    }

77    /// Returns the default WebAssembly features supported by this backend for a given target.
78    ///
79    /// These are the features that will be enabled by default without any user configuration.
80    pub fn default_features_for_backend(backend: &crate::BackendKind, target: &Target) -> Features {
81        match backend {
82            #[cfg(feature = "cranelift")]
83            crate::BackendKind::Cranelift => {
84                wasmer_compiler_cranelift::Cranelift::default().default_features_for_target(target)
85            }
86            #[cfg(feature = "llvm")]
87            crate::BackendKind::LLVM => {
88                wasmer_compiler_llvm::LLVM::default().default_features_for_target(target)
89            }
90            #[cfg(feature = "singlepass")]
91            crate::BackendKind::Singlepass => wasmer_compiler_singlepass::Singlepass::default()
92                .default_features_for_target(target),
93            #[cfg(feature = "v8")]
94            crate::BackendKind::V8 => {
95                // Get V8-specific features
96                crate::backend::v8::engine::Engine::default_features()
97            }
98            #[cfg(feature = "wamr")]
99            crate::BackendKind::Wamr => {
100                // Get WAMR-specific features
101                crate::backend::wamr::engine::Engine::default_features()
102            }
103            #[cfg(feature = "wasmi")]
104            crate::BackendKind::Wasmi => {
105                // Get WASMI-specific features
106                crate::backend::wasmi::engine::Engine::default_features()
107            }
108            #[cfg(feature = "js")]
109            crate::BackendKind::Js => {
110                // Get JS-specific features
111                crate::backend::js::engine::Engine::default_features()
112            }
113            #[cfg(feature = "jsc")]
114            crate::BackendKind::Jsc => {
115                // Get JSC-specific features
116                crate::backend::jsc::engine::Engine::default_features()
117            }
118            // Default case
119            _ => Features::default(),
120        }
121    }

171    #[cfg(all(feature = "sys", not(target_arch = "wasm32")))]
172    /// Deserializes a WebAssembly module which was previously serialized with
173    /// `Module::serialize`,
174    ///
175    /// # Note
176    /// You should almost always prefer [`Self::deserialize`].
177    ///
178    /// # Errors
179    /// Not every implementer supports serializing and deserializing modules.
180    /// Currently, only the `sys` engines support it, and only when the target
181    /// architecture is not `wasm32`.
182    ///
183    /// # Safety
184    /// See [`Artifact::deserialize_unchecked`].
185    unsafe fn deserialize_unchecked(
186        &self,
187        bytes: impl IntoBytes,
188    ) -> Result<Arc<Artifact>, DeserializeError> {
189        unsafe { self.be.deserialize_unchecked(bytes) }
190    }

241    /// Add suggested optimizations to this engine.
242    ///
243    /// # Note
244    ///
245    /// Not every backend supports every optimization. This function may fail (i.e. not set the
246    /// suggested optimizations) silently if the underlying engine backend does not support one or
247    /// more optimizations.
248    pub fn with_opts(
249        &mut self,
250        suggested_opts: &UserCompilerOptimizations,
251    ) -> Result<(), CompileError> {
252        match self.be {
253            #[cfg(feature = "sys")]
254            BackendEngine::Sys(ref mut e) => e.with_opts(suggested_opts),
255            _ => Ok(()),
256        }
257    }
