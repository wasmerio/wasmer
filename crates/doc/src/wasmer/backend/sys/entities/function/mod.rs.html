37    pub(crate) fn new_with_env<FT, F, T: Send + 'static>(
38        store: &mut impl AsStoreMut,
39        env: &FunctionEnv<T>,
40        ty: FT,
41        func: F,
42    ) -> Self
43    where
44        FT: Into<FunctionType>,
45        F: Fn(FunctionEnvMut<T>, &[Value]) -> Result<Vec<Value>, RuntimeError>
46            + 'static
47            + Send
48            + Sync,
49    {
50        let function_type = ty.into();
51        let func_ty = function_type.clone();
52        let func_env = env.clone().into_sys();
53        let raw_store = store.as_store_mut().as_raw() as *mut u8;
54        let wrapper = move |values_vec: *mut RawValue| -> Result<(), RuntimeError> {
55            unsafe {
56                let mut store = StoreMut::from_raw(raw_store as *mut StoreInner);
57                let mut args = Vec::with_capacity(func_ty.params().len());
58
59                for (i, ty) in func_ty.params().iter().enumerate() {
60                    args.push(Value::from_raw(
61                        &mut store,
62                        *ty,
63                        values_vec.add(i).read_unaligned(),
64                    ));
65                }
66                let store_mut = StoreMut::from_raw(raw_store as *mut StoreInner);
67                let env = env::FunctionEnvMut {
68                    store_mut,
69                    func_env: func_env.clone(),
70                }
71                .into();
72                let returns = func(env, &args)?;
73
74                // We need to dynamically check that the returns
75                // match the expected types, as well as expected length.
76                let return_types = returns.iter().map(|ret| ret.ty());
77                if return_types.ne(func_ty.results().iter().copied()) {
78                    return Err(RuntimeError::new(format!(
79                        "Dynamic function returned wrong signature. Expected {:?} but got {:?}",
80                        func_ty.results(),
81                        returns.iter().map(|ret| ret.ty())
82                    )));
83                }
84                for (i, ret) in returns.iter().enumerate() {
85                    values_vec.add(i).write_unaligned(ret.as_raw(&store));
86                }
87            }
88            Ok(())
89        };
90        let mut host_data = Box::new(VMDynamicFunctionContext {
91            address: std::ptr::null(),
92            ctx: DynamicFunction { func: wrapper },
93        });
94        host_data.address = host_data.ctx.func_body_ptr();
95
96        // We don't yet have the address with the Wasm ABI signature.
97        // The engine linker will replace the address with one pointing to a
98        // generated dynamic trampoline.
99        let func_ptr = std::ptr::null() as VMFunctionCallback;
100        let type_index = store
101            .as_store_mut()
102            .engine()
103            .as_sys()
104            .register_signature(&function_type);
105        let vmctx = VMFunctionContext {
106            host_env: host_data.as_ref() as *const _ as *mut c_void,
107        };
108        let call_trampoline = host_data.ctx.call_trampoline_address();
109        let anyfunc = VMCallerCheckedAnyfunc {
110            func_ptr,
111            type_index,
112            vmctx,
113            call_trampoline,
114        };
115
116        let vm_function = VMFunction {
117            anyfunc: MaybeInstanceOwned::Host(Box::new(UnsafeCell::new(anyfunc))),
118            kind: VMFunctionKind::Dynamic,
119            signature: function_type,
120            host_data,
121        };
122        Self {
123            handle: StoreHandle::new(store.as_store_mut().objects_mut().as_sys_mut(), vm_function),
124        }
125    }

224    fn call_wasm(
225        &self,
226        store: &mut impl AsStoreMut,
227        trampoline: VMTrampoline,
228        params: &[Value],
229        results: &mut [Value],
230    ) -> Result<(), RuntimeError> {
231        let format_types_for_error_message = |items: &[Value]| {
232            items
233                .iter()
234                .map(|param| param.ty().to_string())
235                .collect::<Vec<String>>()
236                .join(", ")
237        };
238        // TODO: Avoid cloning the signature here, it's expensive.
239        let signature = self.ty(store);
240        if signature.params().len() != params.len() {
241            return Err(RuntimeError::new(format!(
242                "Parameters of type [{}] did not match signature {}",
243                format_types_for_error_message(params),
244                &signature
245            )));
246        }
247        if signature.results().len() != results.len() {
248            return Err(RuntimeError::new(format!(
249                "Results of type [{}] did not match signature {}",
250                format_types_for_error_message(results),
251                &signature,
252            )));
253        }
254
255        let mut values_vec = vec![RawValue { i32: 0 }; max(params.len(), results.len())];
256
257        // Store the argument values into `values_vec`.
258        let param_tys = signature.params().iter();
259        for ((arg, slot), ty) in params.iter().zip(&mut values_vec).zip(param_tys) {
260            if arg.ty() != *ty {
261                let param_types = format_types_for_error_message(params);
262                return Err(RuntimeError::new(format!(
263                    "Parameters of type [{}] did not match signature {}",
264                    param_types, &signature,
265                )));
266            }
267            if !arg.is_from_store(store) {
268                return Err(RuntimeError::new("cross-`Store` values are not supported"));
269            }
270            *slot = arg.as_raw(store);
271        }
272
273        // Invoke the call
274        self.call_wasm_raw(store, trampoline, values_vec, results)?;
275        Ok(())
276    }

278    fn call_wasm_raw(
279        &self,
280        store: &mut impl AsStoreMut,
281        trampoline: VMTrampoline,
282        mut params: Vec<RawValue>,
283        results: &mut [Value],
284    ) -> Result<(), RuntimeError> {
285        // Call the trampoline.
286        let result = {
287            let mut r;
288            // TODO: This loop is needed for asyncify. It will be refactored with https://github.com/wasmerio/wasmer/issues/3451
289            loop {
290                let storeref = store.as_store_ref();
291                let vm_function = self.handle.get(storeref.objects().as_sys());
292                let config = storeref.engine().tunables().vmconfig();
293                r = unsafe {
294                    wasmer_call_trampoline(
295                        store.as_store_ref().signal_handler(),
296                        config,
297                        vm_function.anyfunc.as_ptr().as_ref().vmctx,
298                        trampoline,
299                        vm_function.anyfunc.as_ptr().as_ref().func_ptr,
300                        params.as_mut_ptr() as *mut u8,
301                    )
302                };
303                let store_mut = store.as_store_mut();
304                if let Some(callback) = store_mut.inner.on_called.take() {
305                    match callback(store_mut) {
306                        Ok(wasmer_types::OnCalledAction::InvokeAgain) => {
307                            continue;
308                        }
309                        Ok(wasmer_types::OnCalledAction::Finish) => {
310                            break;
311                        }
312                        Ok(wasmer_types::OnCalledAction::Trap(trap)) => {
313                            return Err(RuntimeError::user(trap));
314                        }
315                        Err(trap) => return Err(RuntimeError::user(trap)),
316                    }
317                }
318                break;
319            }
320            r
321        };
322        if let Err(error) = result {
323            return Err(error.into());
324        }
325
326        // Load the return values out of `values_vec`.
327        let signature = self.ty(store);
328        for (index, &value_type) in signature.results().iter().enumerate() {
329            unsafe {
330                results[index] = Value::from_raw(store, value_type, params[index]);
331            }
332        }
333
334        Ok(())
335    }

341    pub(crate) fn call(
342        &self,
343        store: &mut impl AsStoreMut,
344        params: &[Value],
345    ) -> Result<Box<[Value]>, RuntimeError> {
346        let trampoline = unsafe {
347            self.handle
348                .get(store.as_store_ref().objects().as_sys())
349                .anyfunc
350                .as_ptr()
351                .as_ref()
352                .call_trampoline
353        };
354        let mut results = vec![Value::null(); self.result_arity(store)];
355        self.call_wasm(store, trampoline, params, &mut results)?;
356        Ok(results.into_boxed_slice())
357    }

379    pub(crate) fn vm_funcref(&self, store: &impl AsStoreRef) -> VMFuncRef {
380        let vm_function = self.handle.get(store.as_store_ref().objects().as_sys());
381        if vm_function.kind == VMFunctionKind::Dynamic {
382            panic!("dynamic functions cannot be used in tables or as funcrefs");
383        }
384        VMFuncRef(vm_function.anyfunc.as_ptr())
385    }

387    pub(crate) unsafe fn from_vm_funcref(store: &mut impl AsStoreMut, funcref: VMFuncRef) -> Self {
388        let signature = {
389            let anyfunc = unsafe { funcref.0.as_ref() };
390            store
391                .as_store_ref()
392                .engine()
393                .as_sys()
394                .lookup_signature(anyfunc.type_index)
395                .expect("Signature not found in store")
396        };
397        let vm_function = VMFunction {
398            anyfunc: MaybeInstanceOwned::Instance(funcref.0),
399            signature,
400            // All functions in tables are already Static (as dynamic functions
401            // are converted to use the trampolines with static signatures).
402            kind: wasmer_vm::VMFunctionKind::Static,
403            host_data: Box::new(()),
404        };
405        Self {
406            handle: StoreHandle::new(store.objects_mut().as_sys_mut(), vm_function),
407        }
408    }

431/// Host state for a dynamic function.
432pub(crate) struct DynamicFunction<F> {
433    func: F,
434}
435
436impl<F> DynamicFunction<F>
437where
438    F: Fn(*mut RawValue) -> Result<(), RuntimeError> + 'static,
439{
440    // This function wraps our func, to make it compatible with the
441    // reverse trampoline signature
442    unsafe extern "C" fn func_wrapper(
443        this: &mut VMDynamicFunctionContext<Self>,
444        values_vec: *mut RawValue,
445    ) {
446        let result = on_host_stack(|| {
447            panic::catch_unwind(AssertUnwindSafe(|| {
448                (this.ctx.func)(values_vec).map_err(Box::new)
449            }))
450        });
451
452        // IMPORTANT: DO NOT ALLOCATE ON THE STACK,
453        // AS WE ARE IN THE WASM STACK, NOT ON THE HOST ONE.
454        // See: https://github.com/wasmerio/wasmer/pull/5700
455        match result {
456            Ok(Ok(())) => {}
457            Ok(Err(trap)) => unsafe { raise_user_trap(trap) },
458            Err(panic) => unsafe { resume_panic(panic) },
459        }
460    }
461
462    fn func_body_ptr(&self) -> VMFunctionCallback {
463        Self::func_wrapper as VMFunctionCallback
464    }
465
466    fn call_trampoline_address(&self) -> VMTrampoline {
467        Self::call_trampoline
468    }
469
470    unsafe extern "C" fn call_trampoline(
471        vmctx: *mut VMContext,
472        _body: VMFunctionCallback,
473        args: *mut RawValue,
474    ) {
475        // The VMFunctionCallback is null here: it is only filled in later
476        // by the engine linker.
477        unsafe {
478            let dynamic_function = &mut *(vmctx as *mut VMDynamicFunctionContext<Self>);
479            Self::func_wrapper(dynamic_function, args);
480        }
481    }
482}

519macro_rules! impl_host_function {
520    ([$c_struct_representation:ident] $c_struct_name:ident, $( $x:ident ),* ) => {
521        paste::paste! {
522        #[allow(non_snake_case)]
523        pub(crate) fn [<gen_fn_callback_ $c_struct_name:lower _no_env>]
524            <$( $x: FromToNativeWasmType, )* Rets: WasmTypeList, RetsAsResult: IntoResult<Rets>, Func: Fn($( $x , )*) -> RetsAsResult + 'static>
525            (this: &Func) -> crate::backend::sys::vm::VMFunctionCallback {
526            /// This is a function that wraps the real host
527            /// function. Its address will be used inside the
528            /// runtime.
529            unsafe extern "C" fn func_wrapper<$( $x, )* Rets, RetsAsResult, Func>( env: &StaticFunction<Func, ()>, $( $x: <$x::Native as NativeWasmType>::Abi, )* ) -> Rets::CStruct
530            where
531                $( $x: FromToNativeWasmType, )*
532                Rets: WasmTypeList,
533                RetsAsResult: IntoResult<Rets>,
534                Func: Fn($( $x , )*) -> RetsAsResult + 'static,
535            {
536                let mut store = unsafe { StoreMut::from_raw(env.raw_store as *mut _) };
537                let result = on_host_stack(|| {
538                    panic::catch_unwind(AssertUnwindSafe(|| {
539                        $(
540                            let $x = unsafe {
541                                FromToNativeWasmType::from_native(NativeWasmTypeInto::from_abi(&mut store, $x))
542                            };
543                        )*
544                        (env.func)($($x),* ).into_result().map_err(Box::new)
545                    }))
546                });
547
548                // IMPORTANT: DO NOT ALLOCATE ON THE STACK,
549                // AS WE ARE IN THE WASM STACK, NOT ON THE HOST ONE.
550                // See: https://github.com/wasmerio/wasmer/pull/5700
551                match result {
552                    Ok(Ok(result)) => {
553                        unsafe {
554                            return result.into_c_struct(&mut store);
555                        }
556                    },
557                    Ok(Err(trap)) => unsafe {
558                        raise_user_trap(trap)
559                    },
560                    Err(panic) => unsafe {
561                        resume_panic(panic)
562                    },
563                }
564            }
565
566            func_wrapper::< $( $x, )* Rets, RetsAsResult, Func > as _
567
568        }
569
570        #[allow(non_snake_case)]
571        pub(crate) fn [<gen_call_trampoline_address_ $c_struct_name:lower _no_env>]
572            <$( $x: FromToNativeWasmType, )* Rets: WasmTypeList>
573            () -> crate::backend::sys::vm::VMTrampoline {
574
575            unsafe extern "C" fn call_trampoline<$( $x: FromToNativeWasmType, )* Rets: WasmTypeList>
576            (
577                vmctx: *mut crate::backend::sys::vm::VMContext,
578                body: crate::backend::sys::vm::VMFunctionCallback,
579                args: *mut RawValue,
580            ) {
581                let mut _n = 0;
582
583                unsafe {
584                    let body: unsafe extern "C" fn(vmctx: *mut crate::backend::sys::vm::VMContext, $( $x: <$x::Native as NativeWasmType>::Abi, )*) -> Rets::CStruct = std::mem::transmute(body);
585                    $(
586                        let $x = *args.add(_n).cast();
587                        _n += 1;
588                    )*
589                    let results = body(vmctx, $( $x ),*);
590                    Rets::write_c_struct_to_ptr(results, args);
591                }
592            }
593
594            call_trampoline::<$( $x, )* Rets> as _
595
596        }

598        #[allow(non_snake_case)]
599        pub(crate) fn [<gen_fn_callback_ $c_struct_name:lower>]
600            <$( $x: FromToNativeWasmType, )* Rets: WasmTypeList, RetsAsResult: IntoResult<Rets>, T: Send + 'static,  Func: Fn(FunctionEnvMut<T>, $( $x , )*) -> RetsAsResult + 'static>
601            (this: &Func) -> crate::backend::sys::vm::VMFunctionCallback {
602            /// This is a function that wraps the real host
603            /// function. Its address will be used inside the
604            /// runtime.
605            unsafe extern "C" fn func_wrapper<T: Send + 'static, $( $x, )* Rets, RetsAsResult, Func>( env: &StaticFunction<Func, T>, $( $x: <$x::Native as NativeWasmType>::Abi, )* ) -> Rets::CStruct
606                where
607                $( $x: FromToNativeWasmType, )*
608                Rets: WasmTypeList,
609                RetsAsResult: IntoResult<Rets>,
610                Func: Fn(FunctionEnvMut<T>, $( $x , )*) -> RetsAsResult + 'static,
611            {
612
613                let mut store = unsafe { StoreMut::from_raw(env.raw_store as *mut _) };
614                let result = wasmer_vm::on_host_stack(|| {
615                    panic::catch_unwind(AssertUnwindSafe(|| {
616                        $(
617                            let $x = unsafe {
618                                FromToNativeWasmType::from_native(NativeWasmTypeInto::from_abi(&mut store, $x))
619                            };
620                        )*
621                        let store_mut = unsafe { StoreMut::from_raw(env.raw_store as *mut _) };
622                        let f_env = crate::backend::sys::function::env::FunctionEnvMut {
623                            store_mut,
624                            func_env: env.env.as_sys().clone(),
625                        }.into();
626                        (env.func)(f_env, $($x),* ).into_result().map_err(Box::new)
627                    }))
628                });

651        }

662            ) {

665                    let mut _n = 0;
