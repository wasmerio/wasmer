14            /// Call the typed func and return results.
15            #[allow(unused_mut)]
16            #[allow(clippy::too_many_arguments)]
17            pub fn call_sys(&self, store: &mut impl AsStoreMut, $( $x: $x, )* ) -> Result<Rets, RuntimeError> {
18                let anyfunc = unsafe {
19                    *self.func.as_sys()
20                        .handle
21                        .get(store.as_store_ref().objects().as_sys())
22                        .anyfunc
23                        .as_ptr()
24                        .as_ref()
25                };
26                // Ensure all parameters come from the same context.
27                if $(!FromToNativeWasmType::is_from_store(&$x, store) ||)* false {
28                    return Err(RuntimeError::new(
29                        "cross-`Store` values are not supported",
30                    ));
31                }
32                // TODO: when `const fn` related features mature more, we can declare a single array
33                // of the correct size here.
34                let mut params_list = [ $( $x.to_native().into_raw(store) ),* ];
35                let mut rets_list_array = Rets::empty_array();
36                let rets_list: &mut [RawValue] = rets_list_array.as_mut();
37                let using_rets_array;
38                let args_rets: &mut [RawValue] = if params_list.len() > rets_list.len() {
39                    using_rets_array = false;
40                    params_list.as_mut()
41                } else {
42                    using_rets_array = true;
43                    for (i, &arg) in params_list.iter().enumerate() {
44                        rets_list[i] = arg;
45                    }
46                    rets_list.as_mut()
47                };
48
49                let mut r;
50                loop {
51                    let storeref = store.as_store_ref();
52                    let config = storeref.engine().tunables().vmconfig();
53                    r = unsafe {
54                        wasmer_vm::wasmer_call_trampoline(
55                            store.as_store_ref().signal_handler(),
56                            config,
57                            anyfunc.vmctx,
58                            anyfunc.call_trampoline,
59                            anyfunc.func_ptr,
60                            args_rets.as_mut_ptr() as *mut u8,
61                        )
62                    };
63                    let store_mut = store.as_store_mut();
64                    if let Some(callback) = store_mut.inner.on_called.take() {
65                        match callback(store_mut) {
66                            Ok(wasmer_types::OnCalledAction::InvokeAgain) => { continue; }
67                            Ok(wasmer_types::OnCalledAction::Finish) => { break; }
68                            Ok(wasmer_types::OnCalledAction::Trap(trap)) => { return Err(RuntimeError::user(trap)) },
69                            Err(trap) => { return Err(RuntimeError::user(trap)) },
70                        }
71                    }
72                    break;
73                }
74                r?;
75
76                let num_rets = rets_list.len();
77                if !using_rets_array && num_rets > 0 {
78                    let src_pointer = params_list.as_ptr();
79                    let rets_list = &mut rets_list_array.as_mut()[0] as *mut RawValue;
80                    unsafe {
81                        // TODO: we can probably remove this copy by doing some clever `transmute`s.
82                        // we know it's not overlapping because `using_rets_array` is false
83                        std::ptr::copy_nonoverlapping(src_pointer,
84                                                        rets_list,
85                                                        num_rets);
86                    }
87                }
88                Ok(unsafe { Rets::from_array(store, rets_list_array) })
89                // TODO: When the Host ABI and Wasm ABI are the same, we could do this instead:
90                // but we can't currently detect whether that's safe.
91                //
92                // let results = unsafe {
93                //     wasmer_vm::catch_traps_with_result(self.vmctx, || {
94                //         let f = std::mem::transmute::<_, unsafe extern "C" fn( *mut VMContext, $( $x, )*) -> Rets::CStruct>(self.address());
95                //         // We always pass the vmctx
96                //         f( self.vmctx, $( $x, )* )
97                //     }).map_err(RuntimeError::from_trap)?
98                // };
99                // Ok(Rets::from_c_struct(results))
100            }
