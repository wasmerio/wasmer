12/// An opaque reference to a function environment.
13/// The function environment data is owned by the `Store`.
14pub struct FunctionEnv<T> {
15    pub(crate) handle: StoreHandle<VMFunctionEnvironment>,
16    marker: PhantomData<T>,
17}
18
19impl<T> FunctionEnv<T> {
20    /// Make a new FunctionEnv
21    pub fn new(store: &mut impl AsStoreMut, value: T) -> Self
22    where
23        T: Any + Send + 'static + Sized,
24    {
25        Self {
26            handle: StoreHandle::new(
27                store.as_store_mut().objects_mut().as_sys_mut(),
28                VMFunctionEnvironment::new(value),
29            ),
30            marker: PhantomData,
31        }
32    }
33
34    /// Get the data as reference
35    pub fn as_ref<'a>(&self, store: &'a impl AsStoreRef) -> &'a T
36    where
37        T: Any + Send + 'static + Sized,
38    {
39        self.handle
40            .get(store.as_store_ref().objects().as_sys())
41            .as_ref()
42            .downcast_ref::<T>()
43            .unwrap()
44    }
45
46    #[allow(dead_code)] // This function is only used in js
47    pub(crate) fn from_handle(handle: StoreHandle<VMFunctionEnvironment>) -> Self {
48        Self {
49            handle,
50            marker: PhantomData,
51        }
52    }
53
54    /// Get the data as mutable
55    pub fn as_mut<'a>(&self, store: &'a mut impl AsStoreMut) -> &'a mut T
56    where
57        T: Any + Send + 'static + Sized,
58    {
59        self.handle
60            .get_mut(store.objects_mut().as_sys_mut())
61            .as_mut()
62            .downcast_mut::<T>()
63            .unwrap()
64    }
65
66    /// Convert it into a `FunctionEnvMut`
67    pub fn into_mut(self, store: &mut impl AsStoreMut) -> FunctionEnvMut<'_, T>
68    where
69        T: Any + Send + 'static + Sized,
70    {
71        FunctionEnvMut {
72            store_mut: store.as_store_mut(),
73            func_env: self,
74        }
75    }
76}

78impl<T> crate::FunctionEnv<T> {
79    /// Consume self into [`crate::backend::sys::function::FunctionEnv`].
80    pub fn into_sys(self) -> FunctionEnv<T> {
81        match self.0 {
82            crate::BackendFunctionEnv::Sys(s) => s,
83            _ => panic!("Not a `sys` function env!"),
84        }
85    }
86
87    /// Convert a reference to self into a reference to [`crate::backend::sys::function::FunctionEnv`].
88    pub fn as_sys(&self) -> &FunctionEnv<T> {
89        match self.0 {
90            crate::BackendFunctionEnv::Sys(ref s) => s,
91            _ => panic!("Not a `sys` function env!"),
92        }
93    }
94
95    /// Convert a mutable reference to self into a mutable reference [`crate::backend::sys::function::FunctionEnv`].
96    pub fn as_sys_mut(&mut self) -> &mut FunctionEnv<T> {
97        match self.0 {
98            crate::BackendFunctionEnv::Sys(ref mut s) => s,
99            _ => panic!("Not a `sys` function env!"),
100        }
101    }
102}

128/// A temporary handle to a [`FunctionEnv`].
129pub struct FunctionEnvMut<'a, T: 'a> {
130    pub(crate) store_mut: StoreMut<'a>,
131    pub(crate) func_env: FunctionEnv<T>,
132}
133
134impl<T> Debug for FunctionEnvMut<'_, T>
135where
136    T: Send + Debug + 'static,
137{
138    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
139        self.func_env.as_ref(&self.store_mut).fmt(f)
140    }
141}
142
143impl<T: Send + 'static> FunctionEnvMut<'_, T> {
144    /// Returns a reference to the host state in this function environement.
145    pub fn data(&self) -> &T {
146        self.func_env.as_ref(&self.store_mut)
147    }
148
149    /// Returns a mutable- reference to the host state in this function environement.
150    pub fn data_mut(&mut self) -> &mut T {
151        self.func_env.as_mut(&mut self.store_mut)
152    }
153
154    /// Borrows a new immmutable reference
155    pub fn as_ref(&self) -> FunctionEnv<T> {
156        self.func_env.clone()
157    }
158
159    /// Borrows a new mutable reference
160    pub fn as_mut(&mut self) -> FunctionEnvMut<'_, T> {
161        FunctionEnvMut {
162            store_mut: self.store_mut.as_store_mut(),
163            func_env: self.func_env.clone(),
164        }
165    }
166
167    /// Borrows a new mutable reference of both the attached Store and host state
168    pub fn data_and_store_mut(&mut self) -> (&mut T, StoreMut<'_>) {
169        let data = self.func_env.as_mut(&mut self.store_mut) as *mut T;
170        // telling the borrow check to close his eyes here
171        // this is still relatively safe to do as func_env are
172        // stored in a specific vec of Store, separate from the other objects
173        // and not really directly accessible with the StoreMut
174        let data = unsafe { &mut *data };
175        (data, self.store_mut.as_store_mut())
176    }
177}
