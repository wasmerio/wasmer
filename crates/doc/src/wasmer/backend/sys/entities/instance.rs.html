33impl From<wasmer_compiler::InstantiationError> for InstantiationError {
34    fn from(other: wasmer_compiler::InstantiationError) -> Self {
35        match other {
36            wasmer_compiler::InstantiationError::Link(e) => Self::Link(e.into()),
37            wasmer_compiler::InstantiationError::Start(e) => Self::Start(e.into()),
38            wasmer_compiler::InstantiationError::CpuFeature(e) => Self::CpuFeature(e),
39        }
40    }
41}

43impl Instance {
44    #[allow(clippy::result_large_err)]
45    pub(crate) fn new(
46        store: &mut impl AsStoreMut,
47        module: &Module,
48        imports: &Imports,
49    ) -> Result<(Self, Exports), InstantiationError> {
50        let externs = imports
51            .imports_for_module(module)
52            .map_err(InstantiationError::Link)?;
53        let mut handle = module.as_sys().instantiate(store, &externs)?;
54        let exports = Self::get_exports(store, module, handle.as_sys_mut());
55
56        let instance = Self {
57            _handle: StoreHandle::new(store.objects_mut().as_sys_mut(), handle.into_sys()),
58        };
59
60        Ok((instance, exports))
61    }
62
63    #[allow(clippy::result_large_err)]
64    pub(crate) fn new_by_index(
65        store: &mut impl AsStoreMut,
66        module: &Module,
67        externs: &[Extern],
68    ) -> Result<(Self, Exports), InstantiationError> {
69        let externs = externs.to_vec();
70        let mut handle = module.as_sys().instantiate(store, &externs)?;
71        let exports = Self::get_exports(store, module, handle.as_sys_mut());
72        let instance = Self {
73            _handle: StoreHandle::new(
74                store.as_store_mut().objects_mut().as_sys_mut(),
75                handle.into_sys(),
76            ),
77        };
78
79        Ok((instance, exports))
80    }
81
82    fn get_exports(
83        store: &mut impl AsStoreMut,
84        module: &Module,
85        handle: &mut VMInstance,
86    ) -> Exports {
87        module
88            .exports()
89            .map(|export| {
90                let name = export.name().to_string();
91                let export = handle.lookup(&name).expect("export");
92                let extern_ = Extern::from_vm_extern(store, crate::vm::VMExtern::Sys(export));
93                (name, extern_)
94            })
95            .collect::<Exports>()
96    }
97}

99impl crate::BackendInstance {
