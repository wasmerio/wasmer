28#[derive(Debug, Clone)]
29#[repr(transparent)]
30/// A WebAssembly `extern ref` in the `sys` runtime.
31pub(crate) struct ExceptionRef {
32    handle: StoreHandle<wasmer_vm::VMExceptionObj>,
33}
34
35impl ExceptionRef {
36    /// Make a new extern reference
37    pub fn new<T>(store: &mut impl AsStoreMut, value: T) -> Self
38    where
39        T: Any + Send + Sync + 'static + Sized,
40    {
41        Self {
42            handle: crate::backend::sys::store::StoreHandle::new(
43                store.objects_mut().as_sys_mut(),
44                wasmer_vm::VMExceptionObj::new(value),
45            ),
46        }
47    }
48
49    /// Try to downcast to the given value.
50    pub fn downcast<'a, T>(&self, store: &'a impl AsStoreRef) -> Option<&'a T>
51    where
52        T: Any + Send + Sync + 'static + Sized,
53    {
54        self.handle
55            .get(store.as_store_ref().objects().as_sys())
56            .as_ref()
57            .downcast_ref::<T>()
58    }
59
60    /// Create a [`VMExceptionRef`] from [`Self`].
61    pub(crate) fn vm_exceptionref(&self) -> VMExceptionRef {
62        wasmer_vm::VMExceptionRef(self.handle.internal_handle())
63    }
64
65    /// Create an instance of [`Self`] from a [`VMExceptionRef`].
66    pub(crate) unsafe fn from_vm_exceptionref(
67        store: &mut impl AsStoreMut,
68        vm_exceptionref: VMExceptionRef,
69    ) -> Self {
70        Self {
71            handle: unsafe {
72                StoreHandle::from_internal(store.objects_mut().id(), vm_exceptionref.0)
73            },
74        }
75    }
76
77    /// Checks whether this `ExceptionRef` can be used with the given context.
78    ///
79    /// Primitive (`i32`, `i64`, etc) and null funcref/exceptionref values are not
80    /// tied to a context and can be freely shared between contexts.
81    ///
82    /// exceptionref and funcref values are tied to a context and can only be used
83    /// with that context.
84    pub fn is_from_store(&self, store: &impl AsStoreRef) -> bool {
85        self.handle.store_id() == store.as_store_ref().objects().id()
86    }
87}
