19#[cfg_attr(feature = "artifact-size", derive(loupe::MemoryUsage))]
20/// A WebAssembly `module` in the `sys` runtime.
21pub struct Module {
22    // The field ordering here is actually significant because of the drop
23    // order: we want to drop the artifact before dropping the engine.
24    //
25    // The reason for this is that dropping the Artifact will de-register the
26    // trap handling metadata from the global registry. This must be done before
27    // the code memory for the artifact is freed (which happens when the store
28    // is dropped) since there is a chance that this memory could be reused by
29    // another module which will try to register its own trap information.
30    //
31    // Note that in Rust, the drop order for struct fields is from top to
32    // bottom: the opposite of C++.
33    //
34    // In the future, this code should be refactored to properly describe the
35    // ownership of the code and its metadata.
36    artifact: Arc<Artifact>,
37}

39impl Module {
40    pub(crate) fn from_binary(
41        engine: &impl AsEngineRef,
42        binary: &[u8],
43    ) -> Result<Self, CompileError> {
44        Self::validate(engine, binary)?;
45        unsafe { Self::from_binary_unchecked(engine, binary) }
46    }

48    pub(crate) unsafe fn from_binary_unchecked(
49        engine: &impl AsEngineRef,

93        let artifact = unsafe {

95                .as_engine_ref()

151    #[allow(clippy::result_large_err)]
152    pub(crate) fn instantiate(
153        &self,
154        store: &mut impl AsStoreMut,
155        imports: &[crate::Extern],
156    ) -> Result<VMInstance, InstantiationError> {
157        if !self.artifact.allocated() {
158            // Return an error mentioning that the artifact is compiled for a different
159            // platform.
160            return Err(InstantiationError::DifferentArchOS);
161        }
162        // Ensure all imports come from the same context.
163        for import in imports {
164            if !import.is_from_store(store) {
165                return Err(InstantiationError::DifferentStores);
166            }
167        }
168        let signal_handler = store.as_store_ref().signal_handler();
169        let mut store_mut = store.as_store_mut();
170        let store_ptr = store_mut.inner as *mut _;
171        let (engine, objects) = store_mut.engine_and_objects_mut();
172        let config = engine.tunables().vmconfig();
173        unsafe {
174            let mut instance_handle = self.artifact.instantiate(
175                engine.tunables(),
176                &imports
177                    .iter()
178                    .map(|e| crate::Extern::to_vm_extern(e).into_sys())
179                    .collect::<Vec<_>>(),
180                objects.as_sys_mut(),
181            )?;
182
183            let store_id = objects.id();
184            let store_install_guard = StoreContext::ensure_installed(store_ptr);
185
186            // After the instance handle is created, we need to initialize
187            // the data, call the start function and so. However, if any
188            // of this steps traps, we still need to keep the instance alive
189            // as some of the Instance elements may have placed in other
190            // instance tables.
191            self.artifact
192                .finish_instantiation(config, signal_handler, &mut instance_handle)?;
193
194            drop(store_install_guard);
195
196            Ok(VMInstance::Sys(instance_handle))
197        }
198    }

229impl crate::Module {
230    /// Consume [`self`] into a reference [`crate::backend::sys::module::Module`].
231    pub fn into_sys(self) -> crate::backend::sys::module::Module {
232        match self.0 {
233            BackendModule::Sys(s) => s,
234            _ => panic!("Not a `sys` module!"),
235        }
236    }
237
238    /// Convert a reference to [`self`] into a reference [`crate::backend::sys::module::Module`].
239    pub fn as_sys(&self) -> &crate::backend::sys::module::Module {
240        match self.0 {
241            BackendModule::Sys(ref s) => s,
242            _ => panic!("Not a `sys` module!"),
243        }
244    }
245
246    /// Convert a mutable reference to [`self`] into a mutable reference [`crate::backend::sys::module::Module`].
247    pub fn as_sys_mut(&mut self) -> &mut crate::backend::sys::module::Module {
248        match self.0 {
249            BackendModule::Sys(ref mut s) => s,
250            _ => panic!("Not a `sys` module!"),
251        }
252    }
