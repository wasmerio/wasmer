69    impl VMTinyMemory {
70        pub fn new() -> Result<Self, MemoryError> {
71            let sz = 18 * WASM_PAGE_SIZE;
72            let memory = vec![0; sz];
73            let mut ret = Self {
74                mem: memory,
75                memory_definition: None,
76            };
77            ret.memory_definition = Some(UnsafeCell::new(VMMemoryDefinition {
78                base: ret.mem.as_ptr() as _,
79                current_length: sz,
80            }));
81            Ok(ret)
82        }
83    }

85    impl LinearMemory for VMTinyMemory {
86        fn ty(&self) -> MemoryType {
87            MemoryType {
88                minimum: Pages::from(18u32),
89                maximum: Some(Pages::from(18u32)),
90                shared: false,
91            }
92        }
93        fn size(&self) -> Pages {
94            Pages::from(18u32)
95        }
96        fn style(&self) -> MemoryStyle {
97            MemoryStyle::Static {
98                bound: Pages::from(18u32),
99                offset_guard_size: 0,
100            }
101        }
102        fn grow(&mut self, delta: Pages) -> Result<Pages, MemoryError> {
103            Err(MemoryError::CouldNotGrow {
104                current: Pages::from(100u32),
105                attempted_delta: delta,
106            })
107        }
108
109        fn grow_at_least(&mut self, min_size: u64) -> Result<(), MemoryError> {
110            let cur_size = self.size().0 as u64 * WASM_PAGE_SIZE as u64;
111            if min_size > cur_size {
112                let delta = min_size - cur_size;
113                return Err(MemoryError::CouldNotGrow {
114                    current: Pages::from(100u32),
115                    attempted_delta: Pages(delta as u32),
116                });
117            }
118            Ok(())
119        }
120
121        fn reset(&mut self) -> Result<(), MemoryError> {
122            self.mem.fill(0);
123            Ok(())
124        }
125
126        fn vmmemory(&self) -> NonNull<VMMemoryDefinition> {
127            unsafe {
128                NonNull::new(
129                    self.memory_definition
130                        .as_ref()
131                        .unwrap()
132                        .get()
133                        .as_mut()
134                        .unwrap() as _,
135                )
136                .unwrap()
137            }
138        }
139
140        fn try_clone(&self) -> Result<Box<dyn LinearMemory + 'static>, MemoryError> {
141            Err(MemoryError::InvalidMemory {
142                reason: "VMTinyMemory can not be cloned".to_string(),
143            })
144        }
145
146        fn copy(&mut self) -> Result<Box<dyn LinearMemory + 'static>, MemoryError> {
147            let mem = self.mem.clone();
148            Ok(Box::new(Self {
149                memory_definition: Some(UnsafeCell::new(VMMemoryDefinition {
150                    base: mem.as_ptr() as _,
151                    current_length: mem.len(),
152                })),
153                mem,
154            }))
155        }
156        /*

159        unsafe fn initialize_with_data(&self, _start: usize, _data: &[u8]) -> Result<(), Trap> {
160            Ok(())
161        }
162        */
163    }

171    struct TinyTunables;
172    impl Tunables for TinyTunables {
173        fn memory_style(&self, _memory: &MemoryType) -> MemoryStyle {
174            MemoryStyle::Static {
175                bound: Pages::from(18u32),
176                offset_guard_size: 0,
177            }
178        }
179
180        /// Construct a `TableStyle` for the provided `TableType`
181        fn table_style(&self, _table: &TableType) -> TableStyle {
182            TableStyle::CallerChecksSignature
183        }
184        fn create_host_memory(
185            &self,
186            _ty: &MemoryType,
187            _style: &MemoryStyle,
188        ) -> Result<VMMemory, MemoryError> {
189            let memory = VMTinyMemory::new().unwrap();
190            Ok(VMMemory::from_custom(memory))
191        }
192        unsafe fn create_vm_memory(
193            &self,
194            _ty: &MemoryType,
195            _style: &MemoryStyle,
196            vm_definition_location: NonNull<VMMemoryDefinition>,
197        ) -> Result<VMMemory, MemoryError> {
198            let memory = VMTinyMemory::new().unwrap();
199            // now, it's important to update vm_definition_location with the memory information!
200            let mut ptr = vm_definition_location;
201            let md = unsafe { ptr.as_mut() };
202            let unsafecell = memory.memory_definition.as_ref().unwrap();
203            let def = unsafe { unsafecell.get().as_ref().unwrap() };
204            md.base = def.base;
205            md.current_length = def.current_length;
206            Ok(memory.into())
207        }
208
209        /// Create a table owned by the host given a [`TableType`] and a [`TableStyle`].
210        fn create_host_table(&self, ty: &TableType, style: &TableStyle) -> Result<VMTable, String> {
211            VMTable::new(ty, style)
212        }
213
214        /// Create a table owned by the VM given a [`TableType`] and a [`TableStyle`].
215        ///
216        /// # Safety
217        /// - `vm_definition_location` must point to a valid location in VM memory.
218        unsafe fn create_vm_table(
219            &self,
220            ty: &TableType,
221            style: &TableStyle,
222            vm_definition_location: NonNull<VMTableDefinition>,
223        ) -> Result<VMTable, String> {
224            unsafe { VMTable::from_definition(ty, style, vm_definition_location) }
225        }
226
227        // Will use a minimum stack size of 8kb, not the 1Mb default
228        fn vmconfig(&self) -> &wasmer_vm::VMConfig {
229            &VMConfig {
230                wasm_stack_size: Some(8 * 1024),
231            }
232        }
233    }
