136/// Automatically create a match statement, repeating the expression for each backend.
137#[macro_use]
138macro_rules! match_rt {
139    (on $self:expr => $var:ident { $stmt:expr }) => {
140        match $self {
141            #[cfg(feature = "sys")]
142            Self::Sys($var) => $stmt,
143            #[cfg(feature = "wamr")]
144            Self::Wamr($var) => $stmt,
145            #[cfg(feature = "wasmi")]
146            Self::Wasmi($var) => $stmt,
147            #[cfg(feature = "v8")]
148            Self::V8($var) => $stmt,
149            #[cfg(feature = "js")]
150            Self::Js($var) => $stmt,
151            #[cfg(feature = "jsc")]
152            Self::Jsc($var) => $stmt,
153        }
154    };
155
156    (on $value:expr ; $match:expr => $var:ident { $stmt:expr }) => {
157        match $self {
158            #[cfg(feature = "sys")]
159            Self::Sys($var) => $stmt,
160            #[cfg(feature = "wamr")]
161            Self::Wamr($var) => $stmt,
162            #[cfg(feature = "wasmi")]
163            Self::Wasmi($var) => $stmt,
164            #[cfg(feature = "v8")]
165            Self::V8($var) => $stmt,
166            #[cfg(feature = "js")]
167            Self::Js($var) => $stmt,
168            #[cfg(feature = "jsc")]
169            Self::Jsc($var) => $stmt,
170        }
171    };
172}
