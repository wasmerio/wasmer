1//! Native Functions.
2//!
3//! This module creates the helper [`TypedFunction`] that let us call WebAssembly
4//! functions with the native ABI, that is:
5//!
6//! ```ignore
7//! let add_one = instance.exports.get_function("function_name")?;
8//! let add_one_native: TypedFunction<i32, i32> = add_one.native().unwrap();
9//! ```
10use crate::{
11    AsStoreMut, BackendStore, FromToNativeWasmType, Function, NativeWasmTypeInto, RuntimeError,
12    WasmTypeList, store::AsStoreRef,
13};
14use std::marker::PhantomData;
15use wasmer_types::RawValue;
16
17/// A WebAssembly function that can be called natively
18/// (using the Native ABI).
19#[derive(Clone, Debug)]
20pub struct TypedFunction<Args, Rets> {
21    pub(crate) func: Function,
22    _phantom: PhantomData<fn(Args) -> Rets>,
23}
24
25unsafe impl<Args, Rets> Send for TypedFunction<Args, Rets> {}
26unsafe impl<Args, Rets> Sync for TypedFunction<Args, Rets> {}
27
28impl<Args, Rets> TypedFunction<Args, Rets>
29where
30    Args: WasmTypeList,
31    Rets: WasmTypeList,
32{
33    #[allow(dead_code)]
34    pub(crate) fn new(_store: &impl AsStoreRef, func: Function) -> Self {
35        Self {
36            func,
37            _phantom: PhantomData,
38        }
39    }
40
41    pub(crate) fn into_function(self) -> Function {
42        self.func
43    }
44}
45
46macro_rules! impl_native_traits {
47    (  $( $x:ident ),* ) => {
48        paste::paste!{
49        #[allow(unused_parens, non_snake_case)]
50        impl<$( $x , )* Rets> TypedFunction<( $( $x ),* ), Rets>
51        where
52            $( $x: FromToNativeWasmType, )*
53            Rets: WasmTypeList,
54        {
55            /// Call the typed func and return results.
56            #[allow(unused_mut)]
57            #[allow(clippy::too_many_arguments)]
58            pub fn call(&self, store: &mut impl AsStoreMut, $( $x: $x, )* ) -> Result<Rets, RuntimeError> where $( $x: FromToNativeWasmType, )*
59
60            {
61                $(
62                    let [<p_ $x>] = $x;
63                )*
64                match store.as_store_mut().inner.store {
65                    #[cfg(feature = "sys")]
66                    BackendStore::Sys(_) => self.call_sys(store, $([<p_ $x>]),*),
67                    #[cfg(feature = "wamr")]
68                    BackendStore::Wamr(_) => self.call_wamr(store, $([<p_ $x>]),*),
69                    #[cfg(feature = "wasmi")]
70                    BackendStore::Wasmi(_) => self.call_wasmi(store, $([<p_ $x>]),*),
71                    #[cfg(feature = "v8")]
72                    BackendStore::V8(_) => self.call_v8(store, $([<p_ $x>]),*),
73                    #[cfg(feature = "js")]
74                    BackendStore::Js(_) => self.call_js(store, $([<p_ $x>]),*),
75                    #[cfg(feature = "jsc")]
76                    BackendStore::Jsc(_) => self.call_jsc(store, $([<p_ $x>]),*),
77
78                }
79            }
80
81            #[doc(hidden)]
82            #[allow(missing_docs)]
83            #[allow(unused_mut)]
84            #[allow(clippy::too_many_arguments)]
85            pub fn call_raw(&self, store: &mut impl AsStoreMut, mut params_list: Vec<RawValue> ) -> Result<Rets, RuntimeError> {
86                match store.as_store_mut().inner.store {
87                    #[cfg(feature = "sys")]
88                    BackendStore::Sys(_) => self.call_raw_sys(store, params_list),
89                    #[cfg(feature = "wamr")]
90                    BackendStore::Wamr(_) => self.call_raw_wamr(store, params_list),
91                    #[cfg(feature = "wasmi")]
92                    BackendStore::Wasmi(_) => self.call_raw_wasmi(store, params_list),
93                    #[cfg(feature = "v8")]
94                    BackendStore::V8(_) => self.call_raw_v8(store, params_list),
95                    #[cfg(feature = "js")]
96                    BackendStore::Js(_) => self.call_raw_js(store, params_list),
97                    #[cfg(feature = "jsc")]
98                    BackendStore::Jsc(_) => self.call_raw_jsc(store, params_list),
99                }
100            }
101        }
102        }
103    };
104}
