37// TrapInformation can be stored in the "Undefined Instruction" itself.
38// On x86_64, 0xC? select a "Register" for the Mod R/M part of "ud1" (so with no other bytes after)
39// On Arm64, the udf alows for a 16bits values, so we'll use the same 0xC? to store the trapinfo
40static MAGIC: u8 = 0xc0;

65/// Sets the process-wide default stack size for new Wasmer coroutines.
66/// The value is clamped to [8 KiB, MAX_STACK_SIZE].
67pub fn set_stack_size(size: usize) {
68    DEFAULT_STACK_SIZE.store(
69        size.clamp(ByteSize::kib(8).as_u64() as usize, MAX_STACK_SIZE),
70        Ordering::Relaxed,
71    );
72}
73
74/// Returns the process-wide default stack size in bytes.
75pub fn get_stack_size() -> usize {
76    DEFAULT_STACK_SIZE.load(Ordering::Relaxed)
77}

79/// Pool of pre-allocated coroutine stacks to avoid repeated mmap syscalls.
80static STACK_POOL: LazyLock<crossbeam_queue::SegQueue<DefaultStack>> =
81    LazyLock::new(crossbeam_queue::SegQueue::new);
82
83/// Drains the coroutine stack pool at the moment it runs.
84///
85/// This is intended to be called before retrying with a larger stack size so
86/// that the pool does not keep serving cached undersized stacks.
87///
88/// Note that `STACK_POOL` is a global, concurrently used queue. Other threads
89/// may push stacks back into the pool (for example, when their Wasm execution
90/// finishes) while or after this function is running. As a result, this
91/// function provides only a best-effort drain of the pool: there is no
92/// guarantee that no undersized stacks exist immediately after it returns
93/// unless the caller ensures, via external synchronization, that no other
94/// Wasm executions can return stacks to the pool while this function runs.
95pub fn drain_stack_pool() {
96    while STACK_POOL.pop().is_some() {}
97}

109// Process an IllegalOpcode to see if it has a TrapCode payload
110unsafe fn process_illegal_op(addr: usize) -> Option<TrapCode> {
111    let mut val: Option<u8> = None;
112    unsafe {
113        if cfg!(target_arch = "x86_64") {
114            val = if read(addr as *mut u8) & 0xf0 == 0x40
115                && read((addr + 1) as *mut u8) == 0x0f
116                && read((addr + 2) as *mut u8) == 0xb9
117            {
118                Some(read((addr + 3) as *mut u8))
119            } else if read(addr as *mut u8) == 0x0f && read((addr + 1) as *mut u8) == 0xb9 {
120                Some(read((addr + 2) as *mut u8))
121            } else {
122                None
123            }
124        }
125        if cfg!(target_arch = "aarch64") {
126            val = if read_unaligned(addr as *mut u32) & 0xffff0000 == 0 {
127                Some(read(addr as *mut u8))
128            } else {
129                None
130            }
131        }
132        if cfg!(target_arch = "riscv64") {
133            let addr = addr as *mut u32;
134            // Check if 'unimp' instruction
135            val = if read(addr) == 0xc0001073 {
136                // Read from the instruction we emitted: 'addi a0, xzero, $payload'
137                // and take the encoded immediate value (upper 12-bits).
138                let prev_insn = read(addr.sub(1));
139                if (prev_insn & 0xffff) == 0x0513 {
140                    Some((prev_insn >> 20) as u8)
141                } else {
142                    None
143                }
144            } else {
145                None
146            };
147        }
148    }
149
150    // The direct encoding of a trap into the instruction is unused on RISC-V:
151    if cfg!(target_arch = "x86_64") || cfg!(target_arch = "aarch64") {
152        val = val.and_then(|val| {
153            if val & MAGIC == MAGIC {
154                Some(val & 0xf)
155            } else {
156                None
157            }
158        });
159    }
160
161    match val {
162        None => None,
163        Some(val) => match val {
164            0 => Some(TrapCode::StackOverflow),
165            1 => Some(TrapCode::HeapAccessOutOfBounds),
166            2 => Some(TrapCode::HeapMisaligned),
167            3 => Some(TrapCode::TableAccessOutOfBounds),
168            4 => Some(TrapCode::IndirectCallToNull),
169            5 => Some(TrapCode::BadSignature),
170            6 => Some(TrapCode::IntegerOverflow),
171            7 => Some(TrapCode::IntegerDivisionByZero),
172            8 => Some(TrapCode::BadConversionToInteger),
173            9 => Some(TrapCode::UnreachableCodeReached),
174            10 => Some(TrapCode::UnalignedAtomic),
175            _ => None,
176        },
177    }
178}

187        unsafe fn platform_init() { unsafe {
188            let register = |slot: &mut MaybeUninit<libc::sigaction>, signal: i32| {
189                let mut handler: libc::sigaction = mem::zeroed();
190                // The flags here are relatively careful, and they are...
191                //
192                // SA_SIGINFO gives us access to information like the program
193                // counter from where the fault happened.
194                //
195                // SA_ONSTACK allows us to handle signals on an alternate stack,
196                // so that the handler can run in response to running out of
197                // stack space on the main stack. Rust installs an alternate
198                // stack with sigaltstack, so we rely on that.
199                //
200                // SA_NODEFER allows us to reenter the signal handler if we
201                // crash while handling the signal, and fall through to the
202                // Breakpad handler by testing handlingSegFault.
203                handler.sa_flags = libc::SA_SIGINFO | libc::SA_NODEFER | libc::SA_ONSTACK;
204                handler.sa_sigaction = trap_handler as *const () as usize;
205                libc::sigemptyset(&mut handler.sa_mask);
206                if libc::sigaction(signal, &handler, slot.as_mut_ptr()) != 0 {
207                    panic!(
208                        "unable to install signal handler: {}",
209                        io::Error::last_os_error(),
210                    );
211                }
212            };
213
214            // Allow handling OOB with signals on all architectures
215            register(&mut PREV_SIGSEGV, libc::SIGSEGV);
216
217            // Handle `unreachable` instructions which execute `ud2` right now
218            register(&mut PREV_SIGILL, libc::SIGILL);
219
220            // x86 uses SIGFPE to report division by zero
221            if cfg!(target_arch = "x86") || cfg!(target_arch = "x86_64") {
222                register(&mut PREV_SIGFPE, libc::SIGFPE);
223            }
224
225            // On ARM, handle Unaligned Accesses.
226            // On Darwin, guard page accesses are raised as SIGBUS.
227            if cfg!(target_arch = "arm") || cfg!(target_vendor = "apple") {
228                register(&mut PREV_SIGBUS, libc::SIGBUS);
229            }
230
231            // This is necessary to support debugging under LLDB on Darwin.
232            // For more details see https://github.com/mono/mono/commit/8e75f5a28e6537e56ad70bf870b86e22539c2fb7
233            #[cfg(target_vendor = "apple")]
234            {
235                use mach2::exception_types::*;
236                use mach2::kern_return::*;
237                use mach2::port::*;
238                use mach2::thread_status::*;
239                use mach2::traps::*;
240                use mach2::mach_types::*;
241
242                unsafe extern "C" {
243                    fn task_set_exception_ports(
244                        task: task_t,
245                        exception_mask: exception_mask_t,
246                        new_port: mach_port_t,
247                        behavior: exception_behavior_t,
248                        new_flavor: thread_state_flavor_t,
249                    ) -> kern_return_t;
250                }
251
252                #[allow(non_snake_case)]
253                #[cfg(target_arch = "x86_64")]
254                let MACHINE_THREAD_STATE = x86_THREAD_STATE64;
255                #[allow(non_snake_case)]
256                #[cfg(target_arch = "aarch64")]
257                let MACHINE_THREAD_STATE = 6;
258
259                task_set_exception_ports(
260                    mach_task_self(),
261                    EXC_MASK_BAD_ACCESS | EXC_MASK_ARITHMETIC | EXC_MASK_BAD_INSTRUCTION,
262                    MACH_PORT_NULL,
263                    EXCEPTION_STATE_IDENTITY as exception_behavior_t,
264                    MACHINE_THREAD_STATE,
265                );
266            }
267        }}

269        unsafe extern "C" fn trap_handler(
270            signum: libc::c_int,
271            siginfo: *mut libc::siginfo_t,
272            context: *mut libc::c_void,
273        ) { unsafe {
274            let previous = match signum {
275                libc::SIGSEGV => &PREV_SIGSEGV,
276                libc::SIGBUS => &PREV_SIGBUS,
277                libc::SIGFPE => &PREV_SIGFPE,
278                libc::SIGILL => &PREV_SIGILL,
279                _ => panic!("unknown signal: {signum}"),
280            };
281            // We try to get the fault address associated to this signal
282            let maybe_fault_address = match signum {
283                libc::SIGSEGV | libc::SIGBUS => {
284                    Some((*siginfo).si_addr() as usize)
285                }
286                _ => None,
287            };
288            let trap_code = match signum {
289                // check if it was cased by a UD and if the Trap info is a payload to it
290                libc::SIGILL => {
291                    let addr = (*siginfo).si_addr() as usize;
292                    process_illegal_op(addr)
293                }
294                _ => None,
295            };
296            let ucontext = &mut *(context as *mut ucontext_t);
297            let (pc, sp) = get_pc_sp(ucontext);
298            let handled = TrapHandlerContext::handle_trap(
299                pc,
300                sp,
301                maybe_fault_address,
302                trap_code,
303                |regs| update_context(ucontext, regs),
304                |handler| handler(signum, siginfo, context),
305            );
306
307            if handled {
308                return;
309            }
310
311            // This signal is not for any compiled wasm code we expect, so we
312            // need to forward the signal to the next handler. If there is no
313            // next handler (SIG_IGN or SIG_DFL), then it's time to crash. To do
314            // this, we set the signal back to its original disposition and
315            // return. This will cause the faulting op to be re-executed which
316            // will crash in the normal way. If there is a next handler, call
317            // it. It will either crash synchronously, fix up the instruction
318            // so that execution can continue and return, or trigger a crash by
319            // returning the signal to it's original disposition and returning.
320            let previous = &*previous.as_ptr();
321            if previous.sa_flags & libc::SA_SIGINFO != 0 {
322                mem::transmute::<
323                    usize,
324                    extern "C" fn(libc::c_int, *mut libc::siginfo_t, *mut libc::c_void),
325                >(previous.sa_sigaction)(signum, siginfo, context)
326            } else if previous.sa_sigaction == libc::SIG_DFL
327            {
328                libc::sigaction(signum, previous, ptr::null_mut());
329            } else if previous.sa_sigaction != libc::SIG_IGN {
330                mem::transmute::<usize, extern "C" fn(libc::c_int)>(
331                    previous.sa_sigaction
332                )(signum)
333            }
334        }}

336        unsafe fn get_pc_sp(context: &ucontext_t) -> (usize, usize) {
337            let (pc, sp);
338            cfg_if::cfg_if! {
339                if #[cfg(all(
340                    any(target_os = "linux", target_os = "android"),
341                    target_arch = "x86_64",
342                ))] {
343                    pc = context.uc_mcontext.gregs[libc::REG_RIP as usize] as usize;
344                    sp = context.uc_mcontext.gregs[libc::REG_RSP as usize] as usize;
345                } else if #[cfg(all(
346                    any(target_os = "linux", target_os = "android"),
347                    target_arch = "x86",
348                ))] {
349                    pc = context.uc_mcontext.gregs[libc::REG_EIP as usize] as usize;
350                    sp = context.uc_mcontext.gregs[libc::REG_ESP as usize] as usize;
351                } else if #[cfg(all(target_os = "freebsd", target_arch = "x86"))] {
352                    pc = context.uc_mcontext.mc_eip as usize;
353                    sp = context.uc_mcontext.mc_esp as usize;
354                } else if #[cfg(all(target_os = "freebsd", target_arch = "x86_64"))] {
355                    pc = context.uc_mcontext.mc_rip as usize;
356                    sp = context.uc_mcontext.mc_rsp as usize;
357                } else if #[cfg(all(target_vendor = "apple", target_arch = "x86_64"))] {
358                    let mcontext = unsafe { &*context.uc_mcontext };
359                    pc = mcontext.__ss.__rip as usize;
360                    sp = mcontext.__ss.__rsp as usize;
361                } else if #[cfg(all(
362                        any(target_os = "linux", target_os = "android"),
363                        target_arch = "aarch64",
364                    ))] {
365                    pc = context.uc_mcontext.pc as usize;
366                    sp = context.uc_mcontext.sp as usize;
367                } else if #[cfg(all(
368                    any(target_os = "linux", target_os = "android"),
369                    target_arch = "arm",
370                ))] {
371                    pc = context.uc_mcontext.arm_pc as usize;
372                    sp = context.uc_mcontext.arm_sp as usize;
373                } else if #[cfg(all(
374                    any(target_os = "linux", target_os = "android"),
375                    any(target_arch = "riscv64", target_arch = "riscv32"),
376                ))] {
377                    pc = context.uc_mcontext.__gregs[libc::REG_PC] as usize;
378                    sp = context.uc_mcontext.__gregs[libc::REG_SP] as usize;
379                } else if #[cfg(all(target_vendor = "apple", target_arch = "aarch64"))] {
380                    let mcontext = unsafe { &*context.uc_mcontext };
381                    pc = mcontext.__ss.__pc as usize;
382                    sp = mcontext.__ss.__sp as usize;
383                } else if #[cfg(all(target_os = "freebsd", target_arch = "aarch64"))] {
384                    pc = context.uc_mcontext.mc_gpregs.gp_elr as usize;
385                    sp = context.uc_mcontext.mc_gpregs.gp_sp as usize;
386                } else if #[cfg(all(target_os = "linux", target_arch = "loongarch64"))] {
387                    pc = context.uc_mcontext.__gregs[1] as usize;
388                    sp = context.uc_mcontext.__gregs[3] as usize;
389                } else if #[cfg(all(target_os = "linux", target_arch = "powerpc64"))] {
390                    pc = (*context.uc_mcontext.regs).nip as usize;
391                    sp = (*context.uc_mcontext.regs).gpr[1] as usize;
392                } else {
393                    compile_error!("Unsupported platform");
394                }
395            };
396            (pc, sp)
397        }

399        unsafe fn update_context(context: &mut ucontext_t, regs: TrapHandlerRegs) {
400            cfg_if::cfg_if! {
401                if #[cfg(all(
402                        any(target_os = "linux", target_os = "android"),
403                        target_arch = "x86_64",
404                    ))] {

407                    context.uc_mcontext.gregs[libc::REG_RSP as usize] = rsp as i64;

409                    context.uc_mcontext.gregs[libc::REG_RDI as usize] = rdi as i64;
410                    context.uc_mcontext.gregs[libc::REG_RSI as usize] = rsi as i64;

412                    any(target_os = "linux", target_os = "android"),
413                    target_arch = "x86",
414                ))] {
415                    let TrapHandlerRegs { eip, esp, ebp, ecx, edx } = regs;

417                    context.uc_mcontext.gregs[libc::REG_ESP as usize] = esp as i32;

421                } else if #[cfg(all(target_vendor = "apple", target_arch = "x86_64"))] {
422                    let TrapHandlerRegs { rip, rsp, rbp, rdi, rsi } = regs;
423                    let mcontext = unsafe { &mut *context.uc_mcontext };
424                    mcontext.__ss.__rip = rip;
425                    mcontext.__ss.__rsp = rsp;

433                    context.uc_mcontext.mc_ebp = ebp as libc::register_t;
434                    context.uc_mcontext.mc_ecx = ecx as libc::register_t;
435                    context.uc_mcontext.mc_edx = edx as libc::register_t;
436                } else if #[cfg(all(target_os = "freebsd", target_arch = "x86_64"))] {
437                    let TrapHandlerRegs { rip, rsp, rbp, rdi, rsi } = regs;
438                    context.uc_mcontext.mc_rip = rip as libc::register_t;
439                    context.uc_mcontext.mc_rsp = rsp as libc::register_t;
440                    context.uc_mcontext.mc_rbp = rbp as libc::register_t;
441                    context.uc_mcontext.mc_rdi = rdi as libc::register_t;
442                    context.uc_mcontext.mc_rsi = rsi as libc::register_t;
443                } else if #[cfg(all(
444                        any(target_os = "linux", target_os = "android"),
445                        target_arch = "aarch64",
446                    ))] {
447                    let TrapHandlerRegs { pc, sp, x0, x1, x29, lr } = regs;
448                    context.uc_mcontext.pc = pc;
449                    context.uc_mcontext.sp = sp;
450                    context.uc_mcontext.regs[0] = x0;
451                    context.uc_mcontext.regs[1] = x1;
452                    context.uc_mcontext.regs[29] = x29;
453                    context.uc_mcontext.regs[30] = lr;
454                } else if #[cfg(all(
455                        any(target_os = "linux", target_os = "android"),
456                        target_arch = "arm",
457                    ))] {
458                    let TrapHandlerRegs {
459                        pc,
460                        r0,
461                        r1,
462                        r7,
463                        r11,
464                        r13,
465                        r14,
466                        cpsr_thumb,
467                        cpsr_endian,
468                    } = regs;
469                    context.uc_mcontext.arm_pc = pc;
470                    context.uc_mcontext.arm_r0 = r0;
471                    context.uc_mcontext.arm_r1 = r1;
472                    context.uc_mcontext.arm_r7 = r7;
473                    context.uc_mcontext.arm_fp = r11;
474                    context.uc_mcontext.arm_sp = r13;
475                    context.uc_mcontext.arm_lr = r14;
476                    if cpsr_thumb {
477                        context.uc_mcontext.arm_cpsr |= 0x20;
478                    } else {
479                        context.uc_mcontext.arm_cpsr &= !0x20;
480                    }
481                    if cpsr_endian {
482                        context.uc_mcontext.arm_cpsr |= 0x200;
483                    } else {
484                        context.uc_mcontext.arm_cpsr &= !0x200;
485                    }
486                } else if #[cfg(all(
487                    any(target_os = "linux", target_os = "android"),
488                    any(target_arch = "riscv64", target_arch = "riscv32"),
489                ))] {
490                    let TrapHandlerRegs { pc, ra, sp, a0, a1, s0 } = regs;
491                    context.uc_mcontext.__gregs[libc::REG_PC] = pc as libc::c_ulong;
492                    context.uc_mcontext.__gregs[libc::REG_RA] = ra as libc::c_ulong;
493                    context.uc_mcontext.__gregs[libc::REG_SP] = sp as libc::c_ulong;
494                    context.uc_mcontext.__gregs[libc::REG_A0] = a0 as libc::c_ulong;
495                    context.uc_mcontext.__gregs[libc::REG_A0 + 1] = a1 as libc::c_ulong;
496                    context.uc_mcontext.__gregs[libc::REG_S0] = s0 as libc::c_ulong;
497                } else if #[cfg(all(target_vendor = "apple", target_arch = "aarch64"))] {
498                    let TrapHandlerRegs { pc, sp, x0, x1, x29, lr } = regs;
499                    let mcontext = unsafe { &mut *context.uc_mcontext };
500                    mcontext.__ss.__pc = pc;
501                    mcontext.__ss.__sp = sp;
502                    mcontext.__ss.__x[0] = x0;
503                    mcontext.__ss.__x[1] = x1;
504                    mcontext.__ss.__fp = x29;
505                    mcontext.__ss.__lr = lr;
506                } else if #[cfg(all(target_os = "freebsd", target_arch = "aarch64"))] {
507                    let TrapHandlerRegs { pc, sp, x0, x1, x29, lr } = regs;
508                    context.uc_mcontext.mc_gpregs.gp_elr = pc as libc::register_t;
509                    context.uc_mcontext.mc_gpregs.gp_sp = sp as libc::register_t;
510                    context.uc_mcontext.mc_gpregs.gp_x[0] = x0 as libc::register_t;
511                    context.uc_mcontext.mc_gpregs.gp_x[1] = x1 as libc::register_t;
512                    context.uc_mcontext.mc_gpregs.gp_x[29] = x29 as libc::register_t;
513                    context.uc_mcontext.mc_gpregs.gp_lr = lr as libc::register_t;
514                } else if #[cfg(all(target_os = "linux", target_arch = "loongarch64"))] {
515                    let TrapHandlerRegs { pc, sp, a0, a1, fp, ra } = regs;
516                    context.uc_mcontext.__pc = pc;
517                    context.uc_mcontext.__gregs[1] = ra;
518                    context.uc_mcontext.__gregs[3] = sp;
519                    context.uc_mcontext.__gregs[4] = a0;
520                    context.uc_mcontext.__gregs[5] = a1;
521                    context.uc_mcontext.__gregs[22] = fp;
522                } else if #[cfg(all(target_os = "linux", target_arch = "powerpc64"))] {
523                    let TrapHandlerRegs { pc, sp, r3, r4, r31, lr } = regs;
524                    (*context.uc_mcontext.regs).nip = pc;
525                    (*context.uc_mcontext.regs).gpr[1] = sp;
526                    (*context.uc_mcontext.regs).gpr[3] = r3;
527                    (*context.uc_mcontext.regs).gpr[4] = r4;
528                    (*context.uc_mcontext.regs).gpr[31] = r31;
529                    (*context.uc_mcontext.regs).link = lr;
530                } else {
531                    compile_error!("Unsupported platform");
532                }
533            };
534        }
535    } else if #[cfg(target_os = "windows")] {
536        use windows_sys::Win32::System::Diagnostics::Debug::{
537            AddVectoredExceptionHandler,
538            CONTEXT,
539            EXCEPTION_CONTINUE_EXECUTION,
540            EXCEPTION_CONTINUE_SEARCH,
541            EXCEPTION_POINTERS,
542        };
543        use windows_sys::Win32::Foundation::{
544            EXCEPTION_ACCESS_VIOLATION,
545            EXCEPTION_ILLEGAL_INSTRUCTION,
546            EXCEPTION_INT_DIVIDE_BY_ZERO,
547            EXCEPTION_INT_OVERFLOW,
548            EXCEPTION_STACK_OVERFLOW,
549        };
550
551        unsafe fn platform_init() {
552            unsafe {
553                // our trap handler needs to go first, so that we can recover from
554                // wasm faults and continue execution, so pass `1` as a true value
555                // here.
556                let handler = AddVectoredExceptionHandler(1, Some(exception_handler));
557                if handler.is_null() {
558                    panic!("failed to add exception handler: {}", io::Error::last_os_error());
559                }
560            }
561        }
562
563        unsafe extern "system" fn exception_handler(
564            exception_info: *mut EXCEPTION_POINTERS
565        ) -> i32 {
566            unsafe {
567                // Check the kind of exception, since we only handle a subset within
568                // wasm code. If anything else happens we want to defer to whatever
569                // the rest of the system wants to do for this exception.
570                let record = &*(*exception_info).ExceptionRecord;
571                if record.ExceptionCode != EXCEPTION_ACCESS_VIOLATION &&
572                    record.ExceptionCode != EXCEPTION_ILLEGAL_INSTRUCTION &&
573                    record.ExceptionCode != EXCEPTION_STACK_OVERFLOW &&
574                    record.ExceptionCode != EXCEPTION_INT_DIVIDE_BY_ZERO &&
575                    record.ExceptionCode != EXCEPTION_INT_OVERFLOW
576                {
577                    return EXCEPTION_CONTINUE_SEARCH;
578                }
579
580                // FIXME: this is what the previous C++ did to make sure that TLS
581                // works by the time we execute this trap handling code. This isn't
582                // exactly super easy to call from Rust though and it's not clear we
583                // necessarily need to do so. Leaving this here in case we need this
584                // in the future, but for now we can probably wait until we see a
585                // strange fault before figuring out how to reimplement this in
586                // Rust.
587                //
588                // if (!NtCurrentTeb()->Reserved1[sThreadLocalArrayPointerIndex]) {
589                //     return EXCEPTION_CONTINUE_SEARCH;
590                // }
591
592                let context = &mut *(*exception_info).ContextRecord;
593                let (pc, sp) = get_pc_sp(context);
594
595                // We try to get the fault address associated to this exception.
596                let maybe_fault_address = match record.ExceptionCode {
597                    EXCEPTION_ACCESS_VIOLATION => Some(record.ExceptionInformation[1]),
598                    EXCEPTION_STACK_OVERFLOW => Some(sp),
599                    _ => None,
600                };
601                let trap_code = match record.ExceptionCode {
602                    // check if it was cased by a UD and if the Trap info is a payload to it
603                    EXCEPTION_ILLEGAL_INSTRUCTION => {
604                        process_illegal_op(pc)
605                    }
606                    _ => None,
607                };
608                // This is basically the same as the unix version above, only with a
609                // few parameters tweaked here and there.
610                let handled = TrapHandlerContext::handle_trap(
611                    pc,
612                    sp,
613                    maybe_fault_address,
614                    trap_code,
615                    |regs| update_context(context, regs),
616                    |handler| handler(exception_info),
617                );
618
619                if handled {
620                    EXCEPTION_CONTINUE_EXECUTION
621                } else {
622                    EXCEPTION_CONTINUE_SEARCH
623                }
624            }
625        }
626
627        unsafe fn get_pc_sp(context: &CONTEXT) -> (usize, usize) {
628            let (pc, sp);
629            cfg_if::cfg_if! {
630                if #[cfg(target_arch = "x86_64")] {
631                    pc = context.Rip as usize;
632                    sp = context.Rsp as usize;
633                } else if #[cfg(target_arch = "x86")] {
634                    pc = context.Eip as usize;
635                    sp = context.Esp as usize;
636                } else {
637                    compile_error!("Unsupported platform");
638                }
639            };
640            (pc, sp)
641        }
642
643        unsafe fn update_context(context: &mut CONTEXT, regs: TrapHandlerRegs) {
644            cfg_if::cfg_if! {
645                if #[cfg(target_arch = "x86_64")] {
646                    let TrapHandlerRegs { rip, rsp, rbp, rdi, rsi } = regs;
647                    context.Rip = rip;
648                    context.Rsp = rsp;
649                    context.Rbp = rbp;
650                    context.Rdi = rdi;
651                    context.Rsi = rsi;
652                } else if #[cfg(target_arch = "x86")] {
653                    let TrapHandlerRegs { eip, esp, ebp, ecx, edx } = regs;
654                    context.Eip = eip;
655                    context.Esp = esp;
656                    context.Ebp = ebp;
657                    context.Ecx = ecx;
658                    context.Edx = edx;
659                } else {
660                    compile_error!("Unsupported platform");
661                }
662            };
663        }
664    }

725/// Call the wasm function pointed to by `callee`.
726///
727/// * `vmctx` - the callee vmctx argument
728/// * `caller_vmctx` - the caller vmctx argument
729/// * `trampoline` - the jit-generated trampoline whose ABI takes 4 values, the
730///   callee vmctx, the caller vmctx, the `callee` argument below, and then the
731///   `values_vec` argument.
732/// * `callee` - the third argument to the `trampoline` function
733/// * `values_vec` - points to a buffer which holds the incoming arguments, and to
734///   which the outgoing return values will be written.
735///
736/// # Safety
737///
738/// Wildly unsafe because it calls raw function pointers and reads/writes raw
739/// function pointers.
740pub unsafe fn wasmer_call_trampoline(
741    trap_handler: Option<*const TrapHandlerFn<'static>>,
742    config: &VMConfig,
743    vmctx: VMFunctionContext,
744    trampoline: VMTrampoline,
745    callee: *const VMFunctionBody,
746    values_vec: *mut u8,
747) -> Result<(), Trap> {
748    unsafe {
749        catch_traps(trap_handler, config, move || {
750            mem::transmute::<
751                unsafe extern "C" fn(
752                    *mut VMContext,
753                    *const VMFunctionBody,
754                    *mut wasmer_types::RawValue,
755                ),
756                extern "C" fn(VMFunctionContext, *const VMFunctionBody, *mut u8),
757            >(trampoline)(vmctx, callee, values_vec);
758        })
759    }
760}

762/// Catches any wasm traps that happen within the execution of `closure`,
763/// returning them as a `Result`.
764///
765/// # Safety
766///
767/// Highly unsafe since `closure` won't have any dtors run.
768pub unsafe fn catch_traps<F, R: 'static>(
769    trap_handler: Option<*const TrapHandlerFn<'static>>,
770    config: &VMConfig,
771    closure: F,
772) -> Result<R, Trap>
773where
774    F: FnOnce() -> R + 'static,
775{
776    // Ensure that per-thread initialization is done.
777    lazy_per_thread_init()?;
778    let stack_size = config
779        .wasm_stack_size
780        .unwrap_or_else(|| DEFAULT_STACK_SIZE.load(Ordering::Relaxed));
781    on_wasm_stack(stack_size, trap_handler, closure).map_err(UnwindReason::into_trap)
782}

784// We need two separate thread-local variables here:
785// - YIELDER is set within the new stack and is used to unwind back to the root
786//   of the stack from inside it.
787// - TRAP_HANDLER is set from outside the new stack and is solely used from
788//   signal handlers. It must be atomic since it is used by signal handlers.
789//
790// We also do per-thread signal stack initialization on the first time
791// TRAP_HANDLER is accessed.
792thread_local! {
793    static YIELDER: Cell<Option<NonNull<Yielder<(), UnwindReason>>>> = const { Cell::new(None) };
794    static TRAP_HANDLER: AtomicPtr<TrapHandlerContext> = const { AtomicPtr::new(ptr::null_mut()) };
795}

797/// Read-only information that is used by signal handlers to handle and recover
798/// from traps.
799#[allow(clippy::type_complexity)]
800struct TrapHandlerContext {
801    inner: *const u8,
802    handle_trap: fn(
803        *const u8,
804        usize,
805        usize,
806        Option<usize>,
807        Option<TrapCode>,
808        &mut dyn FnMut(TrapHandlerRegs),
809    ) -> bool,
810    custom_trap: Option<*const TrapHandlerFn<'static>>,
811}
812struct TrapHandlerContextInner<T> {
813    /// Information about the currently running coroutine. This is used to
814    /// reset execution to the root of the coroutine when a trap is handled.
815    coro_trap_handler: CoroutineTrapHandler<Result<T, UnwindReason>>,
816}
817
818impl TrapHandlerContext {
819    /// Runs the given function with a trap handler context. The previous
820    /// trap handler context is preserved and restored afterwards.
821    fn install<T, R>(
822        custom_trap: Option<*const TrapHandlerFn<'static>>,
823        coro_trap_handler: CoroutineTrapHandler<Result<T, UnwindReason>>,
824        f: impl FnOnce() -> R,
825    ) -> R {
826        // Type-erase the trap handler function so that it can be placed in TLS.
827        fn func<T>(
828            ptr: *const u8,
829            pc: usize,
830            sp: usize,
831            maybe_fault_address: Option<usize>,
832            trap_code: Option<TrapCode>,
833            update_regs: &mut dyn FnMut(TrapHandlerRegs),
834        ) -> bool {
835            unsafe {
836                (*(ptr as *const TrapHandlerContextInner<T>)).handle_trap(
837                    pc,
838                    sp,
839                    maybe_fault_address,
840                    trap_code,
841                    update_regs,
842                )
843            }
844        }
845        let inner = TrapHandlerContextInner { coro_trap_handler };
846        let ctx = Self {
847            inner: &inner as *const _ as *const u8,
848            handle_trap: func::<T>,
849            custom_trap,
850        };
851
852        compiler_fence(Ordering::Release);
853        let prev = TRAP_HANDLER.with(|ptr| {
854            let prev = ptr.load(Ordering::Relaxed);
855            ptr.store(&ctx as *const Self as *mut Self, Ordering::Relaxed);
856            prev
857        });
858
859        defer! {
860            TRAP_HANDLER.with(|ptr| ptr.store(prev, Ordering::Relaxed));
861            compiler_fence(Ordering::Acquire);
862        }
863
864        f()
865    }
866
867    /// Attempts to handle the trap if it's a wasm trap.
868    unsafe fn handle_trap(
869        pc: usize,
870        sp: usize,
871        maybe_fault_address: Option<usize>,
872        trap_code: Option<TrapCode>,
873        mut update_regs: impl FnMut(TrapHandlerRegs),
874        call_handler: impl Fn(&TrapHandlerFn<'static>) -> bool,
875    ) -> bool {
876        unsafe {
877            let ptr = TRAP_HANDLER.with(|ptr| ptr.load(Ordering::Relaxed));
878            if ptr.is_null() {
879                return false;
880            }
881
882            let ctx = &*ptr;
883
884            // Check if this trap is handled by a custom trap handler.
885            if let Some(trap_handler) = ctx.custom_trap
886                && call_handler(&*trap_handler)
887            {
888                return true;
889            }
890
891            (ctx.handle_trap)(
892                ctx.inner,
893                pc,
894                sp,
895                maybe_fault_address,
896                trap_code,
897                &mut update_regs,
898            )
899        }
900    }
901}
902
903impl<T> TrapHandlerContextInner<T> {
904    unsafe fn handle_trap(
905        &self,
906        pc: usize,
907        sp: usize,
908        maybe_fault_address: Option<usize>,
909        trap_code: Option<TrapCode>,
910        update_regs: &mut dyn FnMut(TrapHandlerRegs),
911    ) -> bool {
912        unsafe {
913            // Check if this trap occurred while executing on the Wasm stack. We can
914            // only recover from traps if that is the case.
915            if !self.coro_trap_handler.stack_ptr_in_bounds(sp) {
916                return false;
917            }
918
919            let signal_trap = trap_code.or_else(|| {
920                maybe_fault_address.map(|addr| {
921                    if self.coro_trap_handler.stack_ptr_in_bounds(addr) {
922                        TrapCode::StackOverflow
923                    } else {
924                        TrapCode::HeapAccessOutOfBounds
925                    }
926                })
927            });
928
929            // Don't try to generate a backtrace for stack overflows: unwinding
930            // information is often not precise enough to properly describe what is
931            // happenning during a function prologue, which can lead the unwinder to
932            // read invalid memory addresses.
933            //
934            // See: https://github.com/rust-lang/backtrace-rs/pull/357
935            let backtrace = if signal_trap == Some(TrapCode::StackOverflow) {
936                Backtrace::from(vec![])
937            } else {
938                Backtrace::new_unresolved()
939            };
940
941            // Set up the register state for exception return to force the
942            // coroutine to return to its caller with UnwindReason::WasmTrap.
943            let unwind = UnwindReason::WasmTrap {
944                backtrace,
945                signal_trap,
946                pc,
947            };
948            let regs = self
949                .coro_trap_handler
950                .setup_trap_handler(move || Err(unwind));
951            update_regs(regs);
952            true
953        }
954    }
955}

957enum UnwindReason {
958    /// A panic caused by the host
959    Panic(Box<dyn Any + Send>),
960    /// A custom error triggered by the user
961    UserTrap(Box<dyn Error + Send + Sync>),
962    /// A Trap triggered by a wasm libcall
963    LibTrap(Trap),
964    /// A trap caused by the Wasm generated code
965    WasmTrap {
966        backtrace: Backtrace,
967        pc: usize,
968        signal_trap: Option<TrapCode>,
969    },
970}
971
972impl UnwindReason {
973    fn into_trap(self) -> Trap {
974        match self {
975            Self::UserTrap(data) => Trap::User(data),
976            Self::LibTrap(trap) => trap,
977            Self::WasmTrap {
978                backtrace,
979                pc,
980                signal_trap,
981            } => Trap::wasm(pc, backtrace, signal_trap),
982            Self::Panic(panic) => std::panic::resume_unwind(panic),
983        }
984    }
985}

987unsafe fn unwind_with(reason: UnwindReason) -> ! {
988    unsafe {
989        let yielder = YIELDER
990            .with(|cell| cell.replace(None))
991            .expect("not running on Wasm stack");
992
993        yielder.as_ref().suspend(reason);
994
995        // on_wasm_stack will forcibly reset the coroutine stack after yielding.
996        unreachable!();
997    }
998}

1000/// Runs the given function on a separate stack so that its stack usage can be
1001/// bounded. Stack overflows and other traps can be caught and execution
1002/// returned to the root of the stack.
1003fn on_wasm_stack<F: FnOnce() -> T + 'static, T: 'static>(
1004    stack_size: usize,
1005    trap_handler: Option<*const TrapHandlerFn<'static>>,
1006    f: F,
1007) -> Result<T, UnwindReason> {
1008    // Reuse a cached stack from the module-level STACK_POOL when available,
1009    // otherwise allocate a new one.
1010    // Note: the pool does not track stack sizes — a pooled stack may be
1011    // smaller than `stack_size` if the default was raised between calls.
1012    // Callers that need a guaranteed size must drain the pool first via
1013    // `drain_stack_pool()` after updating the default with `set_stack_size()`
1014    // and before relying on the new value returned by `get_stack_size()`.
1015    let stack = STACK_POOL
1016        .pop()
1017        .unwrap_or_else(|| DefaultStack::new(stack_size).unwrap());
1018    let mut stack = scopeguard::guard(stack, |stack| STACK_POOL.push(stack));
1019
1020    // Create a coroutine with a new stack to run the function on.
1021    let coro = ScopedCoroutine::with_stack(&mut *stack, move |yielder, ()| {
1022        // Save the yielder to TLS so that it can be used later.
1023        YIELDER.with(|cell| cell.set(Some(yielder.into())));
1024
1025        Ok(f())
1026    });
1027
1028    // Ensure that YIELDER is reset on exit even if the coroutine panics,
1029    defer! {
1030        YIELDER.with(|cell| cell.set(None));
1031    }
1032
1033    coro.scope(|mut coro_ref| {
1034        // Set up metadata for the trap handler for the duration of the coroutine
1035        // execution. This is restored to its previous value afterwards.
1036        TrapHandlerContext::install(trap_handler, coro_ref.trap_handler(), || {
1037            match coro_ref.resume(()) {
1038                CoroutineResult::Yield(trap) => {
1039                    // This came from unwind_with which requires that there be only
1040                    // Wasm code on the stack.
1041                    unsafe {
1042                        coro_ref.force_reset();
1043                    }
1044                    Err(trap)
1045                }
1046                CoroutineResult::Return(result) => result,
1047            }
1048        })
1049    })
1050}

1052/// When executing on the Wasm stack, temporarily switch back to the host stack
1053/// to perform an operation that should not be constrainted by the Wasm stack
1054/// limits.
1055///
1056/// This is particularly important since the usage of the Wasm stack is under
1057/// the control of untrusted code. Malicious code could artificially induce a
1058/// stack overflow in the middle of a sensitive host operations (e.g. growing
1059/// a memory) which would be hard to recover from.
1060pub fn on_host_stack<F: FnOnce() -> T, T>(f: F) -> T {
1061    // Reset YIEDER to None for the duration of this call to indicate that we
1062    // are no longer on the Wasm stack.
1063    let yielder_ptr = YIELDER.with(|cell| cell.replace(None));
1064
1065    // If we are already on the host stack, execute the function directly. This
1066    // happens if a host function is called directly from the API.
1067    let yielder = match yielder_ptr {
1068        Some(ptr) => unsafe { ptr.as_ref() },
1069        None => return f(),
1070    };
1071
1072    // Restore YIELDER upon exiting normally or unwinding.
1073    defer! {
1074        YIELDER.with(|cell| cell.set(yielder_ptr));
1075    }
1076
1077    // on_parent_stack requires the closure to be Send so that the Yielder
1078    // cannot be called from the parent stack. This is not a problem for us
1079    // since we don't expose the Yielder.
1080    struct SendWrapper<T>(T);
1081    unsafe impl<T> Send for SendWrapper<T> {}
1082    let wrapped = SendWrapper(f);
1083    yielder.on_parent_stack(move || {
1084        let wrapped = wrapped;
1085        (wrapped.0)()
1086    })
1087}

1090pub fn lazy_per_thread_init() -> Result<(), Trap> {
1091    // We need additional space on the stack to handle stack overflow
1092    // exceptions. Rust's initialization code sets this to 0x5000 but this
1093    // seems to be insufficient in practice.
1094    use windows_sys::Win32::System::Threading::SetThreadStackGuarantee;

1099    Ok(())
1100}

1102/// A module for registering a custom alternate signal stack (sigaltstack).
1103///
1104/// Rust's libstd installs an alternate stack with size `SIGSTKSZ`, which is not
1105/// always large enough for our signal handling code. Override it by creating
1106/// and registering our own alternate stack that is large enough and has a guard
1107/// page.
1108#[cfg(unix)]
1109pub fn lazy_per_thread_init() -> Result<(), Trap> {
1110    use std::ptr::null_mut;
1111
1112    thread_local! {
1113        /// Thread-local state is lazy-initialized on the first time it's used,
1114        /// and dropped when the thread exits.
1115        static TLS: Tls = unsafe { init_sigstack() };
1116    }
1117
1118    /// The size of the sigaltstack (not including the guard, which will be
1119    /// added). Make this large enough to run our signal handlers.
1120    const MIN_STACK_SIZE: usize = ByteSize::kib(64).as_u64() as usize;
1121
1122    enum Tls {
1123        OutOfMemory,
1124        Allocated {
1125            mmap_ptr: *mut libc::c_void,
1126            mmap_size: usize,
1127        },
1128        BigEnough,
1129    }
1130
1131    unsafe fn init_sigstack() -> Tls {
1132        unsafe {
1133            // Check to see if the existing sigaltstack, if it exists, is big
1134            // enough. If so we don't need to allocate our own.
1135            let mut old_stack = mem::zeroed();
1136            let r = libc::sigaltstack(ptr::null(), &mut old_stack);
1137            assert_eq!(r, 0, "learning about sigaltstack failed");
1138            if old_stack.ss_flags & libc::SS_DISABLE == 0 && old_stack.ss_size >= MIN_STACK_SIZE {
1139                return Tls::BigEnough;
1140            }
1141
1142            // ... but failing that we need to allocate our own, so do all that
1143            // here.
1144            let page_size: usize = region::page::size();
1145            let guard_size = page_size;
1146            let alloc_size = guard_size + MIN_STACK_SIZE;
1147
1148            let ptr = libc::mmap(
1149                null_mut(),
1150                alloc_size,
1151                libc::PROT_NONE,
1152                libc::MAP_PRIVATE | libc::MAP_ANON,
1153                -1,
1154                0,
1155            );
1156            if ptr == libc::MAP_FAILED {
1157                return Tls::OutOfMemory;
1158            }
1159
1160            // Prepare the stack with readable/writable memory and then register it
1161            // with `sigaltstack`.
1162            let stack_ptr = (ptr as usize + guard_size) as *mut libc::c_void;
1163            let r = libc::mprotect(
1164                stack_ptr,
1165                MIN_STACK_SIZE,
1166                libc::PROT_READ | libc::PROT_WRITE,
1167            );
1168            assert_eq!(r, 0, "mprotect to configure memory for sigaltstack failed");
1169            let new_stack = libc::stack_t {
1170                ss_sp: stack_ptr,
1171                ss_flags: 0,
1172                ss_size: MIN_STACK_SIZE,
1173            };
1174            let r = libc::sigaltstack(&new_stack, ptr::null_mut());
1175            assert_eq!(r, 0, "registering new sigaltstack failed");
1176
1177            Tls::Allocated {
1178                mmap_ptr: ptr,
1179                mmap_size: alloc_size,
1180            }
1181        }
1182    }
1183
1184    // Ensure TLS runs its initializer and return an error if it failed to
1185    // set up a separate stack for signal handlers.
1186    return TLS.with(|tls| {
1187        if let Tls::OutOfMemory = tls {
1188            Err(Trap::oom())
1189        } else {
1190            Ok(())
1191        }
1192    });
1193
1194    impl Drop for Tls {
1195        fn drop(&mut self) {
1196            let (ptr, size) = match self {
1197                Self::Allocated {
1198                    mmap_ptr,
1199                    mmap_size,
1200                } => (*mmap_ptr, *mmap_size),
1201                _ => return,
1202            };
1203            unsafe {
1204                // Deallocate the stack memory.
1205                let r = libc::munmap(ptr, size);
1206                debug_assert_eq!(r, 0, "munmap failed during thread shutdown");
1207            }
1208        }
1209    }
1210}

1278    /// The stack pool is not size-aware, so after a stack size increase it keeps
1279    /// serving cached undersized stacks. `drain_stack_pool()` breaks the cycle.
1280    ///
1281    /// 1. A call fills the pool with 500 KiB stacks (simulating normal execution).
1282    /// 2. The caller doubles the default to 1 MiB (simulating overflow retry).
1283    /// 3. WITHOUT draining, the pool still hands back a 500 KiB stack — the
1284    ///    retry would overflow again, creating an infinite loop.
1285    /// 4. After `drain_stack_pool()`, the pool is empty and the next allocation
1286    ///    must use the new, larger size.
1287    #[test]
1288    fn pool_returns_stale_stack_without_drain() {
1289        let _lock = GLOBAL_STATE.lock().unwrap();
1290        let _restore = RestoreStackSize(get_stack_size());
1291        drain_stack_pool();
1292
1293        // --- Phase 1: simulate normal execution that returns a 500 KiB stack ---
1294        let small_size = ByteSize::kib(500).as_u64() as usize;
1295        let small_stack = DefaultStack::new(small_size).unwrap();
1296        STACK_POOL.push(small_stack);
1297
1298        // --- Phase 2: "overflow detected" — caller doubles the default ---
1299        let big_size = ByteSize::mib(1).as_u64() as usize;
1300        set_stack_size(big_size);
1301        assert_eq!(get_stack_size(), big_size);
1302
1303        // --- Phase 3: WITHOUT drain, pool still returns the old small stack ---
1304        // This is the bug: the caller asked for a bigger stack but the pool
1305        // serves a cached undersized one, causing the retry to overflow again.
1306        let stale = STACK_POOL.pop();
1307        assert!(
1308            stale.is_some(),
1309            "pool should still contain the old stack (the bug scenario)"
1310        );
1311
1312        // --- Phase 4: with drain, pool is empty — next alloc uses new size ---
1313        STACK_POOL.push(stale.unwrap());
1314        drain_stack_pool();
1315        assert!(
1316            STACK_POOL.pop().is_none(),
1317            "after drain, pool must be empty so a fresh stack is allocated at the new size"
1318        );
1319    }
