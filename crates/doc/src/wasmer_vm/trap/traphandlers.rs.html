75// Process an IllegalOpcode to see if it has a TrapCode payload
76unsafe fn process_illegal_op(addr: usize) -> Option<TrapCode> {
77    unsafe {
78        let mut val: Option<u8> = None;
79        if cfg!(target_arch = "x86_64") {
80            val = if read(addr as *mut u8) & 0xf0 == 0x40
81                && read((addr + 1) as *mut u8) == 0x0f
82                && read((addr + 2) as *mut u8) == 0xb9
83            {
84                Some(read((addr + 3) as *mut u8))
85            } else if read(addr as *mut u8) == 0x0f && read((addr + 1) as *mut u8) == 0xb9 {
86                Some(read((addr + 2) as *mut u8))
87            } else {
88                None
89            }
90        }
91        if cfg!(target_arch = "aarch64") {
92            val = if read_unaligned(addr as *mut u32) & 0xffff0000 == 0 {
93                Some(read(addr as *mut u8))
94            } else {
95                None
96            }
97        }
98        match val.and_then(|val| {
99            if val & MAGIC == MAGIC {
100                Some(val & 0xf)
101            } else {
102                None
103            }
104        }) {
105            None => None,
106            Some(val) => match val {
107                0 => Some(TrapCode::StackOverflow),
108                1 => Some(TrapCode::HeapAccessOutOfBounds),
109                2 => Some(TrapCode::HeapMisaligned),
110                3 => Some(TrapCode::TableAccessOutOfBounds),
111                4 => Some(TrapCode::IndirectCallToNull),
112                5 => Some(TrapCode::BadSignature),
113                6 => Some(TrapCode::IntegerOverflow),
114                7 => Some(TrapCode::IntegerDivisionByZero),
115                8 => Some(TrapCode::BadConversionToInteger),
116                9 => Some(TrapCode::UnreachableCodeReached),
117                10 => Some(TrapCode::UnalignedAtomic),
118                _ => None,
119            },
120        }
121    }
122}

131        unsafe fn platform_init() { unsafe {
132            let register = |slot: &mut MaybeUninit<libc::sigaction>, signal: i32| {
133                let mut handler: libc::sigaction = mem::zeroed();
134                // The flags here are relatively careful, and they are...
135                //
136                // SA_SIGINFO gives us access to information like the program
137                // counter from where the fault happened.
138                //
139                // SA_ONSTACK allows us to handle signals on an alternate stack,
140                // so that the handler can run in response to running out of
141                // stack space on the main stack. Rust installs an alternate
142                // stack with sigaltstack, so we rely on that.
143                //
144                // SA_NODEFER allows us to reenter the signal handler if we
145                // crash while handling the signal, and fall through to the
146                // Breakpad handler by testing handlingSegFault.
147                handler.sa_flags = libc::SA_SIGINFO | libc::SA_NODEFER | libc::SA_ONSTACK;
148                handler.sa_sigaction = trap_handler as *const () as usize;
149                libc::sigemptyset(&mut handler.sa_mask);
150                if libc::sigaction(signal, &handler, slot.as_mut_ptr()) != 0 {
151                    panic!(
152                        "unable to install signal handler: {}",
153                        io::Error::last_os_error(),
154                    );
155                }
156            };
157
158            // Allow handling OOB with signals on all architectures
159            register(&mut PREV_SIGSEGV, libc::SIGSEGV);
160
161            // Handle `unreachable` instructions which execute `ud2` right now
162            register(&mut PREV_SIGILL, libc::SIGILL);
163
164            // x86 uses SIGFPE to report division by zero
165            if cfg!(target_arch = "x86") || cfg!(target_arch = "x86_64") {
166                register(&mut PREV_SIGFPE, libc::SIGFPE);
167            }
168
169            // On ARM, handle Unaligned Accesses.
170            // On Darwin, guard page accesses are raised as SIGBUS.
171            if cfg!(target_arch = "arm") || cfg!(target_vendor = "apple") {
172                register(&mut PREV_SIGBUS, libc::SIGBUS);
173            }
174
175            // This is necessary to support debugging under LLDB on Darwin.
176            // For more details see https://github.com/mono/mono/commit/8e75f5a28e6537e56ad70bf870b86e22539c2fb7
177            #[cfg(target_vendor = "apple")]
178            {
179                use mach2::exception_types::*;
180                use mach2::kern_return::*;
181                use mach2::port::*;
182                use mach2::thread_status::*;
183                use mach2::traps::*;
184                use mach2::mach_types::*;
185
186                unsafe extern "C" {
187                    fn task_set_exception_ports(
188                        task: task_t,
189                        exception_mask: exception_mask_t,
190                        new_port: mach_port_t,
191                        behavior: exception_behavior_t,
192                        new_flavor: thread_state_flavor_t,
193                    ) -> kern_return_t;
194                }
195
196                #[allow(non_snake_case)]
197                #[cfg(target_arch = "x86_64")]
198                let MACHINE_THREAD_STATE = x86_THREAD_STATE64;
199                #[allow(non_snake_case)]
200                #[cfg(target_arch = "aarch64")]
201                let MACHINE_THREAD_STATE = 6;
202
203                task_set_exception_ports(
204                    mach_task_self(),
205                    EXC_MASK_BAD_ACCESS | EXC_MASK_ARITHMETIC | EXC_MASK_BAD_INSTRUCTION,
206                    MACH_PORT_NULL,
207                    EXCEPTION_STATE_IDENTITY as exception_behavior_t,
208                    MACHINE_THREAD_STATE,
209                );
210            }
211        }}

213        unsafe extern "C" fn trap_handler(
214            signum: libc::c_int,
215            siginfo: *mut libc::siginfo_t,
216            context: *mut libc::c_void,
217        ) { unsafe {
218            let previous = match signum {
219                libc::SIGSEGV => &PREV_SIGSEGV,
220                libc::SIGBUS => &PREV_SIGBUS,
221                libc::SIGFPE => &PREV_SIGFPE,
222                libc::SIGILL => &PREV_SIGILL,
223                _ => panic!("unknown signal: {signum}"),
224            };
225            // We try to get the fault address associated to this signal
226            let maybe_fault_address = match signum {
227                libc::SIGSEGV | libc::SIGBUS => {
228                    Some((*siginfo).si_addr() as usize)
229                }
230                _ => None,
231            };
232            let trap_code = match signum {
233                // check if it was cased by a UD and if the Trap info is a payload to it
234                libc::SIGILL => {
235                    let addr = (*siginfo).si_addr() as usize;
236                    process_illegal_op(addr)
237                }
238                _ => None,
239            };
240            let ucontext = &mut *(context as *mut ucontext_t);
241            let (pc, sp) = get_pc_sp(ucontext);
242            let handled = TrapHandlerContext::handle_trap(
243                pc,
244                sp,
245                maybe_fault_address,
246                trap_code,
247                |regs| update_context(ucontext, regs),
248                |handler| handler(signum, siginfo, context),
249            );
250
251            if handled {
252                return;
253            }
254
255            // This signal is not for any compiled wasm code we expect, so we
256            // need to forward the signal to the next handler. If there is no
257            // next handler (SIG_IGN or SIG_DFL), then it's time to crash. To do
258            // this, we set the signal back to its original disposition and
259            // return. This will cause the faulting op to be re-executed which
260            // will crash in the normal way. If there is a next handler, call
261            // it. It will either crash synchronously, fix up the instruction
262            // so that execution can continue and return, or trigger a crash by
263            // returning the signal to it's original disposition and returning.
264            let previous = &*previous.as_ptr();
265            if previous.sa_flags & libc::SA_SIGINFO != 0 {
266                mem::transmute::<
267                    usize,
268                    extern "C" fn(libc::c_int, *mut libc::siginfo_t, *mut libc::c_void),
269                >(previous.sa_sigaction)(signum, siginfo, context)
270            } else if previous.sa_sigaction == libc::SIG_DFL
271            {
272                libc::sigaction(signum, previous, ptr::null_mut());
273            } else if previous.sa_sigaction != libc::SIG_IGN {
274                mem::transmute::<usize, extern "C" fn(libc::c_int)>(
275                    previous.sa_sigaction
276                )(signum)
277            }
278        }}

280        unsafe fn get_pc_sp(context: &ucontext_t) -> (usize, usize) {
281            let (pc, sp);
282            cfg_if::cfg_if! {
283                if #[cfg(all(
284                    any(target_os = "linux", target_os = "android"),
285                    target_arch = "x86_64",
286                ))] {
287                    pc = context.uc_mcontext.gregs[libc::REG_RIP as usize] as usize;
288                    sp = context.uc_mcontext.gregs[libc::REG_RSP as usize] as usize;
289                } else if #[cfg(all(
290                    any(target_os = "linux", target_os = "android"),
291                    target_arch = "x86",
292                ))] {
293                    pc = context.uc_mcontext.gregs[libc::REG_EIP as usize] as usize;
294                    sp = context.uc_mcontext.gregs[libc::REG_ESP as usize] as usize;
295                } else if #[cfg(all(target_os = "freebsd", target_arch = "x86"))] {
296                    pc = context.uc_mcontext.mc_eip as usize;
297                    sp = context.uc_mcontext.mc_esp as usize;
298                } else if #[cfg(all(target_os = "freebsd", target_arch = "x86_64"))] {
299                    pc = context.uc_mcontext.mc_rip as usize;
300                    sp = context.uc_mcontext.mc_rsp as usize;
301                } else if #[cfg(all(target_vendor = "apple", target_arch = "x86_64"))] {
302                    let mcontext = unsafe { &*context.uc_mcontext };
303                    pc = mcontext.__ss.__rip as usize;
304                    sp = mcontext.__ss.__rsp as usize;
305                } else if #[cfg(all(
306                        any(target_os = "linux", target_os = "android"),
307                        target_arch = "aarch64",
308                    ))] {
309                    pc = context.uc_mcontext.pc as usize;
310                    sp = context.uc_mcontext.sp as usize;
311                } else if #[cfg(all(
312                    any(target_os = "linux", target_os = "android"),
313                    target_arch = "arm",
314                ))] {
315                    pc = context.uc_mcontext.arm_pc as usize;
316                    sp = context.uc_mcontext.arm_sp as usize;
317                } else if #[cfg(all(
318                    any(target_os = "linux", target_os = "android"),
319                    any(target_arch = "riscv64", target_arch = "riscv32"),
320                ))] {
321                    pc = context.uc_mcontext.__gregs[libc::REG_PC] as usize;
322                    sp = context.uc_mcontext.__gregs[libc::REG_SP] as usize;
323                } else if #[cfg(all(target_vendor = "apple", target_arch = "aarch64"))] {
324                    let mcontext = unsafe { &*context.uc_mcontext };
325                    pc = mcontext.__ss.__pc as usize;
326                    sp = mcontext.__ss.__sp as usize;
327                } else if #[cfg(all(target_os = "freebsd", target_arch = "aarch64"))] {
328                    pc = context.uc_mcontext.mc_gpregs.gp_elr as usize;
329                    sp = context.uc_mcontext.mc_gpregs.gp_sp as usize;
330                } else if #[cfg(all(target_os = "linux", target_arch = "loongarch64"))] {
331                    pc = context.uc_mcontext.__gregs[1] as usize;
332                    sp = context.uc_mcontext.__gregs[3] as usize;
333                } else if #[cfg(all(target_os = "linux", target_arch = "powerpc64"))] {
334                    pc = (*context.uc_mcontext.regs).nip as usize;
335                    sp = (*context.uc_mcontext.regs).gpr[1] as usize;
336                } else {
337                    compile_error!("Unsupported platform");
338                }
339            };
340            (pc, sp)
341        }

507        unsafe extern "system" fn exception_handler(
508            exception_info: *mut EXCEPTION_POINTERS
509        ) -> i32 {
510            unsafe {
511                // Check the kind of exception, since we only handle a subset within
512                // wasm code. If anything else happens we want to defer to whatever
513                // the rest of the system wants to do for this exception.
514                let record = &*(*exception_info).ExceptionRecord;
515                if record.ExceptionCode != EXCEPTION_ACCESS_VIOLATION &&
516                    record.ExceptionCode != EXCEPTION_ILLEGAL_INSTRUCTION &&
517                    record.ExceptionCode != EXCEPTION_STACK_OVERFLOW &&
518                    record.ExceptionCode != EXCEPTION_INT_DIVIDE_BY_ZERO &&
519                    record.ExceptionCode != EXCEPTION_INT_OVERFLOW
520                {
521                    return EXCEPTION_CONTINUE_SEARCH;
522                }
523
524                // FIXME: this is what the previous C++ did to make sure that TLS
525                // works by the time we execute this trap handling code. This isn't
526                // exactly super easy to call from Rust though and it's not clear we
527                // necessarily need to do so. Leaving this here in case we need this
528                // in the future, but for now we can probably wait until we see a
529                // strange fault before figuring out how to reimplement this in
530                // Rust.
531                //
532                // if (!NtCurrentTeb()->Reserved1[sThreadLocalArrayPointerIndex]) {
533                //     return EXCEPTION_CONTINUE_SEARCH;
534                // }
535
536                let context = &mut *(*exception_info).ContextRecord;
537                let (pc, sp) = get_pc_sp(context);
538
539                // We try to get the fault address associated to this exception.
540                let maybe_fault_address = match record.ExceptionCode {
541                    EXCEPTION_ACCESS_VIOLATION => Some(record.ExceptionInformation[1]),
542                    EXCEPTION_STACK_OVERFLOW => Some(sp),
543                    _ => None,
544                };
545                let trap_code = match record.ExceptionCode {
546                    // check if it was cased by a UD and if the Trap info is a payload to it
547                    EXCEPTION_ILLEGAL_INSTRUCTION => {
548                        process_illegal_op(pc)
549                    }
550                    _ => None,
551                };
552                // This is basically the same as the unix version above, only with a
553                // few parameters tweaked here and there.
554                let handled = TrapHandlerContext::handle_trap(
555                    pc,
556                    sp,
557                    maybe_fault_address,
558                    trap_code,
559                    |regs| update_context(context, regs),
560                    |handler| handler(exception_info),
561                );
562
563                if handled {
564                    EXCEPTION_CONTINUE_EXECUTION
565                } else {
566                    EXCEPTION_CONTINUE_SEARCH
567                }
568            }
569        }
570
571        unsafe fn get_pc_sp(context: &CONTEXT) -> (usize, usize) {
572            let (pc, sp);
573            cfg_if::cfg_if! {
574                if #[cfg(target_arch = "x86_64")] {
575                    pc = context.Rip as usize;
576                    sp = context.Rsp as usize;
577                } else if #[cfg(target_arch = "x86")] {
578                    pc = context.Rip as usize;
579                    sp = context.Rsp as usize;
580                } else {
581                    compile_error!("Unsupported platform");
582                }
583            };
584            (pc, sp)
585        }
586
587        unsafe fn update_context(context: &mut CONTEXT, regs: TrapHandlerRegs) {
588            cfg_if::cfg_if! {
589                if #[cfg(target_arch = "x86_64")] {
590                    let TrapHandlerRegs { rip, rsp, rbp, rdi, rsi } = regs;
591                    context.Rip = rip;
592                    context.Rsp = rsp;
593                    context.Rbp = rbp;
594                    context.Rdi = rdi;
595                    context.Rsi = rsi;
596                } else if #[cfg(target_arch = "x86")] {
597                    let TrapHandlerRegs { eip, esp, ebp, ecx, edx } = regs;
598                    context.Eip = eip;
599                    context.Esp = esp;
600                    context.Ebp = ebp;
601                    context.Ecx = ecx;
602                    context.Edx = edx;
603                } else {
604                    compile_error!("Unsupported platform");
605                }
606            };
607        }
608    }
609}

669/// Call the wasm function pointed to by `callee`.
670///
671/// * `vmctx` - the callee vmctx argument
672/// * `caller_vmctx` - the caller vmctx argument
673/// * `trampoline` - the jit-generated trampoline whose ABI takes 4 values, the
674///   callee vmctx, the caller vmctx, the `callee` argument below, and then the
675///   `values_vec` argument.
676/// * `callee` - the third argument to the `trampoline` function
677/// * `values_vec` - points to a buffer which holds the incoming arguments, and to
678///   which the outgoing return values will be written.
679///
680/// # Safety
681///
682/// Wildly unsafe because it calls raw function pointers and reads/writes raw
683/// function pointers.
684pub unsafe fn wasmer_call_trampoline(
685    trap_handler: Option<*const TrapHandlerFn<'static>>,
686    config: &VMConfig,
687    vmctx: VMFunctionContext,
688    trampoline: VMTrampoline,
689    callee: *const VMFunctionBody,
690    values_vec: *mut u8,
691) -> Result<(), Trap> {
692    unsafe {
693        catch_traps(trap_handler, config, move || {
694            mem::transmute::<
695                unsafe extern "C" fn(
696                    *mut VMContext,
697                    *const VMFunctionBody,
698                    *mut wasmer_types::RawValue,
699                ),
700                extern "C" fn(VMFunctionContext, *const VMFunctionBody, *mut u8),
701            >(trampoline)(vmctx, callee, values_vec);
702        })
703    }
704}

706/// Catches any wasm traps that happen within the execution of `closure`,
707/// returning them as a `Result`.
708///
709/// # Safety
710///
711/// Highly unsafe since `closure` won't have any dtors run.
712pub unsafe fn catch_traps<F, R: 'static>(
713    trap_handler: Option<*const TrapHandlerFn<'static>>,
714    config: &VMConfig,
715    closure: F,
716) -> Result<R, Trap>
717where
718    F: FnOnce() -> R + 'static,
719{
720    // Ensure that per-thread initialization is done.
721    lazy_per_thread_init()?;
722    let stack_size = config
723        .wasm_stack_size
724        .unwrap_or_else(|| DEFAULT_STACK_SIZE.load(Ordering::Relaxed));
725    on_wasm_stack(stack_size, trap_handler, closure).map_err(UnwindReason::into_trap)
726}

728// We need two separate thread-local variables here:
729// - YIELDER is set within the new stack and is used to unwind back to the root
730//   of the stack from inside it.
731// - TRAP_HANDLER is set from outside the new stack and is solely used from
732//   signal handlers. It must be atomic since it is used by signal handlers.
733//
734// We also do per-thread signal stack initialization on the first time
735// TRAP_HANDLER is accessed.
736thread_local! {
737    static YIELDER: Cell<Option<NonNull<Yielder<(), UnwindReason>>>> = const { Cell::new(None) };
738    static TRAP_HANDLER: AtomicPtr<TrapHandlerContext> = const { AtomicPtr::new(ptr::null_mut()) };
739}

741/// Read-only information that is used by signal handlers to handle and recover
742/// from traps.
743#[allow(clippy::type_complexity)]
744struct TrapHandlerContext {
745    inner: *const u8,
746    handle_trap: fn(
747        *const u8,
748        usize,
749        usize,
750        Option<usize>,
751        Option<TrapCode>,
752        &mut dyn FnMut(TrapHandlerRegs),
753    ) -> bool,
754    custom_trap: Option<*const TrapHandlerFn<'static>>,
755}
756struct TrapHandlerContextInner<T> {
757    /// Information about the currently running coroutine. This is used to
758    /// reset execution to the root of the coroutine when a trap is handled.
759    coro_trap_handler: CoroutineTrapHandler<Result<T, UnwindReason>>,
760}
761
762impl TrapHandlerContext {
763    /// Runs the given function with a trap handler context. The previous
764    /// trap handler context is preserved and restored afterwards.
765    fn install<T, R>(
766        custom_trap: Option<*const TrapHandlerFn<'static>>,
767        coro_trap_handler: CoroutineTrapHandler<Result<T, UnwindReason>>,
768        f: impl FnOnce() -> R,
769    ) -> R {
770        // Type-erase the trap handler function so that it can be placed in TLS.
771        fn func<T>(
772            ptr: *const u8,
773            pc: usize,
774            sp: usize,
775            maybe_fault_address: Option<usize>,
776            trap_code: Option<TrapCode>,
777            update_regs: &mut dyn FnMut(TrapHandlerRegs),
778        ) -> bool {
779            unsafe {
780                (*(ptr as *const TrapHandlerContextInner<T>)).handle_trap(
781                    pc,
782                    sp,
783                    maybe_fault_address,
784                    trap_code,
785                    update_regs,
786                )
787            }
788        }
789        let inner = TrapHandlerContextInner { coro_trap_handler };
790        let ctx = Self {
791            inner: &inner as *const _ as *const u8,
792            handle_trap: func::<T>,
793            custom_trap,
794        };
795
796        compiler_fence(Ordering::Release);
797        let prev = TRAP_HANDLER.with(|ptr| {
798            let prev = ptr.load(Ordering::Relaxed);
799            ptr.store(&ctx as *const Self as *mut Self, Ordering::Relaxed);
800            prev
801        });
802
803        defer! {
804            TRAP_HANDLER.with(|ptr| ptr.store(prev, Ordering::Relaxed));
805            compiler_fence(Ordering::Acquire);
806        }
807
808        f()
809    }
810
811    /// Attempts to handle the trap if it's a wasm trap.
812    unsafe fn handle_trap(
813        pc: usize,
814        sp: usize,
815        maybe_fault_address: Option<usize>,
816        trap_code: Option<TrapCode>,
817        mut update_regs: impl FnMut(TrapHandlerRegs),
818        call_handler: impl Fn(&TrapHandlerFn<'static>) -> bool,
819    ) -> bool {
820        unsafe {
821            let ptr = TRAP_HANDLER.with(|ptr| ptr.load(Ordering::Relaxed));
822            if ptr.is_null() {
823                return false;
824            }
825
826            let ctx = &*ptr;
827
828            // Check if this trap is handled by a custom trap handler.
829            if let Some(trap_handler) = ctx.custom_trap
830                && call_handler(&*trap_handler)
831            {
832                return true;
833            }
834
835            (ctx.handle_trap)(
836                ctx.inner,
837                pc,
838                sp,
839                maybe_fault_address,
840                trap_code,
841                &mut update_regs,
842            )
843        }
844    }
845}
846
847impl<T> TrapHandlerContextInner<T> {
848    unsafe fn handle_trap(
849        &self,
850        pc: usize,
851        sp: usize,
852        maybe_fault_address: Option<usize>,
853        trap_code: Option<TrapCode>,
854        update_regs: &mut dyn FnMut(TrapHandlerRegs),
855    ) -> bool {
856        unsafe {
857            // Check if this trap occurred while executing on the Wasm stack. We can
858            // only recover from traps if that is the case.
859            if !self.coro_trap_handler.stack_ptr_in_bounds(sp) {
860                return false;
861            }
862
863            let signal_trap = trap_code.or_else(|| {
864                maybe_fault_address.map(|addr| {
865                    if self.coro_trap_handler.stack_ptr_in_bounds(addr) {
866                        TrapCode::StackOverflow
867                    } else {
868                        TrapCode::HeapAccessOutOfBounds
869                    }
870                })
871            });
872
873            // Don't try to generate a backtrace for stack overflows: unwinding
874            // information is often not precise enough to properly describe what is
875            // happenning during a function prologue, which can lead the unwinder to
876            // read invalid memory addresses.
877            //
878            // See: https://github.com/rust-lang/backtrace-rs/pull/357
879            let backtrace = if signal_trap == Some(TrapCode::StackOverflow) {
880                Backtrace::from(vec![])
881            } else {
882                Backtrace::new_unresolved()
883            };
884
885            // Set up the register state for exception return to force the
886            // coroutine to return to its caller with UnwindReason::WasmTrap.
887            let unwind = UnwindReason::WasmTrap {
888                backtrace,
889                signal_trap,
890                pc,
891            };
892            let regs = self
893                .coro_trap_handler
894                .setup_trap_handler(move || Err(unwind));
895            update_regs(regs);
896            true
897        }
898    }
899}

901enum UnwindReason {
902    /// A panic caused by the host
903    Panic(Box<dyn Any + Send>),
904    /// A custom error triggered by the user
905    UserTrap(Box<dyn Error + Send + Sync>),
906    /// A Trap triggered by a wasm libcall
907    LibTrap(Trap),
908    /// A trap caused by the Wasm generated code
909    WasmTrap {
910        backtrace: Backtrace,
911        pc: usize,
912        signal_trap: Option<TrapCode>,
913    },
914}
915
916impl UnwindReason {
917    fn into_trap(self) -> Trap {
918        match self {
919            Self::UserTrap(data) => Trap::User(data),
920            Self::LibTrap(trap) => trap,
921            Self::WasmTrap {
922                backtrace,
923                pc,
924                signal_trap,
925            } => Trap::wasm(pc, backtrace, signal_trap),
926            Self::Panic(panic) => std::panic::resume_unwind(panic),
927        }
928    }
929}

931unsafe fn unwind_with(reason: UnwindReason) -> ! {
932    unsafe {
933        let yielder = YIELDER
934            .with(|cell| cell.replace(None))
935            .expect("not running on Wasm stack");
936
937        yielder.as_ref().suspend(reason);
938
939        // on_wasm_stack will forcibly reset the coroutine stack after yielding.
940        unreachable!();
941    }
942}

944/// Runs the given function on a separate stack so that its stack usage can be
945/// bounded. Stack overflows and other traps can be caught and execution
946/// returned to the root of the stack.
947fn on_wasm_stack<F: FnOnce() -> T + 'static, T: 'static>(
948    stack_size: usize,
949    trap_handler: Option<*const TrapHandlerFn<'static>>,
950    f: F,
951) -> Result<T, UnwindReason> {
952    // Allocating a new stack is pretty expensive since it involves several
953    // system calls. We therefore keep a cache of pre-allocated stacks which
954    // allows them to be reused multiple times.
955    // FIXME(Amanieu): We should refactor this to avoid the lock.
956    static STACK_POOL: LazyLock<crossbeam_queue::SegQueue<DefaultStack>> =
957        LazyLock::new(crossbeam_queue::SegQueue::new);
958
959    let stack = STACK_POOL
960        .pop()
961        .unwrap_or_else(|| DefaultStack::new(stack_size).unwrap());
962    let mut stack = scopeguard::guard(stack, |stack| STACK_POOL.push(stack));
963
964    // Create a coroutine with a new stack to run the function on.
965    let coro = ScopedCoroutine::with_stack(&mut *stack, move |yielder, ()| {
966        // Save the yielder to TLS so that it can be used later.
967        YIELDER.with(|cell| cell.set(Some(yielder.into())));
968
969        Ok(f())
970    });
971
972    // Ensure that YIELDER is reset on exit even if the coroutine panics,
973    defer! {
974        YIELDER.with(|cell| cell.set(None));
975    }
976
977    coro.scope(|mut coro_ref| {
978        // Set up metadata for the trap handler for the duration of the coroutine
979        // execution. This is restored to its previous value afterwards.
980        TrapHandlerContext::install(trap_handler, coro_ref.trap_handler(), || {
981            match coro_ref.resume(()) {
982                CoroutineResult::Yield(trap) => {
983                    // This came from unwind_with which requires that there be only
984                    // Wasm code on the stack.
985                    unsafe {
986                        coro_ref.force_reset();
987                    }
988                    Err(trap)
989                }
990                CoroutineResult::Return(result) => result,
991            }
992        })
993    })
994}

996/// When executing on the Wasm stack, temporarily switch back to the host stack
997/// to perform an operation that should not be constrainted by the Wasm stack
998/// limits.
999///
1000/// This is particularly important since the usage of the Wasm stack is under
1001/// the control of untrusted code. Malicious code could artificially induce a
1002/// stack overflow in the middle of a sensitive host operations (e.g. growing
1003/// a memory) which would be hard to recover from.
1004pub fn on_host_stack<F: FnOnce() -> T, T>(f: F) -> T {
1005    // Reset YIEDER to None for the duration of this call to indicate that we
1006    // are no longer on the Wasm stack.
1007    let yielder_ptr = YIELDER.with(|cell| cell.replace(None));
1008
1009    // If we are already on the host stack, execute the function directly. This
1010    // happens if a host function is called directly from the API.
1011    let yielder = match yielder_ptr {
1012        Some(ptr) => unsafe { ptr.as_ref() },
1013        None => return f(),
1014    };
1015
1016    // Restore YIELDER upon exiting normally or unwinding.
1017    defer! {
1018        YIELDER.with(|cell| cell.set(yielder_ptr));
1019    }
1020
1021    // on_parent_stack requires the closure to be Send so that the Yielder
1022    // cannot be called from the parent stack. This is not a problem for us
1023    // since we don't expose the Yielder.
1024    struct SendWrapper<T>(T);
1025    unsafe impl<T> Send for SendWrapper<T> {}
1026    let wrapped = SendWrapper(f);
1027    yielder.on_parent_stack(move || {
1028        let wrapped = wrapped;
1029        (wrapped.0)()
1030    })
1031}

1046/// A module for registering a custom alternate signal stack (sigaltstack).
1047///
1048/// Rust's libstd installs an alternate stack with size `SIGSTKSZ`, which is not
1049/// always large enough for our signal handling code. Override it by creating
1050/// and registering our own alternate stack that is large enough and has a guard
1051/// page.
1052#[cfg(unix)]
1053pub fn lazy_per_thread_init() -> Result<(), Trap> {
1054    use std::ptr::null_mut;
1055
1056    thread_local! {
1057        /// Thread-local state is lazy-initialized on the first time it's used,
1058        /// and dropped when the thread exits.
1059        static TLS: Tls = unsafe { init_sigstack() };
1060    }
1061
1062    /// The size of the sigaltstack (not including the guard, which will be
1063    /// added). Make this large enough to run our signal handlers.
1064    const MIN_STACK_SIZE: usize = 16 * 4096;
1065
1066    enum Tls {
1067        OutOfMemory,
1068        Allocated {
1069            mmap_ptr: *mut libc::c_void,
1070            mmap_size: usize,
1071        },
1072        BigEnough,
1073    }
1074
1075    unsafe fn init_sigstack() -> Tls {
1076        unsafe {
1077            // Check to see if the existing sigaltstack, if it exists, is big
1078            // enough. If so we don't need to allocate our own.
1079            let mut old_stack = mem::zeroed();
1080            let r = libc::sigaltstack(ptr::null(), &mut old_stack);
1081            assert_eq!(r, 0, "learning about sigaltstack failed");
1082            if old_stack.ss_flags & libc::SS_DISABLE == 0 && old_stack.ss_size >= MIN_STACK_SIZE {
1083                return Tls::BigEnough;
1084            }
1085
1086            // ... but failing that we need to allocate our own, so do all that
1087            // here.
1088            let page_size: usize = region::page::size();
1089            let guard_size = page_size;
1090            let alloc_size = guard_size + MIN_STACK_SIZE;
1091
1092            let ptr = libc::mmap(
1093                null_mut(),
1094                alloc_size,
1095                libc::PROT_NONE,
1096                libc::MAP_PRIVATE | libc::MAP_ANON,
1097                -1,
1098                0,
1099            );
1100            if ptr == libc::MAP_FAILED {
1101                return Tls::OutOfMemory;
1102            }
1103
1104            // Prepare the stack with readable/writable memory and then register it
1105            // with `sigaltstack`.
1106            let stack_ptr = (ptr as usize + guard_size) as *mut libc::c_void;
1107            let r = libc::mprotect(
1108                stack_ptr,
1109                MIN_STACK_SIZE,
1110                libc::PROT_READ | libc::PROT_WRITE,
1111            );
1112            assert_eq!(r, 0, "mprotect to configure memory for sigaltstack failed");
1113            let new_stack = libc::stack_t {
1114                ss_sp: stack_ptr,
1115                ss_flags: 0,
1116                ss_size: MIN_STACK_SIZE,
1117            };
1118            let r = libc::sigaltstack(&new_stack, ptr::null_mut());
1119            assert_eq!(r, 0, "registering new sigaltstack failed");
1120
1121            Tls::Allocated {
1122                mmap_ptr: ptr,
1123                mmap_size: alloc_size,
1124            }
1125        }
1126    }
1127
1128    // Ensure TLS runs its initializer and return an error if it failed to
1129    // set up a separate stack for signal handlers.
1130    return TLS.with(|tls| {
1131        if let Tls::OutOfMemory = tls {
1132            Err(Trap::oom())
1133        } else {
1134            Ok(())
1135        }
1136    });
1137
1138    impl Drop for Tls {
1139        fn drop(&mut self) {
1140            let (ptr, size) = match self {
1141                Self::Allocated {
1142                    mmap_ptr,
1143                    mmap_size,
1144                } => (*mmap_ptr, *mmap_size),
1145                _ => return,
1146            };
1147            unsafe {
1148                // Deallocate the stack memory.
1149                let r = libc::munmap(ptr, size);
1150                debug_assert_eq!(r, 0, "munmap failed during thread shutdown");
1151            }
1152        }
1153    }
