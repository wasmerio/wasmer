6/// We can only validate types that have a well defined layout.
7fn check_repr(input: &DeriveInput) {
8    if input.attrs.iter().any(|attr| {
9        attr.path().is_ident("repr") && {
10            let mut valid = false;
11            let _ = attr.parse_nested_meta(|meta| {
12                if meta.path.is_ident("C") || meta.path.is_ident("transparent") {
13                    valid = true;
14                }
15                Ok(())
16            });
17            valid
18        }
19    }) {
20        return;
21    }
22
23    abort!(
24        input,
25        "ValueType can only be derived for #[repr(C)] or #[repr(transparent)] structs"
26    )
27}

29/// Zero out any padding bytes between fields.
30fn zero_padding(fields: &Fields) -> TokenStream {
31    let names: Vec<_> = fields
32        .iter()
33        .enumerate()
34        .map(|(i, field)| match &field.ident {
35            Some(ident) => Member::Named(ident.clone()),
36            None => Member::Unnamed(i.into()),
37        })
38        .collect();
39
40    let mut out = TokenStream::new();
41    for i in 0..fields.len() {
42        let name = &names[i];
43        let start = quote! {
44            &self.#name as *const _ as usize - self as *const _ as usize
45        };
46        let len = quote! {
47            ::core::mem::size_of_val(&self.#name)
48        };
49        let end = quote! {
50            #start + #len
51        };
52
53        // Zero out padding bytes within the current field.
54        //
55        // This also ensures that all fields implement ValueType.
56        out.extend(quote! {
57            ::wasmer_types::ValueType::zero_padding_bytes(&self.#name, &mut _bytes[#start..(#start + #len)]);
58        });
59
60        let padding_end = if i == fields.len() - 1 {
61            // Zero out padding bytes between the last field and the end of the struct.
62            let total_size = quote! {
63                ::core::mem::size_of_val(self)
64            };
65            total_size
66        } else {
67            // Zero out padding bytes between the current field and the next one.
68            let next_name = &names[i + 1];
69            let next_start = quote! {
70                &self.#next_name as *const _ as usize - self as *const _ as usize
71            };
72            next_start
73        };
74        out.extend(quote! {
75            for i in #end..#padding_end {
76                _bytes[i] = ::core::mem::MaybeUninit::new(0);
77            }
78        });
79    }
80    out
81}

83pub fn impl_value_type(input: &DeriveInput) -> TokenStream {
84    check_repr(input);
85
86    let struct_name = &input.ident;
87    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
88    let fields = match &input.data {
89        Data::Struct(ds) => &ds.fields,
90        _ => abort!(input, "ValueType can only be derived for structs"),
91    };
92
93    let zero_padding = zero_padding(fields);
94
95    quote! {
96        unsafe impl #impl_generics ::wasmer_types::ValueType for #struct_name #ty_generics #where_clause {
97            #[inline]
98            fn zero_padding_bytes(&self, _bytes: &mut [::core::mem::MaybeUninit<u8>]) {
99                #zero_padding
100            }
101        }
102    }
103}
