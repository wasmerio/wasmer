1//! Logging functions for the debug feature.
2
3use is_terminal::IsTerminal;
4use tracing::level_filters::LevelFilter;
5use tracing_subscriber::{EnvFilter, Layer, fmt, layer::SubscriberExt, util::SubscriberInitExt};
6
7const WHITELISTED_LOG_TARGETS: &[&str] = &["wasmer", "wasmer_wasix", "virtual_fs"];
8
9/// Control the output generated by the CLI.
10#[derive(Debug, Default, Clone, PartialEq, clap::Parser)]
11pub struct Output {
12    /// Generate verbose output (repeat for more verbosity)
13    #[clap(short, long, action = clap::ArgAction::Count, global = true, conflicts_with = "quiet")]
14    pub verbose: u8,
15    /// Do not print progress messages.
16    #[clap(short, long, global = true, conflicts_with = "verbose")]
17    pub quiet: bool,
18    /// The format to use when generating logs.
19    #[clap(long, global = true, env, default_value = "text")]
20    pub log_format: LogFormat,
21    /// Which span events to log.
22    #[clap(long, global = true, env, default_value = "close")]
23    pub log_events: LogEvents,
24    /// When to display colored output.
25    #[clap(long, default_value_t = clap::ColorChoice::Auto, global = true)]
26    pub color: clap::ColorChoice,
27}
28
29impl Output {
30    /// Has the `--verbose` flag been set?
31    pub fn is_verbose(&self) -> bool {
32        self.verbose > 0
33    }
34
35    /// Initialize logging based on the `$RUST_LOG` environment variable and
36    /// command-line flags.
37    pub fn initialize_logging(&self) {
38        let filter_layer = self.log_filter();
39
40        let fmt_layer = fmt::layer()
41            .with_target(true)
42            .with_ansi(self.should_emit_colors())
43            .with_thread_ids(true)
44            .with_writer(std::io::stderr);
45
46        let fmt_layer = match self.log_events {
47            LogEvents::New => fmt_layer.with_span_events(fmt::format::FmtSpan::NEW),
48            LogEvents::Close => fmt_layer.with_span_events(fmt::format::FmtSpan::CLOSE),
49            LogEvents::All => {
50                fmt_layer.with_span_events(fmt::format::FmtSpan::NEW | fmt::format::FmtSpan::CLOSE)
51            }
52        };
53
54        let registry = tracing_subscriber::registry();
55
56        #[cfg(feature = "tokio-subscriber")]
57        let registry = registry.with(console_subscriber::spawn());
58
59        match self.log_format {
60            LogFormat::Text => registry
61                .with(
62                    fmt_layer
63                        .compact()
64                        .with_target(true)
65                        .with_filter(filter_layer),
66                )
67                .init(),
68            LogFormat::Json => registry
69                .with(fmt_layer.json().with_target(true).with_filter(filter_layer))
70                .init(),
71        };
72    }
73
74    fn log_filter(&self) -> EnvFilter {
75        let default_filters = [
76            LevelFilter::OFF,
77            LevelFilter::WARN,
78            LevelFilter::INFO,
79            LevelFilter::DEBUG,
80        ];
81
82        // First, we set up the default log level.
83        let default_level = default_filters
84            .get(self.verbose as usize)
85            .copied()
86            .unwrap_or(LevelFilter::TRACE);
87        let mut filter = EnvFilter::builder()
88            .with_default_directive(default_level.into())
89            .from_env_lossy();
90
91        // Next we add level-specific directives, where verbosity=0 means don't
92        // override anything. Note that these are shifted one level up so we'll
93        // get something like RUST_LOG="warn,wasmer_wasix=info"
94        let specific_filters = [LevelFilter::WARN, LevelFilter::INFO, LevelFilter::DEBUG];
95        if self.verbose > 0 {
96            let level = specific_filters
97                .get(self.verbose as usize)
98                .copied()
99                .unwrap_or(LevelFilter::TRACE);
100
101            for target in WHITELISTED_LOG_TARGETS {
102                let directive = format!("{target}={level}").parse().unwrap();
103                filter = filter.add_directive(directive);
104            }
105        }
106
107        filter
108    }
109
110    /// Check whether we should emit ANSI escape codes for log formatting.
111    ///
112    /// The `tracing-subscriber` crate doesn't have native support for
113    /// "--color=always|never|auto", so we implement a poor man's version.
114    ///
115    /// For more, see https://github.com/tokio-rs/tracing/issues/2388
116    fn should_emit_colors(&self) -> bool {
117        match self.color {
118            clap::ColorChoice::Auto => std::io::stderr().is_terminal(),
119            clap::ColorChoice::Always => true,
120            clap::ColorChoice::Never => false,
121        }
122    }
123
124    /// Get the draw target to be used with the `indicatif` crate.
125    ///
126    /// Progress indicators won't draw anything if the user passed the `--quiet`
127    /// flag.
128    pub fn draw_target(&self) -> indicatif::ProgressDrawTarget {
129        if self.quiet {
130            return indicatif::ProgressDrawTarget::hidden();
131        }
132
133        indicatif::ProgressDrawTarget::stderr()
134    }
135}

137/// The format used when generating logs.
138#[derive(Debug, Default, Copy, Clone, PartialEq, clap::ValueEnum)]
139pub enum LogFormat {
140    /// Human-readable logs.
141    #[default]
142    Text,
143    /// Machine-readable logs.
144    Json,
145}
146
147/// Which span events to log.
148#[derive(Debug, Default, Copy, Clone, PartialEq, clap::ValueEnum)]
149pub enum LogEvents {
150    // Only log at the start of a new span.
151    New,
152    // Only log at the end of a span.
153    #[default]
154    Close,
155    // Log at the start and end of a span.
156    All,
157}
