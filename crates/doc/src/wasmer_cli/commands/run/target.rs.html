20/// We've been given the path for a file... What does it contain and how should
21/// that be run?
22#[derive(Debug, Clone)]
23pub enum TargetOnDisk {
24    WebAssemblyBinary,
25    Wat,
26    LocalWebc,
27    Artifact,
28}
29
30impl TargetOnDisk {
31    pub fn from_file(path: &Path) -> Result<TargetOnDisk, Error> {
32        // Normally the first couple hundred bytes is enough to figure
33        // out what type of file this is.
34        let mut buffer = [0_u8; 512];
35
36        let mut f = File::open(path)
37            .with_context(|| format!("Unable to open \"{}\" for reading", path.display()))?;
38        let bytes_read = f.read(&mut buffer)?;
39
40        let leading_bytes = &buffer[..bytes_read];
41
42        if wasmer::is_wasm(leading_bytes) {
43            return Ok(TargetOnDisk::WebAssemblyBinary);
44        }
45
46        if webc::detect(leading_bytes).is_ok() {
47            return Ok(TargetOnDisk::LocalWebc);
48        }
49
50        #[cfg(feature = "compiler")]
51        if ArtifactBuild::is_deserializable(leading_bytes) {
52            return Ok(TargetOnDisk::Artifact);
53        }
54
55        // If we can't figure out the file type based on its content, fall back
56        // to checking the extension.
57
58        match path.extension().and_then(|s| s.to_str()) {
59            Some("wat") => Ok(TargetOnDisk::Wat),
60            Some("wasm") => Ok(TargetOnDisk::WebAssemblyBinary),
61            Some("webc") => Ok(TargetOnDisk::LocalWebc),
62            Some("wasmu") => Ok(TargetOnDisk::WebAssemblyBinary),
63            _ => bail!("Unable to determine how to execute \"{}\"", path.display()),
64        }
65    }
66}

68#[derive(Debug, Clone)]
69pub enum ExecutableTarget {
70    WebAssembly {
71        module: Module,
72        module_hash: ModuleHash,
73        path: PathBuf,
74    },
75    Package(Box<BinaryPackage>),
76}
77
78impl ExecutableTarget {
79    /// Try to load a Wasmer package from a directory containing a `wasmer.toml`
80    /// file.
81    #[tracing::instrument(level = "debug", skip_all)]
82    pub fn from_dir(
83        dir: &Path,
84        runtime: &Arc<dyn Runtime + Send + Sync>,
85        pb: &ProgressBar,
86    ) -> Result<Self, Error> {
87        pb.set_message(format!("Loading \"{}\" into memory", dir.display()));
88        pb.set_message("Resolving dependencies");
89        let inner_runtime = runtime.clone();
90        let pkg = runtime.task_manager().spawn_and_block_on({
91            let path = dir.to_path_buf();
92
93            async move { BinaryPackage::from_dir(&path, inner_runtime.as_ref()).await }
94        })??;
95
96        Ok(ExecutableTarget::Package(Box::new(pkg)))
97    }
98
99    /// Try to load a file into something that can be used to run it.
100    #[tracing::instrument(level = "debug", skip_all)]
101    pub fn from_file(
102        path: &Path,
103        runtime: &Arc<dyn Runtime + Send + Sync>,
104        pb: &ProgressBar,
105    ) -> Result<Self, Error> {
106        pb.set_message(format!("Loading from \"{}\"", path.display()));
107
108        match TargetOnDisk::from_file(path)? {
109            TargetOnDisk::WebAssemblyBinary | TargetOnDisk::Wat => {
110                let wasm = std::fs::read(path)?;
111                let module_data = HashedModuleData::new(wasm);
112                let module_hash = *module_data.hash();
113
114                pb.set_message("Compiling to WebAssembly");
115                let module = runtime
116                    .load_hashed_module_sync(module_data, None)
117                    .with_context(|| format!("Unable to compile \"{}\"", path.display()))?;
118
119                Ok(ExecutableTarget::WebAssembly {
120                    module,
121                    module_hash,
122                    path: path.to_path_buf(),
123                })
124            }
125            TargetOnDisk::Artifact => {
126                let engine = runtime.engine();
127                pb.set_message("Deserializing pre-compiled WebAssembly module");
128                let module = unsafe { Module::deserialize_from_file(&engine, path)? };
129
130                let module_hash = module.info().hash.ok_or_else(|| {
131                    anyhow::Error::msg("module hash is not present in the artifact")
132                })?;
133
134                Ok(ExecutableTarget::WebAssembly {
135                    module,
136                    module_hash,
137                    path: path.to_path_buf(),
138                })
139            }
140            TargetOnDisk::LocalWebc => {
141                let container = wasmer_package::utils::from_disk(path)?;
142                pb.set_message("Resolving dependencies");
143
144                let inner_runtime = runtime.clone();
145                let pkg = runtime.task_manager().spawn_and_block_on(async move {
146                    BinaryPackage::from_webc(&container, inner_runtime.as_ref()).await
147                })??;
148                Ok(ExecutableTarget::Package(Box::new(pkg)))
149            }
150        }
151    }
152}
