271    pub fn prepare(
272        &self,
273        module: &Module,
274        program_name: String,
275        args: Vec<String>,
276        rt: Arc<dyn Runtime + Send + Sync>,
277    ) -> Result<WasiEnvBuilder> {
278        let args = args.into_iter().map(|arg| arg.into_bytes());
279
280        let map_commands = self
281            .map_commands
282            .iter()
283            .map(|map| map.split_once('=').unwrap())
284            .map(|(a, b)| (a.to_string(), b.to_string()))
285            .collect::<HashMap<_, _>>();
286
287        let mut uses = Vec::new();
288        for name in &self.uses {
289            let specifier = PackageSpecifier::from_str(name)
290                .with_context(|| format!("Unable to parse \"{name}\" as a package specifier"))?;
291            let pkg = {
292                let inner_rt = rt.clone();
293                rt.task_manager()
294                    .spawn_and_block_on(async move {
295                        BinaryPackage::from_registry(&specifier, &*inner_rt).await
296                    })
297                    .with_context(|| format!("Unable to load \"{name}\""))??
298            };
299            uses.push(pkg);
300        }
301
302        let mut builder = WasiEnv::builder(program_name)
303            .runtime(Arc::clone(&rt))
304            .args(args)
305            .envs(self.env_vars.clone())
306            .uses(uses)
307            .map_commands(map_commands);
308
309        let mut builder = {
310            // If we preopen anything from the host then shallow copy it over
311            let root_fs = RootFileSystemBuilder::new()
312                .with_tty(Box::new(DeviceFile::new(__WASI_STDIN_FILENO)))
313                .build();
314
315            let mut mapped_dirs = Vec::new();
316
317            // Process the --dirs flag and merge it with --mapdir.
318            let mut have_current_dir = false;
319            for dir in &self.pre_opened_directories {
320                let mapping = if dir == Path::new(".") {
321                    if have_current_dir {
322                        bail!(
323                            "Cannot pre-open the current directory twice: --dir=. must only be specified once"
324                        );
325                    }
326                    have_current_dir = true;
327
328                    let current_dir =
329                        std::env::current_dir().context("could not determine current directory")?;
330
331                    MappedDirectory {
332                        host: current_dir,
333                        guest: MAPPED_CURRENT_DIR_DEFAULT_PATH.to_string(),
334                    }
335                } else {
336                    let resolved = dir.canonicalize().with_context(|| {
337                        format!(
338                            "could not canonicalize path for argument '--dir {}'",
339                            dir.display()
340                        )
341                    })?;
342
343                    if &resolved != dir {
344                        bail!(
345                            "Invalid argument '--dir {}': path must either be absolute, or '.'",
346                            dir.display(),
347                        );
348                    }
349
350                    let guest = resolved
351                        .to_str()
352                        .with_context(|| {
353                            format!(
354                                "invalid argument '--dir {}': path must be valid utf-8",
355                                dir.display(),
356                            )
357                        })?
358                        .to_string();
359
360                    MappedDirectory {
361                        host: resolved,
362                        guest,
363                    }
364                };
365
366                mapped_dirs.push(mapping);
367            }
368
369            for MappedDirectory { host, guest } in &self.mapped_dirs {
370                let resolved_host = host.canonicalize().with_context(|| {
371                    format!(
372                        "could not canonicalize path for argument '--mapdir {}:{}'",
373                        host.display(),
374                        guest,
375                    )
376                })?;
377
378                let mapping = if guest == "." {
379                    if have_current_dir {
380                        bail!(
381                            "Cannot pre-open the current directory twice: '--mapdir=?:.' / '--dir=.' must only be specified once"
382                        );
383                    }
384                    have_current_dir = true;
385
386                    MappedDirectory {
387                        host: resolved_host,
388                        guest: MAPPED_CURRENT_DIR_DEFAULT_PATH.to_string(),
389                    }
390                } else {
391                    MappedDirectory {
392                        host: resolved_host,
393                        guest: guest.clone(),
394                    }
395                };
396                mapped_dirs.push(mapping);
397            }
398
399            if !mapped_dirs.is_empty() {
400                // TODO: should we expose the common ancestor instead of root?
401                let fs_backing: Arc<dyn FileSystem + Send + Sync> =
402                    Arc::new(PassthruFileSystem::new_arc(default_fs_backing()));
403                for MappedDirectory { host, guest } in self.mapped_dirs.clone() {
404                    let host = if !host.is_absolute() {
405                        Path::new("/").join(host)
406                    } else {
407                        host
408                    };
409                    root_fs.mount(guest.into(), &fs_backing, host)?;
410                }
411            }
412
413            if let Some(cwd) = self.cwd.as_ref() {
414                if !cwd.starts_with("/") {
415                    bail!("The argument to --cwd must be an absolute path");
416                }
417                builder = builder.current_dir(cwd.clone());
418            }
419
420            // Open the root of the new filesystem
421            builder = builder
422                .sandbox_fs(root_fs)
423                .preopen_dir(Path::new("/"))
424                .unwrap();
425
426            if have_current_dir {
427                builder.map_dir(".", MAPPED_CURRENT_DIR_DEFAULT_PATH)?
428            } else {
429                builder.map_dir(".", "/")?
430            }
431        };
432
433        *builder.capabilities_mut() = self.capabilities();
434
435        #[cfg(feature = "journal")]
436        {
437            for trigger in self.snapshot_on.iter().cloned() {
438                builder.add_snapshot_trigger(trigger);
439            }
440            if let Some(interval) = self.snapshot_interval {
441                builder.with_snapshot_interval(std::time::Duration::from_millis(interval));
442            }
443            if self.stop_after_snapshot {
444                builder.with_stop_running_after_snapshot(true);
445            }
446            let (r, w) = self.build_journals()?;
447            for journal in r {
448                builder.add_read_only_journal(journal);
449            }
450            for journal in w {
451                builder.add_writable_journal(journal);
452            }
453            builder.with_skip_stdio_during_bootstrap(self.skip_stdio_during_bootstrap);
454        }
455
456        Ok(builder)
457    }

459    #[cfg(feature = "journal")]
460    #[allow(clippy::type_complexity)]
461    pub fn build_journals(
462        &self,
463    ) -> anyhow::Result<(Vec<Arc<DynReadableJournal>>, Vec<Arc<DynJournal>>)> {
464        let mut readable = Vec::new();
465        for journal in self.read_only_journals.clone() {
466            if matches!(std::fs::metadata(&journal), Err(e) if e.kind() == std::io::ErrorKind::NotFound)
467            {
468                bail!("Read-only journal file does not exist: {journal:?}");
469            }
470
471            readable
472                .push(Arc::new(LogFileJournal::new_readonly(journal)?) as Arc<DynReadableJournal>);
473        }
474
475        let mut writable = Vec::new();
476        for journal in self.writable_journals.clone() {
477            if self.enable_compaction {
478                let mut journal = CompactingLogFileJournal::new(journal)?;
479                if !self.without_compact_on_drop {
480                    journal = journal.with_compact_on_drop()
481                }
482                if self.with_compact_on_growth.is_normal() && self.with_compact_on_growth != 0f32 {
483                    journal = journal.with_compact_on_factor_size(self.with_compact_on_growth);
484                }
485                writable.push(Arc::new(journal) as Arc<DynJournal>);
486            } else {
487                writable.push(Arc::new(LogFileJournal::new(journal)?));
488            }
489        }
490        Ok((readable, writable))
491    }

498    pub fn build_mapped_directories(
499        &self,
500        is_wasix: bool,
501    ) -> Result<(bool, Vec<MappedDirectory>), anyhow::Error> {
502        let mut mapped_dirs = Vec::new();
503
504        // Process the --dirs flag and merge it with --mapdir.
505        let mut have_current_dir = false;
506        for dir in &self.pre_opened_directories {
507            let mapping = if dir == Path::new(".") {
508                if have_current_dir {
509                    bail!(
510                        "Cannot pre-open the current directory twice: --dir=. must only be specified once"
511                    );
512                }
513                have_current_dir = true;
514
515                let current_dir =
516                    std::env::current_dir().context("could not determine current directory")?;
517
518                MappedDirectory {
519                    host: current_dir,
520                    guest: if is_wasix {
521                        MAPPED_CURRENT_DIR_DEFAULT_PATH.to_string()
522                    } else {
523                        "/".to_string()
524                    },
525                }
526            } else {
527                if dir == Path::new("/") && is_wasix {
528                    tracing::warn!(
529                        "Pre-opening the root directory with --dir=/ breaks WASIX modules' filesystems"
530                    );
531                }
532
533                let resolved = dir.canonicalize().with_context(|| {
534                    format!(
535                        "could not canonicalize path for argument '--dir {}'",
536                        dir.display()
537                    )
538                })?;
539
540                if &resolved != dir {
541                    bail!(
542                        "Invalid argument '--dir {}': path must either be absolute, or '.'",
543                        dir.display(),
544                    );
545                }
546
547                let guest = resolved
548                    .to_str()
549                    .with_context(|| {
550                        format!(
551                            "invalid argument '--dir {}': path must be valid utf-8",
552                            dir.display(),
553                        )
554                    })?
555                    .to_string();
556
557                MappedDirectory {
558                    host: resolved,
559                    guest,
560                }
561            };
562
563            mapped_dirs.push(mapping);
564        }
565
566        for MappedDirectory { host, guest } in &self.mapped_dirs {
567            if guest == "/" && is_wasix {
568                // Note: it appears we canonicalize the path before this point and showing the value of
569                // `host` in the error message may throw users off, so we use a placeholder.
570                tracing::warn!(
571                    "Mounting on the guest's virtual root with --mapdir /:<HOST_PATH> breaks WASIX modules' filesystems"
572                );
573            }
574            let resolved_host = host.canonicalize().with_context(|| {
575                format!(
576                    "could not canonicalize path for argument '--mapdir {}:{}'",
577                    guest,
578                    host.display(),
579                )
580            })?;
581
582            let mapping = if guest == "." {
583                if have_current_dir {
584                    bail!(
585                        "Cannot pre-open the current directory twice: '--mapdir=?:.' / '--dir=.' must only be specified once"
586                    );
587                }
588                have_current_dir = true;
589
590                MappedDirectory {
591                    host: resolved_host,
592                    guest: if is_wasix {
593                        MAPPED_CURRENT_DIR_DEFAULT_PATH.to_string()
594                    } else {
595                        "/".to_string()
596                    },
597                }
598            } else {
599                MappedDirectory {
600                    host: resolved_host,
601                    guest: guest.clone(),
602                }
603            };
604            mapped_dirs.push(mapping);
605        }
606
607        Ok((have_current_dir, mapped_dirs))
608    }

610    pub fn build_mapped_commands(&self) -> Result<Vec<MappedCommand>, anyhow::Error> {
611        self.map_commands
612            .iter()
613            .map(|item| {
614                let (a, b) = item.split_once('=').with_context(|| {
615                    format!(
616                        "Invalid --map-command flag: expected <ALIAS>=<HOST_PATH>, got '{item}'"
617                    )
618                })?;
619
620                let a = a.trim();
621                let b = b.trim();
622
623                if a.is_empty() {
624                    bail!("Invalid --map-command flag - alias cannot be empty: '{item}'");
625                }
626                // TODO(theduke): check if host command exists, and canonicalize PathBuf.
627                if b.is_empty() {
628                    bail!("Invalid --map-command flag - host path cannot be empty: '{item}'");
629                }
630
631                Ok(MappedCommand {
632                    alias: a.to_string(),
633                    target: b.to_string(),
634                })
635            })
636            .collect::<Result<Vec<_>, anyhow::Error>>()
637    }

653    pub fn prepare_runtime<I>(
654        &self,
655        engine: Engine,
656        env: &WasmerEnv,
657        pkg_cache_path: &Path,
658        rt_or_handle: I,
659        preferred_webc_version: webc::Version,
660    ) -> Result<impl Runtime + Send + Sync + use<I>>
661    where
662        I: Into<RuntimeOrHandle>,
663    {
664        let tokio_task_manager = Arc::new(TokioTaskManager::new(rt_or_handle.into()));
665        let mut rt = PluggableRuntime::new(tokio_task_manager.clone());
666
667        let has_networking = self.networking.is_some()
668            || capabilities::get_cached_capability(pkg_cache_path)
669                .ok()
670                .is_some_and(|v| v.enable_networking);
671
672        let ruleset = self
673            .networking
674            .clone()
675            .flatten()
676            .map(|ruleset| Ruleset::from_str(&ruleset))
677            .transpose()?;
678
679        let network = if let Some(ruleset) = ruleset {
680            virtual_net::host::LocalNetworking::with_ruleset(ruleset)
681        } else {
682            virtual_net::host::LocalNetworking::default()
683        };
684
685        if has_networking {
686            rt.set_networking_implementation(network);
687        } else {
688            let net = super::capabilities::net::AskingNetworking::new(
689                pkg_cache_path.to_path_buf(),
690                Arc::new(network),
691            );
692
693            rt.set_networking_implementation(net);
694        }
695
696        #[cfg(feature = "journal")]
697        {
698            let (r, w) = self.build_journals()?;
699            for journal in r {
700                rt.add_read_only_journal(journal);
701            }
702            for journal in w {
703                rt.add_writable_journal(journal);
704            }
705        }
706
707        if !self.no_tty {
708            let tty = Arc::new(SysTty);
709            tty.reset();
710            rt.set_tty(tty);
711        }
712
713        let client =
714            wasmer_wasix::http::default_http_client().context("No HTTP client available")?;
715        let client = Arc::new(client);
716
717        let package_loader = self
718            .prepare_package_loader(env, client.clone())
719            .context("Unable to prepare the package loader")?;
720
721        let registry = self.prepare_source(env, client, preferred_webc_version)?;
722
723        if !self.disable_cache {
724            let cache_dir = env.cache_dir().join("compiled");
725            let module_cache = wasmer_wasix::runtime::module_cache::in_memory()
726                .with_fallback(FileSystemCache::new(cache_dir, tokio_task_manager));
727            rt.set_module_cache(module_cache);
728        }
729
730        rt.set_package_loader(package_loader)
731            .set_source(registry)
732            .set_engine(engine);
733
734        Ok(rt)
735    }

737    /// Helper function for instantiating a module with Wasi imports for the `Run` command.
738    pub fn instantiate(
739        &self,
740        module: &Module,
741        module_hash: ModuleHash,
742        program_name: String,
743        args: Vec<String>,
744        runtime: Arc<dyn Runtime + Send + Sync>,
745        store: &mut Store,
746    ) -> Result<(WasiFunctionEnv, Instance)> {
747        let builder = self.prepare(module, program_name, args, runtime)?;
748        let (instance, wasi_env) = builder.instantiate_ext(module.clone(), module_hash, store)?;
749
750        Ok((wasi_env, instance))
751    }

765    fn prepare_package_loader(
766        &self,
767        env: &WasmerEnv,
768        client: Arc<dyn HttpClient + Send + Sync>,
769    ) -> Result<BuiltinPackageLoader> {
770        let checkout_dir = env.cache_dir().join("checkouts");
771        let tokens = tokens_by_authority(env)?;
772
773        let loader = BuiltinPackageLoader::new()
774            .with_cache_dir(checkout_dir)
775            .with_shared_http_client(client)
776            .with_tokens(tokens);
777
778        Ok(loader)

781    fn prepare_source(
782        &self,
783        env: &WasmerEnv,
784        client: Arc<dyn HttpClient + Send + Sync>,
785        preferred_webc_version: webc::Version,
786    ) -> Result<MultiSource> {
787        let mut source = MultiSource::default();
788
789        // Note: This should be first so our "preloaded" sources get a chance to
790        // override the main registry.
791        let mut preloaded = InMemorySource::new();
792        for path in &self.include_webcs {
793            preloaded
794                .add_webc(path)
795                .with_context(|| format!("Unable to load \"{}\"", path.display()))?;
796        }
797        source.add_source(preloaded);
798
799        let graphql_endpoint = self.graphql_endpoint(env)?;
800        let cache_dir = env.cache_dir().join("queries");
801        let mut wapm_source = BackendSource::new(graphql_endpoint, Arc::clone(&client))
802            .with_local_cache(cache_dir, WAPM_SOURCE_CACHE_TIMEOUT)
803            .with_preferred_webc_version(preferred_webc_version);
804        if let Some(token) = env
805            .config()?
806            .registry
807            .get_login_token_for_registry(wapm_source.registry_endpoint().as_str())
808        {
809            wapm_source = wapm_source.with_auth_token(token);
810        }
811        source.add_source(wapm_source);
812
813        let cache_dir = env.cache_dir().join("downloads");
814        source.add_source(WebSource::new(cache_dir, client));
815
816        source.add_source(FileSystemSource::default());
817
818        Ok(source)
819    }

821    fn graphql_endpoint(&self, env: &WasmerEnv) -> Result<Url> {
822        if let Ok(endpoint) = env.registry_endpoint() {
823            return Ok(endpoint);
824        }
825
826        let config = env.config()?;
827        let graphql_endpoint = config.registry.get_graphql_url();
828        let graphql_endpoint = graphql_endpoint
829            .parse()
830            .with_context(|| format!("Unable to parse \"{graphql_endpoint}\" as a URL"))?;
831
832        Ok(graphql_endpoint)
833    }

840fn tokens_by_authority(env: &WasmerEnv) -> Result<HashMap<String, String>> {
841    let mut tokens = HashMap::new();
842    let config = env.config()?;
843
844    for credentials in config.registry.tokens {
845        if let Ok(url) = Url::parse(&credentials.registry)
846            && url.has_authority()
847        {
848            tokens.insert(url.authority().to_string(), credentials.token);
849        }
850    }
851
852    if let (Ok(current_registry), Some(token)) = (env.registry_endpoint(), env.token())
853        && current_registry.has_authority()
854    {
855        tokens.insert(current_registry.authority().to_string(), token);
856    }
857
858    // Note: The global wasmer.toml config file stores URLs for the GraphQL
859    // endpoint, however that's often on the backend (i.e.
860    // https://registry.wasmer.io/graphql) and we also want to use the same API
861    // token when sending requests to the frontend (e.g. downloading a package
862    // using the `Accept: application/webc` header).
863    //
864    // As a workaround to avoid needing to query *all* backends to find out
865    // their frontend URL every time the `wasmer` CLI runs, we'll assume that
866    // when a backend is called something like `registry.wasmer.io`, the
867    // frontend will be at `wasmer.io`. This works everywhere except for people
868    // developing the backend locally... Sorry, Ayush.
869
870    let mut frontend_tokens = HashMap::new();
871    for (hostname, token) in &tokens {
872        if let Some(frontend_url) = hostname.strip_prefix("registry.")
873            && !tokens.contains_key(frontend_url)
874        {
875            frontend_tokens.insert(frontend_url.to_string(), token.clone());
876        }
877    }
878    tokens.extend(frontend_tokens);
879
880    Ok(tokens)
881}
