10/// Flags that specify what should be filtered out
11#[derive(Debug, Clone, Copy, PartialOrd, Ord, PartialEq, Eq, Hash)]
12pub enum FilterOut {
13    /// Filters out all the memory events
14    Memory,
15    /// Filters out all the thread stacks
16    Threads,
17    /// Filters out all the file system operations
18    FileSystem,
19    /// Filters out all core syscalls
20    Core,
21    /// Filters out the snapshots
22    Snapshots,
23    /// Filters out the networking
24    Networking,
25}
26
27impl FromStr for FilterOut {
28    type Err = String;
29
30    fn from_str(s: &str) -> Result<Self, Self::Err> {
31        Ok(match s.to_lowercase().as_str() {
32            "mem" | "memory" => Self::Memory,
33            "thread" | "threads" => Self::Threads,
34            "fs" | "file" | "filesystem" | "file-system" => Self::FileSystem,
35            "core" => Self::Core,
36            "snap" | "snapshot" | "snapshots" => Self::Snapshots,
37            "net" | "network" | "networking" => Self::Networking,
38            t => return Err(format!("unknown filter type - {t}")),
39        })
40    }
41}

43/// Rewrites a journal log removing events that match the
44/// filter parameters.
45#[derive(Debug, Parser)]
46pub struct CmdJournalFilter {
47    /// Path to the journal that will be read
48    #[clap(index = 1)]
49    source_path: PathBuf,
50    /// Path to the journal that will be the output of the filter
51    #[clap(index = 2)]
52    target_path: PathBuf,
53    /// Filters to be applied to the output journal, filter options are
54    /// - 'mem' | 'memory' -> removes all WASM memory related events
55    /// - 'thread' | 'threads' -> removes all events related to the state of the threads
56    /// - 'fs' | 'file' -> removes file system mutation events
57    /// - 'core' -> removes core operating system operations such as TTY
58    /// - 'snap' | 'snapshot' -> removes the snapshots from the journal
59    /// - 'net' | 'network' -> removes network socket and interface events
60    #[clap(short, long = "filter")]
61    filters: Vec<FilterOut>,
62}
63
64impl CliCommand for CmdJournalFilter {
65    type Output = ();
66
67    fn run(self) -> Result<(), anyhow::Error> {
68        // Create a new file name that will be the temp new file
69        // while its written
70        let mut temp_filename = self
71            .target_path
72            .file_name()
73            .ok_or_else(|| {
74                anyhow::format_err!(
75                    "The path is not a valid filename - {}",
76                    self.target_path.to_string_lossy()
77                )
78            })?
79            .to_string_lossy()
80            .to_string();
81        temp_filename.insert_str(0, ".staging.");
82        let temp_path = self.target_path.with_file_name(&temp_filename);
83        std::fs::remove_file(&temp_path).ok();
84
85        // Load the source journal and the target journal (in the temp location)
86        let source = LogFileJournal::new(self.source_path)?;
87        let target = LogFileJournal::new(temp_path.clone())?;
88
89        // Put a filter on the farget
90        let mut builder = FilteredJournalBuilder::new();
91        for filter in self.filters {
92            builder = match filter {
93                FilterOut::Memory => builder.with_ignore_memory(true),
94                FilterOut::Threads => builder.with_ignore_threads(true),
95                FilterOut::FileSystem => builder.with_ignore_fs(true),
96                FilterOut::Core => builder.with_ignore_core(true),
97                FilterOut::Snapshots => builder.with_ignore_snapshots(true),
98                FilterOut::Networking => builder.with_ignore_networking(true),
99            }
100        }
101        let target = builder.build(target);
102
103        // Copy the journal over and rename the temp file to the target file
104        copy_journal(&source, &target)?;
105        drop(target);
106        std::fs::rename(temp_path, self.target_path.clone())?;
107
108        // Now print the outcome
109        let journal = LogFileJournal::new(&self.target_path)?;
110        let printer = PrintingJournal::default();
111        copy_journal(&journal, &printer)?;
112        Ok(())
113    }
114}
