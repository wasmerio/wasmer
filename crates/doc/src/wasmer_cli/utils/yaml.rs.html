1//! Format-preserving YAML editing for `app.yaml`.
2//!
3//! Serializing a `serde_yaml::Value` back to a string drops the user's
4//! formatting (comments, key order, blank lines, quoting).
5//! [`apply_app_config_to_yaml`] instead edits the original text in place with
6//! [`yaml_edit`], rewriting only the nodes whose value changed.
7
8use std::str::FromStr;
9
10use anyhow::Context as _;
11use serde_yaml::Value;
12use yaml_edit::{Document, Mapping, YamlNode};
13
14/// Apply `target` onto the original app YAML `text`, preserving the formatting
15/// of everything that did not change.
16///
17/// * A node is rewritten only when its value differs from `target`; comments,
18///   order, blank lines and quoting are otherwise untouched.
19/// * Top-level keys are synced (added, updated, or removed); nested mappings are
20///   merged without removing keys. See [`merge_into_mapping`] for why.
21/// * A `null` in `target` is never added (avoids `name: null` noise); an
22///   existing key can still be set to it.
23pub(crate) fn apply_app_config_to_yaml(text: &str, target: &Value) -> anyhow::Result<String> {
24    let doc = Document::from_str(text)
25        .map_err(|e| anyhow::anyhow!("could not parse YAML for format-preserving edit: {e}"))?;
26
27    match (doc.as_mapping(), target) {
28        (Some(mapping), Value::Mapping(target_mapping)) => {
29            merge_into_mapping(&mapping, target_mapping, true)?;
30            let out = doc.to_string();
31            // Restore the leading comment block that `yaml_edit` drops (see
32            // `leading_trivia`).
33            let header = leading_trivia(text);
34            if !header.is_empty() && !out.starts_with(header) {
35                Ok(format!("{header}{out}"))
36            } else {
37                Ok(out)
38            }
39        }
40        // The document root is not a mapping (or the target is not a mapping).
41        // We have no formatting to preserve in a meaningful way, so fall back to
42        // a plain serialization of the target.
43        _ => Ok(serde_yaml::to_string(target)?),
44    }
45}

47/// Recursively merge `target` into `mapping`. `remove_missing` (true only at the
48/// top level) drops keys absent from `target`.
49///
50/// Only the top level removes, because `AppConfigV1`'s `#[serde(flatten)] extra`
51/// re-emits unknown top-level keys: a key missing from `target` there was cleared
52/// on purpose (e.g. `app_id`/`name` on owner change). Nested sub-structs have no
53/// such catch-all and drop fields they don't model, so removing nested keys would
54/// delete the user's forward-compatible settings.
55fn merge_into_mapping(
56    mapping: &Mapping,
57    target: &serde_yaml::Mapping,
58    remove_missing: bool,
59) -> anyhow::Result<()> {
60    for (key, value) in target {
61        let Some(key) = scalar_key(key) else {
62            continue;
63        };
64
65        if let Some(existing) = mapping.get(&key) {
66            if node_matches(&existing.to_string(), value) {
67                // Semantically identical - keep the original formatting.
68                continue;
69            }
70
71            // Recurse so we only touch what changed within the nested mapping.
72            if let (Some(child), Value::Mapping(target_child)) = (mapping.get_mapping(&key), value)
73            {
74                merge_into_mapping(&child, target_child, false)?;
75            } else {
76                set_value(mapping, &key, value)
77                    .with_context(|| format!("could not update app YAML key `{key}`"))?;
78            }
79        } else if !value.is_null() {
80            // Don't introduce `key: null` noise for absent keys.
81            set_value(mapping, &key, value)
82                .with_context(|| format!("could not add app YAML key `{key}`"))?;
83        }
84    }
85
86    if remove_missing {
87        for key in stale_top_level_keys(mapping, target) {
88            mapping.remove(key.as_str());
89        }
90    }
91
92    Ok(())
93}

95/// Set `mapping[key] = value`, dispatching scalars to typed setters and
96/// structured values (mappings/sequences) to a parsed node so block style is
97/// preserved.
98fn set_value(mapping: &Mapping, key: &str, value: &Value) -> anyhow::Result<()> {
99    match value {
100        Value::Null => set_via_node(mapping, key, value)?,
101        Value::Bool(b) => mapping.set(key, *b),
102        Value::String(s) => mapping.set(key, s.as_str()),
103        Value::Number(n) => {
104            if let Some(i) = n.as_i64() {
105                mapping.set(key, i);
106            } else if let Some(u) = n.as_u64() {
107                mapping.set(key, u);
108            } else if let Some(f) = n.as_f64() {
109                mapping.set(key, f);
110            } else {
111                set_via_node(mapping, key, value)?;
112            }
113        }
114        Value::Sequence(_) | Value::Mapping(_) | Value::Tagged(_) => {
115            set_via_node(mapping, key, value)?;
116        }
117    }
118
119    Ok(())
120}

122/// Set a structured value by serializing it and parsing it back into a node, so
123/// `yaml_edit` reconstructs the proper block/flow representation.
124fn set_via_node(mapping: &Mapping, key: &str, value: &Value) -> anyhow::Result<()> {
125    // There is no direct cast from `serde_yaml::Value` to a `yaml_edit` node, and
126    // a recursive `AsYaml` converter would be more code and easier to get wrong
127    // than this. Serialize the value under a temporary key, parse it back, and
128    // graft the resulting node.
129    let mut wrapper = serde_yaml::Mapping::new();
130    wrapper.insert(Value::String(key.to_string()), value.clone());
131    let rendered = serde_yaml::to_string(&Value::Mapping(wrapper))
132        .context("could not serialize app YAML value")?;
133
134    let node = Document::from_str(&rendered)
135        .map_err(|e| anyhow::anyhow!("could not parse serialized app YAML value: {e}"))?
136        .as_mapping()
137        .and_then(|m| m.get(key))
138        .context("serialized app YAML value did not contain the expected key")?;
139
140    mapping.set(key, node);
141    Ok(())
142}

144/// Whether the rendered text of an existing node is semantically equal to
145/// `target` (ignoring formatting/quoting differences).
146fn node_matches(node_text: &str, target: &Value) -> bool {
147    serde_yaml::from_str::<Value>(node_text)
148        .map(|v| &v == target)
149        .unwrap_or(false)
150}

152/// Return the leading run of blank and comment-only lines at the top of `text`.
153///
154/// `yaml_edit` has a bug that drops this block (as of 0.2), so we
155/// capture it here to splice back after editing.
156fn leading_trivia(text: &str) -> &str {
157    let mut end = 0;
158    for line in text.split_inclusive('\n') {
159        let trimmed = line.trim_start();
160        if trimmed.is_empty() || trimmed.starts_with('#') {
161            end += line.len();
162        } else {
163            break;
164        }
165    }
166    &text[..end]
167}

169fn stale_top_level_keys(mapping: &Mapping, target: &serde_yaml::Mapping) -> Vec<String> {
170    mapping
171        .entries()
172        .filter_map(|entry| {
173            let key = entry.key_node()?;
174            let is_target_key = target
175                .keys()
176                .filter_map(scalar_key)
177                .any(|target_key| key.yaml_eq(&target_key));
178            if is_target_key {
179                None
180            } else {
181                scalar_key_node(&key)
182            }
183        })
184        .collect()
185}
186
187/// Extract a scalar mapping key as a string. Non-scalar keys
188/// are skipped.
189fn scalar_key(key: &Value) -> Option<String> {
190    match key {
191        Value::String(s) => Some(s.clone()),
192        Value::Bool(b) => Some(b.to_string()),
193        Value::Number(n) => Some(n.to_string()),
194        _ => None,
195    }
196}
197
198fn scalar_key_node(key: &YamlNode) -> Option<String> {
199    key.as_scalar().map(|s| s.as_string())
200}

202/// Convenience wrapper: read the original text from `path`, apply `target`
203/// onto it, and return the format-preserved result. If the file does not exist
204/// or cannot be read, falls back to a plain serialization of `target`.
205pub(crate) fn apply_app_config_to_yaml_file(
206    path: &std::path::Path,
207    target: &Value,
208) -> anyhow::Result<String> {
209    match std::fs::read_to_string(path) {
210        Ok(text) => apply_app_config_to_yaml(&text, target)
211            .with_context(|| format!("could not edit YAML file '{}'", path.display())),
212        Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(serde_yaml::to_string(target)?),
213        Err(e) => Err(e).with_context(|| format!("could not read YAML file '{}'", path.display())),
214    }
215}

331    /// End-to-end check of the real deploy flow: parse a commented `app.yaml`
332    /// into an [`AppConfigV1`], mutate a field the way `wasmer app deploy` does
333    /// (assign the backend `app_id`), serialize it with `to_yaml_value`, and
334    /// apply it back. The backend assignment must land while every comment,
335    /// blank line, key order and nested structure is left intact.
336    #[test]
337    fn deploy_flow_preserves_formatting_when_assigning_app_id() {
338        use wasmer_config::app::AppConfigV1;
339
340        let original = r#"# Wasmer app configuration
341kind: wasmer.io/App.v0
342name: my-cool-app  # human readable name
343owner: alice
344package: .
345
346# how much memory the app gets
347capabilities:
348  memory:
349    limit: 512MB
350
351env:
352  LOG_LEVEL: debug
353"#;
354
355        // Parse exactly like the CLI does.
356        let mut config = AppConfigV1::parse_yaml(original).unwrap();
357        assert!(config.app_id.is_none());
358
359        // The backend assigns an id on first deploy; this is the only change.
360        config.app_id = Some("da_abc123".to_string());
361
362        let target = config.clone().to_yaml_value().unwrap();
363        let out = apply_app_config_to_yaml(original, &target).unwrap();
364
365        // The new field is added (appended, not reordered into the middle)...
366        assert!(out.contains("app_id: da_abc123"), "app_id missing:\n{out}");
367
368        // ...and all comments, ordering, blank lines and nesting survive.
369        assert!(out.starts_with("# Wasmer app configuration\n"), "{out}");
370        assert!(
371            out.contains("name: my-cool-app  # human readable name"),
372            "{out}"
373        );
374        assert!(out.contains("# how much memory the app gets"), "{out}");
375        assert!(out.contains("  memory:\n    limit: "), "{out}");
376        assert!(out.contains("env:\n  LOG_LEVEL: debug"), "{out}");
377        // The blank lines between sections are kept.
378        assert!(out.contains("package: .\n\n"), "{out}");
379
380        // The `limit` node is the one exception: `ByteSize` is normalized when it
381        // round-trips through the typed struct (`512MB` -> `488.3 MiB`, the same
382        // amount). Only that node changes; the surrounding formatting does not.
383        assert!(!out.contains("512MB"), "{out}");
384        assert!(out.contains("488.3 MiB"), "{out}");
385
386        // The result is still a valid app config with the expected fields.
387        // (We don't assert full struct equality because `ByteSize`'s own
388        // display round-trip is mildly lossy: 512MB -> "488.3 MiB" -> a value a
389        // few hundred bytes off. That predates this change.)
390        let reparsed = AppConfigV1::parse_yaml(&out).unwrap();
391        assert_eq!(reparsed.app_id.as_deref(), Some("da_abc123"));
392        assert_eq!(reparsed.name, config.name);
393        assert_eq!(reparsed.owner, config.owner);
394        assert_eq!(reparsed.env, config.env);
395    }
