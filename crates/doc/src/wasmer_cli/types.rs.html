11/// Render the full name (`namespace/name`) of a package version's package.
12fn package_full_name(pv: &SearchPackageVersion) -> String {
13    match &pv.package.namespace {
14        Some(ns) => format!("{ns}/{}", pv.package.package_name),
15        None => pv.package.package_name.clone(),
16    }
17}

19fn cron_job_target_summary(target: &CronJobTarget) -> String {
20    match target {
21        CronJobTarget::FetchCronJobTarget(target) => {
22            format!("{} {}", target.method, target.path)
23        }
24        CronJobTarget::ExecuteCronJobTarget(target) => {
25            let package = target.package_name.as_deref().unwrap_or("-");
26            let command = target.command.as_deref().unwrap_or("-");
27            format!("{package} {command}")
28        }
29        CronJobTarget::Unknown => "unknown".to_string(),
30    }
31}
32
33fn cron_job_kind(kind: CronJobKind) -> &'static str {
34    match kind {
35        CronJobKind::Fetch => "fetch",
36        CronJobKind::Execute => "execute",
37    }
38}
39
40fn cron_invocation_result_summary(result: &Option<CronJobInvocationResult>) -> String {
41    match result {
42        Some(CronJobInvocationResult::ExecuteCronJobInvocationResult(result)) => result
43            .exit_code
44            .map(|code| format!("exit_code={code}"))
45            .unwrap_or_else(|| "execute".to_string()),
46        Some(CronJobInvocationResult::FetchCronJobInvocationResult(result)) => result
47            .status_code
48            .map(|code| format!("status_code={code}"))
49            .unwrap_or_else(|| "fetch".to_string()),
50        Some(CronJobInvocationResult::Unknown) => "unknown".to_string(),
51        None => "-".to_string(),
52    }
53}

197impl CliRender for CronJobLog {
198    fn render_item_table(&self) -> String {
199        let mut table = Table::new();
200        table.add_rows([
201            vec!["Timestamp".to_string(), self.datetime.0.clone()],
202            vec!["Message".to_string(), self.message.clone()],
203        ]);
204        table.to_string()
205    }
206
207    fn render_list_table(items: &[Self]) -> String {
208        let mut table = Table::new();
209        table.load_preset(comfy_table::presets::NOTHING);
210        table.set_content_arrangement(comfy_table::ContentArrangement::Dynamic);
211
212        for item in items {
213            let message = match item.stream {
214                Some(LogStream::Stderr) => item.message.clone().yellow(),
215                Some(LogStream::Runtime) => item.message.clone().cyan(),
216                Some(LogStream::Stdout) | None => item.message.clone().bold(),
217            };
218            table.add_row([
219                comfy_table::Cell::new(format!("[{}]", item.datetime.0))
220                    .set_alignment(comfy_table::CellAlignment::Right),
221                comfy_table::Cell::new(message),
222            ]);
223        }
224        table.to_string()
225    }
226}

279impl CliRender for DnsDomainWithRecords {
280    fn render_item_table(&self) -> String {
281        let mut output = String::new();
282        let mut table = Table::new();
283        table
284            .load_preset(comfy_table::presets::UTF8_FULL_CONDENSED)
285            .set_header(vec![
286                "Type".to_string(),
287                "Name".to_string(),
288                "TTL".to_string(),
289                "Value".to_string(),
290            ]);
291        let mut rows: Vec<Vec<String>> = vec![];
292        if let Some(ref records) = self.records {
293            records.iter().flatten().for_each(|record| {
294                rows.push(vec![
295                    record.record_type().to_string(),
296                    record.name().unwrap_or("<no name>").to_string(),
297                    record
298                        .ttl()
299                        .expect("expected a TTL value for record")
300                        .to_string(),
301                    record.text().to_string(),
302                ]);
303            });
304        }
305
306        table.add_rows(rows);
307        output += &table.to_string();
308        output
309    }
310
311    fn render_list_table(items: &[Self]) -> String {
312        if items.is_empty() {
313            return String::new();
314        }
315        let mut table = Table::new();
316        table.set_header(vec!["Domain".to_string()]);
317        table.add_rows(items.iter().map(|ns| vec![ns.name.clone()]));
318        table.to_string()
319    }

490fn format_disk_size_opt(value: Option<wasmer_backend_api::types::BigInt>) -> String {
491    let value = value.and_then(|x| {
492        let y: Option<u64> = x.0.try_into().ok();
493        y
494    });
495
496    if let Some(v) = value {
497        let s = bytesize::ByteSize(v);
498        s.to_string()
499    } else {
500        "n/a".to_string()
501    }
502}

504impl CliRender for Deployment {
505    fn render_item_table(&self) -> String {
506        match self {
507            Deployment::NakedDeployment(naked) => naked.render_item_table(),
508            Deployment::AutobuildRepository(build) => build.render_item_table(),
509            Deployment::Other => "unknown deployment type".to_string(),
510        }
511    }
512
513    fn render_list_table(items: &[Self]) -> String {
514        let mut table = Table::new();
515        table.set_header(vec![
516            "Id".to_string(),
517            "Type".to_string(),
518            "Created at".to_string(),
519            "Status".to_string(),
520            "App version".to_string(),
521        ]);
522
523        let rows = items
524            .iter()
525            .map(|item| match item {
526                Deployment::NakedDeployment(naked) => {
527                    vec![
528                        naked.id.inner().to_string(),
529                        "Manual".to_string(),
530                        naked.created_at.0.clone(),
531                        String::new(),
532                        naked
533                            .app_version
534                            .as_ref()
535                            .map_or_else(|| "n/a".to_string(), |x| x.version.clone()),
536                    ]
537                }
538                Deployment::AutobuildRepository(build) => {
539                    vec![
540                        build.id.inner().to_string(),
541                        "Autobuild".to_string(),
542                        build.status.as_str().to_string(),
543                        build.created_at.0.clone(),
544                    ]
545                }
546                Deployment::Other => vec![
547                    String::new(),
548                    "Unknown".to_string(),
549                    String::new(),
550                    String::new(),
551                ],
552            })
553            .collect::<Vec<_>>();
554        table.add_rows(rows);
555
556        table.to_string()
557    }
558}
