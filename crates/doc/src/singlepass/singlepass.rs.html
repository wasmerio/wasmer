3use libfuzzer_sys::fuzz_target;
4mod misc;
5use misc::{SinglePassFuzzModule, ignore_compilation_error, ignore_runtime_error, save_wasm_file};
6use wasmer::{Instance, Module, Store, imports};
7use wasmer_compiler::EngineBuilder;
8use wasmer_compiler_singlepass::Singlepass;
9
10fuzz_target!(|module: SinglePassFuzzModule| {
11    let wasm_bytes = module.0.to_bytes();
12
13    let compiler = Singlepass::default();
14    let mut store = Store::new(EngineBuilder::new(compiler));
15    // Save early (and always) as we might hit a crash or a validation error.
16    save_wasm_file(&wasm_bytes);
17    let module = Module::new(&store, &wasm_bytes);
18    let module = match module {
19        Ok(m) => m,
20        Err(e) => {
21            if ignore_compilation_error(&e.to_string()) {
22                return;
23            }
24            panic!("{}", e);
25        }
26    };
27
28    match Instance::new(&mut store, &module, &imports! {}) {
29        Ok(_) => {}
30        Err(e) => {
31            if ignore_runtime_error(&e.to_string()) {
32                return;
33            }
34            panic!("{}", e);
35        }
36    }
37});
