3use libfuzzer_sys::{arbitrary::Arbitrary, fuzz_target};
4mod misc;
5use misc::{ignore_compilation_error, ignore_runtime_error, save_wasm_file};

10struct LLVMPassFuzzModule(wasm_smith::Module);
11
12impl Arbitrary<'_> for LLVMPassFuzzModule {
13    fn arbitrary(
14        u: &mut libfuzzer_sys::arbitrary::Unstructured,
15    ) -> libfuzzer_sys::arbitrary::Result<Self> {
16        let mut config = wasm_smith::Config::arbitrary(u)?;
17        config.min_imports = 0;
18        config.max_imports = 0;
19        config.max_memory32_bytes = 65535 * 4096;
20        config.min_funcs = 1;
21        config.max_funcs = std::cmp::max(config.min_funcs, config.max_funcs);
22        config.min_exports = 1;
23        config.max_exports = std::cmp::max(config.min_exports, config.max_exports);
24        config.gc_enabled = false;
25        config.memory64_enabled = false;
26        config.max_memories = 1;
27        config.tail_call_enabled = false;
28        config.relaxed_simd_enabled = true;
29        config.wide_arithmetic_enabled = true;
30        config.extended_const_enabled = true;
31        config.tail_call_enabled = true;
32        Ok(Self(wasm_smith::Module::new(config, u)?))
33    }
34}
35
36impl std::fmt::Debug for LLVMPassFuzzModule {
37    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
38        f.write_str(&wasmprinter::print_bytes(self.0.to_bytes()).unwrap())
39    }
40}

42fuzz_target!(|module: LLVMPassFuzzModule| {
43    let wasm_bytes = module.0.to_bytes();
44
45    let mut compiler = LLVM::default();
46    compiler.canonicalize_nans(true);
47    compiler.enable_verifier();
48    let mut store = Store::new(EngineBuilder::new(compiler));
49    // Save early (and always) as we might hit a crash or a validation error in the LLVM library.
50    save_wasm_file(&wasm_bytes);
51
52    let module = match Module::new(&store, &wasm_bytes) {
53        Err(e) => {
54            if ignore_compilation_error(&e.to_string()) {
55                return;
56            }
57            panic!("{}", e);
58        }
59        Ok(module) => module,
60    };
61
62    match Instance::new(&mut store, &module, &imports! {}) {
63        Ok(_) => {}
64        Err(e) => {
65            if ignore_runtime_error(&e.to_string()) {
66                return;
67            }
68            panic!("{}", e);
69        }
70    }
71});
