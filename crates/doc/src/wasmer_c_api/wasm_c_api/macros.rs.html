60                // Note that we're careful to not create a slice with a null

62                // behavior in Rust.
63                if self.size == 0 {
64                    &[]
65                } else {
66                    assert!(!self.data.is_null());
67                    unsafe { std::slice::from_raw_parts(self.data, self.size) }
68                }

77                } else {

299macro_rules! c_try {
300    ($expr:expr; otherwise ()) => {{
301        let res: Result<_, _> = $expr;
302        match res {
303            Ok(val) => val,
304            Err(err) => {
305                crate::error::update_last_error(err);
306                return;
307            }
308        }
309    }};
310    ($expr:expr; otherwise $return:expr) => {{
311        let res: Result<_, _> = $expr;
312        match res {
313            Ok(val) => val,
314            Err(err) => {
315                crate::error::update_last_error(err);
316                return $return;
317            }
318        }
319    }};
320    ($expr:expr) => {{
321        c_try!($expr; otherwise None)
322    }};
323    ($expr:expr, $e:expr) => {{
324        let opt: Option<_> = $expr;
325        c_try!(opt.ok_or_else(|| $e))
326    }};
