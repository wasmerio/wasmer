6pub(crate) struct WasmFunctionType {
7    pub(crate) function_type: FunctionType,
8    params: wasm_valtype_vec_t,
9    results: wasm_valtype_vec_t,
10}
11
12impl WasmFunctionType {
13    pub(crate) fn new(function_type: FunctionType) -> Self {
14        let params: Vec<_> = function_type
15            .params()
16            .iter()
17            .map(|&valtype| Some(Box::new(valtype.into())))
18            .collect();
19        let results: Vec<_> = function_type
20            .results()
21            .iter()
22            .map(|&valtype| Some(Box::new(valtype.into())))
23            .collect();
24
25        Self {
26            function_type,
27            params: params.into(),
28            results: results.into(),
29        }
30    }
31}
32
33impl Clone for WasmFunctionType {
34    fn clone(&self) -> Self {

36    }
37}

40    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {

46#[derive(Debug, Clone)]
47#[repr(transparent)]
48pub struct wasm_functype_t {
49    pub(crate) extern_type: wasm_externtype_t,
50}
51
52impl wasm_functype_t {
53    pub(crate) fn new(function_type: FunctionType) -> Self {
54        Self {
55            extern_type: wasm_externtype_t::new(ExternType::Function(function_type)),
56        }
57    }
58
59    pub(crate) fn inner(&self) -> &WasmFunctionType {
60        match &self.extern_type.inner {
61            WasmExternType::Function(wasm_function_type) => wasm_function_type,
62            _ => {
63                unreachable!("Data corruption: `wasm_functype_t` does not contain a function type")
64            }
65        }
66    }
67}

72#[unsafe(no_mangle)]
73pub unsafe extern "C" fn wasm_functype_new(
74    params: Option<&mut wasm_valtype_vec_t>,
75    results: Option<&mut wasm_valtype_vec_t>,
76) -> Option<Box<wasm_functype_t>> {
77    let params = params?;
78    let results = results?;
79
80    let params_as_valtype: Vec<Type> = params
81        .take()
82        .into_iter()
83        .map(|val| val.as_ref().unwrap().as_ref().into())
84        .collect::<Vec<_>>();
85    let results_as_valtype: Vec<Type> = results
86        .take()
87        .into_iter()
88        .map(|val| val.as_ref().unwrap().as_ref().into())
89        .collect::<Vec<_>>();
90
91    Some(Box::new(wasm_functype_t::new(FunctionType::new(
92        params_as_valtype,
93        results_as_valtype,
94    ))))
95}
