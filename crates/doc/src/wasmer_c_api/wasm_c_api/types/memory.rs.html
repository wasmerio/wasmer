4#[derive(Debug, Clone)]
5pub(crate) struct WasmMemoryType {
6    pub(crate) memory_type: MemoryType,
7    limits: wasm_limits_t,
8}
9
10impl WasmMemoryType {
11    pub(crate) fn new(memory_type: MemoryType) -> Self {
12        let limits = wasm_limits_t {
13            min: memory_type.minimum.0 as _,
14            max: memory_type
15                .maximum
16                .map(|max| max.0 as _)
17                .unwrap_or(LIMITS_MAX_SENTINEL),
18        };
19
20        Self {
21            memory_type,
22            limits,
23        }
24    }
25}

28#[derive(Debug, Clone)]
29#[repr(transparent)]
30pub struct wasm_memorytype_t {
31    pub(crate) extern_type: wasm_externtype_t,
32}
33
34impl wasm_memorytype_t {
35    pub(crate) fn new(memory_type: MemoryType) -> Self {
36        Self {
37            extern_type: wasm_externtype_t::new(ExternType::Memory(memory_type)),
38        }
39    }
40
41    pub(crate) fn inner(&self) -> &WasmMemoryType {
42        match &self.extern_type.inner {
43            WasmExternType::Memory(wasm_memory_type) => wasm_memory_type,
44            _ => {
45                unreachable!("Data corruption: `wasm_memorytype_t` does not contain a memory type")
46            }
47        }
48    }
49}

53#[unsafe(no_mangle)]
54pub unsafe extern "C" fn wasm_memorytype_new(limits: &wasm_limits_t) -> Box<wasm_memorytype_t> {
55    let min_pages = Pages(limits.min as _);
56    let max_pages = if limits.max == LIMITS_MAX_SENTINEL {
57        None
58    } else {
59        Some(Pages(limits.max as _))
60    };
61
62    Box::new(wasm_memorytype_t::new(MemoryType::new(
63        min_pages, max_pages, false,
64    )))
65}
