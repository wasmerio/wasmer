8pub enum wasm_valkind_enum {

10    WASM_I64 = 1,
11    WASM_F32 = 2,
12    WASM_F64 = 3,

14    WASM_FUNCREF = 129,

18impl From<Type> for wasm_valkind_enum {
19    fn from(other: Type) -> Self {
20        match other {
21            Type::I32 => Self::WASM_I32,
22            Type::I64 => Self::WASM_I64,
23            Type::F32 => Self::WASM_F32,
24            Type::F64 => Self::WASM_F64,
25            Type::V128 => todo!("no v128 type in Wasm C API yet!"),
26            Type::ExternRef => Self::WASM_EXTERNREF,
27            Type::FuncRef => Self::WASM_FUNCREF,
28            Type::ExceptionRef => Self::WASM_EXNREF,
29        }
30    }
31}

33impl From<wasm_valkind_enum> for Type {
34    fn from(other: wasm_valkind_enum) -> Self {
35        use wasm_valkind_enum::*;
36        match other {
37            WASM_I32 => Type::I32,
38            WASM_I64 => Type::I64,
39            WASM_F32 => Type::F32,
40            WASM_F64 => Type::F64,
41            WASM_EXTERNREF => Type::ExternRef,

43            WASM_EXNREF => Type::ExternRef,
44        }
45    }
46}
