9const LIMITS_MAX_SENTINEL: u32 = u32::MAX;

11#[derive(Debug, Clone)]
12pub(crate) struct WasmTableType {
13    pub(crate) _table_type: TableType,
14    limits: wasm_limits_t,
15    content: wasm_valtype_t,
16}
17
18impl WasmTableType {
19    pub(crate) fn new(table_type: TableType) -> Self {
20        let limits = wasm_limits_t {
21            min: table_type.minimum as _,
22            max: table_type.maximum.unwrap_or(LIMITS_MAX_SENTINEL),
23        };
24        let content = table_type.ty.into();
25
26        Self {
27            _table_type: table_type,
28            limits,
29            content,
30        }
31    }
32}

34#[allow(non_camel_case_types)]
35#[derive(Debug, Clone)]
36#[repr(transparent)]
37pub struct wasm_tabletype_t {
38    pub(crate) extern_type: wasm_externtype_t,
39}
40
41impl wasm_tabletype_t {
42    pub(crate) fn new(table_type: TableType) -> Self {
43        Self {
44            extern_type: wasm_externtype_t::new(ExternType::Table(table_type)),
45        }
46    }
47
48    pub(crate) fn inner(&self) -> &WasmTableType {
49        match &self.extern_type.inner {
50            WasmExternType::Table(wasm_table_type) => wasm_table_type,
51            _ => unreachable!("Data corruption: `wasm_tabletype_t` does not contain a table type"),
52        }
53    }
54}

58#[unsafe(no_mangle)]
59pub unsafe extern "C" fn wasm_tabletype_new(
60    valtype: Option<Box<wasm_valtype_t>>,
61    limits: &wasm_limits_t,
62) -> Option<Box<wasm_tabletype_t>> {
63    let valtype = valtype?;
64    let max_elements = if limits.max == LIMITS_MAX_SENTINEL {
65        None
66    } else {
67        Some(limits.max as _)
68    };
69    let table_type = Box::new(wasm_tabletype_t::new(TableType::new(
70        (*valtype).into(),
71        limits.min as _,
72        max_elements,
73    )));
74
75    unsafe {
76        wasm_valtype_delete(Some(valtype));
77    }
78
79    Some(table_type)
80}
