1//! Implementation of the `wasm_ref_t` reference surface of the wasm-c-api.
2//!
3//! A `wasm_ref_t` is a boxed `{ Value, store handle }` mirroring
4//! [`wasm_extern_t`], carrying either an `externref` or a `funcref`. A `NULL`
5//! `wasm_ref_t*` is the null reference; non-null references are always the
6//! `Some(_)` variant.
7//!
8//! NOTE: references here are non-owning. The underlying extern object lives in
9//! the store arena until the store is dropped, so every `wasm_ref_t` (and its
10//! host info) leaks until store death. Reference lifetime management (a proper
11//! reference GC) is a planned follow-up.
12
13use std::cell::Cell;
14use std::os::raw::c_void;
15
16use wasmer_api::{ExternRef, Value};
17
18use super::super::store::{StoreRef, WeakStoreRef, wasm_store_t};
19
20/// Host-info payload carried inside a foreign [`ExternRef`].
21///
22/// Stored as the `Any` payload of the `ExternRef` so that every `wasm_ref_t`
23/// handle (and every copy) pointing at the same extern object observes the same
24/// host info — no store-side map is needed.
25pub(crate) struct Foreign {
26    host_info: Cell<*mut c_void>,
27    finalizer: Cell<Option<extern "C" fn(*mut c_void)>>,
28}
29
30// SAFETY: wasm-c-api stores are single-threaded (a documented invariant of the
31// C API), and `externref`/`exnref` are not shareable across threads. The raw
32// host-info pointer and finalizer are therefore never accessed concurrently.
33unsafe impl Send for Foreign {}
34unsafe impl Sync for Foreign {}
35
36impl Foreign {
37    fn new() -> Self {
38        Self {
39            host_info: Cell::new(std::ptr::null_mut()),
40            finalizer: Cell::new(None),
41        }
42    }
43}
44
45impl Drop for Foreign {
46    fn drop(&mut self) {
47        if let Some(finalizer) = self.finalizer.get() {
48            finalizer(self.host_info.get());
49        }
50    }
51}

53/// A boxed WebAssembly reference (`externref` or `funcref`).
54///
55/// The store handle is [`WeakStoreRef`], not a strong `StoreRef`: some
56/// references (e.g. from `wasm_func_as_ref`) are never freed by the embedder,
57/// and a strong handle would pin the whole store. Operations upgrade the handle
58/// and safely no-op if the store has already been dropped.
59#[allow(non_camel_case_types)]
60#[derive(Clone)]
61pub struct wasm_ref_t {
62    pub(crate) inner: Value,
63    pub(crate) store: WeakStoreRef,
64}
65
66impl wasm_ref_t {
67    /// Box a reference value. Returns `None` (i.e. the null `wasm_ref_t*`) for
68    /// null references and non-reference values.
69    pub(crate) fn new(store: StoreRef, inner: Value) -> Option<Box<wasm_ref_t>> {
70        match &inner {
71            Value::ExternRef(Some(_)) | Value::FuncRef(Some(_)) => Some(Box::new(wasm_ref_t {
72                inner,
73                store: store.downgrade(),
74            })),
75            _ => None,
76        }
77    }
78}
79
80/// Runs `f` against the [`Foreign`] payload of `ref_`, if it carries one and
81/// the owning store is still alive.
82fn with_foreign<R>(ref_: &wasm_ref_t, f: impl FnOnce(&Foreign) -> R) -> Option<R> {
83    match &ref_.inner {
84        Value::ExternRef(Some(e)) => {
85            let store = ref_.store.upgrade()?;
86            let store_ref = unsafe { store.store() };
87            e.downcast::<Foreign>(&store_ref).map(f)
88        }
89        _ => None,
90    }
91}
92
93/// Whether two reference values point at the same underlying object.
94pub(crate) fn refs_same(a: &Value, b: &Value) -> bool {
95    match (a, b) {
96        (Value::ExternRef(Some(x)), Value::ExternRef(Some(y))) => x.ptr_eq(y),
97        (Value::ExternRef(None), Value::ExternRef(None)) => true,
98        (Value::FuncRef(Some(x)), Value::FuncRef(Some(y))) => x == y,
99        (Value::FuncRef(None), Value::FuncRef(None)) => true,
100        _ => false,
101    }
102}

108#[unsafe(no_mangle)]
109pub unsafe extern "C" fn wasm_ref_delete(_ref: Option<Box<wasm_ref_t>>) {}
110
111#[unsafe(no_mangle)]
112pub unsafe extern "C" fn wasm_ref_copy(ref_: Option<&wasm_ref_t>) -> Option<Box<wasm_ref_t>> {
113    Some(Box::new(ref_?.clone()))
114}
115
116#[unsafe(no_mangle)]
117pub unsafe extern "C" fn wasm_ref_same(
118    ref1: Option<&wasm_ref_t>,
119    ref2: Option<&wasm_ref_t>,
120) -> bool {
121    match (ref1, ref2) {
122        (Some(a), Some(b)) => refs_same(&a.inner, &b.inner),
123        (None, None) => true,
124        _ => false,
125    }
126}
127
128#[unsafe(no_mangle)]
129pub unsafe extern "C" fn wasm_ref_get_host_info(ref_: Option<&wasm_ref_t>) -> *mut c_void {
130    ref_.and_then(|r| with_foreign(r, |f| f.host_info.get()))
131        .unwrap_or(std::ptr::null_mut())
132}
133
134#[unsafe(no_mangle)]
135pub unsafe extern "C" fn wasm_ref_set_host_info(ref_: Option<&wasm_ref_t>, info: *mut c_void) {
136    if let Some(r) = ref_ {
137        with_foreign(r, |f| {
138            f.finalizer.set(None);
139            f.host_info.set(info);
140        });
141    }
142}
143
144#[unsafe(no_mangle)]
145pub unsafe extern "C" fn wasm_ref_set_host_info_with_finalizer(
146    ref_: Option<&wasm_ref_t>,
147    info: *mut c_void,
148    finalizer: Option<extern "C" fn(*mut c_void)>,
149) {
150    if let Some(r) = ref_ {
151        with_foreign(r, |f| {
152            f.host_info.set(info);
153            f.finalizer.set(finalizer);
154        });
155    }
156}

165#[allow(non_camel_case_types)]
166pub type wasm_foreign_t = wasm_ref_t;

168#[unsafe(no_mangle)]
169pub unsafe extern "C" fn wasm_foreign_new(
170    store: Option<&mut wasm_store_t>,
171) -> Option<Box<wasm_foreign_t>> {
172    let store = store?;
173    let extern_ref = {
174        let mut store_mut = unsafe { store.inner.store_mut() };
175        ExternRef::new(&mut store_mut, Foreign::new())
176    };
177    Some(Box::new(wasm_ref_t {
178        inner: Value::ExternRef(Some(extern_ref)),
179        store: store.inner.downgrade(),
180    }))
181}

253// Externref creation is only implemented on the `sys` backend.
254#[cfg(all(test, feature = "sys"))]
255mod tests {
256    use super::*;
257    use crate::wasm_c_api::engine::wasm_engine_new;
258    use crate::wasm_c_api::store::wasm_store_new;
259
260    #[test]
261    fn foreign_ref_host_info_copy_and_same() {
262        unsafe {
263            let engine = wasm_engine_new();
264            let mut store = wasm_store_new(Some(&engine)).expect("store is created");
265
266            let foreign = wasm_foreign_new(Some(&mut store)).expect("foreign ref");
267
268            // Host info defaults to null and round-trips through set/get.
269            assert!(wasm_ref_get_host_info(Some(&foreign)).is_null());
270            wasm_ref_set_host_info(Some(&foreign), 42usize as *mut c_void);
271            assert_eq!(
272                wasm_ref_get_host_info(Some(&foreign)),
273                42usize as *mut c_void
274            );
275
276            // A copy is a distinct box referring to the same extern object, so
277            // it is `same` and observes the same (shared) host info.
278            let copy = wasm_ref_copy(Some(&foreign)).expect("copy");
279            assert!(wasm_ref_same(Some(&foreign), Some(&copy)));
280            assert_eq!(wasm_ref_get_host_info(Some(&copy)), 42usize as *mut c_void);
281
282            // Independently minted refs are not the same; null identities hold.
283            let other = wasm_foreign_new(Some(&mut store)).expect("other ref");
284            assert!(!wasm_ref_same(Some(&foreign), Some(&other)));
285            assert!(wasm_ref_same(None, None));
286            assert!(!wasm_ref_same(Some(&foreign), None));
287        }
288    }
