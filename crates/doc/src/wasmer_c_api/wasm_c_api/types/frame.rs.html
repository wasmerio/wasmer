52#[repr(C)]
53#[allow(non_camel_case_types)]
54#[derive(Debug, Clone)]
55pub struct wasm_name_t {
56    pub name: *mut c_char,
57}
58
59#[unsafe(no_mangle)]
60pub unsafe extern "C" fn wasm_frame_module_name(frame: &wasm_frame_t) -> wasm_name_t {
61    let module_name =
62        Some(frame.info.module_name()).and_then(|f| Some(CString::new(f).ok()?.into_raw()));
63
64    match module_name {
65        Some(s) => wasm_name_t { name: s },
66        None => wasm_name_t {
67            name: core::ptr::null_mut(),
68        },
69    }
70}
71
72#[unsafe(no_mangle)]
73pub unsafe extern "C" fn wasm_frame_func_name(frame: &wasm_frame_t) -> wasm_name_t {
74    let func_name = frame
75        .info
76        .function_name()
77        .and_then(|f| Some(CString::new(f).ok()?.into_raw()));
78
79    match func_name {
80        Some(s) => wasm_name_t { name: s },
81        None => wasm_name_t {
82            name: core::ptr::null_mut(),
83        },
84    }
85}
86
87#[unsafe(no_mangle)]
88pub unsafe extern "C" fn wasm_name_delete(name: Option<&mut wasm_name_t>) {
89    if let Some(s) = name
90        && !s.name.is_null()
91    {
92        let _ = unsafe { CString::from_raw(s.name) };
93    }
94}

100fn test_frame_name() {
101    use std::ffi::CStr;
102    use wasmer_types::SourceLoc;
103
104    let info = wasm_frame_t {
105        info: FrameInfo::new(
106            "module_name".to_string(),
107            5,
108            Some("function_name".to_string()),
109            SourceLoc::new(10),
110            SourceLoc::new(20),
111        ),
112    };
113
114    unsafe {
115        let mut wasm_frame_func_name = wasm_frame_func_name(&info);
116        let s = CStr::from_ptr(wasm_frame_func_name.name);
117        assert_eq!(s.to_str().unwrap(), "function_name");
118        wasm_name_delete(Some(&mut wasm_frame_func_name));
119
120        let mut wasm_frame_module_name = wasm_frame_module_name(&info);
121        let s = CStr::from_ptr(wasm_frame_module_name.name);
122        assert_eq!(s.to_str().unwrap(), "module_name");
123        wasm_name_delete(Some(&mut wasm_frame_module_name));
124    }
