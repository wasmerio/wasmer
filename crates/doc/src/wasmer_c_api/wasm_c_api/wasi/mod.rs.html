39#[unsafe(no_mangle)]
40pub unsafe extern "C" fn wasi_config_new(
41    program_name: *const c_char,
42) -> Option<Box<wasi_config_t>> {
43    debug_assert!(!program_name.is_null());
44
45    let name_c_str = unsafe { CStr::from_ptr(program_name) };
46    let prog_name = c_try!(name_c_str.to_str());
47
48    let runtime = tokio::runtime::Builder::new_multi_thread()
49        .enable_all()
50        .build()
51        .unwrap();
52    let _guard = runtime.enter();
53
54    Some(Box::new(wasi_config_t {
55        inherit_stdout: true,
56        inherit_stderr: true,
57        inherit_stdin: true,
58        builder: WasiEnv::builder(prog_name).fs(default_fs_backing()),
59        runtime: Some(runtime),
60    }))
61}

90#[unsafe(no_mangle)]
91pub unsafe extern "C" fn wasi_config_preopen_dir(
92    config: &mut wasi_config_t,
93    dir: *const c_char,
94) -> bool {
95    let dir_cstr = unsafe { CStr::from_ptr(dir) };
96    let dir_bytes = dir_cstr.to_bytes();
97    let dir_str = match std::str::from_utf8(dir_bytes) {
98        Ok(dir_str) => dir_str,
99        Err(e) => {
100            update_last_error(e);
101            return false;
102        }
103    };
104
105    if let Err(e) = config.builder.add_preopen_dir(dir_str) {
106        update_last_error(e);
107        return false;
108    }
109
110    true
111}

205/// Initializes the `imports` with an import object that links to
206/// the custom file system
207#[cfg(feature = "webc_runner")]
208#[unsafe(no_mangle)]
209pub unsafe extern "C" fn wasi_env_with_filesystem(
210    config: Box<wasi_config_t>,
211    store: Option<&mut wasm_store_t>,
212    module: Option<&wasm_module_t>,
213    fs: Option<&wasi_filesystem_t>,
214    imports: Option<&mut wasm_extern_vec_t>,
215    package: *const c_char,
216) -> Option<Box<wasi_env_t>> {
217    unsafe { wasi_env_with_filesystem_inner(config, store, module, fs, imports, package) }
218}

220#[cfg(feature = "webc_runner")]
221unsafe fn wasi_env_with_filesystem_inner(
222    config: Box<wasi_config_t>,
223    store: Option<&mut wasm_store_t>,
224    module: Option<&wasm_module_t>,
225    fs: Option<&wasi_filesystem_t>,
226    imports: Option<&mut wasm_extern_vec_t>,
227    package: *const c_char,
228) -> Option<Box<wasi_env_t>> {
229    let store = &mut store?.inner;
230    let fs = fs.as_ref()?;
231    let package_str = unsafe { CStr::from_ptr(package) };
232    let package = package_str.to_str().unwrap_or("");
233    let module = &module.as_ref()?.inner;
234    let imports = imports?;
235    let fs_bytes = unsafe { &*(fs.ptr as *const u8) };
236
237    let (wasi_env, import_object) = {
238        let mut store_mut = unsafe { store.store_mut() };
239        prepare_webc_env(
240            config,
241            &mut store_mut,
242            module,
243            fs_bytes, // cast wasi_filesystem_t.ptr as &'static [u8]
244            fs.size,
245            package,
246        )?
247    };
248
249    imports_set_buffer(store, module, import_object, imports)?;
250
251    Some(Box::new(wasi_env_t {
252        inner: wasi_env,
253        store: store.clone(),
254    }))
255}

257#[cfg(feature = "webc_runner")]
258fn prepare_webc_env(
259    mut config: Box<wasi_config_t>,
260    store: &mut impl AsStoreMut,
261    module: &Module,
262    bytes: &'static u8,
263    len: usize,
264    package_name: &str,
265) -> Option<(WasiFunctionEnv, Imports)> {
266    use virtual_fs::static_fs::StaticFileSystem;
267    use webc::v1::{FsEntryType, WebC};
268
269    let store_mut = store.as_store_mut();
270    let runtime = config.runtime.take();
271
272    let runtime = runtime.unwrap_or_else(|| {
273        tokio::runtime::Builder::new_multi_thread()
274            .enable_all()
275            .build()
276            .unwrap()
277    });
278
279    let handle = runtime.handle().clone();
280    let _guard = handle.enter();
281    let mut rt = PluggableRuntime::new(Arc::new(TokioTaskManager::new(runtime)));
282    rt.set_engine(store_mut.engine().clone());
283
284    let slice = unsafe { std::slice::from_raw_parts(bytes, len) };
285    let volumes = WebC::parse_volumes_from_fileblock(slice).ok()?;
286    let top_level_dirs = volumes
287        .into_iter()
288        .flat_map(|(_, volume)| {
289            volume
290                .header
291                .top_level
292                .iter()
293                .filter(|entry| entry.fs_type == FsEntryType::Dir)
294                .cloned()
295                .map(|e| e.text.to_string())
296                .collect::<Vec<_>>()
297                .into_iter()
298        })
299        .collect::<Vec<_>>();
300
301    let filesystem = Box::new(StaticFileSystem::init(slice, package_name)?);
302    let mut builder = config.builder.runtime(Arc::new(rt));
303
304    if !config.inherit_stdout {
305        builder.set_stdout(Box::new(Pipe::channel().0));
306    }
307
308    if !config.inherit_stderr {
309        builder.set_stderr(Box::new(Pipe::channel().0));
310    }
311
312    builder.set_fs(filesystem);
313
314    for f_name in top_level_dirs.iter() {
315        builder
316            .add_preopen_build(|p| p.directory(f_name).read(true).write(true).create(true))
317            .ok()?;
318    }
319    let env = builder.finalize(store).ok()?;
320
321    let import_object = env.import_object(store, module).ok()?;
322    Some((env, import_object))
323}

332/// Create a new WASI environment.
333///
334/// It take ownership over the `wasi_config_t`.
335#[unsafe(no_mangle)]
336pub unsafe extern "C" fn wasi_env_new(
337    store: Option<&mut wasm_store_t>,
338    mut config: Box<wasi_config_t>,
339) -> Option<Box<wasi_env_t>> {
340    let store = &mut store?.inner;
341    let mut store_mut = unsafe { store.store_mut() };
342
343    let runtime = config.runtime.take();
344
345    let runtime = runtime.unwrap_or_else(|| {
346        tokio::runtime::Builder::new_multi_thread()
347            .enable_all()
348            .build()
349            .unwrap()
350    });
351
352    let handle = runtime.handle().clone();
353    let _guard = handle.enter();
354    let mut rt = PluggableRuntime::new(Arc::new(TokioTaskManager::new(runtime)));
355    rt.set_engine(store_mut.engine().clone());
356
357    if !config.inherit_stdout {
358        config.builder.set_stdout(Box::new(Pipe::channel().0));
359    }
360
361    if !config.inherit_stderr {
362        config.builder.set_stderr(Box::new(Pipe::channel().0));
363    }
364
365    // TODO: impl capturer for stdin
366
367    let env = c_try!(
368        config
369            .builder
370            .runtime(Arc::new(rt))
371            .finalize(&mut store_mut)
372    );
373
374    Some(Box::new(wasi_env_t {
375        inner: env,
376        store: store.clone(),
377    }))
378}

399#[unsafe(no_mangle)]
400pub unsafe extern "C" fn wasi_env_read_stdout(
401    env: &mut wasi_env_t,
402    buffer: *mut c_char,
403    buffer_len: usize,
404) -> isize {
405    let inner_buffer = unsafe { slice::from_raw_parts_mut(buffer as *mut _, buffer_len) };
406    let store = unsafe { env.store.store() };
407
408    let stdout = {
409        let data = env.inner.data(&store);
410        data.stdout()
411    };
412
413    if let Ok(mut stdout) = stdout {
414        if let Some(stdout) = stdout.as_mut() {
415            read_inner(stdout, inner_buffer)
416        } else {
417            update_last_error("could not find a file handle for `stdout`");
418            -1
419        }
420    } else {
421        update_last_error("could not find a file handle for `stdout`");
422        -1
423    }
424}

451fn read_inner(
452    wasi_file: &mut Box<dyn VirtualFile + Send + Sync + 'static>,
453    inner_buffer: &mut [u8],
454) -> isize {
455    InlineWaker::block_on(async {
456        match wasi_file.read(inner_buffer).await {
457            Ok(a) => a as isize,
458            Err(err) => {
459                update_last_error(format!("failed to read wasi_file: {err}"));
460                -1
461            }
462        }
463    })
464}

466/// The version of WASI. This is determined by the imports namespace
467/// string.
468#[derive(Debug, Clone, Copy, PartialEq, Eq)]
469#[repr(C)]
470#[allow(non_camel_case_types)]
471pub enum wasi_version_t {
472    /// An invalid version.
473    INVALID_VERSION = -1,
474
475    /// Latest version.
476    ///
477    /// It's a “floating” version, i.e. it's an alias to the latest
478    /// version (for the moment, `Snapshot1`). Using this version is a
479    /// way to ensure that modules will run only if they come with the
480    /// latest WASI version (in case of security issues for instance),
481    /// by just updating the runtime.
482    ///
483    /// Note that this version is never returned by an API. It is
484    /// provided only by the user.
485    LATEST = 0,
486
487    /// `wasi_unstable`.
488    SNAPSHOT0 = 1,
489
490    /// `wasi_snapshot_preview1`.
491    SNAPSHOT1 = 2,
492
493    /// `wasix_32v1`.
494    WASIX32V1 = 3,
495
496    /// `wasix_64v1`.
497    WASIX64V1 = 4,
498}
499
500impl From<WasiVersion> for wasi_version_t {
501    fn from(other: WasiVersion) -> Self {
502        match other {
503            WasiVersion::Snapshot0 => wasi_version_t::SNAPSHOT0,
504            WasiVersion::Snapshot1 => wasi_version_t::SNAPSHOT1,
505            WasiVersion::Wasix32v1 => wasi_version_t::WASIX32V1,
506            WasiVersion::Wasix64v1 => wasi_version_t::WASIX64V1,
507            WasiVersion::Latest => wasi_version_t::LATEST,
508        }
509    }
510}
511
512impl TryFrom<wasi_version_t> for WasiVersion {
513    type Error = &'static str;
514
515    fn try_from(other: wasi_version_t) -> Result<Self, Self::Error> {
516        Ok(match other {
517            wasi_version_t::INVALID_VERSION => return Err("Invalid WASI version cannot be used"),
518            wasi_version_t::SNAPSHOT0 => WasiVersion::Snapshot0,
519            wasi_version_t::SNAPSHOT1 => WasiVersion::Snapshot1,
520            wasi_version_t::WASIX32V1 => WasiVersion::Wasix32v1,
521            wasi_version_t::WASIX64V1 => WasiVersion::Wasix64v1,
522            wasi_version_t::LATEST => WasiVersion::Latest,
523        })
524    }
525}

534/// Non-standard function to get the imports needed for the WASI
535/// implementation ordered as expected by the `wasm_module_t`.
536#[unsafe(no_mangle)]
537pub unsafe extern "C" fn wasi_get_imports(
538    _store: Option<&wasm_store_t>,
539    wasi_env: Option<&mut wasi_env_t>,
540    module: Option<&wasm_module_t>,
541    imports: &mut wasm_extern_vec_t,
542) -> bool {
543    unsafe { wasi_get_imports_inner(wasi_env, module, imports) }.is_some()
544}
545
546unsafe fn wasi_get_imports_inner(
547    wasi_env: Option<&mut wasi_env_t>,
548    module: Option<&wasm_module_t>,
549    imports: &mut wasm_extern_vec_t,
550) -> Option<()> {
551    let wasi_env = wasi_env?;
552    let store = &mut wasi_env.store;
553    let module = module?;
554
555    let mut import_object = {
556        let mut store_mut = unsafe { store.store_mut() };
557        c_try!(wasi_env.inner.import_object(&mut store_mut, &module.inner))
558    };
559
560    let shared_memory = module.inner.imports().memories().next().map(|a| *a.ty());
561
562    let spawn_type = match shared_memory {
563        Some(ty) => wasmer_wasix::runtime::SpawnType::CreateMemoryOfType(ty),
564        None => wasmer_wasix::runtime::SpawnType::CreateMemory,
565    };
566
567    let tasks = {
568        let store_ref = unsafe { store.store() };
569        wasi_env
570            .inner
571            .data(&store_ref)
572            .runtime
573            .task_manager()
574            .clone()
575    };
576    let memory = {
577        let mut store_mut = unsafe { store.store_mut() };
578        tasks.build_memory(&mut store_mut, &spawn_type).unwrap()
579    };
580
581    if let Some(memory) = memory {
582        import_object.define("env", "memory", memory);
583    }
584
585    imports_set_buffer(store, &module.inner, import_object, imports)?;
586
587    Some(())
588}

590pub(crate) fn imports_set_buffer(
591    store: &StoreRef,
592    module: &wasmer_api::Module,
593    import_object: wasmer_api::Imports,
594    imports: &mut wasm_extern_vec_t,
595) -> Option<()> {
596    imports.set_buffer(c_try!(
597        module
598            .imports()
599            .map(|import_type| {
600                let ext = import_object
601                    .get_export(import_type.module(), import_type.name())
602                    .ok_or_else(|| {
603                        format!(
604                            "Failed to resolve import \"{}\" \"{}\"",
605                            import_type.module(),
606                            import_type.name()
607                        )
608                    })?;
609
610                Ok(Some(Box::new(wasm_extern_t::new(store.clone(), ext))))
611            })
612            .collect::<Result<Vec<_>, String>>()
613    ));
614
615    Some(())
616}
