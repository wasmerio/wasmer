23/// Configuration specific for `sys` (Cranelift, LLVM, Singlepass) engines.
24///
25/// This is a Wasmer-specific type with Wasmer-specific functions for manipulating it.
26///
27/// cbindgen:ignore
28#[repr(C)]
29#[derive(Debug, Default)]
30pub(crate) struct wasmer_sys_engine_config_t {
31    pub nan_canonicalization: bool,
32}
33
34/// Updates the configuration to enable NaN canonicalization.
35///
36/// This is a Wasmer-specific function.
37///
38/// # Example
39///
40/// ```rust
41/// # use wasmer_inline_c::assert_c;
42/// # fn main() {
43/// #    (assert_c! {
44/// # #include "tests/wasmer.h"
45/// #
46/// int main() {
47///     // Create the configuration.
48///     wasm_config_t* config = wasm_config_new();
49///
50///     // Enable NaN canonicalization.
51///     wasm_config_sys_canonicalize_nans(config, true);
52///
53///     // Create the engine.
54///     wasm_engine_t* engine = wasm_engine_new_with_config(config);
55///
56///     // Check we have an engine!
57///     assert(engine);
58///
59///     // Free everything.
60///     wasm_engine_delete(engine);
61///
62///     return 0;
63/// }
64/// #    })
65/// #    .success();
66/// # }
67/// ```
68#[unsafe(no_mangle)]
69pub extern "C" fn wasm_config_sys_canonicalize_nans(config: &mut wasm_config_t, enable: bool) {
70    if let wasmer_backend_config_kind_t::Sys(ref mut c) = config.backend_config.inner {
71        c.nan_canonicalization = enable;
72    } else {
73        let sys_config = wasmer_sys_engine_config_t {
74            nan_canonicalization: enable,
75        };
76        config.backend_config.inner = wasmer_backend_config_kind_t::Sys(sys_config);
77    }
78}

80/// Create a new  [`wasm_engine_t`] backed by a `sys` engine.
81pub fn wasm_sys_engine_new_with_config(config: wasm_config_t) -> Option<Box<wasm_engine_t>> {
82    if !matches!(config.backend, wasmer_backend_t::LLVM)
83        && !matches!(config.backend, wasmer_backend_t::SINGLEPASS)
84        && !matches!(config.backend, wasmer_backend_t::CRANELIFT)
85        && !matches!(config.backend, wasmer_backend_t::HEADLESS)
86        && !config.backend_config.inner.is_sys()
87    {
88        update_last_error(format!(
89            "Cannot create a new `sys` engine with a non-sys-specific config! {config:?}"
90        ));
91        return None;
92    }
93
94    let backend = config.backend;
95    #[allow(unused)]
96    let sys_config = match config.backend_config.inner.try_into_sys() {
97        Err(_) => {
98            update_last_error(format!(
99                "Could not create new `sys` engine with the selected {backend:?} backend."
100            ));
101            return None;
102        }
103        Ok(s) => s,
104    };
105
106    cfg_if! {
107    if #[cfg(feature = "compiler")] {
108                #[allow(unused_mut)]
109                let mut compiler_config: Box<dyn CompilerConfig> = match &backend {
110                    wasmer_backend_t::CRANELIFT => {
111                        cfg_if! {
112                            if #[cfg(feature = "cranelift")] {
113                                Box::<wasmer_compiler_cranelift::Cranelift>::default()
114                            } else {
115                                update_last_error("Wasmer has not been compiled with the `cranelift` feature.");
116                                return None;
117                            }
118                        }
119                    },
120                    wasmer_backend_t::LLVM => {
121                        cfg_if! {
122                            if #[cfg(feature = "llvm")] {
123                                Box::<wasmer_compiler_llvm::LLVM>::default()
124                            } else {
125                                update_last_error("Wasmer has not been compiled with the `llvm` feature.");
126                                return None;
127                            }
128                        }
129                    },
130                    wasmer_backend_t::SINGLEPASS => {
131                        cfg_if! {
132                            if #[cfg(feature = "singlepass")] {
133                                Box::<wasmer_compiler_singlepass::Singlepass>::default()
134                            } else {
135                                update_last_error("Wasmer has not been compiled with the `singlepass` feature.");
136                                return None;
137                            }
138                        }
139                    },
140                    _ => panic!("not a `sys` backend!")
141                };
142
143                #[cfg(feature = "middlewares")]
144                for middleware in config.backend_config.middlewares {
145                    compiler_config.push_middleware(middleware.inner.clone());
146                }
147
148                if sys_config.nan_canonicalization {
149                    compiler_config.canonicalize_nans(true);
150                }
151
152                let inner: Engine =
153                             {
154                                let mut builder = EngineBuilder::new(compiler_config);
155
156                                if let Some(target) = config.backend_config.target {
157                                    builder = builder.set_target(Some(target.inner));
158                                }
159
160                                if let Some(features) = config.features {
161                                    builder = builder.set_features(Some(features.inner));
162                                }
163
164                                builder.engine().into()
165                            };
166                Some(Box::new(wasm_engine_t { inner }))
167            } else if #[cfg(feature = "compiler-headless")] {
168                let inner: Engine =
169                         {
170                                let mut builder = EngineBuilder::headless();
171
172                                if let Some(target) = config.backend_config.target {
173                                    builder = builder.set_target(Some(target.inner));
174                                }
175
176                                if let Some(features) = config.features {
177                                    builder = builder.set_features(Some(features.inner));
178                                }
179
180                                builder.engine().into()
181                        };
182
183                Some(Box::new(wasm_engine_t { inner }))
184            } else {
185                update_last_error("No backend enabled for the `sys` engine: enable one of `compiler` or `compiler-headless`!");
186                return None;
187            }
188        }
189}

191impl wasmer_backend_config_kind_t {
192    /// Returns `true` if the wasmer_engine_config_t is [`Sys`].
193    ///
194    /// [`Sys`]: wasmer_engine_config_t::Sys
195    #[must_use]
196    pub(super) fn is_sys(&self) -> bool {
197        matches!(self, Self::Sys(..))
198    }
199
200    pub(super) fn try_into_sys(self) -> Result<wasmer_sys_engine_config_t, Self> {
201        if let Self::Sys(v) = self {
202            Ok(v)
203        } else {
204            Err(self)
205        }
206    }
