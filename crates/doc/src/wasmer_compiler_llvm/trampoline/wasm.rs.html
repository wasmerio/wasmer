42const FUNCTION_SECTION_ELF: &str = "__TEXT,wasmer_trmpl"; // Needs to be between 1 and 16 chars
43const FUNCTION_SECTION_MACHO: &str = "wasmer_trmpl"; // Needs to be between 1 and 16 chars

46    pub fn new(
47        target_machine: TargetMachine,
48        binary_fmt: BinaryFormat,
49    ) -> Result<Self, CompileError> {
50        let abi = get_abi(&target_machine);
51        Ok(Self {
52            ctx: Context::create(),
53            target_machine,
54            abi,
55            func_section: match binary_fmt {
56                BinaryFormat::Elf => FUNCTION_SECTION_ELF.to_string(),
57                BinaryFormat::Macho => FUNCTION_SECTION_MACHO.to_string(),
58                _ => {
59                    return Err(CompileError::UnsupportedTarget(format!(
60                        "Unsupported binary format: {binary_fmt:?}",
61                    )));
62                }
63            },
64            binary_fmt,
65        })
66    }

68    pub fn trampoline_to_module(
69        &self,
70        ty: &FuncType,
71        config: &LLVM,
72        name: &str,
73        compile_info: &CompileModuleInfo,
74    ) -> Result<Module<'_>, CompileError> {
75        // The function type, used for the callbacks.
76        let function = CompiledKind::FunctionCallTrampoline(ty.clone());
77        let module = self.ctx.create_module("");
78        let target_machine = &self.target_machine;
79        let target_triple = target_machine.get_triple();
80        let target_data = target_machine.get_target_data();
81        module.set_triple(&target_triple);
82        module.set_data_layout(&target_data.get_data_layout());
83        let intrinsics = Intrinsics::declare(&module, &self.ctx, &target_data, &self.binary_fmt);
84
85        let func_kind = if config.enable_g0m0_opt {
86            Some(G0M0FunctionKind::Local)
87        } else {
88            None
89        };
90
91        let (callee_ty, callee_attrs) =
92            self.abi
93                .func_type_to_llvm(&self.ctx, &intrinsics, None, ty, func_kind)?;
94        let trampoline_ty = intrinsics.void_ty.fn_type(
95            &[
96                intrinsics.ptr_ty.into(), // vmctx ptr
97                intrinsics.ptr_ty.into(), // callee function address
98                intrinsics.ptr_ty.into(), // in/out values ptr
99            ],
100            false,
101        );
102
103        let trampoline_func = module.add_function(name, trampoline_ty, Some(Linkage::External));
104        trampoline_func
105            .as_global_value()
106            .set_section(Some(&self.func_section));
107        trampoline_func
108            .as_global_value()
109            .set_linkage(Linkage::DLLExport);
110        trampoline_func
111            .as_global_value()
112            .set_dll_storage_class(DLLStorageClass::Export);
113        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.uwtable);
114        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.frame_pointer);
115        self.generate_trampoline(
116            config,
117            compile_info,
118            trampoline_func,
119            ty,
120            callee_ty,
121            &callee_attrs,
122            &self.ctx,
123            &intrinsics,
124        )?;
125
126        if let Some(ref callbacks) = config.callbacks {
127            callbacks.preopt_ir(&function, &module);
128        }
129
130        let mut passes = vec![];
131
132        if config.enable_verifier {
133            passes.push("verify");
134        }
135
136        passes.push("instcombine");
137        module
138            .run_passes(
139                passes.join(",").as_str(),
140                target_machine,
141                PassBuilderOptions::create(),
142            )
143            .unwrap();
144
145        if let Some(ref callbacks) = config.callbacks {
146            callbacks.postopt_ir(&function, &module);
147        }
148
149        // -- Uncomment to enable dumping intermediate LLVM objects
150        //module
151        //    .print_to_file(format!(
152        //        "{}/obj_trmpl.ll",
153        //        std::env!("LLVM_EH_TESTS_DUMP_DIR")
154        //    ))
155        //    .unwrap();
156        Ok(module)
157    }

159    pub fn trampoline(
160        &self,
161        ty: &FuncType,
162        config: &LLVM,
163        name: &str,
164        compile_info: &CompileModuleInfo,
165    ) -> Result<FunctionBody, CompileError> {
166        let module = self.trampoline_to_module(ty, config, name, compile_info)?;
167        let function = CompiledKind::FunctionCallTrampoline(ty.clone());
168        let target_machine = &self.target_machine;
169
170        let memory_buffer = target_machine
171            .write_to_memory_buffer(&module, FileType::Object)
172            .unwrap();
173
174        if let Some(ref callbacks) = config.callbacks {
175            callbacks.obj_memory_buffer(&function, &memory_buffer);
176            let asm_buffer = target_machine
177                .write_to_memory_buffer(&module, FileType::Assembly)
178                .unwrap();
179            callbacks.asm_memory_buffer(&function, &asm_buffer);
180        }
181
182        let mem_buf_slice = memory_buffer.as_slice();
183
184        // Use a dummy function index to detect relocations against the trampoline
185        // function's address, which shouldn't exist and are not supported.
186        // Note, we just drop all custom sections, and verify that the function
187        // body itself has no relocations at all. This value should never be
188        // touched at all. However, it is set up so that if we do touch it (maybe
189        // due to someone changing the code later on), it'll explode, which is desirable!
190        let dummy_reloc_target =
191            RelocationTarget::DynamicTrampoline(FunctionIndex::from_u32(u32::MAX - 1));
192
193        // Note: we don't count .gcc_except_table here because native-to-wasm
194        // trampolines are not supposed to generate any LSDA sections. We *want* them
195        // to terminate libunwind's stack searches.
196        let CompiledFunction {
197            compiled_function,
198            custom_sections,
199            eh_frame_section_indices,
200            mut compact_unwind_section_indices,
201            ..
202        } = load_object_file(
203            mem_buf_slice,
204            &self.func_section,
205            dummy_reloc_target,
206            |name: &str| {
207                Err(CompileError::Codegen(format!(
208                    "trampoline generation produced reference to unknown function {name}",
209                )))
210            },
211            self.binary_fmt,
212        )?;
213        let mut all_sections_are_eh_sections = true;
214        let mut unwind_section_indices = eh_frame_section_indices;
215        unwind_section_indices.append(&mut compact_unwind_section_indices);
216        if unwind_section_indices.len() != custom_sections.len() {
217            all_sections_are_eh_sections = false;
218        } else {
219            unwind_section_indices.sort_unstable();
220            for (idx, section_idx) in unwind_section_indices.iter().enumerate() {
221                if idx as u32 != section_idx.as_u32() {
222                    all_sections_are_eh_sections = false;
223                    break;
224                }
225            }
226        }
227        if !all_sections_are_eh_sections {
228            return Err(CompileError::Codegen(
229                "trampoline generation produced non-eh custom sections".into(),
230            ));
231        }
232        if !compiled_function.relocations.is_empty() {
233            return Err(CompileError::Codegen(
234                "trampoline generation produced relocations".into(),
235            ));
236        }
237        // Ignore CompiledFunctionFrameInfo. Extra frame info isn't a problem.
238
239        Ok(FunctionBody {
240            body: compiled_function.body.body,
241            unwind_info: compiled_function.body.unwind_info,
242        })
243    }

245    pub fn dynamic_trampoline_to_module(
246        &self,
247        ty: &FuncType,
248        config: &LLVM,
249        name: &str,
250    ) -> Result<Module<'_>, CompileError> {
251        // The function type, used for the callbacks
252        let function = CompiledKind::DynamicFunctionTrampoline(ty.clone());
253        let module = self.ctx.create_module("");
254        let target_machine = &self.target_machine;
255        let target_data = target_machine.get_target_data();
256        let target_triple = target_machine.get_triple();
257        module.set_triple(&target_triple);
258        module.set_data_layout(&target_data.get_data_layout());
259        let intrinsics = Intrinsics::declare(&module, &self.ctx, &target_data, &self.binary_fmt);
260
261        let (trampoline_ty, trampoline_attrs) =
262            self.abi
263                .func_type_to_llvm(&self.ctx, &intrinsics, None, ty, None)?;
264        let trampoline_func = module.add_function(name, trampoline_ty, Some(Linkage::External));
265        trampoline_func.set_personality_function(intrinsics.personality);
266        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.frame_pointer);
267        for (attr, attr_loc) in trampoline_attrs {
268            trampoline_func.add_attribute(attr_loc, attr);
269        }
270        trampoline_func
271            .as_global_value()
272            .set_section(Some(&self.func_section));
273        trampoline_func
274            .as_global_value()
275            .set_linkage(Linkage::DLLExport);
276        trampoline_func
277            .as_global_value()
278            .set_dll_storage_class(DLLStorageClass::Export);
279        self.generate_dynamic_trampoline(trampoline_func, ty, &self.ctx, &intrinsics)?;
280
281        if let Some(ref callbacks) = config.callbacks {
282            callbacks.preopt_ir(&function, &module);
283        }
284
285        let mut passes = vec![];
286
287        if config.enable_verifier {
288            passes.push("verify");
289        }
290
291        passes.push("early-cse");
292        module
293            .run_passes(
294                passes.join(",").as_str(),
295                target_machine,
296                PassBuilderOptions::create(),
297            )
298            .unwrap();
299
300        if let Some(ref callbacks) = config.callbacks {
301            callbacks.postopt_ir(&function, &module);
302        }
303
304        Ok(module)
305    }

307    #[allow(clippy::too_many_arguments)]
308    pub fn dynamic_trampoline(
309        &self,
310        ty: &FuncType,
311        config: &LLVM,
312        name: &str,
313        dynamic_trampoline_index: u32,
314        final_module_custom_sections: &mut PrimaryMap<SectionIndex, CustomSection>,
315        eh_frame_section_bytes: &mut Vec<u8>,
316        eh_frame_section_relocations: &mut Vec<Relocation>,
317        compact_unwind_section_bytes: &mut Vec<u8>,
318        compact_unwind_section_relocations: &mut Vec<Relocation>,
319    ) -> Result<FunctionBody, CompileError> {
320        let function = CompiledKind::DynamicFunctionTrampoline(ty.clone());
321        let target_machine = &self.target_machine;
322
323        let module = self.dynamic_trampoline_to_module(ty, config, name)?;
324
325        let memory_buffer = target_machine
326            .write_to_memory_buffer(&module, FileType::Object)
327            .unwrap();
328
329        if let Some(ref callbacks) = config.callbacks {
330            callbacks.obj_memory_buffer(&function, &memory_buffer);
331            let asm_buffer = target_machine
332                .write_to_memory_buffer(&module, FileType::Assembly)
333                .unwrap();
334            callbacks.asm_memory_buffer(&function, &asm_buffer)
335        }
336
337        let mem_buf_slice = memory_buffer.as_slice();
338        let CompiledFunction {
339            compiled_function,
340            custom_sections,
341            eh_frame_section_indices,
342            compact_unwind_section_indices,
343            gcc_except_table_section_indices,
344        } = load_object_file(
345            mem_buf_slice,
346            &self.func_section,
347            RelocationTarget::DynamicTrampoline(FunctionIndex::from_u32(dynamic_trampoline_index)),
348            |name: &str| {
349                Err(CompileError::Codegen(format!(
350                    "trampoline generation produced reference to unknown function {name}",
351                )))
352            },
353            self.binary_fmt,
354        )?;
355
356        if !compiled_function.relocations.is_empty() {
357            return Err(CompileError::Codegen(
358                "trampoline generation produced relocations".into(),
359            ));
360        }
361        // Ignore CompiledFunctionFrameInfo. Extra frame info isn't a problem.
362
363        // Also append EH-related sections to the final module, since we expect
364        // dynamic trampolines to participate in unwinding
365        {
366            let first_section = final_module_custom_sections.len() as u32;
367            for (section_index, mut custom_section) in custom_sections.into_iter() {
368                for reloc in &mut custom_section.relocations {
369                    if let RelocationTarget::CustomSection(index) = reloc.reloc_target {
370                        reloc.reloc_target = RelocationTarget::CustomSection(
371                            SectionIndex::from_u32(first_section + index.as_u32()),
372                        )
373                    }
374
375                    if reloc.kind.needs_got() {
376                        return Err(CompileError::Codegen(
377                            "trampoline generation produced GOT relocation".into(),
378                        ));
379                    }
380                }
381
382                if eh_frame_section_indices.contains(&section_index) {
383                    let offset = eh_frame_section_bytes.len() as u32;
384                    for reloc in &mut custom_section.relocations {
385                        reloc.offset += offset;
386                    }
387                    eh_frame_section_bytes.extend_from_slice(custom_section.bytes.as_slice());
388                    // Terminate the eh_frame info with a zero-length CIE.
389                    eh_frame_section_bytes.extend_from_slice(&[0, 0, 0, 0]);
390                    eh_frame_section_relocations.extend(custom_section.relocations);
391                    // TODO: we do this to keep the count right, remove it.
392                    final_module_custom_sections.push(CustomSection {
393                        protection: CustomSectionProtection::Read,
394                        alignment: None,
395                        bytes: SectionBody::new_with_vec(vec![]),
396                        relocations: vec![],
397                    });
398                } else if compact_unwind_section_indices.contains(&section_index) {
399                    let offset = compact_unwind_section_bytes.len() as u32;
400                    for reloc in &mut custom_section.relocations {
401                        reloc.offset += offset;
402                    }
403                    compact_unwind_section_bytes.extend_from_slice(custom_section.bytes.as_slice());
404                    compact_unwind_section_relocations.extend(custom_section.relocations);
405                    // TODO: we do this to keep the count right, remove it.
406                    final_module_custom_sections.push(CustomSection {
407                        protection: CustomSectionProtection::Read,
408                        alignment: None,
409                        bytes: SectionBody::new_with_vec(vec![]),
410                        relocations: vec![],
411                    });
412                } else if gcc_except_table_section_indices.contains(&section_index) {
413                    final_module_custom_sections.push(custom_section);
414                } else {
415                    return Err(CompileError::Codegen(
416                        "trampoline generation produced non-eh custom sections".into(),
417                    ));
418                }
419            }
420        }
421
422        Ok(FunctionBody {
423            body: compiled_function.body.body,
424            unwind_info: compiled_function.body.unwind_info,
425        })
426    }

428    #[allow(clippy::too_many_arguments)]
429    fn generate_trampoline<'ctx>(
430        &self,
431        config: &LLVM,
432        compile_info: &CompileModuleInfo,
433        trampoline_func: FunctionValue,
434        func_sig: &FuncType,
435        llvm_func_type: FunctionType,
436        func_attrs: &[(Attribute, AttributeLoc)],
437        context: &'ctx Context,
438        intrinsics: &Intrinsics<'ctx>,
439    ) -> Result<(), CompileError> {
440        let entry_block = context.append_basic_block(trampoline_func, "entry");
441        let builder = context.create_builder();
442        builder.position_at_end(entry_block);
443
444        let (callee_vmctx_ptr, func_ptr, args_rets_ptr) =
445            match *trampoline_func.get_params().as_slice() {
446                [callee_vmctx_ptr, func_ptr, args_rets_ptr] => (
447                    callee_vmctx_ptr,
448                    func_ptr.into_pointer_value(),
449                    args_rets_ptr.into_pointer_value(),
450                ),
451                _ => {
452                    return Err(CompileError::Codegen(
453                        "trampoline function unimplemented".to_string(),
454                    ));
455                }
456            };
457
458        let mut args_vec: Vec<BasicMetadataValueEnum> =
459            Vec::with_capacity(if config.enable_g0m0_opt {
460                func_sig.params().len() + 3
461            } else {
462                func_sig.params().len() + 1
463            });
464
465        if self.abi.is_sret(func_sig)? {
466            let basic_types: Vec<_> = func_sig
467                .results()
468                .iter()
469                .map(|&ty| type_to_llvm(intrinsics, ty))
470                .collect::<Result<_, _>>()?;
471
472            let sret_ty = context.struct_type(&basic_types, false);
473            args_vec.push(err!(builder.build_alloca(sret_ty, "sret")).into());
474        }
475
476        args_vec.push(callee_vmctx_ptr.into());
477
478        if config.enable_g0m0_opt {
479            let wasm_module = &compile_info.module;
480            let memory_styles = &compile_info.memory_styles;
481            let callee_vmctx_ptr_value = callee_vmctx_ptr.into_pointer_value();
482            // get value of G0, get a pointer to M0's base
483
484            let offsets = wasmer_vm::VMOffsets::new(8, wasm_module);
485
486            let global_index = wasmer_types::GlobalIndex::from_u32(0);
487            let global_type = wasm_module.globals[global_index];
488            let global_value_type = global_type.ty;
489            let global_mutability = global_type.mutability;
490
491            let offset =
492                if let Some(local_global_index) = wasm_module.local_global_index(global_index) {
493                    offsets.vmctx_vmglobal_definition(local_global_index)
494                } else {
495                    offsets.vmctx_vmglobal_import(global_index)
496                };
497            let offset = intrinsics.i32_ty.const_int(offset.into(), false);
498            let global_ptr = {
499                let global_ptr_ptr = unsafe {
500                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
501                };
502                let global_ptr_ptr =
503                    err!(builder.build_bit_cast(global_ptr_ptr, intrinsics.ptr_ty, ""))
504                        .into_pointer_value();
505
506                err!(builder.build_load(intrinsics.ptr_ty, global_ptr_ptr, "")).into_pointer_value()
507            };
508
509            let global_ptr = err!(builder.build_bit_cast(
510                global_ptr,
511                type_to_llvm_ptr(intrinsics, global_value_type)?,
512                "",
513            ))
514            .into_pointer_value();
515
516            let global_value = match global_mutability {
517                wasmer_types::Mutability::Const => {
518                    err!(builder.build_load(
519                        type_to_llvm(intrinsics, global_value_type)?,
520                        global_ptr,
521                        "g0",
522                    ))
523                }
524                wasmer_types::Mutability::Var => {
525                    err!(builder.build_load(
526                        type_to_llvm(intrinsics, global_value_type)?,
527                        global_ptr,
528                        ""
529                    ))
530                }
531            };
532
533            global_value.set_name("trmpl_g0");
534            args_vec.push(global_value.into());
535
536            // load mem
537            let memory_index = wasmer_types::MemoryIndex::from_u32(0);
538            let memory_definition_ptr = if let Some(local_memory_index) =
539                wasm_module.local_memory_index(memory_index)
540            {
541                let offset = offsets.vmctx_vmmemory_definition(local_memory_index);
542                let offset = intrinsics.i32_ty.const_int(offset.into(), false);
543                unsafe {
544                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
545                }
546            } else {
547                let offset = offsets.vmctx_vmmemory_import(memory_index);
548                let offset = intrinsics.i32_ty.const_int(offset.into(), false);
549                let memory_definition_ptr_ptr = unsafe {
550                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
551                };
552                let memory_definition_ptr_ptr =
553                    err!(builder.build_bit_cast(memory_definition_ptr_ptr, intrinsics.ptr_ty, "",))
554                        .into_pointer_value();
555
556                err!(builder.build_load(intrinsics.ptr_ty, memory_definition_ptr_ptr, ""))
557                    .into_pointer_value()
558            };
559            let memory_definition_ptr =
560                err!(builder.build_bit_cast(memory_definition_ptr, intrinsics.ptr_ty, "",))
561                    .into_pointer_value();
562            let base_ptr = err!(builder.build_struct_gep(
563                intrinsics.vmmemory_definition_ty,
564                memory_definition_ptr,
565                intrinsics.vmmemory_definition_base_element,
566                "",
567            ));
568
569            let memory_style = &memory_styles[memory_index];
570            let base_ptr = if let MemoryStyle::Dynamic { .. } = memory_style {
571                base_ptr
572            } else {
573                err!(builder.build_load(intrinsics.ptr_ty, base_ptr, "")).into_pointer_value()
574            };
575
576            base_ptr.set_name("trmpl_m0_base_ptr");
577
578            args_vec.push(base_ptr.into());
579        }
580
581        for (i, param_ty) in func_sig.params().iter().enumerate() {
582            let index = intrinsics.i32_ty.const_int(i as _, false);
583            let item_pointer = unsafe {
584                err!(builder.build_in_bounds_gep(
585                    intrinsics.i128_ty,
586                    args_rets_ptr,
587                    &[index],
588                    "arg_ptr"
589                ))
590            };
591
592            let casted_type = type_to_llvm(intrinsics, *param_ty)?;
593            let casted_pointer_type = type_to_llvm_ptr(intrinsics, *param_ty)?;
594
595            let typed_item_pointer = err!(builder.build_pointer_cast(
596                item_pointer,
597                casted_pointer_type,
598                "typed_arg_pointer"
599            ));
600
601            let arg = err!(builder.build_load(casted_type, typed_item_pointer, "arg"));
602            args_vec.push(arg.into());
603        }
604
605        let call_site = err!(builder.build_indirect_call(
606            llvm_func_type,
607            func_ptr,
608            args_vec.as_slice(),
609            "call"
610        ));
611        for (attr, attr_loc) in func_attrs {
612            call_site.add_attribute(*attr_loc, *attr);
613        }
614
615        let rets = self
616            .abi
617            .rets_from_call(&builder, intrinsics, call_site, func_sig)?;
618        for (idx, v) in rets.into_iter().enumerate() {
619            let ptr = unsafe {
620                err!(builder.build_gep(
621                    intrinsics.i128_ty,
622                    args_rets_ptr,
623                    &[intrinsics.i32_ty.const_int(idx as u64, false)],
624                    "",
625                ))
626            };
627            let ptr = err!(builder.build_pointer_cast(
628                ptr,
629                self.ctx.ptr_type(AddressSpace::default()),
630                ""
631            ));
632            err!(builder.build_store(ptr, v));
633        }
634
635        err!(builder.build_return(None));
636        Ok(())
637    }

639    fn generate_dynamic_trampoline<'ctx>(
640        &self,
641        trampoline_func: FunctionValue,
642        func_sig: &FuncType,
643        context: &'ctx Context,
644        intrinsics: &Intrinsics<'ctx>,
645    ) -> Result<(), CompileError> {
646        let entry_block = context.append_basic_block(trampoline_func, "entry");
647        let builder = context.create_builder();
648        builder.position_at_end(entry_block);
649
650        // Allocate stack space for the params and results.
651        let values = err!(builder.build_alloca(
652            intrinsics.i128_ty.array_type(cmp::max(
653                func_sig.params().len().try_into().unwrap(),
654                func_sig.results().len().try_into().unwrap(),
655            )),
656            "",
657        ));
658
659        // Copy params to 'values'.
660        let first_user_param = if self.abi.is_sret(func_sig)? { 2 } else { 1 };
661        for i in 0..func_sig.params().len() {
662            let ptr = unsafe {
663                err!(builder.build_in_bounds_gep(
664                    intrinsics.i128_ty,
665                    values,
666                    &[intrinsics.i32_ty.const_int(i.try_into().unwrap(), false)],
667                    "args",
668                ))
669            };
670            let ptr = err!(builder.build_bit_cast(
671                ptr,
672                type_to_llvm_ptr(intrinsics, func_sig.params()[i])?,
673                ""
674            ))
675            .into_pointer_value();
676            err!(
677                builder.build_store(
678                    ptr,
679                    trampoline_func
680                        .get_nth_param(i as u32 + first_user_param)
681                        .unwrap(),
682                )
683            );
684        }
685
686        let callee_ptr_ty = intrinsics.void_ty.fn_type(
687            &[
688                intrinsics.ptr_ty.into(), // vmctx ptr
689                intrinsics.ptr_ty.into(), // in/out values ptr
690            ],
691            false,
692        );
693        let vmctx = self.abi.get_vmctx_ptr_param(&trampoline_func);
694        let callee_ty =
695            err!(builder.build_bit_cast(vmctx, self.ctx.ptr_type(AddressSpace::default()), ""));
696        let callee =
697            err!(builder.build_load(intrinsics.ptr_ty, callee_ty.into_pointer_value(), ""))
698                .into_pointer_value();
699
700        let values_ptr = err!(builder.build_pointer_cast(values, intrinsics.ptr_ty, ""));
701        err!(builder.build_indirect_call(
702            callee_ptr_ty,
703            callee,
704            &[vmctx.into(), values_ptr.into()],
705            "",
706        ));
707
708        if func_sig.results().is_empty() {
709            err!(builder.build_return(None));
710        } else {
711            let results = func_sig
712                .results()
713                .iter()
714                .enumerate()
715                .map(|(idx, ty)| {
716                    let ptr = unsafe {
717                        err!(builder.build_gep(
718                            intrinsics.i128_ty,
719                            values,
720                            &[intrinsics.i32_ty.const_int(idx.try_into().unwrap(), false)],
721                            "",
722                        ))
723                    };
724                    let ptr = err!(builder.build_pointer_cast(
725                        ptr,
726                        type_to_llvm_ptr(intrinsics, *ty)?,
727                        ""
728                    ));
729                    err_nt!(builder.build_load(type_to_llvm(intrinsics, *ty)?, ptr, ""))
730                })
731                .collect::<Result<Vec<_>, CompileError>>()?;
732
733            if self.abi.is_sret(func_sig)? {
734                let sret = trampoline_func
735                    .get_first_param()
736                    .unwrap()
737                    .into_pointer_value();
738
739                let basic_types: Vec<_> = func_sig
740                    .results()
741                    .iter()
742                    .map(|&ty| type_to_llvm(intrinsics, ty))
743                    .collect::<Result<_, _>>()?;
744                let mut struct_value = context.struct_type(&basic_types, false).get_undef();
745
746                for (idx, value) in results.iter().enumerate() {
747                    let value = err!(builder.build_bit_cast(
748                        *value,
749                        type_to_llvm(intrinsics, func_sig.results()[idx])?,
750                        "",
751                    ));
752                    struct_value =
753                        err!(builder.build_insert_value(struct_value, value, idx as u32, ""))
754                            .into_struct_value();
755                }
756                err!(builder.build_store(sret, struct_value));
757                err!(builder.build_return(None));
758            } else {
759                err!(
760                    builder.build_return(Some(&self.abi.pack_values_for_register_return(
761                        intrinsics,
762                        &builder,
763                        results.as_slice(),
764                        &trampoline_func.get_type(),
765                    )?))
766                );
767            }
768        }
769
770        Ok(())
771    }
772}
