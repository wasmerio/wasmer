37pub struct FuncTrampoline {
38    ctx: Context,
39    target_machine: TargetMachine,
40    target_triple: Triple,
41    abi: Box<dyn Abi>,
42    binary_fmt: BinaryFormat,

46const FUNCTION_SECTION_ELF: &str = "__TEXT,wasmer_trmpl"; // Needs to be between 1 and 16 chars
47const FUNCTION_SECTION_MACHO: &str = "wasmer_trmpl"; // Needs to be between 1 and 16 chars

49impl FuncTrampoline {
50    pub fn new(
51        target_machine: TargetMachine,
52        target_triple: Triple,
53        binary_fmt: BinaryFormat,
54    ) -> Result<Self, CompileError> {
55        let abi = get_abi(&target_machine);
56        Ok(Self {
57            ctx: Context::create(),
58            target_machine,
59            target_triple,
60            abi,
61            func_section: match binary_fmt {
62                BinaryFormat::Elf => FUNCTION_SECTION_ELF.to_string(),
63                BinaryFormat::Macho => FUNCTION_SECTION_MACHO.to_string(),
64                _ => {
65                    return Err(CompileError::UnsupportedTarget(format!(
66                        "Unsupported binary format: {binary_fmt:?}",
67                    )));
68                }
69            },
70            binary_fmt,
71        })
72    }

74    pub fn trampoline_to_module(
75        &self,
76        ty: &FuncType,
77        config: &LLVM,
78        name: &str,
79        compile_info: &CompileModuleInfo,
80    ) -> Result<Module<'_>, CompileError> {
81        // The function type, used for the callbacks.
82        let function = CompiledKind::FunctionCallTrampoline(ty.clone());
83        let module = self.ctx.create_module("");
84        let target_machine = &self.target_machine;
85        let target_triple = target_machine.get_triple();
86        let target_data: inkwell::targets::TargetData = target_machine.get_target_data();
87        module.set_triple(&target_triple);
88        module.set_data_layout(&target_data.get_data_layout());
89        let intrinsics = Intrinsics::declare(
90            &module,
91            &self.ctx,
92            &target_data,
93            &self.target_triple,
94            &self.binary_fmt,
95        );
96
97        let func_kind = if config.enable_g0m0_opt {
98            Some(G0M0FunctionKind::Local)
99        } else {
100            None
101        };
102
103        let (callee_ty, callee_attrs) =
104            self.abi
105                .func_type_to_llvm(&self.ctx, &intrinsics, None, ty, func_kind)?;
106        let trampoline_ty = intrinsics.void_ty.fn_type(
107            &[
108                intrinsics.ptr_ty.into(), // vmctx ptr
109                intrinsics.ptr_ty.into(), // callee function address
110                intrinsics.ptr_ty.into(), // in/out values ptr
111            ],
112            false,
113        );
114
115        let trampoline_func = module.add_function(name, trampoline_ty, Some(Linkage::External));
116        trampoline_func
117            .as_global_value()
118            .set_section(Some(&self.func_section));
119        trampoline_func
120            .as_global_value()
121            .set_linkage(Linkage::DLLExport);
122        trampoline_func
123            .as_global_value()
124            .set_dll_storage_class(DLLStorageClass::Export);
125        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.uwtable);
126        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.frame_pointer);
127        self.generate_trampoline(
128            config,
129            compile_info,
130            trampoline_func,
131            ty,
132            callee_ty,
133            &callee_attrs,
134            &self.ctx,
135            &intrinsics,
136        )?;
137
138        if let Some(ref callbacks) = config.callbacks {
139            callbacks.preopt_ir(&function, &module);
140        }
141
142        let mut passes = vec![];
143
144        if config.enable_verifier {
145            passes.push("verify");
146        }
147
148        passes.push("instcombine");
149        module
150            .run_passes(
151                passes.join(",").as_str(),
152                target_machine,
153                PassBuilderOptions::create(),
154            )
155            .unwrap();
156
157        if let Some(ref callbacks) = config.callbacks {
158            callbacks.postopt_ir(&function, &module);
159        }
160        Ok(module)
161    }

163    pub fn trampoline(
164        &self,
165        ty: &FuncType,
166        config: &LLVM,
167        name: &str,
168        compile_info: &CompileModuleInfo,
169    ) -> Result<FunctionBody, CompileError> {
170        let module = self.trampoline_to_module(ty, config, name, compile_info)?;
171        let function = CompiledKind::FunctionCallTrampoline(ty.clone());
172        let target_machine = &self.target_machine;
173
174        let memory_buffer = target_machine
175            .write_to_memory_buffer(&module, FileType::Object)
176            .unwrap();
177
178        if let Some(ref callbacks) = config.callbacks {
179            callbacks.obj_memory_buffer(&function, &memory_buffer);
180            let asm_buffer = target_machine
181                .write_to_memory_buffer(&module, FileType::Assembly)
182                .unwrap();
183            callbacks.asm_memory_buffer(&function, &asm_buffer);
184        }
185
186        let mem_buf_slice = memory_buffer.as_slice();
187
188        // Use a dummy function index to detect relocations against the trampoline
189        // function's address, which shouldn't exist and are not supported.
190        // Note, we just drop all custom sections, and verify that the function
191        // body itself has no relocations at all. This value should never be
192        // touched at all. However, it is set up so that if we do touch it (maybe
193        // due to someone changing the code later on), it'll explode, which is desirable!
194        let dummy_reloc_target =
195            RelocationTarget::DynamicTrampoline(FunctionIndex::from_u32(u32::MAX - 1));
196
197        // Note: we don't count .gcc_except_table here because native-to-wasm
198        // trampolines are not supposed to generate any LSDA sections. We *want* them
199        // to terminate libunwind's stack searches.
200        let CompiledFunction {
201            compiled_function,
202            custom_sections,
203            eh_frame_section_indices,
204            mut compact_unwind_section_indices,
205            ..
206        } = load_object_file(
207            mem_buf_slice,
208            &self.func_section,
209            dummy_reloc_target,
210            |name: &str| {
211                Err(CompileError::Codegen(format!(
212                    "trampoline generation produced reference to unknown function {name}",
213                )))
214            },
215            self.binary_fmt,
216        )?;
217        let mut all_sections_are_eh_sections = true;
218        let mut unwind_section_indices = eh_frame_section_indices;
219        unwind_section_indices.append(&mut compact_unwind_section_indices);
220        if unwind_section_indices.len() != custom_sections.len() {
221            all_sections_are_eh_sections = false;
222        } else {
223            unwind_section_indices.sort_unstable();
224            for (idx, section_idx) in unwind_section_indices.iter().enumerate() {
225                if idx as u32 != section_idx.as_u32() {
226                    all_sections_are_eh_sections = false;
227                    break;
228                }
229            }
230        }
231        if !all_sections_are_eh_sections {
232            return Err(CompileError::Codegen(
233                "trampoline generation produced non-eh custom sections".into(),
234            ));
235        }
236        if !compiled_function.relocations.is_empty() {
237            return Err(CompileError::Codegen(
238                "trampoline generation produced relocations".into(),
239            ));
240        }
241        // Ignore CompiledFunctionFrameInfo. Extra frame info isn't a problem.
242
243        Ok(FunctionBody {
244            body: compiled_function.body.body,
245            unwind_info: compiled_function.body.unwind_info,
246        })
247    }

249    pub fn dynamic_trampoline_to_module(
250        &self,
251        ty: &FuncType,
252        config: &LLVM,
253        name: &str,
254    ) -> Result<Module<'_>, CompileError> {
255        // The function type, used for the callbacks
256        let function = CompiledKind::DynamicFunctionTrampoline(ty.clone());
257        let module = self.ctx.create_module("");
258        let target_machine = &self.target_machine;
259        let target_data = target_machine.get_target_data();
260        let target_triple = target_machine.get_triple();
261        module.set_triple(&target_triple);
262        module.set_data_layout(&target_data.get_data_layout());
263        let intrinsics = Intrinsics::declare(
264            &module,
265            &self.ctx,
266            &target_data,
267            &self.target_triple,
268            &self.binary_fmt,
269        );
270
271        let (trampoline_ty, trampoline_attrs) =
272            self.abi
273                .func_type_to_llvm(&self.ctx, &intrinsics, None, ty, None)?;
274        let trampoline_func = module.add_function(name, trampoline_ty, Some(Linkage::External));
275        trampoline_func.set_personality_function(intrinsics.personality);
276        trampoline_func.add_attribute(AttributeLoc::Function, intrinsics.frame_pointer);
277        for (attr, attr_loc) in trampoline_attrs {
278            trampoline_func.add_attribute(attr_loc, attr);
279        }
280        trampoline_func
281            .as_global_value()
282            .set_section(Some(&self.func_section));
283        trampoline_func
284            .as_global_value()
285            .set_linkage(Linkage::DLLExport);
286        trampoline_func
287            .as_global_value()
288            .set_dll_storage_class(DLLStorageClass::Export);
289        self.generate_dynamic_trampoline(trampoline_func, ty, &self.ctx, &intrinsics)?;
290
291        if let Some(ref callbacks) = config.callbacks {
292            callbacks.preopt_ir(&function, &module);
293        }
294
295        let mut passes = vec![];
296
297        if config.enable_verifier {
298            passes.push("verify");
299        }
300
301        passes.push("early-cse");
302        module
303            .run_passes(
304                passes.join(",").as_str(),
305                target_machine,
306                PassBuilderOptions::create(),
307            )
308            .unwrap();
309
310        if let Some(ref callbacks) = config.callbacks {
311            callbacks.postopt_ir(&function, &module);
312        }
313
314        Ok(module)
315    }

317    #[allow(clippy::too_many_arguments)]
318    pub fn dynamic_trampoline(
319        &self,
320        ty: &FuncType,
321        config: &LLVM,
322        name: &str,
323        dynamic_trampoline_index: u32,
324        final_module_custom_sections: &mut PrimaryMap<SectionIndex, CustomSection>,
325        eh_frame_section_bytes: &mut Vec<u8>,
326        eh_frame_section_relocations: &mut Vec<Relocation>,
327        compact_unwind_section_bytes: &mut Vec<u8>,
328        compact_unwind_section_relocations: &mut Vec<Relocation>,
329    ) -> Result<FunctionBody, CompileError> {
330        let function = CompiledKind::DynamicFunctionTrampoline(ty.clone());
331        let target_machine = &self.target_machine;
332
333        let module = self.dynamic_trampoline_to_module(ty, config, name)?;
334
335        let memory_buffer = target_machine
336            .write_to_memory_buffer(&module, FileType::Object)
337            .unwrap();
338
339        if let Some(ref callbacks) = config.callbacks {
340            callbacks.obj_memory_buffer(&function, &memory_buffer);
341            let asm_buffer = target_machine
342                .write_to_memory_buffer(&module, FileType::Assembly)
343                .unwrap();
344            callbacks.asm_memory_buffer(&function, &asm_buffer)
345        }
346
347        let mem_buf_slice = memory_buffer.as_slice();
348        let CompiledFunction {
349            compiled_function,
350            custom_sections,
351            eh_frame_section_indices,
352            compact_unwind_section_indices,
353            gcc_except_table_section_indices,
354            data_dw_ref_personality_section_indices,
355        } = load_object_file(
356            mem_buf_slice,
357            &self.func_section,
358            RelocationTarget::DynamicTrampoline(FunctionIndex::from_u32(dynamic_trampoline_index)),
359            |name: &str| {
360                Err(CompileError::Codegen(format!(
361                    "trampoline generation produced reference to unknown function {name}",
362                )))
363            },
364            self.binary_fmt,
365        )?;
366
367        if !compiled_function.relocations.is_empty() {
368            return Err(CompileError::Codegen(
369                "trampoline generation produced relocations".into(),
370            ));
371        }
372        // Ignore CompiledFunctionFrameInfo. Extra frame info isn't a problem.
373
374        // Also append EH-related sections to the final module, since we expect
375        // dynamic trampolines to participate in unwinding
376        {
377            let first_section = final_module_custom_sections.len() as u32;
378            for (section_index, mut custom_section) in custom_sections.into_iter() {
379                for reloc in &mut custom_section.relocations {
380                    if let RelocationTarget::CustomSection(index) = reloc.reloc_target {
381                        reloc.reloc_target = RelocationTarget::CustomSection(
382                            SectionIndex::from_u32(first_section + index.as_u32()),
383                        )
384                    }
385
386                    if reloc.kind.needs_got() {
387                        return Err(CompileError::Codegen(
388                            "trampoline generation produced GOT relocation".into(),
389                        ));
390                    }
391                }
392
393                if eh_frame_section_indices.contains(&section_index) {
394                    let offset = eh_frame_section_bytes.len() as u32;
395                    for reloc in &mut custom_section.relocations {
396                        reloc.offset += offset;
397                    }
398                    eh_frame_section_bytes.extend_from_slice(custom_section.bytes.as_slice());
399                    // Terminate the eh_frame info with a zero-length CIE.
400                    eh_frame_section_bytes.extend_from_slice(&[0, 0, 0, 0]);
401                    eh_frame_section_relocations.extend(custom_section.relocations);
402                    // TODO: we do this to keep the count right, remove it.
403                    final_module_custom_sections.push(CustomSection {
404                        protection: CustomSectionProtection::Read,
405                        alignment: None,
406                        bytes: SectionBody::new_with_vec(vec![]),
407                        relocations: vec![],
408                    });
409                } else if compact_unwind_section_indices.contains(&section_index) {
410                    let offset = compact_unwind_section_bytes.len() as u32;
411                    for reloc in &mut custom_section.relocations {
412                        reloc.offset += offset;
413                    }
414                    compact_unwind_section_bytes.extend_from_slice(custom_section.bytes.as_slice());
415                    compact_unwind_section_relocations.extend(custom_section.relocations);
416                    // TODO: we do this to keep the count right, remove it.
417                    final_module_custom_sections.push(CustomSection {
418                        protection: CustomSectionProtection::Read,
419                        alignment: None,
420                        bytes: SectionBody::new_with_vec(vec![]),
421                        relocations: vec![],
422                    });
423                } else if gcc_except_table_section_indices.contains(&section_index)
424                    || data_dw_ref_personality_section_indices.contains(&section_index)
425                {
426                    final_module_custom_sections.push(custom_section);
427                } else {
428                    return Err(CompileError::Codegen(
429                        "trampoline generation produced non-eh custom sections".into(),
430                    ));
431                }
432            }
433        }
434
435        Ok(FunctionBody {
436            body: compiled_function.body.body,
437            unwind_info: compiled_function.body.unwind_info,
438        })
439    }

441    #[allow(clippy::too_many_arguments)]
442    fn generate_trampoline<'ctx>(
443        &self,
444        config: &LLVM,
445        compile_info: &CompileModuleInfo,
446        trampoline_func: FunctionValue,
447        func_sig: &FuncType,
448        llvm_func_type: FunctionType,
449        func_attrs: &[(Attribute, AttributeLoc)],
450        context: &'ctx Context,
451        intrinsics: &Intrinsics<'ctx>,
452    ) -> Result<(), CompileError> {
453        let entry_block = context.append_basic_block(trampoline_func, "entry");
454        let builder = context.create_builder();
455        builder.position_at_end(entry_block);
456
457        let (callee_vmctx_ptr, func_ptr, args_rets_ptr) =
458            match *trampoline_func.get_params().as_slice() {
459                [callee_vmctx_ptr, func_ptr, args_rets_ptr] => (
460                    callee_vmctx_ptr,
461                    func_ptr.into_pointer_value(),
462                    args_rets_ptr.into_pointer_value(),
463                ),
464                _ => {
465                    return Err(CompileError::Codegen(
466                        "trampoline function unimplemented".to_string(),
467                    ));
468                }
469            };
470
471        let mut args_vec: Vec<BasicMetadataValueEnum> =
472            Vec::with_capacity(if config.enable_g0m0_opt {
473                func_sig.params().len() + 3
474            } else {
475                func_sig.params().len() + 1
476            });
477
478        if self.abi.is_sret(func_sig)? {
479            let basic_types: Vec<_> = func_sig
480                .results()
481                .iter()
482                .map(|&ty| type_to_llvm(intrinsics, ty))
483                .collect::<Result<_, _>>()?;
484
485            let sret_ty = context.struct_type(&basic_types, false);
486            args_vec.push(err!(builder.build_alloca(sret_ty, "sret")).into());
487        }
488
489        args_vec.push(callee_vmctx_ptr.into());
490
491        if config.enable_g0m0_opt {
492            let wasm_module = &compile_info.module;
493            let memory_styles = &compile_info.memory_styles;
494            let callee_vmctx_ptr_value = callee_vmctx_ptr.into_pointer_value();
495            // get value of G0, get a pointer to M0's base
496
497            let offsets = wasmer_vm::VMOffsets::new(8, wasm_module);
498
499            let global_index = wasmer_types::GlobalIndex::from_u32(0);
500            let global_type = wasm_module.globals[global_index];
501            let global_value_type = global_type.ty;
502            let global_mutability = global_type.mutability;
503
504            let offset =
505                if let Some(local_global_index) = wasm_module.local_global_index(global_index) {
506                    offsets.vmctx_vmglobal_definition(local_global_index)
507                } else {
508                    offsets.vmctx_vmglobal_import(global_index)
509                };
510            let offset = intrinsics.i32_ty.const_int(offset.into(), false);
511            let global_ptr = {
512                let global_ptr_ptr = unsafe {
513                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
514                };
515                let global_ptr_ptr =
516                    err!(builder.build_bit_cast(global_ptr_ptr, intrinsics.ptr_ty, ""))
517                        .into_pointer_value();
518
519                err!(builder.build_load(intrinsics.ptr_ty, global_ptr_ptr, "")).into_pointer_value()
520            };
521
522            let global_ptr = err!(builder.build_bit_cast(
523                global_ptr,
524                type_to_llvm_ptr(intrinsics, global_value_type)?,
525                "",
526            ))
527            .into_pointer_value();
528
529            let global_value = match global_mutability {
530                wasmer_types::Mutability::Const => {
531                    err!(builder.build_load(
532                        type_to_llvm(intrinsics, global_value_type)?,
533                        global_ptr,
534                        "g0",
535                    ))
536                }
537                wasmer_types::Mutability::Var => {
538                    err!(builder.build_load(
539                        type_to_llvm(intrinsics, global_value_type)?,
540                        global_ptr,
541                        ""
542                    ))
543                }
544            };
545
546            global_value.set_name("trmpl_g0");
547            args_vec.push(global_value.into());
548
549            // load mem
550            let memory_index = wasmer_types::MemoryIndex::from_u32(0);
551            let memory_definition_ptr = if let Some(local_memory_index) =
552                wasm_module.local_memory_index(memory_index)
553            {
554                let offset = offsets.vmctx_vmmemory_definition(local_memory_index);
555                let offset = intrinsics.i32_ty.const_int(offset.into(), false);
556                unsafe {
557                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
558                }
559            } else {
560                let offset = offsets.vmctx_vmmemory_import(memory_index);
561                let offset = intrinsics.i32_ty.const_int(offset.into(), false);
562                let memory_definition_ptr_ptr = unsafe {
563                    err!(builder.build_gep(intrinsics.i8_ty, callee_vmctx_ptr_value, &[offset], ""))
564                };
565                let memory_definition_ptr_ptr =
566                    err!(builder.build_bit_cast(memory_definition_ptr_ptr, intrinsics.ptr_ty, "",))
567                        .into_pointer_value();
568
569                err!(builder.build_load(intrinsics.ptr_ty, memory_definition_ptr_ptr, ""))
570                    .into_pointer_value()
571            };
572            let memory_definition_ptr =
573                err!(builder.build_bit_cast(memory_definition_ptr, intrinsics.ptr_ty, "",))
574                    .into_pointer_value();
575            let base_ptr = err!(builder.build_struct_gep(
576                intrinsics.vmmemory_definition_ty,
577                memory_definition_ptr,
578                intrinsics.vmmemory_definition_base_element,
579                "",
580            ));
581
582            let memory_style = &memory_styles[memory_index];
583            let base_ptr = if let MemoryStyle::Dynamic { .. } = memory_style {
584                base_ptr
585            } else {
586                err!(builder.build_load(intrinsics.ptr_ty, base_ptr, "")).into_pointer_value()
587            };
588
589            base_ptr.set_name("trmpl_m0_base_ptr");
590
591            args_vec.push(base_ptr.into());
592        }
593
594        for (i, param_ty) in func_sig.params().iter().enumerate() {
595            let index = intrinsics.i32_ty.const_int(i as _, false);
596            let item_pointer = unsafe {
597                err!(builder.build_in_bounds_gep(
598                    intrinsics.i128_ty,
599                    args_rets_ptr,
600                    &[index],
601                    "arg_ptr"
602                ))
603            };
604
605            let casted_type = type_to_llvm(intrinsics, *param_ty)?;
606            let casted_pointer_type = type_to_llvm_ptr(intrinsics, *param_ty)?;
607
608            let typed_item_pointer = err!(builder.build_pointer_cast(
609                item_pointer,
610                casted_pointer_type,
611                "typed_arg_pointer"
612            ));
613
614            let arg = err!(builder.build_load(casted_type, typed_item_pointer, "arg"));
615            args_vec.push(arg.into());
616        }
617
618        let call_site = err!(builder.build_indirect_call(
619            llvm_func_type,
620            func_ptr,
621            args_vec.as_slice(),
622            "call"
623        ));
624        for (attr, attr_loc) in func_attrs {
625            call_site.add_attribute(*attr_loc, *attr);
626        }
627
628        let rets = self
629            .abi
630            .rets_from_call(&builder, intrinsics, call_site, func_sig)?;
631        for (idx, v) in rets.into_iter().enumerate() {
632            let ptr = unsafe {
633                err!(builder.build_gep(
634                    intrinsics.i128_ty,
635                    args_rets_ptr,
636                    &[intrinsics.i32_ty.const_int(idx as u64, false)],
637                    "",
638                ))
639            };
640            let ptr = err!(builder.build_pointer_cast(
641                ptr,
642                self.ctx.ptr_type(AddressSpace::default()),
643                ""
644            ));
645            err!(builder.build_store(ptr, v));
646        }
647
648        err!(builder.build_return(None));
649        Ok(())
650    }

652    fn generate_dynamic_trampoline<'ctx>(
653        &self,
654        trampoline_func: FunctionValue,
655        func_sig: &FuncType,
656        context: &'ctx Context,
657        intrinsics: &Intrinsics<'ctx>,
658    ) -> Result<(), CompileError> {
659        let entry_block = context.append_basic_block(trampoline_func, "entry");
660        let builder = context.create_builder();
661        builder.position_at_end(entry_block);
662
663        // Allocate stack space for the params and results.
664        let values = err!(builder.build_alloca(
665            intrinsics.i128_ty.array_type(cmp::max(
666                func_sig.params().len().try_into().unwrap(),
667                func_sig.results().len().try_into().unwrap(),
668            )),
669            "",
670        ));
671
672        // Copy params to 'values'.
673        let first_user_param = if self.abi.is_sret(func_sig)? { 2 } else { 1 };
674        for i in 0..func_sig.params().len() {
675            let ptr = unsafe {
676                err!(builder.build_in_bounds_gep(
677                    intrinsics.i128_ty,
678                    values,
679                    &[intrinsics.i32_ty.const_int(i.try_into().unwrap(), false)],
680                    "args",
681                ))
682            };
683            let ptr = err!(builder.build_bit_cast(
684                ptr,
685                type_to_llvm_ptr(intrinsics, func_sig.params()[i])?,
686                ""
687            ))
688            .into_pointer_value();
689            err!(
690                builder.build_store(
691                    ptr,
692                    trampoline_func
693                        .get_nth_param(i as u32 + first_user_param)
694                        .unwrap(),
695                )
696            );
697        }
698
699        let callee_ptr_ty = intrinsics.void_ty.fn_type(
700            &[
701                intrinsics.ptr_ty.into(), // vmctx ptr
702                intrinsics.ptr_ty.into(), // in/out values ptr
703            ],
704            false,
705        );
706        let vmctx = self.abi.get_vmctx_ptr_param(&trampoline_func);
707        let callee_ty =
708            err!(builder.build_bit_cast(vmctx, self.ctx.ptr_type(AddressSpace::default()), ""));
709        let callee =
710            err!(builder.build_load(intrinsics.ptr_ty, callee_ty.into_pointer_value(), ""))
711                .into_pointer_value();
712
713        let values_ptr = err!(builder.build_pointer_cast(values, intrinsics.ptr_ty, ""));
714        err!(builder.build_indirect_call(
715            callee_ptr_ty,
716            callee,
717            &[vmctx.into(), values_ptr.into()],
718            "",
719        ));
720
721        if func_sig.results().is_empty() {
722            err!(builder.build_return(None));
723        } else {
724            let results = func_sig
725                .results()
726                .iter()
727                .enumerate()
728                .map(|(idx, ty)| {
729                    let ptr = unsafe {
730                        err!(builder.build_gep(
731                            intrinsics.i128_ty,
732                            values,
733                            &[intrinsics.i32_ty.const_int(idx.try_into().unwrap(), false)],
734                            "",
735                        ))
736                    };
737                    let ptr = err!(builder.build_pointer_cast(
738                        ptr,
739                        type_to_llvm_ptr(intrinsics, *ty)?,
740                        ""
741                    ));
742                    err_nt!(builder.build_load(type_to_llvm(intrinsics, *ty)?, ptr, ""))
743                })
744                .collect::<Result<Vec<_>, CompileError>>()?;
745
746            if self.abi.is_sret(func_sig)? {
747                let sret = trampoline_func
748                    .get_first_param()
749                    .unwrap()
750                    .into_pointer_value();
751
752                let basic_types: Vec<_> = func_sig
753                    .results()
754                    .iter()
755                    .map(|&ty| type_to_llvm(intrinsics, ty))
756                    .collect::<Result<_, _>>()?;
757                let mut struct_value = context.struct_type(&basic_types, false).get_undef();
758
759                for (idx, value) in results.iter().enumerate() {
760                    let value = err!(builder.build_bit_cast(
761                        *value,
762                        type_to_llvm(intrinsics, func_sig.results()[idx])?,
763                        "",
764                    ));
765                    struct_value =
766                        err!(builder.build_insert_value(struct_value, value, idx as u32, ""))
767                            .into_struct_value();
768                }
769                err!(builder.build_store(sret, struct_value));
770                err!(builder.build_return(None));
771            } else {
772                err!(
773                    builder.build_return(Some(&self.abi.pack_values_for_register_return(
774                        intrinsics,
775                        &builder,
776                        results.as_slice(),
777                        &trampoline_func.get_type(),
778                    )?))
779                );
780            }
781        }
782
783        Ok(())
784    }
785}
