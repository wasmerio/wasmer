36pub fn type_to_llvm<'ctx>(
37    intrinsics: &Intrinsics<'ctx>,
38    ty: Type,
39) -> Result<BasicTypeEnum<'ctx>, CompileError> {
40    match ty {
41        Type::I32 => Ok(intrinsics.i32_ty.as_basic_type_enum()),
42        Type::I64 => Ok(intrinsics.i64_ty.as_basic_type_enum()),
43        Type::F32 => Ok(intrinsics.f32_ty.as_basic_type_enum()),
44        Type::F64 => Ok(intrinsics.f64_ty.as_basic_type_enum()),
45        Type::V128 => Ok(intrinsics.i128_ty.as_basic_type_enum()),
46        Type::ExceptionRef => Ok(intrinsics.i32_ty.as_basic_type_enum()),
47        Type::FuncRef | Type::ExternRef => Ok(intrinsics.ptr_ty.as_basic_type_enum()),
48    }
49}

1346    pub fn memory(
1347        &mut self,
1348        index: MemoryIndex,
1349        intrinsics: &Intrinsics<'ctx>,
1350        module: &Module<'ctx>,
1351        memory_styles: &PrimaryMap<MemoryIndex, MemoryStyle>,
1352    ) -> Result<MemoryCache<'ctx>, CompileError> {
1353        let (cached_memories, wasm_module, ctx_ptr_value, cache_builder, offsets) = (
1354            &mut self.cached_memories,
1355            self.wasm_module,
1356            self.ctx_ptr_value,
1357            &self.cache_builder,
1358            &self.offsets,
1359        );
1360        let memory_style = &memory_styles[index];
1361        match cached_memories.get(&index) {
1362            Some(r) => Ok(*r),
1363            None => {
1364                let memory_definition_ptr =
1365                    if let Some(local_memory_index) = wasm_module.local_memory_index(index) {
1366                        let offset = offsets.vmctx_vmmemory_definition(local_memory_index);
1367                        let offset = intrinsics.i32_ty.const_int(offset.into(), false);
1368                        unsafe {
1369                            err!(cache_builder.build_gep(
1370                                intrinsics.i8_ty,
1371                                ctx_ptr_value,
1372                                &[offset],
1373                                ""
1374                            ))
1375                        }
1376                    } else {
1377                        let offset = offsets.vmctx_vmmemory_import(index);
1378                        let offset = intrinsics.i32_ty.const_int(offset.into(), false);
1379                        let memory_definition_ptr_ptr = unsafe {
1380                            err!(cache_builder.build_gep(
1381                                intrinsics.i8_ty,
1382                                ctx_ptr_value,
1383                                &[offset],
1384                                ""
1385                            ))
1386                        };
1387                        let memory_definition_ptr_ptr = err!(cache_builder.build_bit_cast(
1388                            memory_definition_ptr_ptr,
1389                            intrinsics.ptr_ty,
1390                            "",
1391                        ))
1392                        .into_pointer_value();
1393                        let memory_definition_ptr = err!(cache_builder.build_load(
1394                            intrinsics.ptr_ty,
1395                            memory_definition_ptr_ptr,
1396                            ""
1397                        ))
1398                        .into_pointer_value();
1399                        tbaa_label(
1400                            module,
1401                            intrinsics,
1402                            format!("memory {} definition", index.as_u32()),
1403                            memory_definition_ptr.as_instruction_value().unwrap(),
1404                        );
1405                        memory_definition_ptr
1406                    };
1407                let memory_definition_ptr = err!(cache_builder.build_bit_cast(
1408                    memory_definition_ptr,
1409                    intrinsics.ptr_ty,
1410                    "",
1411                ))
1412                .into_pointer_value();
1413                let base_ptr = err!(cache_builder.build_struct_gep(
1414                    intrinsics.vmmemory_definition_ty,
1415                    memory_definition_ptr,
1416                    intrinsics.vmmemory_definition_base_element,
1417                    "",
1418                ));
1419                let value = if let MemoryStyle::Dynamic { .. } = memory_style {
1420                    let current_length_ptr = err!(cache_builder.build_struct_gep(
1421                        intrinsics.vmmemory_definition_ty,
1422                        memory_definition_ptr,
1423                        intrinsics.vmmemory_definition_current_length_element,
1424                        "",
1425                    ));
1426                    MemoryCache::Dynamic {
1427                        ptr_to_base_ptr: base_ptr,
1428                        ptr_to_current_length: current_length_ptr,
1429                    }
1430                } else {
1431                    let base_ptr = err!(cache_builder.build_load(intrinsics.ptr_ty, base_ptr, ""))
1432                        .into_pointer_value();
1433                    tbaa_label(
1434                        module,
1435                        intrinsics,
1436                        format!("memory base_ptr {}", index.as_u32()),
1437                        base_ptr.as_instruction_value().unwrap(),
1438                    );
1439                    MemoryCache::Static { base_ptr }
1440                };
1441
1442                self.cached_memories.insert(index, value);
1443                Ok(*self.cached_memories.get(&index).unwrap())
1444            }
1445        }
1446    }

1448    fn build_table_prepare(
1449        table_index: TableIndex,
1450        intrinsics: &Intrinsics<'ctx>,
1451        module: &Module<'ctx>,
1452        wasm_module: &WasmerCompilerModule,
1453        ctx_ptr_value: PointerValue<'ctx>,
1454        offsets: &VMOffsets,
1455        builder: &Builder<'ctx>,
1456    ) -> Result<(PointerValue<'ctx>, PointerValue<'ctx>), CompileError> {
1457        if let Some(local_table_index) = wasm_module.local_table_index(table_index) {
1458            let offset = intrinsics.i64_ty.const_int(
1459                offsets
1460                    .vmctx_vmtable_definition_base(local_table_index)
1461                    .into(),
1462                false,
1463            );
1464            let ptr_to_base_ptr =
1465                unsafe { err!(builder.build_gep(intrinsics.i8_ty, ctx_ptr_value, &[offset], "")) };
1466            let ptr_to_base_ptr =
1467                err!(builder.build_bit_cast(ptr_to_base_ptr, intrinsics.ptr_ty, ""))
1468                    .into_pointer_value();
1469            let offset = intrinsics.i64_ty.const_int(
1470                offsets
1471                    .vmctx_vmtable_definition_current_elements(local_table_index)
1472                    .into(),
1473                false,
1474            );
1475            let ptr_to_bounds =
1476                unsafe { err!(builder.build_gep(intrinsics.i8_ty, ctx_ptr_value, &[offset], "")) };
1477            let ptr_to_bounds = err!(builder.build_bit_cast(ptr_to_bounds, intrinsics.ptr_ty, ""))
1478                .into_pointer_value();
1479            Ok((ptr_to_base_ptr, ptr_to_bounds))
1480        } else {
1481            let offset = intrinsics.i64_ty.const_int(
1482                offsets.vmctx_vmtable_import_definition(table_index).into(),
1483                false,
1484            );
1485            let definition_ptr_ptr =
1486                unsafe { err!(builder.build_gep(intrinsics.i8_ty, ctx_ptr_value, &[offset], "")) };
1487            let definition_ptr_ptr =
1488                err!(builder.build_bit_cast(definition_ptr_ptr, intrinsics.ptr_ty, "",))
1489                    .into_pointer_value();
1490            let definition_ptr =
1491                err!(builder.build_load(intrinsics.ptr_ty, definition_ptr_ptr, ""))
1492                    .into_pointer_value();
1493            tbaa_label(
1494                module,
1495                intrinsics,
1496                format!("table {} definition", table_index.as_u32()),
1497                definition_ptr.as_instruction_value().unwrap(),
1498            );
1499
1500            let offset = intrinsics
1501                .i64_ty
1502                .const_int(offsets.vmtable_definition_base().into(), false);
1503            let ptr_to_base_ptr =
1504                unsafe { err!(builder.build_gep(intrinsics.i8_ty, definition_ptr, &[offset], "")) };
1505            let ptr_to_base_ptr =
1506                err!(builder.build_bit_cast(ptr_to_base_ptr, intrinsics.ptr_ty, ""))
1507                    .into_pointer_value();
1508            let offset = intrinsics
1509                .i64_ty
1510                .const_int(offsets.vmtable_definition_current_elements().into(), false);
1511            let ptr_to_bounds =
1512                unsafe { err!(builder.build_gep(intrinsics.i8_ty, definition_ptr, &[offset], "")) };
1513            let ptr_to_bounds = err!(builder.build_bit_cast(ptr_to_bounds, intrinsics.ptr_ty, ""))
1514                .into_pointer_value();
1515            Ok((ptr_to_base_ptr, ptr_to_bounds))
1516        }
1517    }

1519    fn table_prepare(
1520        &mut self,
1521        table_index: TableIndex,
1522        intrinsics: &Intrinsics<'ctx>,
1523        module: &Module<'ctx>,
1524        body_builder: &Builder<'ctx>,
1525    ) -> Result<(PointerValue<'ctx>, PointerValue<'ctx>), CompileError> {
1526        let (cached_tables, wasm_module, ctx_ptr_value, cache_builder, offsets) = (
1527            &mut self.cached_tables,
1528            self.wasm_module,
1529            self.ctx_ptr_value,
1530            &self.cache_builder,
1531            &self.offsets,
1532        );
1533
1534        let is_growable = is_table_growable(wasm_module, table_index).ok_or_else(|| {
1535            CompileError::Codegen(format!(
1536                "Table index out of bounds: {}",
1537                table_index.as_u32()
1538            ))
1539        })?;
1540
1541        // If the table is growable, it may change, so we can't cache the pointers; they need to
1542        // go directly in the function body at the point where they're needed
1543        if is_growable {
1544            Self::build_table_prepare(
1545                table_index,
1546                intrinsics,
1547                module,
1548                wasm_module,
1549                ctx_ptr_value,
1550                offsets,
1551                body_builder,
1552            )
1553        } else {
1554            let TableCache {
1555                ptr_to_base_ptr,
1556                ptr_to_bounds,
1557            } = match cached_tables.entry(table_index) {
1558                Entry::Occupied(entry) => entry.get().clone(),
1559                Entry::Vacant(entry) => {
1560                    let (ptr_to_base_ptr, ptr_to_bounds) = Self::build_table_prepare(
1561                        table_index,
1562                        intrinsics,
1563                        module,
1564                        wasm_module,
1565                        ctx_ptr_value,
1566                        offsets,
1567                        cache_builder,
1568                    )?;
1569
1570                    let v = TableCache {
1571                        ptr_to_base_ptr,
1572                        ptr_to_bounds,
1573                    };
1574
1575                    entry.insert(v.clone());
1576
1577                    v
1578                }
1579            };
1580
1581            Ok((ptr_to_base_ptr, ptr_to_bounds))
1582        }
1583    }

1585    pub fn table(
1586        &mut self,
1587        index: TableIndex,
1588        intrinsics: &Intrinsics<'ctx>,
1589        module: &Module<'ctx>,
1590        body_builder: &Builder<'ctx>,
1591    ) -> Result<(PointerValue<'ctx>, IntValue<'ctx>), CompileError> {
1592        let (ptr_to_base_ptr, ptr_to_bounds) =
1593            self.table_prepare(index, intrinsics, module, body_builder)?;
1594
1595        // Safe to unwrap since an out-of-bounds index will be caught be table_prepare
1596        let builder = if is_table_growable(self.wasm_module, index).unwrap() {
1597            &body_builder
1598        } else {
1599            &self.cache_builder
1600        };
1601
1602        let base_ptr = err!(builder.build_load(intrinsics.ptr_ty, ptr_to_base_ptr, "base_ptr"))
1603            .into_pointer_value();
1604        let bounds =
1605            err!(builder.build_load(intrinsics.isize_ty, ptr_to_bounds, "bounds")).into_int_value();
1606        tbaa_label(
1607            module,
1608            intrinsics,
1609            format!("table_base_ptr {}", index.index()),
1610            base_ptr.as_instruction_value().unwrap(),
1611        );
1612        tbaa_label(
1613            module,
1614            intrinsics,
1615            format!("table_bounds {}", index.index()),
1616            bounds.as_instruction_value().unwrap(),
1617        );
1618        Ok((base_ptr, bounds))

2116// Given an instruction that operates on memory, mark the access as not aliasing
2117// other memory accesses which have a different label.
2118pub fn tbaa_label<'ctx>(
2119    module: &Module<'ctx>,
2120    intrinsics: &Intrinsics<'ctx>,
2121    label: String,
2122    instruction: InstructionValue<'ctx>,
2123) {
2124    // To convey to LLVM that two pointers must be pointing to distinct memory,
2125    // we use LLVM's Type Based Aliasing Analysis, or TBAA, to mark the memory
2126    // operations as having different types whose pointers may not alias.
2127    //
2128    // See the LLVM documentation at
2129    //   https://llvm.org/docs/LangRef.html#tbaa-metadata
2130    //
2131    // LLVM TBAA supports many features, but we use it in a simple way, with
2132    // only scalar types that are children of the root node. Every TBAA type we
2133    // declare is NoAlias with the others. See NoAlias, PartialAlias,
2134    // MayAlias and MustAlias in the LLVM documentation:
2135    //   https://llvm.org/docs/AliasAnalysis.html#must-may-and-no-alias-responses
2136
2137    let context = module.get_context();
2138
2139    // `!wasmer_tbaa_root = {}`, the TBAA root node for wasmer.
2140    let tbaa_root = module
2141        .get_global_metadata("wasmer_tbaa_root")
2142        .pop()
2143        .unwrap_or_else(|| {
2144            module
2145                .add_global_metadata("wasmer_tbaa_root", &context.metadata_node(&[]))
2146                .unwrap();
2147            module.get_global_metadata("wasmer_tbaa_root")[0]
2148        });
2149
2150    // Construct (or look up) the type descriptor, for example
2151    //   `!"local 0" = !{!"local 0", !wasmer_tbaa_root}`.
2152    let type_label = context.metadata_string(label.as_str());
2153    let type_tbaa = module
2154        .get_global_metadata(label.as_str())
2155        .pop()
2156        .unwrap_or_else(|| {
2157            module
2158                .add_global_metadata(
2159                    label.as_str(),
2160                    &context.metadata_node(&[type_label.into(), tbaa_root.into()]),
2161                )
2162                .unwrap();
2163            module.get_global_metadata(label.as_str())[0]
2164        });
2165
2166    // Construct (or look up) the access tag, which is a struct of the form
2167    // (base type, access type, offset).
2168    //
2169    // "If BaseTy is a scalar type, Offset must be 0 and BaseTy and AccessTy
2170    // must be the same".
2171    //   -- https://llvm.org/docs/LangRef.html#tbaa-metadata
2172    let label = label + "_memop";
2173    let type_tbaa = module
2174        .get_global_metadata(label.as_str())
2175        .pop()
2176        .unwrap_or_else(|| {
2177            module
2178                .add_global_metadata(
2179                    label.as_str(),
2180                    &context.metadata_node(&[
2181                        type_tbaa.into(),
2182                        type_tbaa.into(),
2183                        intrinsics.i64_zero.into(),
2184                    ]),
2185                )
2186                .unwrap();
2187            module.get_global_metadata(label.as_str())[0]
2188        });
2189
2190    // Attach the access tag to the instruction.
2191    let tbaa_kind = context.get_kind_id("tbaa");
2192    instruction.set_metadata(type_tbaa, tbaa_kind).unwrap();
2193}
