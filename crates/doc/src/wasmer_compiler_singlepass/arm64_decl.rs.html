332/// A machine register under the x86-64 architecture.
333#[derive(Copy, Clone, Debug, Eq, PartialEq)]
334#[allow(clippy::upper_case_acronyms)]
335pub enum ARM64Register {
336    /// General-purpose registers.
337    GPR(GPR),
338    /// NEON (floating point/SIMD) registers.
339    NEON(NEON),
340}
341
342impl CombinedRegister for ARM64Register {
343    /// Returns the index of the register.
344    fn to_index(&self) -> RegisterIndex {
345        match *self {
346            ARM64Register::GPR(x) => RegisterIndex(x as usize),
347            ARM64Register::NEON(x) => RegisterIndex(x as usize + 64),
348        }
349    }
350    /// Convert from a GPR register
351    fn from_gpr(x: u16) -> Self {
352        ARM64Register::GPR(GPR::from_index(x as usize).unwrap())
353    }
354    /// Convert from an SIMD register
355    fn from_simd(x: u16) -> Self {
356        ARM64Register::NEON(NEON::from_index(x as usize).unwrap())
357    }
358}

360/// An allocator that allocates registers for function arguments according to the System V ABI.
361#[derive(Default)]
362pub struct ArgumentRegisterAllocator {
363    n_gprs: usize,
364    n_neons: usize,
365}
366
367impl ArgumentRegisterAllocator {
368    /// Allocates a register for argument type `ty`. Returns `None` if no register is available for this type.
369    pub fn next(
370        &mut self,
371        ty: Type,
372        calling_convention: CallingConvention,
373    ) -> Result<Option<ARM64Register>, CompileError> {
374        let ret = match calling_convention {
375            CallingConvention::SystemV | CallingConvention::AppleAarch64 => {
376                static GPR_SEQ: &[GPR] = &[
377                    GPR::X0,
378                    GPR::X1,
379                    GPR::X2,
380                    GPR::X3,
381                    GPR::X4,
382                    GPR::X5,
383                    GPR::X6,
384                    GPR::X7,
385                ];
386                static NEON_SEQ: &[NEON] = &[
387                    NEON::V0,
388                    NEON::V1,
389                    NEON::V2,
390                    NEON::V3,
391                    NEON::V4,
392                    NEON::V5,
393                    NEON::V6,
394                    NEON::V7,
395                ];
396                match ty {
397                    Type::I32 | Type::I64 => {
398                        if self.n_gprs < GPR_SEQ.len() {
399                            let gpr = GPR_SEQ[self.n_gprs];
400                            self.n_gprs += 1;
401                            Some(ARM64Register::GPR(gpr))
402                        } else {
403                            None
404                        }
405                    }
406                    Type::F32 | Type::F64 => {
407                        if self.n_neons < NEON_SEQ.len() {
408                            let neon = NEON_SEQ[self.n_neons];
409                            self.n_neons += 1;
410                            Some(ARM64Register::NEON(neon))
411                        } else {
412                            None
413                        }
414                    }
415                    _ => {
416                        return Err(CompileError::Codegen(format!(
417                            "No register available for {calling_convention:?} and type {ty}"
418                        )));
419                    }
420                }
421            }
422            _ => {
423                return Err(CompileError::Codegen(format!(
424                    "No register available for {calling_convention:?} and type {ty}"
425                )));
426            }
427        };
428
429        Ok(ret)
430    }
431}
