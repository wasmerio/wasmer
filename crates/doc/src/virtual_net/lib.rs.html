349/// Connected sockets have a persistent connection to a remote peer
350pub trait VirtualConnectedSocket: VirtualSocket + fmt::Debug + Send + Sync + 'static {
351    /// Determines how long the socket will remain in a TIME_WAIT
352    /// after it disconnects (only the one that initiates the close will
353    /// be in a TIME_WAIT state thus the clients should always do this rather
354    /// than the server)
355    fn set_linger(&mut self, linger: Option<Duration>) -> Result<()>;
356
357    /// Returns how long the socket will remain in a TIME_WAIT
358    /// after it disconnects
359    fn linger(&self) -> Result<Option<Duration>>;
360
361    /// Tries to send out a datagram or stream of bytes on this socket
362    fn try_send(&mut self, data: &[u8]) -> Result<usize>;
363
364    // Tries to flush any data in the local buffers
365    fn try_flush(&mut self) -> Result<()>;
366
367    /// Closes the socket
368    fn close(&mut self) -> Result<()>;
369
370    /// Tries to read a packet from the socket
371    fn try_recv(&mut self, buf: &mut [MaybeUninit<u8>], peek: bool) -> Result<usize>;
372}

622pub trait VirtualTcpSocket: VirtualConnectedSocket + fmt::Debug + Send + Sync + 'static {
623    /// Sets the receive buffer size which acts as a throttle for how
624    /// much data is buffered on this side of the pipe
625    fn set_recv_buf_size(&mut self, size: usize) -> Result<()>;
626
627    /// Size of the receive buffer that holds all data that has not
628    /// yet been read
629    fn recv_buf_size(&self) -> Result<usize>;
630
631    /// Sets the size of the send buffer which will hold the bytes of
632    /// data while they are being sent over to the peer
633    fn set_send_buf_size(&mut self, size: usize) -> Result<()>;
634
635    /// Size of the send buffer that holds all data that is currently
636    /// being transmitted.
637    fn send_buf_size(&self) -> Result<usize>;
638
639    /// When NO_DELAY is set the data that needs to be transmitted to
640    /// the peer is sent immediately rather than waiting for a bigger
641    /// batch of data, this reduces latency but increases encapsulation
642    /// overhead.
643    fn set_nodelay(&mut self, reuse: bool) -> Result<()>;
644
645    /// Indicates if the NO_DELAY flag is set which means that data
646    /// is immediately sent to the peer without waiting. This reduces
647    /// latency but increases encapsulation overhead.
648    fn nodelay(&self) -> Result<bool>;
649
650    /// When KEEP_ALIVE is set the connection will periodically send
651    /// an empty data packet to the server to make sure the connection
652    /// stays alive.
653    fn set_keepalive(&mut self, keepalive: bool) -> Result<()>;
654
655    /// Indicates if the KEEP_ALIVE flag is set which means that the
656    /// socket will periodically send an empty data packet to keep
657    /// the connection alive.
658    fn keepalive(&self) -> Result<bool>;
659
660    /// When DONT_ROUTE is set the packet will be sent directly
661    /// to the interface without passing through the routing logic.
662    fn set_dontroute(&mut self, keepalive: bool) -> Result<()>;
663
664    /// Indicates if the packet will pass straight through to
665    /// the interface bypassing the routing logic.
666    fn dontroute(&self) -> Result<bool>;
667
668    /// Returns the address (IP and Port) of the peer socket that this
669    /// is connected to
670    fn addr_peer(&self) -> Result<SocketAddr>;
671
672    /// Shuts down either the READER or WRITER sides of the socket
673    /// connection.
674    fn shutdown(&mut self, how: Shutdown) -> Result<()>;

680#[cfg(feature = "tokio")]
681impl<'a> AsyncRead for Box<dyn VirtualTcpSocket + Sync + 'a> {
682    fn poll_read(
683        self: Pin<&mut Self>,
684        cx: &mut Context<'_>,
685        buf: &mut tokio::io::ReadBuf<'_>,
686    ) -> Poll<std::io::Result<()>> {
687        let this = self.get_mut();
688        let handler: Box<dyn InterestHandler + Send + Sync> = cx.waker().into();
689        if let Err(err) = this.set_handler(handler) {
690            return Poll::Ready(Err(net_error_into_io_err(err)));
691        }
692        let buf_unsafe = unsafe { buf.unfilled_mut() };
693        match this.try_recv(buf_unsafe, false) {
694            Ok(ret) => {
695                unsafe { buf.assume_init(ret) };
696                buf.set_filled(ret);
697                Poll::Ready(Ok(()))
698            }
699            Err(NetworkError::WouldBlock) => Poll::Pending,
700            Err(err) => Poll::Ready(Err(net_error_into_io_err(err))),
701        }
702    }
703}

705#[cfg(feature = "tokio")]
706impl<'a> AsyncWrite for Box<dyn VirtualTcpSocket + Sync + 'a> {
707    fn poll_write(
708        self: Pin<&mut Self>,
709        cx: &mut Context<'_>,
710        buf: &[u8],
711    ) -> Poll<std::io::Result<usize>> {
712        let this = self.get_mut();
713        let handler: Box<dyn InterestHandler + Send + Sync> = cx.waker().into();
714        if let Err(err) = this.set_handler(handler) {
715            return Poll::Ready(Err(net_error_into_io_err(err)));
716        }
717        match this.try_send(buf) {
718            Ok(ret) => Poll::Ready(Ok(ret)),
719            Err(NetworkError::WouldBlock) => Poll::Pending,
720            Err(err) => Poll::Ready(Err(net_error_into_io_err(err))),
721        }
722    }
723
724    fn poll_flush(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<std::io::Result<()>> {
725        let this = self.get_mut();
726        let handler: Box<dyn InterestHandler + Send + Sync> = cx.waker().into();
727        if let Err(err) = this.set_handler(handler) {
728            return Poll::Ready(Err(net_error_into_io_err(err)));
729        }
730        match this.try_flush() {
731            Ok(()) => Poll::Ready(Ok(())),
732            Err(NetworkError::WouldBlock) => Poll::Pending,
733            Err(err) => Poll::Ready(Err(net_error_into_io_err(err))),
734        }
735    }
736
737    fn poll_shutdown(mut self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<std::io::Result<()>> {
738        Poll::Ready(
739            self.shutdown(Shutdown::Write)
740                .map_err(net_error_into_io_err),
741        )
742    }
743}

889pub fn io_err_into_net_error(net_error: std::io::Error) -> NetworkError {
890    use std::io::ErrorKind;
891    match net_error.kind() {
892        ErrorKind::BrokenPipe => NetworkError::BrokenPipe,
893        ErrorKind::AlreadyExists => NetworkError::AlreadyExists,
894        ErrorKind::AddrInUse => NetworkError::AddressInUse,
895        ErrorKind::AddrNotAvailable => NetworkError::AddressNotAvailable,
896        ErrorKind::ConnectionAborted => NetworkError::ConnectionAborted,
897        ErrorKind::ConnectionRefused => NetworkError::ConnectionRefused,
898        ErrorKind::ConnectionReset => NetworkError::ConnectionReset,
899        ErrorKind::Interrupted => NetworkError::Interrupted,
900        ErrorKind::InvalidData => NetworkError::InvalidData,
901        ErrorKind::InvalidInput => NetworkError::InvalidInput,
902        ErrorKind::NotConnected => NetworkError::NotConnected,
903        ErrorKind::PermissionDenied => NetworkError::PermissionDenied,
904        ErrorKind::TimedOut => NetworkError::TimedOut,
905        ErrorKind::UnexpectedEof => NetworkError::UnexpectedEof,
906        ErrorKind::WouldBlock => NetworkError::WouldBlock,
907        ErrorKind::WriteZero => NetworkError::WriteZero,
908        ErrorKind::Unsupported => NetworkError::Unsupported,
909
910        #[cfg(all(target_family = "unix", feature = "libc"))]
911        _ => {
912            if let Some(code) = net_error.raw_os_error() {
913                match code {
914                    libc::EPERM => NetworkError::PermissionDenied,
915                    libc::EBADF => NetworkError::InvalidFd,
916                    libc::ECHILD => NetworkError::InvalidFd,
917                    libc::EMFILE => NetworkError::TooManyOpenFiles,
918                    libc::EINTR => NetworkError::Interrupted,
919                    libc::EIO => NetworkError::IOError,
920                    libc::ENXIO => NetworkError::IOError,
921                    libc::EAGAIN => NetworkError::WouldBlock,
922                    libc::ENOMEM => NetworkError::InsufficientMemory,
923                    libc::EACCES => NetworkError::PermissionDenied,
924                    libc::ENODEV => NetworkError::NoDevice,
925                    libc::EINVAL => NetworkError::InvalidInput,
926                    libc::EPIPE => NetworkError::BrokenPipe,
927                    err => {
928                        tracing::trace!("unknown os error {}", err);
929                        NetworkError::UnknownError
930                    }
931                }
932            } else {
933                NetworkError::UnknownError
934            }
935        }
936        #[cfg(not(all(target_family = "unix", feature = "libc")))]
937        _ => NetworkError::UnknownError,
938    }
939}

941pub fn net_error_into_io_err(net_error: NetworkError) -> std::io::Error {
942    use std::io::ErrorKind;
943    match net_error {
944        NetworkError::InvalidFd => ErrorKind::BrokenPipe.into(),
945        NetworkError::AlreadyExists => ErrorKind::AlreadyExists.into(),
946        NetworkError::Lock => ErrorKind::BrokenPipe.into(),
947        NetworkError::IOError => ErrorKind::BrokenPipe.into(),
948        NetworkError::AddressInUse => ErrorKind::AddrInUse.into(),
949        NetworkError::AddressNotAvailable => ErrorKind::AddrNotAvailable.into(),
950        NetworkError::BrokenPipe => ErrorKind::BrokenPipe.into(),
951        NetworkError::ConnectionAborted => ErrorKind::ConnectionAborted.into(),
952        NetworkError::ConnectionRefused => ErrorKind::ConnectionRefused.into(),
953        NetworkError::ConnectionReset => ErrorKind::ConnectionReset.into(),
954        NetworkError::Interrupted => ErrorKind::Interrupted.into(),
955        NetworkError::InvalidData => ErrorKind::InvalidData.into(),
956        NetworkError::InvalidInput => ErrorKind::InvalidInput.into(),
957        NetworkError::NotConnected => ErrorKind::NotConnected.into(),
958        NetworkError::NoDevice => ErrorKind::BrokenPipe.into(),
959        NetworkError::PermissionDenied => ErrorKind::PermissionDenied.into(),
960        NetworkError::TimedOut => ErrorKind::TimedOut.into(),
961        NetworkError::UnexpectedEof => ErrorKind::UnexpectedEof.into(),
962        NetworkError::WouldBlock => ErrorKind::WouldBlock.into(),
963        NetworkError::WriteZero => ErrorKind::WriteZero.into(),
964        NetworkError::Unsupported => ErrorKind::Unsupported.into(),
965        NetworkError::UnknownError => ErrorKind::BrokenPipe.into(),
966        NetworkError::InsufficientMemory => ErrorKind::OutOfMemory.into(),
967        NetworkError::TooManyOpenFiles => {
968            #[cfg(all(target_family = "unix", feature = "libc"))]
969            {
970                std::io::Error::from_raw_os_error(libc::EMFILE)
971            }
972            #[cfg(not(all(target_family = "unix", feature = "libc")))]
973            {
974                ErrorKind::Other.into()
975            }
976        }
977    }
