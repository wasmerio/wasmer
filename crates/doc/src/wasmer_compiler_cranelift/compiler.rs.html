65/// A compiler that compiles a WebAssembly module with Cranelift, translating the Wasm to Cranelift IR,
66/// optimizing it and then translating to assembly.
67#[derive(Debug)]
68pub struct CraneliftCompiler {
69    config: Cranelift,
70}

83    // Helper function to create an easy scope boundary for the thread pool used
84    // in [`Self::compile_module`].
85    fn compile_module_internal(
86        &self,
87        target: &Target,
88        compile_info: &CompileModuleInfo,
89        module_translation_state: &ModuleTranslationState,
90        function_body_inputs: PrimaryMap<LocalFunctionIndex, FunctionBodyData<'_>>,
91        progress_callback: Option<&CompilationProgressCallback>,
92    ) -> Result<Compilation, CompileError> {
93        let isa = self
94            .config()
95            .isa(target)
96            .map_err(|error| CompileError::Codegen(error.to_string()))?;
97        let frontend_config = isa.frontend_config();
98        #[cfg(feature = "unwind")]
99        let pointer_bytes = frontend_config.pointer_bytes();
100        let memory_styles = &compile_info.memory_styles;
101        let table_styles = &compile_info.table_styles;
102        let module = &compile_info.module;
103        let signatures = module
104            .signatures
105            .iter()
106            .map(|(_sig_index, func_type)| signature_to_cranelift_ir(func_type, frontend_config))
107            .collect::<PrimaryMap<SignatureIndex, ir::Signature>>();
108
109        let total_functions = function_body_inputs.len() as u64;
110        let total_function_call_trampolines = module.signatures.len() as u64;
111        let total_dynamic_trampolines = module.num_imported_functions as u64;
112        let total_steps =
113            total_functions + total_function_call_trampolines + total_dynamic_trampolines;
114        let progress = progress_callback
115            .cloned()
116            .map(|cb| ProgressContext::new(cb, total_steps, "cranelift::functions"));
117
118        // Generate the frametable
119        #[cfg(feature = "unwind")]
120        let dwarf_frametable = if function_body_inputs.is_empty() {
121            // If we have no function body inputs, we don't need to
122            // construct the `FrameTable`. Constructing it, with empty
123            // FDEs will cause some issues in Linux.
124            None
125        } else {
126            match target.triple().default_calling_convention() {
127                Ok(CallingConvention::SystemV) => match isa.create_systemv_cie() {
128                    Some(mut cie) => {
129                        cie.personality = Some((
130                            DW_EH_PE_absptr,
131                            Address::Symbol {
132                                symbol: WriterRelocate::PERSONALITY_SYMBOL,
133                                addend: 0,
134                            },
135                        ));
136                        cie.lsda_encoding = Some(DW_EH_PE_absptr);
137                        let mut dwarf_frametable = FrameTable::default();
138                        let cie_id = dwarf_frametable.add_cie(cie);
139                        Some((dwarf_frametable, cie_id))
140                    }
141                    // Even though we are in a SystemV system, Cranelift doesn't support it
142                    None => None,
143                },
144                _ => None,
145            }
146        };
147
148        // The `compile_function` closure is used for both the sequential and
149        // parallel compilation paths to avoid code duplication.
150        let compile_function =
151            |func_translator: &mut FuncTranslator,
152             (i, input): (&LocalFunctionIndex, &FunctionBodyData)| {
153                let func_index = module.func_index(*i);
154                let mut context = Context::new();
155                let mut func_env = FuncEnvironment::new(
156                    isa.frontend_config(),
157                    module,
158                    &signatures,
159                    memory_styles,
160                    table_styles,
161                );
162                context.func.name = match get_function_name(&mut context.func, func_index) {
163                    ExternalName::User(nameref) => {
164                        if context.func.params.user_named_funcs().is_valid(nameref) {
165                            let name = &context.func.params.user_named_funcs()[nameref];
166                            UserFuncName::User(name.clone())
167                        } else {
168                            UserFuncName::default()
169                        }
170                    }
171                    ExternalName::TestCase(testcase) => UserFuncName::Testcase(testcase),
172                    _ => UserFuncName::default(),
173                };
174                context.func.signature = signatures[module.functions[func_index]].clone();
175                // if generate_debug_info {
176                //     context.func.collect_debug_info();
177                // }
178
179                let mut reader =
180                    MiddlewareBinaryReader::new_with_offset(input.data, input.module_offset);
181                reader.set_middleware_chain(
182                    self.config
183                        .middlewares
184                        .generate_function_middleware_chain(*i),
185                );
186
187                func_translator.translate(
188                    module_translation_state,
189                    &mut reader,
190                    &mut context.func,
191                    &mut func_env,
192                    *i,
193                )?;
194
195                if let Some(callbacks) = self.config.callbacks.as_ref() {
196                    use wasmer_compiler::misc::CompiledKind;
197
198                    callbacks.preopt_ir(
199                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
200                        context.func.display().to_string().as_bytes(),
201                    );
202                }
203
204                let mut code_buf: Vec<u8> = Vec::new();
205                let mut ctrl_plane = Default::default();
206                let func_name_map = context.func.params.user_named_funcs().clone();
207                let result = context
208                    .compile(&*isa, &mut ctrl_plane)
209                    .map_err(|error| CompileError::Codegen(format!("{error:#?}")))?;
210                code_buf.extend_from_slice(result.code_buffer());
211
212                if let Some(callbacks) = self.config.callbacks.as_ref() {
213                    use wasmer_compiler::misc::CompiledKind;
214
215                    callbacks.obj_memory_buffer(
216                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
217                        &code_buf,
218                    );
219                    callbacks.asm_memory_buffer(
220                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
221                        target.triple().architecture,
222                        &code_buf,
223                    )?;
224                }
225
226                let func_relocs = result
227                    .buffer
228                    .relocs()
229                    .iter()
230                    .map(|r| mach_reloc_to_reloc(module, &func_name_map, r))
231                    .collect::<Vec<_>>();
232
233                let traps = result
234                    .buffer
235                    .traps()
236                    .iter()
237                    .map(mach_trap_to_trap)
238                    .collect::<Vec<_>>();
239
240                #[cfg(feature = "unwind")]
241                let function_lsda = if dwarf_frametable.is_some() {
242                    build_function_lsda(
243                        result.buffer.call_sites(),
244                        result.buffer.data().len(),
245                        pointer_bytes,
246                    )
247                } else {
248                    None
249                };
250
251                #[cfg(not(feature = "unwind"))]
252                let function_lsda = ();
253
254                let (unwind_info, fde) = match compiled_function_unwind_info(&*isa, &context)? {
255                    #[cfg(feature = "unwind")]
256                    CraneliftUnwindInfo::Fde(fde) => {
257                        if dwarf_frametable.is_some() {
258                            let fde = fde.to_fde(Address::Symbol {
259                                // The symbol is the kind of relocation.
260                                // "0" is used for functions
261                                symbol: WriterRelocate::FUNCTION_SYMBOL,
262                                // We use the addend as a way to specify the
263                                // function index
264                                addend: i.index() as _,
265                            });
266                            // The unwind information is inserted into the dwarf section
267                            (Some(CompiledFunctionUnwindInfo::Dwarf), Some(fde))
268                        } else {
269                            (None, None)
270                        }
271                    }
272                    #[cfg(feature = "unwind")]
273                    other => (other.maybe_into_to_windows_unwind(), None),
274
275                    // This is a bit hacky, but necessary since gimli is not
276                    // available when the "unwind" feature is disabled.
277                    #[cfg(not(feature = "unwind"))]
278                    other => (other.maybe_into_to_windows_unwind(), None::<()>),
279                };
280
281                let range = reader.range();
282                let address_map = get_function_address_map(&context, range, code_buf.len());
283
284                if let Some(progress) = progress.as_ref() {
285                    progress.notify()?;
286                }
287
288                Ok((
289                    CompiledFunction {
290                        body: FunctionBody {
291                            body: code_buf,
292                            unwind_info,
293                        },
294                        relocations: func_relocs,
295                        frame_info: CompiledFunctionFrameInfo { address_map, traps },
296                    },
297                    fde,
298                    function_lsda,
299                ))
300            };
301
302        #[cfg_attr(not(feature = "unwind"), allow(unused_mut))]
303        let mut custom_sections = PrimaryMap::new();
304
305        #[cfg(not(feature = "rayon"))]
306        let mut func_translator = FuncTranslator::new();
307        #[cfg(not(feature = "rayon"))]
308        let results = function_body_inputs
309            .iter()
310            .collect::<Vec<(LocalFunctionIndex, &FunctionBodyData<'_>)>>()
311            .into_iter()
312            .map(|(i, input)| compile_function(&mut func_translator, (&i, input)))
313            .collect::<Result<Vec<_>, CompileError>>()?;
314        #[cfg(feature = "rayon")]
315        let results = function_body_inputs
316            .iter()
317            .collect::<Vec<(LocalFunctionIndex, &FunctionBodyData<'_>)>>()
318            .par_iter()
319            .map_init(FuncTranslator::new, |func_translator, &(i, input)| {
320                compile_function(func_translator, (&i, input))
321            })
322            .collect::<Result<Vec<_>, CompileError>>()?;
323
324        let mut functions = Vec::with_capacity(function_body_inputs.len());
325        let mut fdes = Vec::with_capacity(function_body_inputs.len());
326        let mut lsda_data = Vec::with_capacity(function_body_inputs.len());
327
328        for (func, fde, lsda) in results {
329            functions.push(func);
330            fdes.push(fde);
331            lsda_data.push(lsda);
332        }
333
334        #[cfg(feature = "unwind")]
335        let (_tag_section_index, lsda_section_index, function_lsda_offsets) =
336            if dwarf_frametable.is_some() {
337                let mut tag_section_index = None;
338                let mut tag_offsets = HashMap::new();
339                if let Some((tag_section, offsets)) = build_tag_section(&lsda_data) {
340                    custom_sections.push(tag_section);
341                    tag_section_index = Some(SectionIndex::new(custom_sections.len() - 1));
342                    tag_offsets = offsets;
343                }
344                let lsda_vec = lsda_data;
345                let (lsda_section, offsets_per_function) =
346                    build_lsda_section(lsda_vec, pointer_bytes, &tag_offsets, tag_section_index);
347                let mut lsda_section_index = None;
348                if let Some(section) = lsda_section {
349                    custom_sections.push(section);
350                    lsda_section_index = Some(SectionIndex::new(custom_sections.len() - 1));
351                }
352                (tag_section_index, lsda_section_index, offsets_per_function)
353            } else {
354                (None, None, vec![None; functions.len()])
355            };
356
357        #[cfg_attr(not(feature = "unwind"), allow(unused_mut))]
358        let mut unwind_info = UnwindInfo::default();
359
360        #[cfg(feature = "unwind")]
361        if let Some((mut dwarf_frametable, cie_id)) = dwarf_frametable {
362            for (func_idx, fde_opt) in fdes.into_iter().enumerate() {
363                if let Some(mut fde) = fde_opt {
364                    let has_lsda = function_lsda_offsets
365                        .get(func_idx)
366                        .and_then(|v| *v)
367                        .is_some();
368                    let lsda_address = if has_lsda {
369                        debug_assert!(
370                            lsda_section_index.is_some(),
371                            "LSDA offsets require an LSDA section"
372                        );
373                        if lsda_section_index.is_some() {
374                            let symbol =
375                                WriterRelocate::lsda_symbol(LocalFunctionIndex::new(func_idx));
376                            Address::Symbol { symbol, addend: 0 }
377                        } else {
378                            Address::Constant(0)
379                        }
380                    } else {
381                        Address::Constant(0)
382                    };
383                    fde.lsda = Some(lsda_address);
384                    dwarf_frametable.add_fde(cie_id, fde);
385                }
386            }
387
388            let mut writer = WriterRelocate::new(target.triple().endianness().ok());
389            if let Some(lsda_section_index) = lsda_section_index {
390                for (func_idx, offset) in function_lsda_offsets.iter().enumerate() {
391                    if let Some(offset) = offset {
392                        writer.register_lsda_symbol(
393                            WriterRelocate::lsda_symbol(LocalFunctionIndex::new(func_idx)),
394                            RelocationTarget::CustomSection(lsda_section_index),
395                            *offset,
396                        );
397                    }
398                }
399            }
400
401            let mut eh_frame = EhFrame(writer);
402            dwarf_frametable.write_eh_frame(&mut eh_frame).unwrap();
403            eh_frame.write(&[0, 0, 0, 0]).unwrap(); // Write a 0 length at the end of the table.
404
405            let eh_frame_section = eh_frame.0.into_section();
406            custom_sections.push(eh_frame_section);
407            unwind_info.eh_frame = Some(SectionIndex::new(custom_sections.len() - 1));
408        };
409
410        #[cfg(not(feature = "unwind"))]
411        let _ = fdes;
412
413        // function call trampolines (only for local functions, by signature)
414        #[cfg(not(feature = "rayon"))]
415        let mut cx = FunctionBuilderContext::new();
416        #[cfg(not(feature = "rayon"))]
417        let function_call_trampolines = module
418            .signatures
419            .values()
420            .collect::<Vec<_>>()
421            .into_iter()
422            .map(|sig| {
423                let trampoline = make_trampoline_function_call(
424                    &self.config().callbacks,
425                    &*isa,
426                    target.triple().architecture,
427                    &mut cx,
428                    sig,
429                )?;
430                if let Some(progress) = progress.as_ref() {
431                    progress.notify()?;
432                }
433                Ok(trampoline)
434            })
435            .collect::<Result<Vec<FunctionBody>, CompileError>>()?
436            .into_iter()
437            .collect();
438        #[cfg(feature = "rayon")]
439        let function_call_trampolines = module
440            .signatures
441            .values()
442            .collect::<Vec<_>>()
443            .par_iter()
444            .map_init(FunctionBuilderContext::new, |cx, sig| {
445                let trampoline = make_trampoline_function_call(
446                    &self.config().callbacks,
447                    &*isa,
448                    target.triple().architecture,
449                    cx,
450                    sig,
451                )?;
452                if let Some(progress) = progress.as_ref() {
453                    progress.notify()?;
454                }
455                Ok(trampoline)
456            })
457            .collect::<Result<Vec<FunctionBody>, CompileError>>()?
458            .into_iter()
459            .collect();
460
461        use wasmer_types::VMOffsets;
462        let offsets = VMOffsets::new_for_trampolines(frontend_config.pointer_bytes());
463        // dynamic function trampolines (only for imported functions)
464        #[cfg(not(feature = "rayon"))]
465        let mut cx = FunctionBuilderContext::new();
466        #[cfg(not(feature = "rayon"))]
467        let dynamic_function_trampolines = module
468            .imported_function_types()
469            .collect::<Vec<_>>()
470            .into_iter()
471            .map(|func_type| {
472                let trampoline = make_trampoline_dynamic_function(
473                    &self.config().callbacks,
474                    &*isa,
475                    target.triple().architecture,
476                    &offsets,
477                    &mut cx,
478                    &func_type,
479                )?;
480                if let Some(progress) = progress.as_ref() {
481                    progress.notify()?;
482                }
483                Ok(trampoline)
484            })
485            .collect::<Result<Vec<_>, CompileError>>()?
486            .into_iter()
487            .collect();
488        #[cfg(feature = "rayon")]
489        let dynamic_function_trampolines = module
490            .imported_function_types()
491            .collect::<Vec<_>>()
492            .par_iter()
493            .map_init(FunctionBuilderContext::new, |cx, func_type| {
494                let trampoline = make_trampoline_dynamic_function(
495                    &self.config().callbacks,
496                    &*isa,
497                    target.triple().architecture,
498                    &offsets,
499                    cx,
500                    func_type,
501                )?;
502                if let Some(progress) = progress.as_ref() {
503                    progress.notify()?;
504                }
505                Ok(trampoline)
506            })
507            .collect::<Result<Vec<_>, CompileError>>()?
508            .into_iter()
509            .collect();
510
511        let got = wasmer_compiler::types::function::GOT::empty();
512
513        Ok(Compilation {
514            functions: functions.into_iter().collect(),
515            custom_sections,
516            function_call_trampolines,
517            dynamic_function_trampolines,
518            unwind_info,
519            got,
520        })
521    }

534        String::from("cranelift")

538    fn get_middlewares(&self) -> &[Arc<dyn ModuleMiddleware>] {
539        &self.config.middlewares

542    /// Compile the module using Cranelift, producing a compilation result with
543    /// associated relocations.
544    fn compile_module(
545        &self,
546        target: &Target,
547        compile_info: &CompileModuleInfo,
548        module_translation_state: &ModuleTranslationState,
549        function_body_inputs: PrimaryMap<LocalFunctionIndex, FunctionBodyData<'_>>,
550        progress_callback: Option<&CompilationProgressCallback>,
551    ) -> Result<Compilation, CompileError> {
552        #[cfg(feature = "rayon")]
553        {
554            let num_threads = self.config.num_threads.get();
555            let pool = rayon::ThreadPoolBuilder::new()
556                .num_threads(num_threads)
557                .build()
558                .unwrap();
559
560            pool.install(|| {
561                self.compile_module_internal(
562                    target,
563                    compile_info,
564                    module_translation_state,
565                    function_body_inputs,
566                    progress_callback,
567                )
568            })
569        }
570
571        #[cfg(not(feature = "rayon"))]
572        {
573            self.compile_module_internal(
574                target,
575                compile_info,
576                module_translation_state,
577                function_body_inputs,
578                progress_callback,
579            )
580        }
581    }
582}

584fn mach_reloc_to_reloc(
585    module: &ModuleInfo,
586    func_index_map: &cranelift_entity::PrimaryMap<ir::UserExternalNameRef, ir::UserExternalName>,
587    reloc: &FinalizedMachReloc,
588) -> Relocation {
589    let FinalizedMachReloc {
590        offset,
591        kind,
592        addend,
593        target,
594    } = &reloc;
595    let name = match target {
596        FinalizedRelocTarget::ExternalName(external_name) => external_name,
597        FinalizedRelocTarget::Func(_) => {
598            unimplemented!("relocations to offset in the same function are not yet supported")
599        }
600    };
601    let reloc_target: RelocationTarget = if let ExternalName::User(extname_ref) = name {
602        let func_index = func_index_map[*extname_ref].index;
603        //debug_assert_eq!(namespace, 0);
604        RelocationTarget::LocalFunc(
605            module
606                .local_func_index(FunctionIndex::from_u32(func_index))
607                .expect("The provided function should be local"),
608        )
609    } else if let ExternalName::LibCall(libcall) = name {
610        RelocationTarget::LibCall(irlibcall_to_libcall(*libcall))
611    } else {
612        panic!("unrecognized external target")
613    };
614    Relocation {
615        kind: irreloc_to_relocationkind(*kind),
616        reloc_target,
617        offset: *offset,
618        addend: *addend,
619    }
620}

622fn mach_trap_to_trap(trap: &MachTrap) -> TrapInformation {
623    let &MachTrap { offset, code } = trap;
624    TrapInformation {
625        code_offset: offset,
626        trap_code: translate_ir_trapcode(code),
627    }
628}

630/// Translates the Cranelift IR TrapCode into generic Trap Code
631fn translate_ir_trapcode(trap: ir::TrapCode) -> TrapCode {
632    if trap == ir::TrapCode::STACK_OVERFLOW {
633        TrapCode::StackOverflow
634    } else if trap == ir::TrapCode::HEAP_OUT_OF_BOUNDS {
635        TrapCode::HeapAccessOutOfBounds
636    } else if trap == crate::TRAP_HEAP_MISALIGNED {
637        TrapCode::UnalignedAtomic
638    } else if trap == crate::TRAP_TABLE_OUT_OF_BOUNDS {
639        TrapCode::TableAccessOutOfBounds
640    } else if trap == crate::TRAP_INDIRECT_CALL_TO_NULL {
641        TrapCode::IndirectCallToNull
642    } else if trap == crate::TRAP_BAD_SIGNATURE {
643        TrapCode::BadSignature
644    } else if trap == ir::TrapCode::INTEGER_OVERFLOW {
645        TrapCode::IntegerOverflow
646    } else if trap == ir::TrapCode::INTEGER_DIVISION_BY_ZERO {
647        TrapCode::IntegerDivisionByZero
648    } else if trap == ir::TrapCode::BAD_CONVERSION_TO_INTEGER {
649        TrapCode::BadConversionToInteger
650    } else if trap == crate::TRAP_UNREACHABLE {
651        TrapCode::UnreachableCodeReached
652    } else if trap == crate::TRAP_INTERRUPT {
653        unimplemented!("Interrupts not supported")
654    } else if trap == crate::TRAP_NULL_REFERENCE || trap == crate::TRAP_NULL_I31_REF {
655        unimplemented!("Null reference not supported")
656    } else {
