56/// A compiler that compiles a WebAssembly module with Cranelift, translating the Wasm to Cranelift IR,
57/// optimizing it and then translating to assembly.
58#[derive(Debug)]
59pub struct CraneliftCompiler {
60    config: Cranelift,
61}
62
63impl CraneliftCompiler {
64    /// Creates a new Cranelift compiler
65    pub fn new(config: Cranelift) -> Self {
66        Self { config }
67    }
68
69    /// Gets the WebAssembly features for this Compiler
70    pub fn config(&self) -> &Cranelift {
71        &self.config
72    }

74    // Helper function to create an easy scope boundary for the thread pool used
75    // in [`Self::compile_module`].
76    fn compile_module_internal(
77        &self,
78        target: &Target,
79        compile_info: &CompileModuleInfo,
80        module_translation_state: &ModuleTranslationState,
81        function_body_inputs: PrimaryMap<LocalFunctionIndex, FunctionBodyData<'_>>,
82    ) -> Result<Compilation, CompileError> {
83        let isa = self
84            .config()
85            .isa(target)
86            .map_err(|error| CompileError::Codegen(error.to_string()))?;
87        let frontend_config = isa.frontend_config();
88        let memory_styles = &compile_info.memory_styles;
89        let table_styles = &compile_info.table_styles;
90        let module = &compile_info.module;
91        let signatures = module
92            .signatures
93            .iter()
94            .map(|(_sig_index, func_type)| signature_to_cranelift_ir(func_type, frontend_config))
95            .collect::<PrimaryMap<SignatureIndex, ir::Signature>>();
96
97        // Generate the frametable
98        #[cfg(feature = "unwind")]
99        let dwarf_frametable = if function_body_inputs.is_empty() {
100            // If we have no function body inputs, we don't need to
101            // construct the `FrameTable`. Constructing it, with empty
102            // FDEs will cause some issues in Linux.
103            None
104        } else {
105            match target.triple().default_calling_convention() {
106                Ok(CallingConvention::SystemV) => {
107                    match isa.create_systemv_cie() {
108                        Some(cie) => {
109                            let mut dwarf_frametable = FrameTable::default();
110                            let cie_id = dwarf_frametable.add_cie(cie);
111                            Some((dwarf_frametable, cie_id))
112                        }
113                        // Even though we are in a SystemV system, Cranelift doesn't support it
114                        None => None,
115                    }
116                }
117                _ => None,
118            }
119        };
120
121        let compile_function =
122            |func_translator: &mut FuncTranslator,
123             (i, input): (&LocalFunctionIndex, &FunctionBodyData)| {
124                let func_index = module.func_index(*i);
125                let mut context = Context::new();
126                let mut func_env = FuncEnvironment::new(
127                    isa.frontend_config(),
128                    module,
129                    &signatures,
130                    memory_styles,
131                    table_styles,
132                );
133                context.func.name = match get_function_name(&mut context.func, func_index) {
134                    ExternalName::User(nameref) => {
135                        if context.func.params.user_named_funcs().is_valid(nameref) {
136                            let name = &context.func.params.user_named_funcs()[nameref];
137                            UserFuncName::User(name.clone())
138                        } else {
139                            UserFuncName::default()
140                        }
141                    }
142                    ExternalName::TestCase(testcase) => UserFuncName::Testcase(testcase),
143                    _ => UserFuncName::default(),
144                };
145                context.func.signature = signatures[module.functions[func_index]].clone();
146                // if generate_debug_info {
147                //     context.func.collect_debug_info();
148                // }
149
150                let mut reader =
151                    MiddlewareBinaryReader::new_with_offset(input.data, input.module_offset);
152                reader.set_middleware_chain(
153                    self.config
154                        .middlewares
155                        .generate_function_middleware_chain(*i),
156                );
157
158                func_translator.translate(
159                    module_translation_state,
160                    &mut reader,
161                    &mut context.func,
162                    &mut func_env,
163                    *i,
164                )?;
165
166                if let Some(callbacks) = self.config.callbacks.as_ref() {
167                    use wasmer_compiler::misc::CompiledKind;
168
169                    callbacks.preopt_ir(
170                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
171                        context.func.display().to_string().as_bytes(),
172                    );
173                }
174
175                let mut code_buf: Vec<u8> = Vec::new();
176                let mut ctrl_plane = Default::default();
177                let func_name_map = context.func.params.user_named_funcs().clone();
178                let result = context
179                    .compile(&*isa, &mut ctrl_plane)
180                    .map_err(|error| CompileError::Codegen(format!("{error:#?}")))?;
181                code_buf.extend_from_slice(result.code_buffer());
182
183                if let Some(callbacks) = self.config.callbacks.as_ref() {
184                    use wasmer_compiler::misc::CompiledKind;
185
186                    callbacks.obj_memory_buffer(
187                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
188                        &code_buf,
189                    );
190                    callbacks.asm_memory_buffer(
191                        &CompiledKind::Local(*i, compile_info.module.get_function_name(func_index)),
192                        target.triple().architecture,
193                        &code_buf,
194                    )?;
195                }
196
197                let func_relocs = result
198                    .buffer
199                    .relocs()
200                    .iter()
201                    .map(|r| mach_reloc_to_reloc(module, &func_name_map, r))
202                    .collect::<Vec<_>>();
203
204                let traps = result
205                    .buffer
206                    .traps()
207                    .iter()
208                    .map(mach_trap_to_trap)
209                    .collect::<Vec<_>>();
210
211                let (unwind_info, fde) = match compiled_function_unwind_info(&*isa, &context)? {
212                    #[cfg(feature = "unwind")]
213                    CraneliftUnwindInfo::Fde(fde) => {
214                        if dwarf_frametable.is_some() {
215                            let fde = fde.to_fde(Address::Symbol {
216                                // The symbol is the kind of relocation.
217                                // "0" is used for functions
218                                symbol: WriterRelocate::FUNCTION_SYMBOL,
219                                // We use the addend as a way to specify the
220                                // function index
221                                addend: i.index() as _,
222                            });
223                            // The unwind information is inserted into the dwarf section
224                            (Some(CompiledFunctionUnwindInfo::Dwarf), Some(fde))
225                        } else {
226                            (None, None)
227                        }
228                    }
229                    #[cfg(feature = "unwind")]
230                    other => (other.maybe_into_to_windows_unwind(), None),
231
232                    // This is a bit hacky, but necessary since gimli is not
233                    // available when the "unwind" feature is disabled.
234                    #[cfg(not(feature = "unwind"))]
235                    other => (other.maybe_into_to_windows_unwind(), None::<()>),
236                };
237
238                let range = reader.range();
239                let address_map = get_function_address_map(&context, range, code_buf.len());
240
241                Ok((
242                    CompiledFunction {
243                        body: FunctionBody {
244                            body: code_buf,
245                            unwind_info,
246                        },
247                        relocations: func_relocs,
248                        frame_info: CompiledFunctionFrameInfo { address_map, traps },
249                    },
250                    fde,
251                ))
252            };
253
254        #[cfg_attr(not(feature = "unwind"), allow(unused_mut))]
255        let mut custom_sections = PrimaryMap::new();
256
257        #[cfg(not(feature = "rayon"))]
258        let mut func_translator = FuncTranslator::new();
259        #[cfg(not(feature = "rayon"))]
260        #[cfg_attr(not(feature = "unwind"), allow(unused_variables))]
261        let (functions, fdes): (Vec<CompiledFunction>, Vec<_>) = function_body_inputs
262            .iter()
263            .collect::<Vec<(LocalFunctionIndex, &FunctionBodyData<'_>)>>()
264            .into_iter()
265            .map(|(i, input)| compile_function(&mut func_translator, (&i, input)))
266            .collect::<Result<Vec<_>, CompileError>>()?
267            .into_iter()
268            .unzip();
269        #[cfg(feature = "rayon")]
270        #[cfg_attr(not(feature = "unwind"), allow(unused_variables))]
271        let (functions, fdes): (Vec<CompiledFunction>, Vec<_>) = function_body_inputs
272            .iter()
273            .collect::<Vec<(LocalFunctionIndex, &FunctionBodyData<'_>)>>()
274            .par_iter()
275            .map_init(FuncTranslator::new, |func_translator, &(i, input)| {
276                compile_function(func_translator, (&i, input))
277            })
278            .collect::<Result<Vec<_>, CompileError>>()?
279            .into_iter()
280            .unzip();
281
282        #[cfg_attr(not(feature = "unwind"), allow(unused_mut))]
283        let mut unwind_info = UnwindInfo::default();
284
285        #[cfg(feature = "unwind")]
286        if let Some((mut dwarf_frametable, cie_id)) = dwarf_frametable {
287            for fde in fdes.into_iter().flatten() {
288                dwarf_frametable.add_fde(cie_id, fde);
289            }
290            let mut eh_frame = EhFrame(WriterRelocate::new(target.triple().endianness().ok()));
291            dwarf_frametable.write_eh_frame(&mut eh_frame).unwrap();
292            eh_frame.write(&[0, 0, 0, 0]).unwrap(); // Write a 0 length at the end of the table.
293
294            let eh_frame_section = eh_frame.0.into_section();
295            custom_sections.push(eh_frame_section);
296            unwind_info.eh_frame = Some(SectionIndex::new(custom_sections.len() - 1));
297        };
298
299        // function call trampolines (only for local functions, by signature)
300        #[cfg(not(feature = "rayon"))]
301        let mut cx = FunctionBuilderContext::new();
302        #[cfg(not(feature = "rayon"))]
303        let function_call_trampolines = module
304            .signatures
305            .values()
306            .collect::<Vec<_>>()
307            .into_iter()
308            .map(|sig| {
309                make_trampoline_function_call(
310                    &self.config().callbacks,
311                    &*isa,
312                    target.triple().architecture,
313                    &mut cx,
314                    sig,
315                )
316            })
317            .collect::<Result<Vec<FunctionBody>, CompileError>>()?
318            .into_iter()
319            .collect();
320        #[cfg(feature = "rayon")]
321        let function_call_trampolines = module
322            .signatures
323            .values()
324            .collect::<Vec<_>>()
325            .par_iter()
326            .map_init(FunctionBuilderContext::new, |cx, sig| {
327                make_trampoline_function_call(
328                    &self.config().callbacks,
329                    &*isa,
330                    target.triple().architecture,
331                    cx,
332                    sig,
333                )
334            })
335            .collect::<Result<Vec<FunctionBody>, CompileError>>()?
336            .into_iter()
337            .collect();
338
339        use wasmer_types::VMOffsets;
340        let offsets = VMOffsets::new_for_trampolines(frontend_config.pointer_bytes());
341        // dynamic function trampolines (only for imported functions)
342        #[cfg(not(feature = "rayon"))]
343        let mut cx = FunctionBuilderContext::new();
344        #[cfg(not(feature = "rayon"))]
345        let dynamic_function_trampolines = module
346            .imported_function_types()
347            .collect::<Vec<_>>()
348            .into_iter()
349            .map(|func_type| {
350                make_trampoline_dynamic_function(
351                    &self.config().callbacks,
352                    &*isa,
353                    target.triple().architecture,
354                    &offsets,
355                    &mut cx,
356                    &func_type,
357                )
358            })
359            .collect::<Result<Vec<_>, CompileError>>()?
360            .into_iter()
361            .collect();
362        #[cfg(feature = "rayon")]
363        let dynamic_function_trampolines = module
364            .imported_function_types()
365            .collect::<Vec<_>>()
366            .par_iter()
367            .map_init(FunctionBuilderContext::new, |cx, func_type| {
368                make_trampoline_dynamic_function(
369                    &self.config().callbacks,
370                    &*isa,
371                    target.triple().architecture,
372                    &offsets,
373                    cx,
374                    func_type,
375                )
376            })
377            .collect::<Result<Vec<_>, CompileError>>()?
378            .into_iter()
379            .collect();
380
381        let got = wasmer_compiler::types::function::GOT::empty();
382
383        Ok(Compilation {
384            functions: functions.into_iter().collect(),
385            custom_sections,
386            function_call_trampolines,
387            dynamic_function_trampolines,
388            unwind_info,
389            got,
390        })
391    }
392}

394impl Compiler for CraneliftCompiler {
395    fn name(&self) -> &str {
396        "cranelift"
397    }
398
399    fn get_perfmap_enabled(&self) -> bool {
400        self.config.enable_perfmap
401    }
402
403    fn deterministic_id(&self) -> String {
404        String::from("cranelift")
405    }
406
407    /// Get the middlewares for this compiler
408    fn get_middlewares(&self) -> &[Arc<dyn ModuleMiddleware>] {
409        &self.config.middlewares
410    }
411
412    /// Compile the module using Cranelift, producing a compilation result with
413    /// associated relocations.
414    fn compile_module(
415        &self,
416        target: &Target,
417        compile_info: &CompileModuleInfo,
418        module_translation_state: &ModuleTranslationState,
419        function_body_inputs: PrimaryMap<LocalFunctionIndex, FunctionBodyData<'_>>,
420    ) -> Result<Compilation, CompileError> {
421        #[cfg(feature = "rayon")]
422        {
423            let num_threads = self.config.num_threads.get();
424            let pool = rayon::ThreadPoolBuilder::new()
425                .num_threads(num_threads)
426                .build()
427                .unwrap();
428
429            pool.install(|| {
430                self.compile_module_internal(
431                    target,
432                    compile_info,
433                    module_translation_state,
434                    function_body_inputs,
435                )
436            })
437        }
438
439        #[cfg(not(feature = "rayon"))]
440        {
441            self.compile_module_internal(
442                target,
443                compile_info,
444                module_translation_state,
445                function_body_inputs,
446            )
447        }
448    }
449}

451fn mach_reloc_to_reloc(
452    module: &ModuleInfo,
453    func_index_map: &cranelift_entity::PrimaryMap<ir::UserExternalNameRef, ir::UserExternalName>,
454    reloc: &FinalizedMachReloc,
455) -> Relocation {
456    let FinalizedMachReloc {
457        offset,
458        kind,
459        addend,
460        target,
461    } = &reloc;
462    let name = match target {
463        FinalizedRelocTarget::ExternalName(external_name) => external_name,
464        FinalizedRelocTarget::Func(_) => {
465            unimplemented!("relocations to offset in the same function are not yet supported")
466        }
467    };
468    let reloc_target: RelocationTarget = if let ExternalName::User(extname_ref) = name {
469        let func_index = func_index_map[*extname_ref].index;
470        //debug_assert_eq!(namespace, 0);
471        RelocationTarget::LocalFunc(
472            module
473                .local_func_index(FunctionIndex::from_u32(func_index))
474                .expect("The provided function should be local"),
475        )
476    } else if let ExternalName::LibCall(libcall) = name {
477        RelocationTarget::LibCall(irlibcall_to_libcall(*libcall))
478    } else {
479        panic!("unrecognized external target")
480    };
481    Relocation {
482        kind: irreloc_to_relocationkind(*kind),
483        reloc_target,
484        offset: *offset,
485        addend: *addend,
486    }
487}

489fn mach_trap_to_trap(trap: &MachTrap) -> TrapInformation {
490    let &MachTrap { offset, code } = trap;
491    TrapInformation {
492        code_offset: offset,
493        trap_code: translate_ir_trapcode(code),
494    }
495}
496
497/// Translates the Cranelift IR TrapCode into generic Trap Code
498fn translate_ir_trapcode(trap: ir::TrapCode) -> TrapCode {
499    if trap == ir::TrapCode::STACK_OVERFLOW {
500        TrapCode::StackOverflow
501    } else if trap == ir::TrapCode::HEAP_OUT_OF_BOUNDS {
502        TrapCode::HeapAccessOutOfBounds
503    } else if trap == crate::TRAP_HEAP_MISALIGNED {
504        TrapCode::UnalignedAtomic
505    } else if trap == crate::TRAP_TABLE_OUT_OF_BOUNDS {
506        TrapCode::TableAccessOutOfBounds
507    } else if trap == crate::TRAP_INDIRECT_CALL_TO_NULL {
508        TrapCode::IndirectCallToNull
509    } else if trap == crate::TRAP_BAD_SIGNATURE {
510        TrapCode::BadSignature
511    } else if trap == ir::TrapCode::INTEGER_OVERFLOW {
512        TrapCode::IntegerOverflow
513    } else if trap == ir::TrapCode::INTEGER_DIVISION_BY_ZERO {
514        TrapCode::IntegerDivisionByZero
515    } else if trap == ir::TrapCode::BAD_CONVERSION_TO_INTEGER {
516        TrapCode::BadConversionToInteger
517    } else if trap == crate::TRAP_UNREACHABLE {
518        TrapCode::UnreachableCodeReached
519    } else if trap == crate::TRAP_INTERRUPT {
520        unimplemented!("Interrupts not supported")
521    } else if trap == crate::TRAP_NULL_REFERENCE || trap == crate::TRAP_NULL_I31_REF {
522        unimplemented!("Null reference not supported")
523    } else {
524        unimplemented!("Trap code {trap:?} not supported")
525    }
