19/// Callbacks to the different Cranelift compilation phases.
20#[derive(Debug, Clone)]
21pub struct CraneliftCallbacks {
22    debug_dir: PathBuf,
23}
24
25impl CraneliftCallbacks {
26    /// Creates a new instance of `CraneliftCallbacks` with the specified debug directory.
27    pub fn new(debug_dir: PathBuf) -> Result<Self, io::Error> {
28        // Create the debug dir in case it doesn't exist
29        std::fs::create_dir_all(&debug_dir)?;
30        Ok(Self { debug_dir })
31    }
32
33    /// Writes the pre-optimization intermediate representation to a debug file.
34    pub fn preopt_ir(&self, kind: &CompiledKind, mem_buffer: &[u8]) {
35        let mut path = self.debug_dir.clone();
36        path.push(format!("{}.preopt.clif", function_kind_to_filename(kind)));
37        let mut file =
38            File::create(path).expect("Error while creating debug file from Cranelift IR");
39        file.write_all(mem_buffer).unwrap();
40    }
41
42    /// Writes the object file memory buffer to a debug file.
43    pub fn obj_memory_buffer(&self, kind: &CompiledKind, mem_buffer: &[u8]) {
44        let mut path = self.debug_dir.clone();
45        path.push(format!("{}.o", function_kind_to_filename(kind)));
46        let mut file =
47            File::create(path).expect("Error while creating debug file from Cranelift object");
48        file.write_all(mem_buffer).unwrap();
49    }
50}

54/// Possible optimization levels for the Cranelift codegen backend.
55#[non_exhaustive]
56#[derive(Clone, Debug)]
57pub enum CraneliftOptLevel {
58    /// No optimizations performed, minimizes compilation time by disabling most
59    /// optimizations.
60    None,
61    /// Generates the fastest possible code, but may take longer.
62    Speed,
63    /// Similar to `speed`, but also performs transformations aimed at reducing
64    /// code size.
65    SpeedAndSize,
66}

124    /// Generates the ISA for the provided target
125    pub fn isa(&self, target: &Target) -> CodegenResult<Arc<dyn TargetIsa>> {
126        let mut builder =
127            lookup(target.triple().clone()).expect("construct Cranelift ISA for triple");
128        // Cpu Features
129        let cpu_features = target.cpu_features();
130        if target.triple().architecture == Architecture::X86_64
131            && !cpu_features.contains(CpuFeature::SSE2)
132        {
133            panic!("x86 support requires SSE2");
134        }
135        if cpu_features.contains(CpuFeature::SSE3) {
136            builder.enable("has_sse3").expect("should be valid flag");
137        }
138        if cpu_features.contains(CpuFeature::SSSE3) {
139            builder.enable("has_ssse3").expect("should be valid flag");
140        }
141        if cpu_features.contains(CpuFeature::SSE41) {
142            builder.enable("has_sse41").expect("should be valid flag");
143        }
144        if cpu_features.contains(CpuFeature::SSE42) {
145            builder.enable("has_sse42").expect("should be valid flag");
146        }
147        if cpu_features.contains(CpuFeature::POPCNT) {
148            builder.enable("has_popcnt").expect("should be valid flag");
149        }
150        if cpu_features.contains(CpuFeature::AVX) {
151            builder.enable("has_avx").expect("should be valid flag");
152        }
153        if cpu_features.contains(CpuFeature::BMI1) {
154            builder.enable("has_bmi1").expect("should be valid flag");
155        }
156        if cpu_features.contains(CpuFeature::BMI2) {
157            builder.enable("has_bmi2").expect("should be valid flag");
158        }
159        if cpu_features.contains(CpuFeature::AVX2) {
160            builder.enable("has_avx2").expect("should be valid flag");
161        }
162        if cpu_features.contains(CpuFeature::AVX512DQ) {
163            builder
164                .enable("has_avx512dq")
165                .expect("should be valid flag");
166        }
167        if cpu_features.contains(CpuFeature::AVX512VL) {
168            builder
169                .enable("has_avx512vl")
170                .expect("should be valid flag");
171        }
172        if cpu_features.contains(CpuFeature::LZCNT) {
173            builder.enable("has_lzcnt").expect("should be valid flag");
174        }
175
176        builder.finish(self.flags())
177    }

179    /// Generates the flags for the compiler
180    pub fn flags(&self) -> settings::Flags {
181        let mut flags = settings::builder();
182
183        // Enable probestack
184        flags
185            .enable("enable_probestack")
186            .expect("should be valid flag");
187
188        // Always use inline stack probes (otherwise the call to Probestack needs to be relocated).
189        flags
190            .set("probestack_strategy", "inline")
191            .expect("should be valid flag");
192
193        if self.enable_pic {
194            flags.enable("is_pic").expect("should be a valid flag");
195        }
196
197        // We set up libcall trampolines in engine-universal.
198        // These trampolines are always reachable through short jumps.
199        flags
200            .enable("use_colocated_libcalls")
201            .expect("should be a valid flag");
202
203        // Allow Cranelift to implicitly spill multi-value returns via a hidden
204        // StructReturn argument when register results are exhausted.
205        flags
206            .enable("enable_multi_ret_implicit_sret")
207            .expect("should be a valid flag");
208
209        // Invert cranelift's default-on verification to instead default off.
210        let enable_verifier = if self.enable_verifier {
211            "true"
212        } else {
213            "false"
214        };
215        flags
216            .set("enable_verifier", enable_verifier)
217            .expect("should be valid flag");
218        flags
219            .set("enable_safepoints", "true")
220            .expect("should be valid flag");
221
222        flags
223            .set(
224                "opt_level",
225                match self.opt_level {
226                    CraneliftOptLevel::None => "none",
227                    CraneliftOptLevel::Speed => "speed",
228                    CraneliftOptLevel::SpeedAndSize => "speed_and_size",
229                },
230            )
231            .expect("should be valid flag");
232
233        let enable_nan_canonicalization = if self.enable_nan_canonicalization {
234            "true"
235        } else {
236            "false"
237        };
238        flags
239            .set("enable_nan_canonicalization", enable_nan_canonicalization)
240            .expect("should be valid flag");
241
242        settings::Flags::new(flags)
243    }
