4//! A trampoline generator for calling Wasm functions easily.
5//!
6//! That way, you can start calling Wasm functions doing things like:
7//! ```ignore
8//! let my_func = instance.exports.get("func");
9//! my_func.call([1, 2])
10//! ```
11use crate::{
12    CraneliftCallbacks,
13    translator::{compiled_function_unwind_info, signature_to_cranelift_ir},
14};
15use cranelift_codegen::{
16    Context,
17    ir::{self, InstBuilder},
18    isa::TargetIsa,
19};
20use cranelift_frontend::{FunctionBuilder, FunctionBuilderContext};
21use std::mem;
22use wasmer_compiler::{misc::CompiledKind, types::function::FunctionBody};
23use wasmer_types::{CompileError, FunctionType};
24
25/// Create a trampoline for invoking a WebAssembly function.
26pub fn make_trampoline_function_call(
27    callbacks: &Option<CraneliftCallbacks>,
28    isa: &dyn TargetIsa,
29    fn_builder_ctx: &mut FunctionBuilderContext,
30    func_type: &FunctionType,
31) -> Result<FunctionBody, CompileError> {
32    let pointer_type = isa.pointer_type();
33    let frontend_config = isa.frontend_config();
34    let signature = signature_to_cranelift_ir(func_type, frontend_config);
35    let mut wrapper_sig = ir::Signature::new(frontend_config.default_call_conv);
36
37    // Add the callee `vmctx` parameter.
38    wrapper_sig.params.push(ir::AbiParam::special(
39        pointer_type,
40        ir::ArgumentPurpose::VMContext,
41    ));
42
43    // Add the `callee_address` parameter.
44    wrapper_sig.params.push(ir::AbiParam::new(pointer_type));
45
46    // Add the `values_vec` parameter.
47    wrapper_sig.params.push(ir::AbiParam::new(pointer_type));
48
49    let mut context = Context::new();
50    context.func = ir::Function::with_name_signature(ir::UserFuncName::user(0, 0), wrapper_sig);
51
52    let value_size = mem::size_of::<u128>();
53    {
54        let mut builder = FunctionBuilder::new(&mut context.func, fn_builder_ctx);
55        let block0 = builder.create_block();
56
57        builder.append_block_params_for_function_params(block0);
58        builder.switch_to_block(block0);
59        builder.seal_block(block0);
60
61        let (vmctx_ptr_val, callee_value, values_vec_ptr_val) = {
62            let params = builder.func.dfg.block_params(block0);
63            (params[0], params[1], params[2])
64        };
65
66        // Load the argument values out of `values_vec`.
67        let mflags = ir::MemFlags::trusted();
68        let callee_args = signature
69            .params
70            .iter()
71            .enumerate()
72            .map(|(i, r)| {
73                match i {
74                    0 => vmctx_ptr_val,
75                    _ =>
76                    // i - 1 because vmctx is not passed through `values_vec`.
77                    {
78                        builder.ins().load(
79                            r.value_type,
80                            mflags,
81                            values_vec_ptr_val,
82                            ((i - 1) * value_size) as i32,
83                        )
84                    }
85                }
86            })
87            .collect::<Vec<_>>();
88
89        let new_sig = builder.import_signature(signature);
90
91        let call = builder
92            .ins()
93            .call_indirect(new_sig, callee_value, &callee_args);
94
95        let results = builder.func.dfg.inst_results(call).to_vec();
96
97        // Store the return values into `values_vec`.
98        let mflags = ir::MemFlags::trusted();
99        for (i, r) in results.iter().enumerate() {
100            builder
101                .ins()
102                .store(mflags, *r, values_vec_ptr_val, (i * value_size) as i32);
103        }
104
105        builder.ins().return_(&[]);
106        builder.finalize()
107    }
108
109    if let Some(callbacks) = callbacks.as_ref() {
110        callbacks.preopt_ir(
111            &CompiledKind::FunctionCallTrampoline(func_type.clone()),
112            context.func.display().to_string().as_bytes(),
113        );
114    }
115
116    let mut code_buf = Vec::new();
117    let mut ctrl_plane = Default::default();
118    let compiled = context
119        .compile(isa, &mut ctrl_plane)
120        .map_err(|error| CompileError::Codegen(error.inner.to_string()))?;
121    code_buf.extend_from_slice(compiled.code_buffer());
122
123    if let Some(callbacks) = callbacks.as_ref() {
124        callbacks.obj_memory_buffer(
125            &CompiledKind::FunctionCallTrampoline(func_type.clone()),
126            &code_buf,
127        );
128    }
129
130    let unwind_info = compiled_function_unwind_info(isa, &context)?.maybe_into_to_windows_unwind();
131
132    Ok(FunctionBody {
133        body: code_buf,
134        unwind_info,
135    })
136}
