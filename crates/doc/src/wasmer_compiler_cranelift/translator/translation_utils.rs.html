14/// Helper function translate a Function signature into Cranelift Ir
15pub fn signature_to_cranelift_ir(
16    signature: &FunctionType,
17    target_config: TargetFrontendConfig,
18) -> ir::Signature {
19    let mut sig = ir::Signature::new(target_config.default_call_conv);
20    sig.params.extend(signature.params().iter().map(|&ty| {
21        let cret_arg: ir::Type = type_to_irtype(ty, target_config)
22            .expect("only numeric types are supported in function signatures");
23        AbiParam::new(cret_arg)
24    }));
25    sig.returns.extend(signature.results().iter().map(|&ty| {
26        let cret_arg: ir::Type = type_to_irtype(ty, target_config)
27            .expect("only numeric types are supported in function signatures");
28        AbiParam::new(cret_arg)
29    }));
30    // The Vmctx signature
31    sig.params.insert(
32        0,
33        AbiParam::special(target_config.pointer_type(), ir::ArgumentPurpose::VMContext),
34    );
35    sig
36}

38/// Helper function translating wasmparser types to Cranelift types when possible.
39pub fn reference_type(target_config: TargetFrontendConfig) -> WasmResult<ir::Type> {
40    match target_config.pointer_type() {
41        ir::types::I32 => Ok(ir::types::R32),
42        ir::types::I64 => Ok(ir::types::R64),
43        _ => Err(WasmError::Unsupported(
44            "unsupported pointer type".to_string(),
45        )),
46    }
47}
48
49/// Helper function translating wasmparser types to Cranelift types when possible.
50pub fn type_to_irtype(ty: Type, target_config: TargetFrontendConfig) -> WasmResult<ir::Type> {
51    match ty {
52        Type::I32 => Ok(ir::types::I32),
53        Type::I64 => Ok(ir::types::I64),
54        Type::F32 => Ok(ir::types::F32),
55        Type::F64 => Ok(ir::types::F64),
56        Type::V128 => Ok(ir::types::I8X16),
57        Type::ExternRef | Type::FuncRef => reference_type(target_config),
58        Type::ExceptionRef => Err(wasm_unsupported!(
59            "exnrefs are not supported yet in cranelift"
60        )),
61        // ty => Err(wasm_unsupported!("type_to_type: wasm type {:?}", ty)),
62    }
63}

65/// Transform Cranelift LibCall into runtime LibCall
66pub fn irlibcall_to_libcall(libcall: ir::LibCall) -> LibCall {
67    match libcall {
68        ir::LibCall::Probestack => LibCall::Probestack,
69        ir::LibCall::CeilF32 => LibCall::CeilF32,
70        ir::LibCall::CeilF64 => LibCall::CeilF64,
71        ir::LibCall::FloorF32 => LibCall::FloorF32,
72        ir::LibCall::FloorF64 => LibCall::FloorF64,
73        ir::LibCall::TruncF32 => LibCall::TruncF32,
74        ir::LibCall::TruncF64 => LibCall::TruncF64,
75        ir::LibCall::NearestF32 => LibCall::NearestF32,
76        ir::LibCall::NearestF64 => LibCall::NearestF64,
77        _ => panic!("Unsupported libcall"),
78    }
79}

81/// Transform Cranelift Reloc to compiler Relocation
82pub fn irreloc_to_relocationkind(reloc: Reloc) -> RelocationKind {
83    match reloc {
84        Reloc::Abs4 => RelocationKind::Abs4,
85        Reloc::Abs8 => RelocationKind::Abs8,
86        Reloc::X86PCRel4 => RelocationKind::X86PCRel4,
87        Reloc::X86CallPCRel4 => RelocationKind::X86CallPCRel4,
88        Reloc::X86CallPLTRel4 => RelocationKind::X86CallPLTRel4,
89        Reloc::X86GOTPCRel4 => RelocationKind::X86GOTPCRel4,
90        Reloc::Arm64Call => RelocationKind::Arm64Call,
91        Reloc::RiscvCallPlt => RelocationKind::RiscvCall,
92        _ => panic!("The relocation {reloc} is not yet supported."),
93    }
94}

96/// Create a `Block` with the given Wasm parameters.
97pub fn block_with_params<'a, PE: TargetEnvironment + ?Sized>(
98    builder: &mut FunctionBuilder,
99    params: impl Iterator<Item = &'a wasmparser::ValType>,
100    environ: &PE,
101) -> WasmResult<ir::Block> {
102    let block = builder.create_block();
103    for ty in params.into_iter() {
104        match ty {
105            wasmparser::ValType::I32 => {
106                builder.append_block_param(block, ir::types::I32);
107            }
108            wasmparser::ValType::I64 => {
109                builder.append_block_param(block, ir::types::I64);
110            }
111            wasmparser::ValType::F32 => {
112                builder.append_block_param(block, ir::types::F32);
113            }
114            wasmparser::ValType::F64 => {
115                builder.append_block_param(block, ir::types::F64);
116            }
117            wasmparser::ValType::Ref(ty) => {
118                if ty.is_extern_ref() || ty.is_func_ref() {
119                    builder.append_block_param(block, environ.reference_type());
120                } else {
121                    return Err(WasmError::Unsupported(format!(
122                        "unsupported reference type: {ty:?}"
123                    )));
124                }
125            }
126            wasmparser::ValType::V128 => {
127                builder.append_block_param(block, ir::types::I8X16);
128            }
129        }
130    }
131    Ok(block)
132}

134/// Turns a `wasmparser` `f32` into a `Cranelift` one.
135pub fn f32_translation(x: wasmparser::Ieee32) -> ir::immediates::Ieee32 {
136    ir::immediates::Ieee32::with_bits(x.bits())
137}

141    ir::immediates::Ieee64::with_bits(x.bits())

144/// Special VMContext value label. It is tracked as 0xffff_fffe label.
145pub fn get_vmctx_value_label() -> ir::ValueLabel {
