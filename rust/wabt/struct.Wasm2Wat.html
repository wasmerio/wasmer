pub struct Wasm2Wat { /* fields omitted */ }

A builder for converting wasm binary to wasm text format.

Examples

extern crate wabt;
use wabt::Wasm2Wat;

fn main() {
    let wasm_text = Wasm2Wat::new()
        .fold_exprs(true)
        .inline_export(true)
        .convert(
            &[
                0, 97, 115, 109, // \0ASM - magic
                1, 0, 0, 0       //  0x01 - version
            ]
        ).unwrap();

}
