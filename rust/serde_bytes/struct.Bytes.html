    let mut cache = HashMap::new();
    cache.insert(3, Bytes::new(b"three"));
    cache.insert(2, Bytes::new(b"two"));
    cache.insert(1, Bytes::new(b"one"));

    bincode::serialize_into(&mut io::stdout(), &cache)
